1//! Anthropic HTTP client with retry + auth header injection.
2//!
3//! `post_messages` is the only outbound entrypoint. Owns a `reqwest::Client`
4//! with TLS + timeout configured by the plugin's `Configure::from_config`.
5//!
6//! Retry policy per spec §4.1:
7//! - 2xx → return immediately.
8//! - 429, 503 → retry with exponential backoff; honor `Retry-After`
9//!   when `retry.respect_retry_after` is true. Exhausted → ClientError::Exhausted.
10//! - 4xx other than 429 → return immediately (caller maps to LlmError).
11//! - 5xx other than 503 → retry (treated as transient).
12//! - Network timeout → retry.
13//! - Other transport error (DNS, TLS, connection refused) → ClientError::Transport, no retry.
14
15use reqwest::{Response, StatusCode};
16use secrecy::{ExposeSecret, SecretString};

23/// HTTP client that knows how to call Anthropic Messages.
24pub(crate) struct AnthropicClient {
25    inner: reqwest::Client,
26    base_url: String,
27    api_key: SecretString,
28    api_version: String,
29    retry: RetryConfig,
30}

51    /// `POST /v1/messages` with retry. The body is a serde_json::Value
52    /// produced by `request::build_messages_body`. The `stream` flag
53    /// adds the `Accept: text/event-stream` header for the streaming
54    /// path.
55    pub(crate) async fn post_messages(
56        &self,
57        body: &serde_json::Value,
58        stream: bool,
59    ) -> Result<Response, ClientError> {
60        let url = format!("{}/v1/messages", self.base_url.trim_end_matches('/'));
61        let mut attempt: u32 = 0;
62
63        loop {
64            attempt += 1;
65
66            let mut req = self
67                .inner
68                .post(&url)
69                .header("x-api-key", self.api_key.expose_secret())
70                .header("anthropic-version", &self.api_version)
71                .header("content-type", "application/json")
72                .json(body);
73            if stream {
74                req = req.header("accept", "text/event-stream");
75            }
76
77            let send_result = req.send().await;
78            match self.classify(send_result, attempt) {
79                Decision::Return(resp) => return Ok(resp),
80                Decision::Error(err) => return Err(err),
81                Decision::Retry { delay_ms, status } => {
82                    if attempt >= self.retry.max_attempts {
83                        return Err(ClientError::Exhausted {
84                            status,
85                            attempts: attempt,
86                        });
87                    }
88                    tracing::warn!(
89                        target: "anthropic_plugin::retry",
90                        attempt,
91                        max = self.retry.max_attempts,
92                        delay_ms,
93                        status = status.as_u16(),
94                        "retrying transient error",
95                    );
96                    sleep(Duration::from_millis(delay_ms)).await;
97                }
98            }
99        }
100    }
101
102    fn classify(&self, res: reqwest::Result<Response>, attempt: u32) -> Decision {
103        match res {
104            Ok(resp) => {
105                let status = resp.status();
106                if status.is_success() {
107                    Decision::Return(resp)
108                } else if is_retryable_status(status) {
109                    let delay_ms = self.retry_delay(&resp, attempt);
110                    Decision::Retry { delay_ms, status }
111                } else {
112                    // Non-retryable 4xx (or unexpected): caller maps.
113                    Decision::Return(resp)
114                }
115            }
116            Err(e) if e.is_timeout() => {
117                let delay_ms = self.backoff_only(attempt);
118                // No status; use 408 (Request Timeout) as a placeholder for the Exhausted variant.
119                Decision::Retry {
120                    delay_ms,
121                    status: StatusCode::REQUEST_TIMEOUT,
122                }
123            }
124            Err(e) => Decision::Error(ClientError::Transport(e)),
125        }
126    }
127
128    fn retry_delay(&self, resp: &Response, attempt: u32) -> u64 {
129        if self.retry.respect_retry_after {
130            if let Some(secs) = resp
131                .headers()
132                .get("retry-after")
133                .and_then(|h| h.to_str().ok())
134                .and_then(|s| s.parse::<u64>().ok())
135            {
136                return secs * 1000;
137            }
138        }
139        self.backoff_only(attempt)
140    }
141
142    fn backoff_only(&self, attempt: u32) -> u64 {
143        // base * 2^(attempt-1), capped at 60s.
144        let shift = (attempt - 1).min(6);
145        let delay = self.retry.base_delay_ms.saturating_mul(1u64 << shift);
146        delay.min(60_000)
147    }
148}
149
150fn is_retryable_status(status: StatusCode) -> bool {
151    matches!(status.as_u16(), 429 | 503) || (status.as_u16() >= 500 && status.as_u16() != 501)
152}
153
154enum Decision {
155    Return(Response),
156    Retry { delay_ms: u64, status: StatusCode },
157    Error(ClientError),
158}

168    /// Spawn a tiny HTTP/1.1 server on 127.0.0.1:0 that replies with
169    /// pre-canned responses for the first N connections, then becomes
170    /// silent. Returns the listener URL + a counter of accepted connections.
171    async fn spawn_canned_server(responses: Vec<&'static str>) -> (String, Arc<AtomicUsize>) {
172        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
173        let port = listener.local_addr().unwrap().port();
174        let url = format!("http://127.0.0.1:{port}");
175        let counter = Arc::new(AtomicUsize::new(0));
176        let counter_clone = counter.clone();
177
178        tokio::spawn(async move {
179            for canned in responses {
180                let (mut sock, _) = listener.accept().await.unwrap();
181                counter_clone.fetch_add(1, Ordering::Relaxed);
182                // Drain enough request bytes to let reqwest finish writing.
183                let mut buf = [0u8; 4096];
184                let _ = sock.read(&mut buf).await;
185                sock.write_all(canned.as_bytes()).await.unwrap();
186                sock.flush().await.unwrap();
187                drop(sock);
188            }
189        });
190
191        (url, counter)
192    }

203    fn rate_limited_body(retry_after: Option<&str>) -> String {
204        let mut s = String::from("HTTP/1.1 429 Too Many Requests\r\n");
205        if let Some(ra) = retry_after {
206            s.push_str(&format!("retry-after: {ra}\r\n"));
207        }
208        s.push_str(
209            "content-type: application/json\r\n\
210             content-length: 0\r\n\
211             connection: close\r\n\
212             \r\n",
213        );
214        s
215    }

225    fn build_client(base_url: String, max_attempts: u32, base_delay_ms: u64) -> AnthropicClient {
226        AnthropicClient::new(
227            reqwest::Client::builder()
228                .timeout(Duration::from_secs(5))
229                .build()
230                .unwrap(),
231            base_url,
232            SecretString::new("sk-ant-test".into()),
233            "2023-06-01".into(),
234            RetryConfig {
235                max_attempts,
236                base_delay_ms,
237                respect_retry_after: true,
238            },
239        )
240    }

242    #[tokio::test]
243    async fn successful_request_returns_response() {
244        let (url, counter) = spawn_canned_server(vec![ok_body()]).await;
245        let client = build_client(url, 3, 0);
246        let resp = client
247            .post_messages(&serde_json::json!({"model": "m", "messages": []}), false)
248            .await
249            .unwrap();
250        assert!(resp.status().is_success());
251        assert_eq!(counter.load(Ordering::Relaxed), 1);
252    }
