1//! JSON-RPC 2.0 message types for serve mode.
2//!
3//! Per spec §5: Request, Response, Notification, ErrorObject.
4//! All types use serde for symmetric serialization.
5
6use serde::{Deserialize, Serialize};
7use serde_json::Value;
8
9/// JSON-RPC 2.0 request id. Per spec, may be integer, string, or null.
10/// We accept integer or string; null is treated as a notification
11/// (handled separately).
12#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
13#[serde(untagged)]
14pub enum RequestId {
15    /// Integer id (most common).
16    Int(i64),
17    /// String id (UUIDs, etc.).
18    Str(String),
19}
20
21/// JSON-RPC 2.0 request.
22#[derive(Debug, Clone, Serialize, Deserialize)]
23pub struct Request {
24    /// Always "2.0".
25    pub jsonrpc: String,
26    /// Request id. Absence means "notification" (handled by [`Notification`]).
27    pub id: RequestId,
28    /// Method name (e.g. "runtime.run").
29    pub method: String,
30    /// Method-specific params object. Absent when method takes no args.
31    #[serde(default, skip_serializing_if = "Option::is_none")]
32    pub params: Option<Value>,
33}
34
35/// JSON-RPC 2.0 successful response.
36#[derive(Debug, Clone, Serialize, Deserialize)]
37pub struct Response {
38    /// Always "2.0".
39    pub jsonrpc: String,
40    /// Matches the originating request id.
41    pub id: RequestId,
42    /// Method-specific result payload.
43    pub result: Value,
44}
45
46/// JSON-RPC 2.0 error response.
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct ErrorResponse {
49    /// Always "2.0".
50    pub jsonrpc: String,
51    /// Matches the originating request id.
52    pub id: RequestId,
53    /// Error payload.
54    pub error: ErrorObject,
55}
56
57/// JSON-RPC 2.0 server-initiated notification (no id).
58#[derive(Debug, Clone, Serialize, Deserialize)]
59pub struct Notification {
60    /// Always "2.0".
61    pub jsonrpc: String,
62    /// Method name (e.g. "runtime.event").
63    pub method: String,
64    /// Method-specific params object.
65    #[serde(default, skip_serializing_if = "Option::is_none")]
66    pub params: Option<Value>,
67}
68
69/// JSON-RPC 2.0 error payload.
70#[derive(Debug, Clone, Serialize, Deserialize)]
71pub struct ErrorObject {
72    /// JSON-RPC error code. See [`super::error_codes`].
73    pub code: i32,
74    /// Human-readable error message.
75    pub message: String,
76    /// Structured machine-actionable payload. Shape depends on `code`.
77    #[serde(default, skip_serializing_if = "Option::is_none")]
78    pub data: Option<Value>,
79}
80
81/// Wire-level outbound message (request response, error response, or notification).
82#[derive(Debug, Clone, Serialize)]
83#[serde(untagged)]
84pub enum Outbound {
85    /// Successful response to a request.
86    Response(Response),
87    /// Error response to a request.
88    Error(ErrorResponse),
89    /// Server-initiated notification.
90    Notification(Notification),
91}

93#[cfg(test)]
94mod tests {
95    use super::*;
96    use serde_json::json;
97
98    #[test]
99    fn parse_request_integer_id() {
100        let raw = r#"{"jsonrpc":"2.0","id":1,"method":"meta.ping"}"#;
101        let req: Request = serde_json::from_str(raw).unwrap();
102        assert_eq!(req.jsonrpc, "2.0");
103        assert_eq!(req.id, RequestId::Int(1));
104        assert_eq!(req.method, "meta.ping");
105        assert!(req.params.is_none());
106    }
107
108    #[test]
109    fn parse_request_string_id() {
110        let raw = r#"{"jsonrpc":"2.0","id":"abc","method":"meta.ping","params":{}}"#;
111        let req: Request = serde_json::from_str(raw).unwrap();
112        assert_eq!(req.id, RequestId::Str("abc".into()));
113    }
114
115    #[test]
116    fn serialize_response_omits_none_data() {
117        let out = Outbound::Response(Response {
118            jsonrpc: "2.0".into(),
119            id: RequestId::Int(1),
120            result: json!({"ok": true}),
121        });
122        let s = serde_json::to_string(&out).unwrap();
123        assert_eq!(s, r#"{"jsonrpc":"2.0","id":1,"result":{"ok":true}}"#);
124    }
125
126    #[test]
127    fn serialize_error_with_data() {
128        let out = Outbound::Error(ErrorResponse {
129            jsonrpc: "2.0".into(),
130            id: RequestId::Int(3),
131            error: ErrorObject {
132                code: -32007,
133                message: "Capability denied".into(),
134                data: Some(json!({"kind": "CapabilityDenial"})),
135            },
136        });
137        let s = serde_json::to_string(&out).unwrap();
138        assert!(s.contains("\"code\":-32007"));
139        assert!(s.contains("\"kind\":\"CapabilityDenial\""));
140    }
141
142    #[test]
143    fn serialize_notification_no_id() {
144        let out = Outbound::Notification(Notification {
145            jsonrpc: "2.0".into(),
146            method: "runtime.event".into(),
147            params: Some(json!({"kind": "TextDelta"})),
148        });
149        let s = serde_json::to_string(&out).unwrap();
150        assert!(!s.contains("\"id\":"));
151        assert!(s.contains("\"method\":\"runtime.event\""));
152    }
153}
