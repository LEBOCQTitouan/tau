1//! JSON-RPC error codes used by serve mode.
2//!
3//! Standard JSON-RPC 2.0 codes plus tau-namespaced codes in the
4//! "Server error" reserved range (-32000 to -32099) per spec §6.
5//!
6//! Some constants are not yet referenced by code in this PR
7//! (`INTERNAL_ERROR`, `PROJECT_ERROR`) — they reserve protocol numbers
8//! for future variants. Suppress the unused-const lint at module level.
9
10#![allow(dead_code)]
11
12// Standard JSON-RPC 2.0 codes.
13/// Invalid JSON received on the wire.
14pub const PARSE_ERROR: i32 = -32700;
15/// Not a valid JSON-RPC 2.0 object.
16pub const INVALID_REQUEST: i32 = -32600;
17/// Method does not exist or is not available.
18pub const METHOD_NOT_FOUND: i32 = -32601;
19/// Invalid method parameter(s).
20pub const INVALID_PARAMS: i32 = -32602;
21/// Internal JSON-RPC error.
22pub const INTERNAL_ERROR: i32 = -32603;
23
24// Tau-namespaced (-32000..-32099).
25/// Handshake `protocol_version` not supported.
26pub const HANDSHAKE_MISMATCH: i32 = -32000;
27/// Request was cancelled by client.
28pub const CANCELLED: i32 = -32001;
29/// Non-`meta.*` call before handshake completed.
30pub const HANDSHAKE_REQUIRED: i32 = -32002;
31/// `meta.handshake` called after a successful handshake.
32pub const ALREADY_HANDSHAKEN: i32 = -32003;
33/// `max_concurrent_runs` cap reached.
34pub const SERVER_BUSY: i32 = -32004;
35/// RuntimeBuilder build error (reserved for future `runtime.reload`).
36pub const PROJECT_ERROR: i32 = -32005;
37/// Generic `RuntimeError` not covered by a more specific code.
38pub const RUNTIME_ERROR: i32 = -32006;
39/// `RuntimeError::CapabilityDenied`.
40pub const CAPABILITY_DENIED: i32 = -32007;
41/// Tool plugin returned error.
42pub const TOOL_ERROR: i32 = -32008;
43/// LLM backend plugin returned error.
44pub const LLM_ERROR: i32 = -32009;
45/// `agent_id` not in this project.
46pub const UNKNOWN_AGENT: i32 = -32010;
