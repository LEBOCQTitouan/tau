15/// Main serve entry point. Builds runtime, spawns tasks, blocks until shutdown.
16pub async fn run(opts: ServeOptions) -> Result<()> {
17    super::tracing_init::install();
18
19    info!(project = %opts.project_path.display(), "serve starting");
20
21    let project = Arc::new(
22        Project::load(&opts.project_path)
23            .await
24            .context("load project")?,
25    );
26
27    let runtime = build_runtime(&project).await.context("build runtime")?;
28
29    let (in_tx, in_rx) = mpsc::channel(64);
30    let (out_tx, out_rx) = mpsc::channel(256);
31
32    // Linux: set PDEATHSIG so we die when parent dies.
33    #[cfg(target_os = "linux")]
34    set_pdeathsig();
35
36    let cancel_reg = CancelRegistry::default();
37    let dispatcher = Dispatcher {
38        project: project.clone(),
39        runtime: Arc::new(runtime),
40        handshake: HandshakeState::default(),
41        cancel_reg: cancel_reg.clone(),
42        max_concurrent: opts.max_concurrent,
43        out_tx: out_tx.clone(),
44    };
45
46    let local_set = LocalSet::new();
47
48    // Reader and writer tasks are Send-friendly — spawn on multi-thread side.
49    let reader_handle = tokio::spawn(framing::reader_task(in_tx));
50    let writer_handle = tokio::spawn(framing::writer_task(out_rx));
51
52    if opts.ready_on_stderr {
53        eprintln!("tau-serve ready");
54    }
55
56    // Run dispatcher loop on the LocalSet so per-request tasks have a
57    // current_thread executor available for non-Send streams.
58    // spawn_local (used inside Dispatcher::spawn_run) works within any
59    // active LocalSet on the current thread — no &LocalSet borrow needed.
60    let dispatch_result = local_set
61        .run_until(async move {
62            let shutdown_signal = wait_for_shutdown_signal();
63            tokio::select! {
64                r = dispatcher.run(in_rx) => r,
65                _ = shutdown_signal => Ok(()),
66            }
67        })
68        .await;
69
70    // Graceful drain.
71    cancel_reg.cancel_all();
72    let grace_result = tokio::time::timeout(opts.shutdown_grace, async {
73        let _ = reader_handle.await;
74    })
75    .await;
76    if grace_result.is_err() {
77        warn!(grace = ?opts.shutdown_grace, "shutdown grace expired");
78    }
79    drop(out_tx);
80    let _ = writer_handle.await;
81
82    info!("serve shutdown complete");
83    dispatch_result?;
84    Ok(())
85}

87/// Build the `Runtime` from a loaded `Project`.
88///
89/// Iterates every agent in the project's `tau.toml`, reads the lockfile to
90/// locate the LLM-backend and tool plugin binaries, and spawns them via
91/// `tau_runtime::plugin_host`. Deduplicates by plugin name so that two
92/// agents sharing the same backend/tool spawn only one subprocess.
93///
94/// Serve mode v1 simplifications (deferred to future iterations):
95/// - Sandbox is not enforced (`sandbox_plan = None`); plugins run
96///   unsandboxed. `PluginHostOptions::sandbox_adapter` is `None`.
97/// - Recorder / trace context: a synthetic trace context is generated
98///   per plugin (no recorder ledger, no run-level trace propagation).
99/// - `[agents.<id>.config]` is forwarded to the LLM backend as JSON;
100///   tools always receive `{}` (per-tool config selectors not yet landed).
101async fn build_runtime(project: &Project) -> Result<tau_runtime::Runtime> {
102    use tau_pkg::LockFile;
103    use tau_plugin_protocol::handshake::TraceContext;
104    use tau_runtime::plugin_host;
105
106    let lockfile_path = project.scope.lockfile_path();
107    let lockfile = LockFile::load(&lockfile_path)
108        .with_context(|| format!("load lockfile {}", lockfile_path.display()))?;
109
110    let mut builder = tau_runtime::Runtime::builder();
111
112    // Dedup by plugin name — multiple agents may reference the same package.
113    let mut seen_llm_backends: std::collections::HashSet<String> = Default::default();
114    let mut seen_tools: std::collections::HashSet<String> = Default::default();
115
116    for entry in project.config.agents.values() {
117        // ---- LLM backend ----
118        if seen_llm_backends.insert(entry.llm_backend.clone()) {
119            let pkg_name: tau_domain::PackageName =
120                entry.llm_backend.parse().with_context(|| {
121                    format!("invalid LLM backend package name {:?}", entry.llm_backend)
122                })?;
123            let pkg = lockfile.find(&pkg_name).ok_or_else(|| {
124                anyhow::anyhow!(
125                    "LLM backend {:?} not installed in scope (run `tau install <url>`)",
126                    entry.llm_backend
127                )
128            })?;
129            let plugin = pkg.plugin.as_ref().ok_or_else(|| {
130                anyhow::anyhow!(
131                    "package {:?} has no [plugin] table in its tau.toml (data-only package \
132                     cannot be used as an llm_backend)",
133                    entry.llm_backend
134                )
135            })?;
136            let trace_context = TraceContext::new(
137                "serve".to_string(),
138                entry.id.clone(),
139                "serve-root".to_string(),
140            );
141            let config = agent_config_to_json(entry);
142            let backend = plugin_host::load_llm_backend(
143                plugin,
144                config,
145                trace_context,
146                plugin_host::PluginHostOptions::default(),
147                None,
148            )
149            .await
150            .with_context(|| format!("load LLM backend {:?}", entry.llm_backend))?;
151            builder = builder.with_dyn_llm_backend(backend);
152        }
153
154        // ---- Tools ----
155        for tool in &entry.requires.tools {
156            let tool_name = tool.name.to_string();
157            if seen_tools.insert(tool_name.clone()) {
158                let pkg = lockfile.find(&tool.name).ok_or_else(|| {
159                    anyhow::anyhow!(
160                        "tool {:?} not installed in scope (run `tau install <url>`)",
161                        tool_name
162                    )
163                })?;
164                let plugin = pkg.plugin.as_ref().ok_or_else(|| {
165                    anyhow::anyhow!(
166                        "package {:?} has no [plugin] table in its tau.toml (data-only package \
167                         cannot be used as a tool)",
168                        tool_name
169                    )
170                })?;
171                let trace_context = TraceContext::new(
172                    "serve".to_string(),
173                    entry.id.clone(),
174                    "serve-root".to_string(),
175                );
176                let loaded = plugin_host::load_tool(
177                    plugin,
178                    serde_json::Value::Object(Default::default()),
179                    trace_context,
180                    plugin_host::PluginHostOptions::default(),
181                    None,
182                )
183                .await
184                .with_context(|| format!("load tool {:?}", tool_name))?;
185                builder = builder.with_dyn_tool(loaded);
186            }
187        }
188    }
189
190    // Use `build_allow_empty` so a project with zero agents still starts
191    // successfully. Serve mode accepts `meta.handshake` and `meta.ping` on
192    // any project; `runtime.run` calls will return -32010 UNKNOWN_AGENT
193    // which is the correct behaviour when no agents are defined.
194    builder.build_allow_empty().context("build Runtime")
195}

197/// Convert an agent's `[agents.<id>.config]` TOML sub-table to a
198/// `serde_json::Value` for the plugin handshake.
199///
200/// Returns `{}` (not `Null`) when no config is set: plugin config structs
201/// typically don't deserialize from JSON null, so an empty object lets every
202/// plugin with default-able config construct its defaults.
203fn agent_config_to_json(entry: &tau_pkg::project::AgentEntry) -> serde_json::Value {
204    if entry.config.is_empty() {
205        return serde_json::Value::Object(serde_json::Map::new());
206    }
207    let mut map = serde_json::Map::with_capacity(entry.config.len());
208    for (k, v) in &entry.config {
209        map.insert(k.clone(), toml_to_json(v.clone()));
210    }
211    serde_json::Value::Object(map)
212}
213
214fn toml_to_json(v: toml::Value) -> serde_json::Value {
215    match v {
216        toml::Value::String(s) => serde_json::Value::String(s),
217        toml::Value::Integer(i) => serde_json::Value::Number(i.into()),
218        toml::Value::Float(f) => serde_json::Number::from_f64(f)
219            .map(serde_json::Value::Number)
220            .unwrap_or(serde_json::Value::Null),
221        toml::Value::Boolean(b) => serde_json::Value::Bool(b),
222        toml::Value::Datetime(dt) => serde_json::Value::String(dt.to_string()),
223        toml::Value::Array(arr) => {
224            serde_json::Value::Array(arr.into_iter().map(toml_to_json).collect())
225        }
226        toml::Value::Table(t) => {
227            serde_json::Value::Object(t.into_iter().map(|(k, v)| (k, toml_to_json(v))).collect())
228        }
229    }
230}

232/// Wait for any of: SIGTERM, SIGINT (Unix), Ctrl-C (all platforms).
233///
234/// Unix uses dedicated `SignalKind::terminate` + `SignalKind::interrupt`
235/// streams (`tokio::signal::unix`). Windows has no `unix` submodule;
236/// we use `tokio::signal::ctrl_c` as the cross-platform interrupt
237/// signal. SIGTERM has no analog on Windows — parent-death detection
238/// there relies on stdin EOF (see lifecycle::run).
239#[cfg(unix)]
240async fn wait_for_shutdown_signal() {
241    use tokio::signal::unix::{signal, SignalKind};
242    let mut term = match signal(SignalKind::terminate()) {
243        Ok(s) => s,
244        Err(_) => return,
245    };
246    let mut int = match signal(SignalKind::interrupt()) {
247        Ok(s) => s,
248        Err(_) => return,
249    };
250    tokio::select! {
251        _ = term.recv() => info!("received SIGTERM"),
252        _ = int.recv() => info!("received SIGINT"),
253    }
254}
255
256#[cfg(not(unix))]
257async fn wait_for_shutdown_signal() {
258    let _ = tokio::signal::ctrl_c().await;
259    info!("received Ctrl-C");
260}

262/// On Linux, ask the kernel to deliver SIGTERM to us when our parent dies.
263#[cfg(target_os = "linux")]
264#[allow(unsafe_code)]
265fn set_pdeathsig() {
266    // SAFETY: prctl is async-signal-safe; the SIGTERM target is the
267    // current process which always exists. This is the only unsafe
268    // block in tau-app — see the crate-level lib.rs note for context.
269    unsafe {
270        libc::prctl(
271            libc::PR_SET_PDEATHSIG,
272            libc::SIGTERM as libc::c_ulong,
273            0,
274            0,
275            0,
276        );
277    }
