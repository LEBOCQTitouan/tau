1//! Tracing subscriber configured to write to stderr only.
2//!
3//! stdout is reserved for the JSON-RPC protocol. Any tracing/logging
4//! sent to stdout would corrupt the protocol stream.
5
6use tracing_subscriber::{fmt, EnvFilter};
7
8/// Install a global tracing subscriber writing to stderr. Honors
9/// `RUST_LOG`. Idempotent — safe to call multiple times.
10pub fn install() {
11    let _ = fmt()
12        .with_writer(std::io::stderr)
13        .with_env_filter(
14            EnvFilter::try_from_default_env().unwrap_or_else(|_| EnvFilter::new("info")),
15        )
16        .try_init();
17}
