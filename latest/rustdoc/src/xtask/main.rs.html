1//! Workspace task runner. Currently exposes:
2//!
3//! - `cargo xtask build-base-image` — builds `tau-plugin-base:dev`.
4//! - `cargo xtask build-plugin-images [--name <bin>]` — builds the base if
5//!   missing, then builds each plugin's image (or just the named one).
6//!
7//! Auto-detects the container runtime (podman first, docker fallback) by
8//! probing `<runtime> --version` with a 2-second timeout. Mirrors the
9//! convention from `tau-sandbox-container::probe`.
10//!
11//! Honors `BUILDX_CACHE_FROM` / `BUILDX_CACHE_TO` env vars: when set, each
12//! `<runtime> build` invocation gets `--cache-from=<value>` /
13//! `--cache-to=<value>` flags appended. CI sets these to
14//! `type=gha` / `type=gha,mode=max` to use buildx's GitHub Actions cache.
15
16use std::process::{Command, Stdio};
17use std::time::Duration;
18
19use anyhow::{bail, Context, Result};
20use clap::{Parser, Subcommand};
21
22#[derive(Parser, Debug)]
23#[command(name = "xtask", about = "Workspace task runner.")]
24struct Cli {
25    #[command(subcommand)]
26    command: Cmd,
27}
28
29#[derive(Subcommand, Debug)]
30enum Cmd {
31    /// Build the shared `tau-plugin-base:dev` image.
32    BuildBaseImage,
33    /// Build per-plugin images (all, or just `--name <bin>`).
34    BuildPluginImages {
35        /// Cargo `[[bin]]` name (e.g. `shell-plugin`). Builds all if omitted.
36        #[arg(long)]
37        name: Option<String>,
38    },
39}
40
41/// All plugins shipped in-tree that get a Dockerfile in Phase 1.
42const PLUGINS: &[&str] = &[
43    "shell-plugin",
44    "fs-read-plugin",
45    "anthropic-plugin",
46    "ollama-plugin",
47    "openai-plugin",
48];
49
50fn main() -> Result<()> {
51    let cli = Cli::parse();
52    let runtime = detect_runtime()?;
53    eprintln!("xtask: using container runtime `{runtime}`");
54
55    match cli.command {
56        Cmd::BuildBaseImage => build_base(&runtime),
57        Cmd::BuildPluginImages { name } => match name {
58            Some(n) => {
59                ensure_base(&runtime)?;
60                build_plugin(&runtime, &n)
61            }
62            None => {
63                ensure_base(&runtime)?;
64                for p in PLUGINS {
65                    build_plugin(&runtime, p)?;
66                }
67                Ok(())
68            }
69        },
70    }
71}

73/// Probe `podman` first, then `docker`. Mirrors PR #40's
74/// `ContainerRuntime::Auto` ordering.
75fn detect_runtime() -> Result<String> {
76    // Honor explicit override first. CI sets TAU_CONTAINER_RUNTIME=docker
77    // because GHA Linux runners ship both podman and docker, and the
78    // BUILDX_CACHE_FROM/TO `type=gha` directive is buildx-specific (podman
79    // rejects it). Local dev leaves the env unset → probe order applies.
80    if let Ok(forced) = std::env::var("TAU_CONTAINER_RUNTIME") {
81        let forced = forced.trim();
82        if !forced.is_empty() {
83            if probe_one(forced) {
84                return Ok(forced.to_string());
85            }
86            bail!("TAU_CONTAINER_RUNTIME=`{forced}` is not on PATH or unresponsive");
87        }
88    }
89    for bin in ["podman", "docker"] {
90        if probe_one(bin) {
91            return Ok(bin.to_string());
92        }
93    }
94    bail!("no container runtime found on PATH (tried podman, docker)")
95}
96
97fn probe_one(bin: &str) -> bool {
98    let res = Command::new(bin)
99        .arg("--version")
100        .stdin(Stdio::null())
101        .stdout(Stdio::null())
102        .stderr(Stdio::null())
103        .spawn();
104    let mut child = match res {
105        Ok(c) => c,
106        Err(_) => return false,
107    };
108    let start = std::time::Instant::now();
109    loop {
110        match child.try_wait() {
111            Ok(Some(status)) => return status.success(),
112            Ok(None) => {
113                if start.elapsed() > Duration::from_secs(2) {
114                    let _ = child.kill();
115                    return false;
116                }
117                std::thread::sleep(Duration::from_millis(50));
118            }
119            Err(_) => return false,
120        }
121    }
122}

124fn ensure_base(runtime: &str) -> Result<()> {
125    // Always rebuild the base; the buildx cache makes warm rebuilds cheap.
126    build_base(runtime)
127}
128
129/// Build args helper. Honors `BUILDX_CACHE_FROM` / `BUILDX_CACHE_TO`
130/// env vars (set by CI). If unset, builds with default (no GHA cache).
131fn build_args(dockerfile: &str, tag: &str) -> Vec<String> {
132    let mut args = vec!["build".to_string()];
133    if let Ok(v) = std::env::var("BUILDX_CACHE_FROM") {
134        if !v.is_empty() {
135            args.push(format!("--cache-from={v}"));
136        }
137    }
138    if let Ok(v) = std::env::var("BUILDX_CACHE_TO") {
139        if !v.is_empty() {
140            args.push(format!("--cache-to={v}"));
141        }
142    }
143    args.push("-f".to_string());
144    args.push(dockerfile.to_string());
145    args.push("-t".to_string());
146    args.push(tag.to_string());
147    args.push(".".to_string());
148    args
149}
150
151fn build_base(runtime: &str) -> Result<()> {
152    eprintln!("xtask: building tau-plugin-base:dev ...");
153    let args = build_args("crates/tau-plugin-base/Dockerfile", "tau-plugin-base:dev");
154    let status = Command::new(runtime)
155        .args(&args)
156        .status()
157        .with_context(|| format!("invoke {runtime} build (base image)"))?;
158    if !status.success() {
159        bail!("`{runtime} build` for tau-plugin-base failed (exit {status})");
160    }
161    Ok(())
162}
163
164fn build_plugin(runtime: &str, bin_name: &str) -> Result<()> {
165    let crate_subdir = bin_name.strip_suffix("-plugin").unwrap_or(bin_name);
166    let dockerfile = format!("crates/tau-plugins/{crate_subdir}/Dockerfile");
167    let tag = format!("tau-plugin-{bin_name}:dev");
168    eprintln!("xtask: building {tag} ...");
169    let args = build_args(&dockerfile, &tag);
170    let status = Command::new(runtime)
171        .args(&args)
172        .status()
173        .with_context(|| format!("invoke {runtime} build for {bin_name}"))?;
174    if !status.success() {
175        bail!("`{runtime} build` for {bin_name} failed (exit {status})");
176    }
177    Ok(())
178}
