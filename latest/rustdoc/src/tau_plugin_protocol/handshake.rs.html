1//! Handshake and shutdown payload types for the plugin protocol.
2//!
3//! See `docs/superpowers/specs/2026-04-28-plugin-loading-design.md`
4//! §4.4 (handshake) and §4.8 (shutdown).
5//!
6//! These are the typed payloads exchanged via the protocol-level
7//! `meta.handshake`, `meta.shutdown`, and `meta.describe` methods.
8//! They serialize through `rmp-serde` when used as the `params[0]` of
9//! a [`crate::Frame::Request`] (host→plugin) or as the `result` of the
10//! corresponding [`crate::Frame::Response`] (plugin→host).
11//!
12//! All payload types are `#[non_exhaustive]` and expose explicit
13//! `::new(...)` constructors so future revisions can add fields without
14//! breaking callers (struct-literal construction is blocked across
15//! crate boundaries by E0639).
16
17use std::collections::BTreeMap;
18
19use serde::{Deserialize, Serialize};
20use tau_domain::PortKind;

22/// Trace context propagated from host to plugin so plugin tracing
23/// events tag the right run / agent / root span.
24///
25/// Carried inside [`HandshakeRequest`] so the plugin SDK can install a
26/// tracing subscriber that injects these IDs as fields on every event.
27///
28/// # Example
29///
30/// ```ignore
31/// // `TraceContext` is `#[non_exhaustive]`; cannot be constructed via
32/// // struct-literal across crate boundaries (E0639). Use
33/// // `TraceContext::new(...)` instead.
34/// ```
35#[non_exhaustive]
36#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
37pub struct TraceContext {
38    /// Run id (typically a ULID) — identifies the host run that
39    /// spawned this plugin invocation.
40    pub run_id: String,
41    /// Agent id — identifies the logical agent within the run.
42    pub agent_id: String,
43    /// Root span id — the parent span under which plugin events nest.
44    pub root_span_id: String,
45}
46
47impl TraceContext {
48    /// Construct a [`TraceContext`].
49    pub fn new(run_id: String, agent_id: String, root_span_id: String) -> Self {
50        Self {
51            run_id,
52            agent_id,
53            root_span_id,
54        }
55    }
56}

58/// Host's `meta.handshake` request payload (the single element of the
59/// request frame's `params` array).
60///
61/// # Example
62///
63/// ```ignore
64/// // `HandshakeRequest` is `#[non_exhaustive]`; use
65/// // `HandshakeRequest::new(...)` instead of struct-literal syntax
66/// // across crate boundaries (E0639).
67/// ```
68#[non_exhaustive]
69#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
70pub struct HandshakeRequest {
71    /// Protocol version string the host speaks (currently
72    /// [`PROTOCOL_VERSION`]).
73    pub protocol_version: String,
74    /// The port the host expects this plugin to provide.
75    pub port: PortKind,
76    /// Trace context for the host run requesting the plugin.
77    pub trace_context: TraceContext,
78    /// Free-form per-plugin configuration drawn from the host's
79    /// configuration system. Defaults to `null` when absent.
80    #[serde(default)]
81    pub config: serde_json::Value,
82}
83
84impl HandshakeRequest {
85    /// Construct a [`HandshakeRequest`].
86    pub fn new(
87        protocol_version: String,
88        port: PortKind,
89        trace_context: TraceContext,
90        config: serde_json::Value,
91    ) -> Self {
92        Self {
93            protocol_version,
94            port,
95            trace_context,
96            config,
97        }
98    }
99}

101/// JSON Schema (or rmpv-typed schema) for one method's params and
102/// result, as advertised in [`HandshakeResponse::schemas`].
103///
104/// # Example
105///
106/// ```ignore
107/// // `MethodSchema` is `#[non_exhaustive]`; use `MethodSchema::new(...)`
108/// // instead of struct-literal syntax across crate boundaries (E0639).
109/// ```
110#[non_exhaustive]
111#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
112pub struct MethodSchema {
113    /// Schema for the method's `params` payload.
114    pub params: serde_json::Value,
115    /// Schema for the method's `result` payload.
116    pub result: serde_json::Value,
117}
118
119impl MethodSchema {
120    /// Construct a [`MethodSchema`].
121    pub fn new(params: serde_json::Value, result: serde_json::Value) -> Self {
122        Self { params, result }
123    }
124}

126/// Plugin's `meta.handshake` response payload (the `result` field of
127/// the response frame).
128///
129/// # Example
130///
131/// ```ignore
132/// // `HandshakeResponse` is `#[non_exhaustive]`; use
133/// // `HandshakeResponse::new(...)` instead of struct-literal syntax
134/// // across crate boundaries (E0639).
135/// ```
136#[non_exhaustive]
137#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
138pub struct HandshakeResponse {
139    /// Protocol version string the plugin speaks.
140    pub protocol_version: String,
141    /// The port the plugin provides.
142    pub provides: PortKind,
143    /// Plugin name (typically the manifest `name`).
144    pub plugin_name: String,
145    /// Plugin version (typically the manifest `version`).
146    pub plugin_version: String,
147    /// Names of the methods the plugin handles, in addition to the
148    /// `meta.*` protocol-level methods.
149    pub methods: Vec<String>,
150    /// Per-method schemas keyed by method name.
151    pub schemas: BTreeMap<String, MethodSchema>,
152}
153
154impl HandshakeResponse {
155    /// Construct a [`HandshakeResponse`].
156    pub fn new(
157        protocol_version: String,
158        provides: PortKind,
159        plugin_name: String,
160        plugin_version: String,
161        methods: Vec<String>,
162        schemas: BTreeMap<String, MethodSchema>,
163    ) -> Self {
164        Self {
165            protocol_version,
166            provides,
167            plugin_name,
168            plugin_version,
169            methods,
170            schemas,
171        }
172    }
173}

175/// Method-name constants for the protocol-level `meta.*` methods.
176pub mod meta {
177    /// `meta.handshake` — host-initiated request, plugin responds.
178    pub const HANDSHAKE_METHOD: &str = "meta.handshake";
179    /// `meta.shutdown` — host-sent notification on host exit.
180    pub const SHUTDOWN_METHOD: &str = "meta.shutdown";
181    /// `meta.describe` — host-sent request, plugin returns method
182    /// schemas (typically the same shape as
183    /// [`super::HandshakeResponse::schemas`]).
184    pub const DESCRIBE_METHOD: &str = "meta.describe";
185}
