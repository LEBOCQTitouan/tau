5//! Wire-format types and framing primitives for the tau plugin protocol.
6//!
7//! Plugins talk to the tau runtime over MessagePack-RPC on stdio with
8//! length-prefixed framing. This crate is shared by the host (in
9//! `tau-runtime::plugin_host`) and the SDK (in `tau-plugin-sdk`); it
10//! contains pure types and IO helpers, no tracing, no process management.
11//!
12//! See `docs/superpowers/specs/2026-04-28-plugin-loading-design.md` §4
13//! and ADR-0008 for the design rationale.
14

16pub mod frame;
