7/// Failures from the framing and codec layers.
8///
9/// `#[non_exhaustive]`: additive variants do not break callers.
10///
11/// # Example
12///
13/// ```ignore
14/// use tau_plugin_protocol::ProtocolError;
15/// let err = ProtocolError::FrameTooLarge { len: 1, max: 0 };
16/// assert!(format!("{err}").contains("frame too large"));
17/// ```
18#[non_exhaustive]
19#[derive(Debug, Error)]
20pub enum ProtocolError {
21    /// Underlying IO error from the transport.
22    #[error("io error: {0}")]
23    Io(#[from] std::io::Error),
24
25    /// The receiving side observed end-of-stream while expecting more
26    /// bytes. For host-side framers, this typically means the plugin
27    /// process exited.
28    #[error("frame truncated: expected {expected} more bytes, got EOF")]
29    FrameTruncated {
30        /// How many more bytes were expected.
31        expected: usize,
32    },
33
34    /// A frame's length-prefix exceeded the configured max.
35    #[error("frame too large: {len} bytes (max {max})")]
36    FrameTooLarge {
37        /// Reported length from the prefix.
38        len: usize,
39        /// Configured maximum.
40        max: usize,
41    },
42
43    /// The frame body failed to decode as MessagePack.
44    #[error("body decode failed: {0}")]
45    BodyDecodeFailed(#[from] rmp_serde::decode::Error),
46
47    /// Body encoding failed.
48    #[error("body encode failed: {0}")]
49    BodyEncodeFailed(#[from] rmp_serde::encode::Error),
50
51    /// A `Frame` slot that must carry rmp-serde-encoded MessagePack
52    /// bytes (e.g. `params` on a request/notification or a non-`None`
53    /// `result` on a response) was passed to [`crate::Frame::encode`]
54    /// as an empty slice. The smallest legitimate payload is a
55    /// one-byte `[0x90]` (empty MessagePack array); empty input would
56    /// otherwise round-trip asymmetrically through `Value::Nil`.
57    ///
58    /// # Example
59    ///
60    /// ```
61    /// use tau_plugin_protocol::{Frame, ProtocolError};
62    /// let frame = Frame::Request { id: 1, method: "m".into(), params: vec![] };
63    /// let err = frame.encode().unwrap_err();
64    /// assert!(matches!(err, ProtocolError::EmptyFrameSlot { slot: "params" }));
65    /// ```
66    #[error("empty frame slot: {slot} must contain rmp-serde-encoded bytes")]
67    EmptyFrameSlot {
68        /// Which slot was empty: `"params"` or `"result"`.
69        slot: &'static str,
70    },
71}
72
73/// MessagePack-RPC error envelope carried in the `error` slot of a
74/// response frame.
75///
76/// The `code` follows JSON-RPC 2.0 conventions (see the constants in
77/// this module). `message` is a short human-readable summary; `data`
78/// carries optional structured payload (e.g. a serialized port-specific
79/// error). Spec §4.7.
80///
81/// # Example
82///
83/// ```ignore
84/// use tau_plugin_protocol::{RpcErrorEnvelope, METHOD_NOT_FOUND};
85/// let env = RpcErrorEnvelope {
86///     code: METHOD_NOT_FOUND,
87///     message: "method not found".into(),
88///     data: None,
89/// };
90/// assert_eq!(env.code, -32601);
91/// ```
92#[non_exhaustive]
93#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
94pub struct RpcErrorEnvelope {
95    /// Numeric error code. See the `*_ERROR` / `*_DENIED` constants.
96    pub code: i32,
97    /// Short, human-readable error description.
98    pub message: String,
99    /// Optional structured payload. For port-specific errors (codes in
100    /// the [`PORT_SPECIFIC_ERROR_BASE`] range) this is the serialized
101    /// `tau-ports` error type.
102    #[serde(default, skip_serializing_if = "Option::is_none")]
103    pub data: Option<rmpv::Value>,
104}
105
106impl RpcErrorEnvelope {
107    /// Construct an [`RpcErrorEnvelope`].
108    pub fn new(code: i32, message: String, data: Option<rmpv::Value>) -> Self {
109        Self {
110            code,
111            message,
112            data,
113        }
114    }
115}
116
117/// Standard JSON-RPC parse-error code.
118pub const PARSE_ERROR: i32 = -32700;
119/// Standard JSON-RPC invalid-request code.
120pub const INVALID_REQUEST: i32 = -32600;
121/// Standard JSON-RPC method-not-found code.
122pub const METHOD_NOT_FOUND: i32 = -32601;
123/// Standard JSON-RPC invalid-params code.
124pub const INVALID_PARAMS: i32 = -32602;
125/// Standard JSON-RPC internal-error code.
126pub const INTERNAL_ERROR: i32 = -32603;
127/// Tau-specific: plugin contract violation.
128pub const PLUGIN_CONTRACT_VIOLATION: i32 = -32000;
129/// Tau-specific: capability check denied this method.
130pub const CAPABILITY_DENIED: i32 = -32001;
131
132/// Reserved range for port-specific recoverable errors. The `data`
133/// field of the envelope carries the serialized tau-ports
134/// `LlmError`/`ToolError`/etc. in this range.
135pub const PORT_SPECIFIC_ERROR_BASE: i32 = -32100;
