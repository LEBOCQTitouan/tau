1//! MessagePack-RPC frame types (Request / Response / Notification).
2//!
3//! The wire shape (see spec §4.2) is a top-level MessagePack array
4//! whose first element is a small integer type discriminator:
5//!
6//! * `[0, msgid, method, params]` — Request
7//! * `[1, msgid, error, result]`  — Response
8//! * `[2, method, params]`         — Notification
9//!
10//! [`Frame`] keeps `params` and `result` as **raw MessagePack bytes**
11//! (the encoded form of the inner value, typically itself an array).
12//! Callers decode their concrete request/response types via `rmp-serde`
13//! on those bytes; this keeps `Frame` itself generic without an
14//! intermediate `serde_json::Value` indirection.
15
16use rmpv::Value;
17
18use crate::error::{ProtocolError, RpcErrorEnvelope};
19
20/// A single MessagePack-RPC frame body.
21///
22/// `#[non_exhaustive]`: future protocol revisions may add variants
23/// without breaking callers.
24///
25/// # Example
26///
27/// ```ignore
28/// use tau_plugin_protocol::Frame;
29/// let frame = Frame::Notification {
30///     method: "stream.chunk".into(),
31///     params: vec![],
32/// };
33/// let bytes = frame.clone().encode().unwrap();
34/// let decoded = Frame::decode(&bytes).unwrap();
35/// assert_eq!(frame, decoded);
36/// ```
37#[non_exhaustive]
38#[derive(Debug, Clone, PartialEq)]
39pub enum Frame {
40    /// Request frame: `[0, id, method, params]`. The `params` field is
41    /// the raw MessagePack-encoded bytes of the inner value (typically
42    /// itself an array).
43    Request {
44        /// Request id; pairs with the matching `Response::id`.
45        id: u32,
46        /// Method name (e.g. `"llm.complete"`).
47        method: String,
48        /// MessagePack-encoded parameters.
49        params: Vec<u8>,
50    },
51    /// Response frame: `[1, id, error, result]`. Exactly one of
52    /// `error` / `result` is `Some` per the spec, but this type does
53    /// not enforce that invariant on the wire — callers should.
54    Response {
55        /// Matches the `Request::id` this is a reply to.
56        id: u32,
57        /// `Some` if the call failed; `None` on success.
58        error: Option<RpcErrorEnvelope>,
59        /// `Some` MessagePack-encoded bytes on success; `None` on error.
60        result: Option<Vec<u8>>,
61    },
62    /// Notification frame: `[2, method, params]`. Fire-and-forget; the
63    /// receiver does not reply.
64    Notification {
65        /// Method name (e.g. `"stream.chunk"`).
66        method: String,
67        /// MessagePack-encoded parameters.
68        params: Vec<u8>,
69    },
70}
71
72const TYPE_REQUEST: i64 = 0;
73const TYPE_RESPONSE: i64 = 1;
74const TYPE_NOTIFICATION: i64 = 2;
75
76impl Frame {
77    /// Decode a frame body (as produced by [`crate::FramedReader`])
78    /// into a typed [`Frame`]. Malformed bodies (non-array, wrong
79    /// arity, unknown type discriminator, wrong member types) return
80    /// [`ProtocolError::BodyDecodeFailed`].
81    pub fn decode(body: &[u8]) -> Result<Frame, ProtocolError> {
82        let mut cursor = body;
83        let value: Value = rmpv::decode::read_value(&mut cursor).map_err(decode_err)?;
84
85        let array = match value {
86            Value::Array(a) => a,
87            _ => return Err(decode_msg("frame body is not a MessagePack array")),
88        };
89
90        let ty = array
91            .first()
92            .and_then(value_as_i64)
93            .ok_or_else(|| decode_msg("frame missing integer type discriminator"))?;
94
95        match ty {
96            TYPE_REQUEST => decode_request(&array),
97            TYPE_RESPONSE => decode_response(&array),
98            TYPE_NOTIFICATION => decode_notification(&array),
99            other => Err(decode_msg(&format!(
100                "unknown frame type discriminator: {other}"
101            ))),
102        }
103    }
104
105    /// Encode this frame to MessagePack-RPC wire bytes.
106    ///
107    /// Returns [`ProtocolError::EmptyFrameSlot`] if `params`
108    /// (Request/Notification) or a `Some(_)` `result` (Response) is
109    /// empty — legitimate callers always pass non-empty
110    /// rmp-serde-encoded bytes (smallest is `[0x90]` for an empty
111    /// array), and accepting empty input would round-trip
112    /// asymmetrically through `Value::Nil`.
113    pub fn encode(self) -> Result<Vec<u8>, ProtocolError> {
114        let value = match self {
115            Frame::Request { id, method, params } => {
116                if params.is_empty() {
117                    return Err(ProtocolError::EmptyFrameSlot { slot: "params" });
118                }
119                Value::Array(vec![
120                    Value::Integer(TYPE_REQUEST.into()),
121                    Value::Integer(u64::from(id).into()),
122                    Value::String(method.into()),
123                    bytes_to_value(&params)?,
124                ])
125            }
126            Frame::Response { id, error, result } => {
127                let error_val = match error {
128                    Some(env) => rmpv::ext::to_value(&env).map_err(encode_err)?,
129                    None => Value::Nil,
130                };
131                let result_val = match result {
132                    Some(bytes) => {
133                        if bytes.is_empty() {
134                            return Err(ProtocolError::EmptyFrameSlot { slot: "result" });
135                        }
136                        bytes_to_value(&bytes)?
137                    }
138                    None => Value::Nil,
139                };
140                Value::Array(vec![
141                    Value::Integer(TYPE_RESPONSE.into()),
142                    Value::Integer(u64::from(id).into()),
143                    error_val,
144                    result_val,
145                ])
146            }
147            Frame::Notification { method, params } => {
148                if params.is_empty() {
149                    return Err(ProtocolError::EmptyFrameSlot { slot: "params" });
150                }
151                Value::Array(vec![
152                    Value::Integer(TYPE_NOTIFICATION.into()),
153                    Value::String(method.into()),
154                    bytes_to_value(&params)?,
155                ])
156            }
157        };
158
159        let mut out = Vec::new();
160        rmpv::encode::write_value(&mut out, &value).map_err(|e| {
161            ProtocolError::BodyEncodeFailed(rmp_serde::encode::Error::InvalidValueWrite(e))
162        })?;
163        Ok(out)
164    }
165}

167fn decode_request(array: &[Value]) -> Result<Frame, ProtocolError> {
168    if array.len() != 4 {
169        return Err(decode_msg(&format!(
170            "request frame must have 4 elements, got {}",
171            array.len()
172        )));
173    }
174    let id = value_as_u32(&array[1])
175        .ok_or_else(|| decode_msg("request msgid is not a u32-compatible integer"))?;
176    let method =
177        value_as_string(&array[2]).ok_or_else(|| decode_msg("request method is not a string"))?;
178    let params = value_to_bytes(&array[3])?;
179    Ok(Frame::Request { id, method, params })
180}
181
182fn decode_response(array: &[Value]) -> Result<Frame, ProtocolError> {
183    if array.len() != 4 {
184        return Err(decode_msg(&format!(
185            "response frame must have 4 elements, got {}",
186            array.len()
187        )));
188    }
189    let id = value_as_u32(&array[1])
190        .ok_or_else(|| decode_msg("response msgid is not a u32-compatible integer"))?;
191    let error = match &array[2] {
192        Value::Nil => None,
193        v => Some(rmpv::ext::from_value::<RpcErrorEnvelope>(v.clone()).map_err(decode_err)?),
194    };
195    let result = match &array[3] {
196        Value::Nil => None,
197        v => Some(value_to_bytes(v)?),
198    };
199    Ok(Frame::Response { id, error, result })
200}
201
202fn decode_notification(array: &[Value]) -> Result<Frame, ProtocolError> {
203    if array.len() != 3 {
204        return Err(decode_msg(&format!(
205            "notification frame must have 3 elements, got {}",
206            array.len()
207        )));
208    }
209    let method = value_as_string(&array[1])
210        .ok_or_else(|| decode_msg("notification method is not a string"))?;
211    let params = value_to_bytes(&array[2])?;
212    Ok(Frame::Notification { method, params })
213}

215/// Re-serialize a single rmpv `Value` back to its MessagePack byte
216/// representation. Used to keep `params`/`result` fields opaque on the
217/// way out of `decode`.
218fn value_to_bytes(value: &Value) -> Result<Vec<u8>, ProtocolError> {
219    let mut out = Vec::new();
220    rmpv::encode::write_value(&mut out, value).map_err(|e| {
221        ProtocolError::BodyEncodeFailed(rmp_serde::encode::Error::InvalidValueWrite(e))
222    })?;
223    Ok(out)
224}
225
226/// Decode raw MessagePack bytes back into an `rmpv::Value`. Used when
227/// re-encoding a `Frame` to splice opaque `params`/`result` blobs into
228/// the outer array. Callers in [`Frame::encode`] reject empty input
229/// before reaching this helper (see [`ProtocolError::EmptyFrameSlot`]),
230/// so this function assumes `bytes` is non-empty rmp-serde-encoded
231/// MessagePack.
232fn bytes_to_value(bytes: &[u8]) -> Result<Value, ProtocolError> {
233    let mut cursor = bytes;
234    rmpv::decode::read_value(&mut cursor).map_err(decode_err)
235}
236
237fn value_as_i64(value: &Value) -> Option<i64> {
238    match value {
239        Value::Integer(i) => i.as_i64(),
240        _ => None,
241    }
242}
243
244fn value_as_u32(value: &Value) -> Option<u32> {
245    let n = match value {
246        Value::Integer(i) => i.as_u64()?,
247        _ => return None,
248    };
249    u32::try_from(n).ok()
250}
251
252fn value_as_string(value: &Value) -> Option<String> {
253    match value {
254        Value::String(s) => s.as_str().map(|s| s.to_owned()),
255        _ => None,
256    }
257}

259/// Wrap an `rmpv` decode error in [`ProtocolError::BodyDecodeFailed`].
260/// Goes through `rmp_serde::decode::Error` to match the existing
261/// variant.
262fn decode_err<E: std::fmt::Display>(err: E) -> ProtocolError {
263    decode_msg(&err.to_string())
264}
265
266fn decode_msg(msg: &str) -> ProtocolError {
267    ProtocolError::BodyDecodeFailed(rmp_serde::decode::Error::Uncategorized(msg.to_owned()))
268}
269
270fn encode_err<E: std::fmt::Display>(err: E) -> ProtocolError {
271    ProtocolError::BodyEncodeFailed(rmp_serde::encode::Error::Syntax(err.to_string()))
272}

278    /// Helper: build a small MessagePack-encoded payload to use as
279    /// `params` / `result`.
280    fn sample_params() -> Vec<u8> {
281        // Encode `["hello", 42]` as MessagePack.
282        let value = Value::Array(vec![
283            Value::String("hello".into()),
284            Value::Integer(42i64.into()),
285        ]);
286        let mut out = Vec::new();
287        rmpv::encode::write_value(&mut out, &value).unwrap();
288        out
289    }
