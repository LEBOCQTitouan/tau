1//! Typed errors for tau-workflow.
2
3use std::path::PathBuf;
4
5/// Errors raised by parsing, validating, running, or persisting a workflow.
6#[derive(Debug, thiserror::Error)]
7#[non_exhaustive]
8pub enum WorkflowError {
9    /// Failed to read or parse a workflow TOML file.
10    #[error("workflow {path:?}: {message}")]
11    ParseFailed {
12        /// The workflow file path.
13        path: PathBuf,
14        /// Human-readable parse detail.
15        message: String,
16    },
17
18    /// A `${steps.<id>.output}` template referenced a step that does not exist
19    /// (or is later in the workflow, which is rejected at parse time).
20    #[error("workflow {workflow:?}: step {step_id:?} references unknown step {missing:?}")]
21    TemplateUnresolved {
22        /// The workflow name.
23        workflow: String,
24        /// The step that contained the bad template.
25        step_id: String,
26        /// The missing step identifier.
27        missing: String,
28    },
29
30    /// A workflow step `agent.run` referenced an agent not declared in tau.toml.
31    #[error("workflow {workflow:?}: step {step_id:?} references unknown agent {agent:?}")]
32    AgentNotFound {
33        /// The workflow name.
34        workflow: String,
35        /// The step id with the bad reference.
36        step_id: String,
37        /// The missing agent id.
38        agent: String,
39    },
40
41    /// A `tool.call` step referenced a tool not declared / not granted by the
42    /// workflow's default agent.
43    #[error("workflow {workflow:?}: step {step_id:?} references unknown tool {tool:?}")]
44    ToolNotFound {
45        /// The workflow name.
46        workflow: String,
47        /// The step id with the bad reference.
48        step_id: String,
49        /// The missing tool id.
50        tool: String,
51    },
52
53    /// A step terminated abnormally. The wrapped source is preserved for
54    /// `Debug` output; the run aborts and subsequent steps are not executed.
55    #[error("workflow step {step_id:?} failed: {source}")]
56    StepFailed {
57        /// The failing step's id.
58        step_id: String,
59        /// Underlying runtime error from `tau_runtime`.
60        #[source]
61        source: Box<dyn std::error::Error + Send + Sync>,
62    },
63
64    /// Persistence I/O failure (disk full, permission denied, etc.).
65    /// The partial JSONL is NOT cleaned up — the user can inspect it.
66    #[error("workflow persistence failed at {path:?}: {source}")]
67    PersistenceError {
68        /// The JSONL file path.
69        path: PathBuf,
70        /// Underlying I/O error.
71        #[source]
72        source: std::io::Error,
73    },
74
75    /// Resume requested but the JSONL file's recorded steps no longer match
76    /// the workflow's current step ids. Use `--force` to override.
77    #[error(
78        "workflow drift: log step ids {logged:?} differ from current workflow step ids {current:?}"
79    )]
80    DriftDetected {
81        /// Step ids found in the JSONL log.
82        logged: Vec<String>,
83        /// Step ids present in the current workflow file.
84        current: Vec<String>,
85    },
86}
