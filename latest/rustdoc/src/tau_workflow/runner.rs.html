1//! Workflow runner: dispatches each step in order.
2//!
3//! For v1, the runner is linear. Each step's output feeds future steps'
4//! `${steps.<id>.output}` templates. A failed step aborts the run.
5//! Persistence is append-only JSONL (see `crate::persistence`).
6

22/// One workflow execution.
23pub struct Runner {
24    runtime: Arc<Runtime>,
25    scope_root: PathBuf,
26}

28/// Per-run options.
29#[derive(Debug, Clone)]
30pub struct RunOpts {
31    /// Caller-supplied input string available as `${input}`.
32    pub input: String,
33    /// Optional pre-existing run id (used by `--resume`). When `None`,
34    /// a fresh ULID is allocated.
35    pub run_id: Option<String>,
36    /// Already-completed step records (from replay). The runner skips
37    /// any step whose id appears here with status Ok.
38    pub completed: Vec<StepRecord>,
39    /// Agent definitions resolved from tau.toml by the caller.
40    pub agents: BTreeMap<String, (AgentDefinition, PackageManifest)>,
41}
42
43/// Outcome of a single run invocation.
44#[derive(Debug, Clone)]
45pub struct RunOutcome {
46    /// The run id (ULID).
47    pub run_id: String,
48    /// Path to the JSONL log on disk.
49    pub log_path: PathBuf,
50    /// Whether every step completed with status Ok.
51    pub success: bool,
52    /// The last step's output (or empty on failure).
53    pub last_output: String,
54}
55
56impl Runner {
57    /// Construct a runner.
58    pub fn new(runtime: Arc<Runtime>, scope_root: PathBuf) -> Self {
59        Self {
60            runtime,
61            scope_root,
62        }
63    }
64
65    /// Run (or resume) a workflow.
66    pub async fn run(
67        &self,
68        workflow: &Workflow,
69        opts: RunOpts,
70    ) -> Result<RunOutcome, WorkflowError> {
71        let run_id = opts
72            .run_id
73            .clone()
74            .unwrap_or_else(|| ulid::Ulid::new().to_string());
75        let log_path = run_log_path(&self.scope_root, &workflow.name, &run_id);
76        let mut log = RunLog::open_for_write(&log_path).await?;
77
78        // Seed prior_outputs from completed records (resume path).
79        let mut prior_outputs: BTreeMap<String, String> = BTreeMap::new();
80        let mut completed_ids: BTreeMap<String, ()> = BTreeMap::new();
81        for record in &opts.completed {
82            if record.status == StepStatus::Ok {
83                prior_outputs.insert(record.step_id.clone(), record.output.clone());
84                completed_ids.insert(record.step_id.clone(), ());
85            }
86        }
87
88        let mut last_output = String::new();
89
90        for (idx, step) in workflow.steps.iter().enumerate() {
91            if completed_ids.contains_key(&step.id) {
92                last_output = prior_outputs.get(&step.id).cloned().unwrap_or_default();
93                continue;
94            }
95
96            let started_at = Utc::now();
97            let (input_str, output_result) = match &step.kind {
98                StepKind::AgentRun { agent, input } => {
99                    let resolved_input = resolve_template(
100                        input,
101                        &opts.input,
102                        &prior_outputs,
103                        &workflow.name,
104                        &step.id,
105                    )?;
106                    let (agent_def, manifest) =
107                        opts.agents
108                            .get(agent)
109                            .ok_or_else(|| WorkflowError::AgentNotFound {
110                                workflow: workflow.name.clone(),
111                                step_id: step.id.clone(),
112                                agent: agent.clone(),
113                            })?;
114                    let initial_message = build_user_message(resolved_input.clone());
115                    let result = self
116                        .runtime
117                        .run(
118                            agent_def.clone(),
119                            manifest.clone(),
120                            initial_message,
121                            tau_runtime::RunOptions::default(),
122                        )
123                        .await;
124                    (resolved_input, agent_outcome_to_string(result))
125                }
126                StepKind::ToolCall { tool, args } => {
127                    // Resolve any template strings inside args (string values only).
128                    let resolved_args =
129                        resolve_args(args, &opts.input, &prior_outputs, &workflow.name, &step.id)?;
130                    let default_agent_id = workflow.default_agent.as_ref().ok_or_else(|| {
131                        WorkflowError::ParseFailed {
132                            path: workflow.source_path.clone(),
133                            message: "tool.call step requires [workflow].default-agent".into(),
134                        }
135                    })?;
136                    let (agent_def, manifest) =
137                        opts.agents.get(default_agent_id).ok_or_else(|| {
138                            WorkflowError::AgentNotFound {
139                                workflow: workflow.name.clone(),
140                                step_id: step.id.clone(),
141                                agent: default_agent_id.clone(),
142                            }
143                        })?;
144                    // Convert serde_json::Value → tau_domain::Value for invoke_tool.
145                    let tau_args = json_to_tau_value(&resolved_args);
146                    let result = self
147                        .runtime
148                        .invoke_tool(agent_def, manifest, tool, tau_args)
149                        .await;
150                    let input_repr = serde_json::to_string(&resolved_args).unwrap_or_default();
151                    (input_repr, tool_outcome_to_string(result))
152                }
153            };
154
155            let ended_at = Utc::now();
156            let duration_ms = (ended_at - started_at).num_milliseconds().max(0) as u64;
157
158            let (output, status, error, detail) = match output_result {
159                Ok(out) => (out, StepStatus::Ok, None, None),
160                Err((err_kind, err_detail)) => (
161                    String::new(),
162                    StepStatus::Failed,
163                    Some(err_kind),
164                    Some(err_detail),
165                ),
166            };
167
168            let record = StepRecord {
169                ts: ended_at,
170                run_id: run_id.clone(),
171                step_id: step.id.clone(),
172                step_index: idx,
173                kind: match &step.kind {
174                    StepKind::AgentRun { .. } => "agent.run".into(),
175                    StepKind::ToolCall { .. } => "tool.call".into(),
176                },
177                input: input_str,
178                output: output.clone(),
179                started_at,
180                ended_at,
181                duration_ms,
182                status,
183                error,
184                detail,
185            };
186            log.append(&record).await?;
187
188            if status == StepStatus::Failed {
189                return Ok(RunOutcome {
190                    run_id,
191                    log_path,
192                    success: false,
193                    last_output: output,
194                });
195            }
196
197            prior_outputs.insert(step.id.clone(), output.clone());
198            last_output = output;
199        }
200
201        Ok(RunOutcome {
202            run_id,
203            log_path,
204            success: true,
205            last_output,
206        })
207    }
208}

210/// Construct a `Message` representing the user's initial input.
211///
212/// Uses `Message::new` with `Address::User` sender and a fresh
213/// `Address::Agent(AgentInstanceId::new())` recipient — matching the
214/// same pattern used by `tau-cli/src/cmd/run.rs`. The kernel replaces
215/// the recipient placeholder with its own `AgentInstanceId` once the
216/// loop starts.
217fn build_user_message(text: String) -> Message {
218    Message::new(
219        Address::User,
220        Address::Agent(AgentInstanceId::new()),
221        MessagePayload::Text { content: text },
222    )
223}

225/// Extract the final assistant text from a `RunOutcome`.
226///
227/// For `RunOutcome::Completed`, returns the text content of `final_message`.
228/// For `RunOutcome::Failed`, returns an error tuple with kind `"agent_failed"`
229/// and the status debug string.
230fn agent_outcome_to_string(
231    result: Result<tau_runtime::RunOutcome, tau_runtime::RuntimeError>,
232) -> Result<String, (String, String)> {
233    match result {
234        Ok(tau_runtime::RunOutcome::Completed { final_message, .. }) => {
235            Ok(extract_text_from_payload(&final_message.payload))
236        }
237        Ok(tau_runtime::RunOutcome::Failed { status, .. }) => {
238            Err(("agent_failed".into(), format!("{status:?}")))
239        }
240        // #[non_exhaustive] forward-compat wildcard
241        Ok(_) => Err((
242            "unknown_outcome".into(),
243            "unknown RunOutcome variant".into(),
244        )),
245        Err(e) => Err(("runtime_error".into(), format!("{e}"))),
246    }
247}

249/// Project a [`MessagePayload`] to a text string. Non-text payloads
250/// fall back to a `Debug`-formatted preview, matching `tau-cli`'s
251/// `format_message_text` helper.
252fn extract_text_from_payload(payload: &MessagePayload) -> String {
253    match payload {
254        MessagePayload::Text { content } => content.clone(),
255        other => format!("{other:?}"),
256    }
257}

259/// Convert a `tau_ports::ToolResult` to a string for the JSONL output column.
260///
261/// On `is_error = true`, the step is surfaced as failed so the run aborts.
262/// On success, the content blocks are serialized as a JSON string.
263fn tool_outcome_to_string(
264    result: Result<tau_ports::ToolResult, tau_runtime::RuntimeError>,
265) -> Result<String, (String, String)> {
266    match result {
267        Ok(tool_result) => {
268            if tool_result.is_error {
269                // Tool returned an error result. Surface as a failed step.
270                return Err((
271                    "tool_error".into(),
272                    serde_json::to_string(&tool_result).unwrap_or_default(),
273                ));
274            }
275            // Success — render the content as a JSON string for the log.
276            Ok(serde_json::to_string(&tool_result).unwrap_or_default())
277        }
278        Err(e) => Err(("runtime_error".into(), format!("{e}"))),
279    }
280}

282/// Convert `serde_json::Value` → `tau_domain::Value`.
283///
284/// `tau_domain::Value` has no `From<serde_json::Value>` impl, so we
285/// pattern-match each variant manually.
286///
287/// - `serde_json::Number`: integers fit in `i64` → `Value::Integer`;
288///   everything else → `Value::Float`.
289/// - `serde_json::Null` → `Value::Null`.
290/// - Array/Object recurse.
291fn json_to_tau_value(value: &serde_json::Value) -> tau_domain::Value {
292    match value {
293        serde_json::Value::Null => tau_domain::Value::Null,
294        serde_json::Value::Bool(b) => tau_domain::Value::Bool(*b),
295        serde_json::Value::Number(n) => {
296            if let Some(i) = n.as_i64() {
297                tau_domain::Value::Integer(i)
298            } else {
299                tau_domain::Value::Float(n.as_f64().unwrap_or(f64::NAN))
300            }
301        }
302        serde_json::Value::String(s) => tau_domain::Value::String(s.clone()),
303        serde_json::Value::Array(arr) => {
304            tau_domain::Value::Array(arr.iter().map(json_to_tau_value).collect())
305        }
306        serde_json::Value::Object(map) => {
307            let mut out = std::collections::BTreeMap::new();
308            for (k, v) in map {
309                out.insert(k.clone(), json_to_tau_value(v));
310            }
311            tau_domain::Value::Object(out)
312        }
313    }
314}

316fn resolve_args(
317    args: &serde_json::Value,
318    input: &str,
319    prior: &BTreeMap<String, String>,
320    workflow: &str,
321    step_id: &str,
322) -> Result<serde_json::Value, WorkflowError> {
323    match args {
324        serde_json::Value::String(s) => {
325            let resolved = resolve_template(s, input, prior, workflow, step_id)?;
326            Ok(serde_json::Value::String(resolved))
327        }
328        serde_json::Value::Array(arr) => {
329            let resolved: Result<Vec<_>, _> = arr
330                .iter()
331                .map(|v| resolve_args(v, input, prior, workflow, step_id))
332                .collect();
333            Ok(serde_json::Value::Array(resolved?))
334        }
335        serde_json::Value::Object(map) => {
336            let mut out = serde_json::Map::with_capacity(map.len());
337            for (k, v) in map {
338                out.insert(k.clone(), resolve_args(v, input, prior, workflow, step_id)?);
339            }
340            Ok(serde_json::Value::Object(out))
341        }
342        other => Ok(other.clone()),
343    }
344}

346/// Compare a workflow's current step ids against a log's recorded step
347/// ids. Returns Err(DriftDetected) when they diverge; returns Ok(()) when
348/// the log's ids are a prefix of the workflow's (the resume case).
349///
350/// "Prefix match" means: every record in `logged_records` (in order)
351/// matches the corresponding step in `workflow.steps`. Trailing steps in
352/// the workflow that aren't yet in the log are fine (that's the work
353/// the resume will do).
354pub fn check_drift(
355    workflow: &Workflow,
356    logged_records: &[StepRecord],
357) -> Result<(), WorkflowError> {
358    if logged_records.len() > workflow.steps.len() {
359        return Err(WorkflowError::DriftDetected {
360            logged: logged_records.iter().map(|r| r.step_id.clone()).collect(),
361            current: workflow.steps.iter().map(|s| s.id.clone()).collect(),
362        });
363    }
364    for (idx, record) in logged_records.iter().enumerate() {
365        if workflow.steps[idx].id != record.step_id {
366            return Err(WorkflowError::DriftDetected {
367                logged: logged_records.iter().map(|r| r.step_id.clone()).collect(),
368                current: workflow.steps.iter().map(|s| s.id.clone()).collect(),
369            });
370        }
371    }
372    Ok(())
373}
