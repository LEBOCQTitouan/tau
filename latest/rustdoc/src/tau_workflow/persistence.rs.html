1//! Append-only JSONL persistence for workflow runs.
2//!
3//! One line per step completion. The run log file lives at
4//! `<scope>/.tau/workflow-runs/<workflow-name>-<run-id>.jsonl`.
5//! Lines are fsync'd after each write so a crash mid-write loses at
6//! most the trailing partial line; replay tolerates that.
7
8use std::path::PathBuf;
9
10use chrono::{DateTime, Utc};
11use serde::{Deserialize, Serialize};
12
13/// One step's completion record, serialized as a single JSONL line.
14#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
15pub struct StepRecord {
16    /// Log-line timestamp (record-emit time).
17    pub ts: DateTime<Utc>,
18    /// ULID of the run this record belongs to.
19    pub run_id: String,
20    /// Step id as declared in the workflow TOML.
21    pub step_id: String,
22    /// Zero-based index of the step in the workflow.
23    pub step_index: usize,
24    /// `"agent.run"` or `"tool.call"`.
25    pub kind: String,
26    /// Resolved input string passed to the step.
27    pub input: String,
28    /// Output text captured from the step.
29    pub output: String,
30    /// Wall-clock start of the step.
31    pub started_at: DateTime<Utc>,
32    /// Wall-clock end of the step.
33    pub ended_at: DateTime<Utc>,
34    /// Duration in milliseconds (`ended_at - started_at`).
35    pub duration_ms: u64,
36    /// `"ok"` or `"failed"`.
37    pub status: StepStatus,
38    /// On `status = "failed"`, an opaque error class for matching.
39    /// `None` on `status = "ok"`.
40    #[serde(skip_serializing_if = "Option::is_none", default)]
41    pub error: Option<String>,
42    /// On `status = "failed"`, a human-readable detail line.
43    /// `None` on `status = "ok"`.
44    #[serde(skip_serializing_if = "Option::is_none", default)]
45    pub detail: Option<String>,
46}
47
48/// Status of a step in a run log.
49#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
50#[serde(rename_all = "lowercase")]
51pub enum StepStatus {
52    /// Step completed successfully.
53    Ok,
54    /// Step terminated abnormally. Run aborted.
55    Failed,
56}
57
58/// Builds the canonical run-log path:
59/// `<scope_root>/.tau/workflow-runs/<workflow_name>-<run_id>.jsonl`.
60pub fn run_log_path(scope_root: &std::path::Path, workflow_name: &str, run_id: &str) -> PathBuf {
61    scope_root
62        .join(".tau")
63        .join("workflow-runs")
64        .join(format!("{workflow_name}-{run_id}.jsonl"))
65}

67use std::path::Path;
68
69use tokio::fs::{File, OpenOptions};
70use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
71
72/// Append-only run log for a single workflow run.
73///
74/// Each `append` writes one JSONL line and fsyncs. On crash, the file
75/// contains all complete lines plus possibly a truncated trailing line;
76/// `replay` skips the trailing partial line.
77pub struct RunLog {
78    file: File,
79    path: PathBuf,
80}
81
82impl RunLog {
83    /// Open or create the run log for append. The parent directory is
84    /// created if missing.
85    pub async fn open_for_write(path: &Path) -> Result<Self, crate::WorkflowError> {
86        if let Some(parent) = path.parent() {
87            tokio::fs::create_dir_all(parent).await.map_err(|e| {
88                crate::WorkflowError::PersistenceError {
89                    path: path.to_path_buf(),
90                    source: e,
91                }
92            })?;
93        }
94        let file = OpenOptions::new()
95            .create(true)
96            .append(true)
97            .open(path)
98            .await
99            .map_err(|e| crate::WorkflowError::PersistenceError {
100                path: path.to_path_buf(),
101                source: e,
102            })?;
103        Ok(Self {
104            file,
105            path: path.to_path_buf(),
106        })
107    }
108
109    /// Append one record + fsync.
110    pub async fn append(&mut self, record: &StepRecord) -> Result<(), crate::WorkflowError> {
111        let mut line =
112            serde_json::to_string(record).map_err(|e| crate::WorkflowError::PersistenceError {
113                path: self.path.clone(),
114                source: std::io::Error::new(std::io::ErrorKind::InvalidData, e),
115            })?;
116        line.push('\n');
117        self.file.write_all(line.as_bytes()).await.map_err(|e| {
118            crate::WorkflowError::PersistenceError {
119                path: self.path.clone(),
120                source: e,
121            }
122        })?;
123        self.file
124            .sync_data()
125            .await
126            .map_err(|e| crate::WorkflowError::PersistenceError {
127                path: self.path.clone(),
128                source: e,
129            })?;
130        Ok(())
131    }
132}
133
134/// Replay a JSONL log into a vector of records. Tolerates a trailing
135/// partial line (truncated mid-write on crash) by skipping it.
136pub async fn replay(path: &Path) -> Result<Vec<StepRecord>, crate::WorkflowError> {
137    let file = File::open(path)
138        .await
139        .map_err(|e| crate::WorkflowError::PersistenceError {
140            path: path.to_path_buf(),
141            source: e,
142        })?;
143    let reader = BufReader::new(file);
144    let mut lines = reader.lines();
145    let mut records = Vec::new();
146    while let Some(line) =
147        lines
148            .next_line()
149            .await
150            .map_err(|e| crate::WorkflowError::PersistenceError {
151                path: path.to_path_buf(),
152                source: e,
153            })?
154    {
155        if line.is_empty() {
156            continue;
157        }
158        match serde_json::from_str::<StepRecord>(&line) {
159            Ok(r) => records.push(r),
160            Err(_) => {
161                // Truncated/corrupt trailing line. Skip silently — the
162                // contract is "tolerate the trailing partial". We do NOT
163                // continue past a corrupt line in the middle of a file;
164                // but BufReader returns lines split by `\n`, so a missing
165                // `\n` only affects the final line.
166                break;
167            }
168        }
169    }
170    Ok(records)
171}

173#[cfg(test)]
174mod tests {
175    use super::*;
176    use chrono::Utc;

178    fn make_record(idx: usize, id: &str) -> StepRecord {
179        let now = Utc::now();
180        StepRecord {
181            ts: now,
182            run_id: "01HKZTEST".into(),
183            step_id: id.into(),
184            step_index: idx,
185            kind: "agent.run".into(),
186            input: format!("input-{idx}"),
187            output: format!("output-{idx}"),
188            started_at: now,
189            ended_at: now,
190            duration_ms: 1,
191            status: StepStatus::Ok,
192            error: None,
193            detail: None,
194        }
195    }
196
197    #[tokio::test]
198    async fn append_then_replay_round_trips() {
199        let dir = tempfile::tempdir().unwrap();
200        let path = dir.path().join("run.jsonl");
201        {
202            let mut log = RunLog::open_for_write(&path).await.unwrap();
203            log.append(&make_record(0, "a")).await.unwrap();
204            log.append(&make_record(1, "b")).await.unwrap();
205        }
206        let records = replay(&path).await.unwrap();
207        assert_eq!(records.len(), 2);
208        assert_eq!(records[0].step_id, "a");
209        assert_eq!(records[1].step_id, "b");
210    }
211
212    #[tokio::test]
213    async fn replay_tolerates_trailing_partial_line() {
214        let dir = tempfile::tempdir().unwrap();
215        let path = dir.path().join("run.jsonl");
216        {
217            let mut log = RunLog::open_for_write(&path).await.unwrap();
218            log.append(&make_record(0, "a")).await.unwrap();
219        }
220        // Append 30 bytes of garbage WITHOUT a trailing newline.
221        let mut f = tokio::fs::OpenOptions::new()
222            .append(true)
223            .open(&path)
224            .await
225            .unwrap();
226        f.write_all(b"{\"step_id\":\"trunc").await.unwrap();
227        f.sync_data().await.unwrap();
228        drop(f);
229
230        let records = replay(&path).await.unwrap();
231        assert_eq!(records.len(), 1, "trailing partial line should be dropped");
232        assert_eq!(records[0].step_id, "a");
233    }
234
235    #[tokio::test]
236    async fn replay_empty_file_returns_empty() {
237        let dir = tempfile::tempdir().unwrap();
238        let path = dir.path().join("run.jsonl");
239        tokio::fs::write(&path, b"").await.unwrap();
240        let records = replay(&path).await.unwrap();
241        assert!(records.is_empty());
242    }
