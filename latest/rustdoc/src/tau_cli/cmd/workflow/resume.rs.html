1//! `tau workflow resume <run-id> [--force]` — continue an interrupted run.
2
3use std::sync::Arc;
4
5use anyhow::Context;
6use tau_pkg::Scope;
7use tau_workflow::{check_drift, persistence::replay, RunOpts, Runner, Workflow};
8
9use crate::cli::WorkflowResumeArgs;
10use crate::output::Output;
11
12/// Run `tau workflow resume`.
13pub async fn run(args: &WorkflowResumeArgs, output: &mut Output) -> anyhow::Result<()> {
14    let cwd = std::env::current_dir()?;
15    let scope = Scope::resolve(&cwd).context("resolving package scope")?;
16
17    // Locate the JSONL log by run id.
18    let runs_dir = scope.path().join(".tau").join("workflow-runs");
19    let log_entry = std::fs::read_dir(&runs_dir)?
20        .filter_map(|e| e.ok())
21        .find(|e| {
22            e.file_name()
23                .to_str()
24                .map(|s| s.contains(&args.run_id) && s.ends_with(".jsonl"))
25                .unwrap_or(false)
26        })
27        .ok_or_else(|| anyhow::anyhow!("no run log for {:?}", args.run_id))?;
28    let log_path = log_entry.path();
29
30    // Workflow name is the prefix of the filename before -<run_id>.jsonl.
31    let workflow_name = log_path
32        .file_stem()
33        .and_then(|s| s.to_str())
34        .and_then(|s| s.rsplit_once('-').map(|(name, _)| name.to_string()))
35        .ok_or_else(|| anyhow::anyhow!("cannot derive workflow name from {log_path:?}"))?;
36
37    let wf_path = cwd
38        .join("workflows")
39        .join(format!("{}.toml", workflow_name));
40    let workflow = Workflow::from_path(&wf_path).with_context(|| format!("parsing {wf_path:?}"))?;
41
42    let records = replay(&log_path).await.context("replaying log")?;
43
44    if let Err(e) = check_drift(&workflow, &records) {
45        if !args.force {
46            return Err(anyhow::anyhow!(
47                "{e}\n\nThe workflow file has changed since the original run. \
48                 Use `--force` to override.\n"
49            ));
50        }
51        tracing::warn!(
52            "workflow drift detected on resume; --force was supplied so proceeding anyway"
53        );
54    }
55
56    // Recover the original input string from the FIRST record (its `input`
57    // field is the user-supplied ${input} at run time).
58    let original_input = records.first().map(|r| r.input.clone()).unwrap_or_default();
59
60    let agents = crate::cmd::workflow::build_agents_map(&workflow, &cwd, &scope)?;
61    let runtime = Arc::new(
62        crate::cmd::workflow::build_runtime_for_workflow(&cwd, &scope, &agents)
63            .await
64            .context("building runtime for workflow")?,
65    );
66    let runner = Runner::new(runtime, scope.path().to_path_buf());
67
68    let outcome = runner
69        .run(
70            &workflow,
71            RunOpts {
72                input: original_input,
73                run_id: Some(args.run_id.clone()),
74                completed: records,
75                agents,
76            },
77        )
78        .await
79        .with_context(|| format!("resuming workflow {workflow_name:?}"))?;
80
81    eprintln!("run_id: {} (resumed)", outcome.run_id);
82    output.human(&outcome.last_output)?;
83
84    if outcome.success {
85        Ok(())
86    } else {
87        Err(anyhow::anyhow!(
88            "workflow {:?} still failed after resume (see `tau workflow log {}`)",
89            workflow_name,
90            outcome.run_id
91        ))
92    }
93}
