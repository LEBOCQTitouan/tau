18/// Dispatch a workflow subcommand.
19pub async fn dispatch(sub: WorkflowSubcommand, output: &mut Output) -> anyhow::Result<()> {
20    match sub {
21        WorkflowSubcommand::List => list::run(output),
22        WorkflowSubcommand::Run(args) => run::run(&args, output).await,
23        WorkflowSubcommand::Log(args) => log::run(&args, output).await,
24        WorkflowSubcommand::Resume(args) => resume::run(&args, output).await,
25    }
26}

28/// Build the `{agent_id → (AgentDefinition, PackageManifest)}` map for every
29/// agent referenced by the workflow (either as an `agent.run` target or as the
30/// default-agent for a `tool.call` step).
31pub(crate) fn build_agents_map(
32    workflow: &Workflow,
33    cwd: &Path,
34    scope: &Scope,
35) -> anyhow::Result<BTreeMap<String, (tau_domain::AgentDefinition, tau_domain::PackageManifest)>> {
36    use std::collections::BTreeSet;
37    let mut needed: BTreeSet<String> = BTreeSet::new();
38    for step in &workflow.steps {
39        match &step.kind {
40            StepKind::AgentRun { agent, .. } => {
41                needed.insert(agent.clone());
42            }
43            StepKind::ToolCall { .. } => {
44                if let Some(a) = workflow.default_agent.clone() {
45                    needed.insert(a);
46                }
47            }
48        }
49    }
50
51    let project_path = cwd.join("tau.toml");
52    let project = crate::config::ProjectConfig::from_path(&project_path)
53        .with_context(|| format!("project tau.toml required at {project_path:?}"))?;
54
55    let mut out = BTreeMap::new();
56    for agent_id in needed {
57        let entry = project.agents.get(&agent_id).ok_or_else(|| {
58            anyhow::anyhow!(
59                "workflow references agent {:?} which is not declared in tau.toml",
60                agent_id
61            )
62        })?;
63        let (agent_def, manifest) = crate::config::build_agent_definition(entry, cwd, scope)
64            .with_context(|| format!("resolving agent {:?}", agent_id))?;
65        out.insert(agent_id, (agent_def, manifest));
66    }
67    Ok(out)
68}

70/// Build a `tau_runtime::Runtime` for use by the workflow runner.
71///
72/// Mirrors the plugin-loading sequence from `crates/tau-cli/src/cmd/run.rs`:
73///
74/// 1. Pick the first agent entry from `agents` to source plugin configuration.
75///    For v1 workflows all agents are assumed to share the same LLM backend
76///    and tools; if they differ, the first agent's plugins are used and the
77///    others will rely on the same runtime instance.
78/// 2. Call `plugin_loader::load_plugins` with default host options (no
79///    `--record-protocol`, no `--no-sandbox` / `--sandbox` override).
80/// 3. Build the runtime via `RuntimeBuilder::build`.
81///
82/// Sandbox selection honours the scope's `[sandbox]` config in `.tau/config.toml`
83/// exactly as it would for `tau run`.
84pub(crate) async fn build_runtime_for_workflow(
85    _cwd: &Path,
86    scope: &Scope,
87    agents: &BTreeMap<String, (tau_domain::AgentDefinition, tau_domain::PackageManifest)>,
88) -> anyhow::Result<tau_runtime::Runtime> {
89    // Pick the first agent entry so we can source plugin config.
90    // In a typical workflow all agents share one LLM backend; even when they
91    // differ, the runtime's kernel is agent-agnostic — the LLM backend plugin
92    // loaded here is what actually gets invoked for every agent.run step.
93    let first_agent_id = agents.keys().next().ok_or_else(|| {
94        anyhow::anyhow!("workflow has no agent-run or tool-call steps requiring an agent")
95    })?;
96
97    // Reload the project config to get the raw AgentEntry (which carries the
98    // llm_backend name, requires.tools, capability_overrides, and config block).
99    let cwd = std::env::current_dir()?;
100    let project_path = cwd.join("tau.toml");
101    let project = crate::config::ProjectConfig::from_path(&project_path)
102        .with_context(|| format!("project tau.toml required at {project_path:?}"))?;
103
104    let entry = project.agents.get(first_agent_id).ok_or_else(|| {
105        anyhow::anyhow!(
106            "agent {:?} not found in tau.toml (internal error: was present at build_agents_map time)",
107            first_agent_id
108        )
109    })?;
110
111    // Resolve + install missing tools for the first agent.
112    // (Subsequent agents with the same tool set are implicitly covered.)
113    // Use a null/discarded output so plugin-loader status lines don't
114    // interleave with the caller's output stream.
115    let mut null_output = crate::output::Output::with_writers(
116        Box::new(std::io::sink()),
117        Box::new(std::io::sink()),
118        false,
119        true,
120        crate::output::ColorChoice::Never,
121    );
122    crate::cmd::resolve_helpers::resolve_and_install_for_agent(
123        entry,
124        scope,
125        false, // no_install = false: auto-install if needed
126        &mut null_output,
127    )?;
128
129    let run_id = format!(
130        "tau-workflow-{}",
131        std::time::SystemTime::now()
132            .duration_since(std::time::UNIX_EPOCH)
133            .map(|d| d.as_nanos())
134            .unwrap_or(0)
135    );
136    let trace_context = tau_plugin_protocol::handshake::TraceContext::new(
137        run_id,
138        first_agent_id.clone(),
139        "workflow".to_string(),
140    );
141
142    // Default host options: no recording, no forced sandbox override.
143    let (host_options, _ledger) = crate::cmd::plugin_loader::build_host_options(None, false, None);
144
145    let loaded = crate::cmd::plugin_loader::load_plugins(entry, scope, trace_context, host_options)
146        .await
147        .with_context(|| format!("loading plugins for agent {:?}", first_agent_id))?;
148
149    loaded
150        .builder
151        .build()
152        .context("failed to build runtime from spawned plugins")
153}
