1//! Closest-match helper for "did you mean…?" suggestions on unknown
2//! skill names.
3//!
4//! Standard Wagner-Fischer dynamic-programming Levenshtein distance,
5//! O(n*m) over the two input strings. Threshold ≤ 2 distance is the
6//! suggestion cutoff (tunable per call).
7
8/// Find the candidate in `candidates` with the smallest edit
9/// distance to `query`. Returns `Some(&candidate)` if the best
10/// candidate is within `max_dist`; `None` otherwise.
11///
12/// On ties (multiple candidates at the same distance), returns the
13/// first encountered — stable for sorted input.
14pub fn closest_match<'a>(
15    query: &str,
16    candidates: &'a [String],
17    max_dist: usize,
18) -> Option<&'a str> {
19    let mut best: Option<(usize, &str)> = None;
20    for c in candidates {
21        let d = distance(query, c);
22        if d > max_dist {
23            continue;
24        }
25        match best {
26            None => best = Some((d, c.as_str())),
27            Some((bd, _)) if d < bd => best = Some((d, c.as_str())),
28            _ => {}
29        }
30    }
31    best.map(|(_, s)| s)
32}
33
34/// Wagner-Fischer Levenshtein distance.
35fn distance(a: &str, b: &str) -> usize {
36    let a: Vec<char> = a.chars().collect();
37    let b: Vec<char> = b.chars().collect();
38    let n = a.len();
39    let m = b.len();
40    if n == 0 {
41        return m;
42    }
43    if m == 0 {
44        return n;
45    }
46    let mut prev: Vec<usize> = (0..=m).collect();
47    let mut curr: Vec<usize> = vec![0; m + 1];
48    for i in 1..=n {
49        curr[0] = i;
50        for j in 1..=m {
51            let cost = if a[i - 1] == b[j - 1] { 0 } else { 1 };
52            curr[j] = (prev[j] + 1).min(curr[j - 1] + 1).min(prev[j - 1] + cost);
53        }
54        std::mem::swap(&mut prev, &mut curr);
55    }
56    prev[m]
57}

59#[cfg(test)]
60mod tests {
61    use super::*;
62
63    #[test]
64    fn exact_match_returns_itself() {
65        let candidates = vec!["critic".to_string(), "proofread".to_string()];
66        assert_eq!(closest_match("critic", &candidates, 2), Some("critic"));
67    }
68
69    #[test]
70    fn single_typo_returns_closest() {
71        // "kritic" → "critic" is 1 edit (k→c).
72        let candidates = vec!["critic".to_string(), "proofread".to_string()];
73        assert_eq!(closest_match("kritic", &candidates, 2), Some("critic"));
74    }
75
76    #[test]
77    fn distance_beyond_threshold_returns_none() {
78        let candidates = vec!["critic".to_string()];
79        // "xyz123" is much further than 2 edits from "critic".
80        assert_eq!(closest_match("xyz123", &candidates, 2), None);
81    }
