1//! `tau session export` — convert a session to jsonl/md/json.
2
3use anyhow::Context;
4use tau_pkg::Scope;
5
6use crate::cli::{ExportFormat, SessionExportArgs};
7use crate::output::Output;
8
9/// Run `tau session export`.
10pub async fn run(args: &SessionExportArgs, output: &mut Output) -> anyhow::Result<()> {
11    let scope = if args.global {
12        Scope::global().context("resolving global scope")?
13    } else {
14        let cwd = std::env::current_dir().context("getting current directory")?;
15        Scope::resolve(&cwd).context("resolving project scope")?
16    };
17
18    let sessions_dir = scope.state_path().join("sessions");
19    let sid = crate::session::resolve_id_prefix(&sessions_dir, &args.id)
20        .map_err(|e| anyhow::anyhow!("{}", e))?;
21    let path = sessions_dir.join(format!("{}.jsonl", sid.as_str()));
22
23    match args.format {
24        ExportFormat::Jsonl => {
25            // Passthrough: read file and emit verbatim.
26            let contents = std::fs::read_to_string(&path).context("reading session file")?;
27            // Use print! to bypass the Output formatting.
28            print!("{}", contents);
29        }
30        ExportFormat::Md => {
31            let (header, entries) =
32                crate::session::SessionReader::read(&path).map_err(|e| anyhow::anyhow!("{}", e))?;
33            let rendered = crate::session::render_session(&header, &entries);
34            print!("{}", rendered);
35        }
36        ExportFormat::Json => {
37            // Single envelope JSON: { header, messages, turn_summaries }.
38            let (header, entries) =
39                crate::session::SessionReader::read(&path).map_err(|e| anyhow::anyhow!("{}", e))?;
40            let messages: Vec<&tau_domain::Message> = entries
41                .iter()
42                .filter_map(|e| match e {
43                    crate::session::SessionEntry::Message(m) => Some(m),
44                    _ => None,
45                })
46                .collect();
47            let turn_summaries: Vec<serde_json::Value> = entries
48                .iter()
49                .filter_map(|e| match e {
50                    crate::session::SessionEntry::TurnSummary {
51                        turn,
52                        stop_reason,
53                        input_tokens,
54                        output_tokens,
55                    } => Some(serde_json::json!({
56                        "turn": turn,
57                        "stop_reason": stop_reason,
58                        "input_tokens": input_tokens,
59                        "output_tokens": output_tokens,
60                    })),
61                    _ => None,
62                })
63                .collect();
64            let envelope = serde_json::json!({
65                "header": header,
66                "messages": messages,
67                "turn_summaries": turn_summaries,
68            });
69            println!(
70                "{}",
71                serde_json::to_string_pretty(&envelope).context("serializing envelope")?
72            );
73        }
74    }
75
76    // Suppress unused warning: output is required by the dispatch signature.
77    let _ = output;
78    Ok(())
79}
