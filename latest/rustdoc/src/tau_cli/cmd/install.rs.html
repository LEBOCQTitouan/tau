1//! `tau install` — install a package from a git URL.
2//!
3//! Per spec §3.13 install row:
4//!
5//! - Parses the URL into a [`PackageSource`].
6//! - Resolves the active [`Scope`] (project or global).
7//! - For `--dry-run`: clones the source into a [`tempfile::TempDir`],
8//!   parses the manifest, prints a preview of what would be installed,
9//!   and writes nothing to the scope.
10//! - Otherwise: delegates to [`tau_pkg::install_with_options`] and
11//!   prints either a one-line summary or, when `--json` is set, a
12//!   structured payload.
13//!
14//! Errors propagate via `anyhow::Error` and exit 2 through the
15//! top-level `dispatch` in `crate::run_main`.
16

30/// Run `tau install`.
31pub async fn run(args: &InstallArgs, output: &mut Output) -> anyhow::Result<()> {
32    // 1. Parse URL into a PackageSource.
33    let source = PackageSource::from_str(&args.url)
34        .map_err(|e| anyhow::anyhow!("invalid package URL {:?}: {}", args.url, e))?;
35
36    // 2. Resolve scope.
37    let scope = if args.global {
38        Scope::global()?
39    } else {
40        let cwd = std::env::current_dir()?;
41        Scope::resolve(&cwd)?
42    };
43
44    // 3. Dry-run path: clone to tmp, parse manifest, print preview.
45    if args.dry_run {
46        let preview = preview_install(&source, &scope)?;
47        emit_dry_run(&preview, output)?;
48        return Ok(());
49    }
50
51    // 4. Real install.
52    output.status(format!("installing from {}...", args.url))?;
53    let installed = install_with_options(&source, &scope, InstallOptions::default())
54        .map_err(|e| anyhow::anyhow!("install failed: {e}"))?;
55
56    if output.is_json() {
57        let payload = serde_json::json!({
58            "name": installed.name.as_str(),
59            "version": installed.version.to_string(),
60            "scope": scope_kind_str(&scope),
61            "path": installed.installed_path,
62        });
63        output.json(&payload)?;
64    } else {
65        output.human(&format!(
66            "installed: {} {} ({})",
67            installed.name,
68            installed.version,
69            scope_kind_str(&scope),
70        ))?;
71    }
72
73    Ok(())
74}

76/// Return `"global"` or `"project"` for the current scope's kind.
77fn scope_kind_str(scope: &Scope) -> &'static str {
78    match scope.kind() {
79        ScopeKind::Global => "global",
80        ScopeKind::Project => "project",
81        // ScopeKind is `#[non_exhaustive]`; cover unknown future variants.
82        _ => "unknown",
83    }
84}

86/// Preview shape used for both human and JSON dry-run output.
87struct DryRunPreview {
88    name: String,
89    version: String,
90    url: String,
91    target_path: PathBuf,
92    kind: String,
93    capabilities: Vec<String>,
94    dependencies: Vec<String>,
95}

97/// Clone the source into a temp dir, parse the manifest, and produce the
98/// preview that the dry-run path emits. The temp dir auto-removes on drop,
99/// so the dry-run leaves no on-disk state behind.
100fn preview_install(source: &PackageSource, scope: &Scope) -> anyhow::Result<DryRunPreview> {
101    let tmp_dir = tempfile::TempDir::new()?;
102
103    // Render the source's URL/scp string for the clone subprocess. v0.1
104    // PackageSource has only the Git variant, but it is `#[non_exhaustive]`,
105    // so we surface a clear error for hypothetical future variants.
106    let (location_str, rev_opt) = match source {
107        PackageSource::Git { location, rev } => (location.to_string(), rev.clone()),
108        other => {
109            anyhow::bail!("unsupported package source for dry-run: {other:?}");
110        }
111    };
112
113    // Shell out to `git` directly. We mirror tau-pkg's clone flags
114    // (protocol.file.allow=always for file:// fixtures + `--branch <rev>
115    // --single-branch` for revision pins) so dry-run accepts the same
116    // sources as the real install path.
117    let dest = tmp_dir.path();
118    let mut cmd = Command::new("git");
119    cmd.env("GIT_CONFIG_COUNT", "1")
120        .env("GIT_CONFIG_KEY_0", "protocol.file.allow")
121        .env("GIT_CONFIG_VALUE_0", "always");
122    cmd.arg("clone");
123    if let Some(rev) = &rev_opt {
124        cmd.arg("--branch").arg(rev).arg("--single-branch");
125    }
126    cmd.arg(&location_str).arg(dest);
127
128    let output = cmd
129        .output()
130        .map_err(|e| anyhow::anyhow!("spawning `git clone`: {e}"))?;
131    if !output.status.success() {
132        anyhow::bail!(
133            "git clone failed (exit {}): {}",
134            output.status.code().unwrap_or(-1),
135            String::from_utf8_lossy(&output.stderr).trim(),
136        );
137    }
138
139    let manifest = read_manifest(&dest.join("tau.toml"))
140        .map_err(|e| anyhow::anyhow!("reading cloned manifest: {e}"))?;
141
142    let target_path = scope.package_dir(manifest.name(), manifest.version());
143
144    Ok(DryRunPreview {
145        name: manifest.name().as_str().to_owned(),
146        version: manifest.version().to_string(),
147        url: source.to_string(),
148        target_path,
149        kind: kind_str(manifest.kind()).to_owned(),
150        capabilities: manifest
151            .capabilities()
152            .iter()
153            .map(capability_kind_str)
154            .collect(),
155        dependencies: dep_strings(&manifest),
156    })
157}

159/// Render a [`PackageKind`] as its canonical kind string (e.g. `"tool"`,
160/// `"llm-backend"`). v0.1 ships only `Custom { kind }`, but the type is
161/// `#[non_exhaustive]`; future typed variants fall back to `Debug`.
162fn kind_str(kind: &PackageKind) -> &str {
163    match kind {
164        PackageKind::Custom { kind } => kind.as_str(),
165        // Forward-compat for typed variants added later.
166        _ => "unknown",
167    }
168}

170/// Render a [`Capability`] as its dot-namespaced kind string (matches the
171/// manifest `kind = "..."` form in [`Capability`]'s serde).
172fn capability_kind_str(cap: &Capability) -> String {
173    match cap {
174        Capability::Filesystem(FsCapability::Read { .. }) => "fs.read".into(),
175        Capability::Filesystem(FsCapability::Write { .. }) => "fs.write".into(),
176        Capability::Filesystem(FsCapability::Exec { .. }) => "fs.exec".into(),
177        Capability::Network(NetCapability::Http { .. }) => "net.http".into(),
178        Capability::Process(ProcessCapability::Spawn { .. }) => "process.spawn".into(),
179        Capability::Agent(AgentCapability::Spawn { .. }) => "agent.spawn".into(),
180        Capability::TaskList { .. } => "task_list".into(),
181        Capability::Plan { .. } => "plan".into(),
182        Capability::Custom { name, .. } => name.clone(),
183        // Forward-compat: future typed variants fall back to debug.
184        other => format!("{other:?}"),
185    }
186}

188/// Render dependencies as `"<name>@<version_req>"` strings.
189fn dep_strings(manifest: &PackageManifest) -> Vec<String> {
190    manifest
191        .dependencies()
192        .iter()
193        .map(|dep| format!("{}@{}", dep.name.as_str(), dep.version_req))
194        .collect()
195}

197/// Emit the dry-run preview to the user. Format per task spec:
198///
199/// ```text
200/// [dry-run] would install: <name> <version>
201/// [dry-run] from:          <url>
202/// [dry-run] to:            <target-path>
203/// [dry-run] kind:          <kind>
204/// [dry-run] capabilities:  <list> (or "none")
205/// [dry-run] dependencies:  <list> (or "none")
206/// [dry-run] no changes written.
207/// ```
208///
209/// With `--json`, emits a single JSON object on stdout instead.
210fn emit_dry_run(preview: &DryRunPreview, output: &mut Output) -> anyhow::Result<()> {
211    if output.is_json() {
212        let payload = serde_json::json!({
213            "dry_run": true,
214            "name": preview.name,
215            "version": preview.version,
216            "url": preview.url,
217            "target_path": preview.target_path,
218            "kind": preview.kind,
219            "capabilities": preview.capabilities,
220            "dependencies": preview.dependencies,
221        });
222        output.json(&payload)?;
223        return Ok(());
224    }
225
226    output.dry_run(format!(
227        "would install: {} {}",
228        preview.name, preview.version
229    ))?;
230    output.dry_run(format!("from:          {}", preview.url))?;
231    output.dry_run(format!("to:            {}", preview.target_path.display()))?;
232    output.dry_run(format!("kind:          {}", preview.kind))?;
233    let caps = if preview.capabilities.is_empty() {
234        "none".to_string()
235    } else {
236        preview.capabilities.join(", ")
237    };
238    output.dry_run(format!("capabilities:  {caps}"))?;
239    let deps = if preview.dependencies.is_empty() {
240        "none".to_string()
241    } else {
242        preview.dependencies.join(", ")
243    };
244    output.dry_run(format!("dependencies:  {deps}"))?;
245    output.dry_run("no changes written.")?;
246    Ok(())
247}
