1//! `tau run` — invoke an agent one-shot.
2//!
3//! Per spec §3.13 (run row) and §3.9.
4//!
5//! Reads the project `tau.toml`, resolves the named agent to an
6//! `(AgentDefinition, PackageManifest)` pair, resolves the agent's
7//! LLM-backend + tool plugins from the per-scope lockfile, spawns
8//! each plugin process via `tau_runtime::plugin_host::load_*`, builds
9//! a [`tau_runtime::Runtime`] populated with the resulting `Arc<dyn Dyn*>` shims,
10//! then builds an initial [`Message`] from `--prompt` / stdin, runs the
11//! agent, and maps the resulting [`RunOutcome`] to stdout/stderr + an
12//! `ExitCode`.
13//!
14//! # Plugin loading
15//!
16//! Per spec §11 (migration): the old `--features test-mock` knob has
17//! been retired. The agent's `llm_backend` package and each
18//! `requires.tools` entry must be installed via `tau install` and
19//! carry a `[plugin]` table in their `tau.toml`; the host then spawns
20//! the recorded binary at run time. See
21//! [`crate::cmd::plugin_loader`] for the resolution logic.
22

36/// Marker error: the agent ran but reported [`RunOutcome::Failed`].
37///
38/// Threaded through the existing `anyhow::Result<()>` dispatch so
39/// `lib::run_main` can downcast it and map to
40/// [`crate::ExitCode::AgentFailed`] (exit code 1) — distinct from
41/// kernel/CLI errors that map to [`crate::ExitCode::Error`] (exit
42/// code 2). See the docstring on [`crate::ExitCode`] and ADR-0006 for
43/// the Outcome/Error dichotomy.
44#[derive(Debug, thiserror::Error)]
45#[error("agent failed (exit code 1)")]
46pub(crate) struct AgentFailed;

48/// Run `tau run`.
49///
50/// `record_protocol` is the optional `--record-protocol <path>` global
51/// flag (Task 20 / spec §9 debug tier). When `Some`, every plugin
52/// frame in either direction is mirrored to the JSONL file at `path`;
53/// the recorders are flushed after the runtime drops to ensure
54/// pending writes reach disk.
55///
56/// `force_passthrough` and `force_adapter_kind` come from the global
57/// `--no-sandbox` / `--sandbox <kind>` flags (Task 7).
58pub async fn run(
59    args: &RunArgs,
60    record_protocol: Option<PathBuf>,
61    force_passthrough: bool,
62    force_adapter_kind: Option<tau_runtime::sandbox::registry::RegistryKind>,
63    output: &mut Output,
64) -> anyhow::Result<()> {
65    let cwd = std::env::current_dir()?;
66    let project_path = cwd.join("tau.toml");
67    let project = ProjectConfig::from_path(&project_path)
68        .with_context(|| format!("project tau.toml required at {project_path:?}"))?;
69
70    let entry = project.agents.get(&args.agent_id).ok_or_else(|| {
71        anyhow::anyhow!(
72            "agent id {:?} not found in project tau.toml (declare it under [agents.{}])",
73            args.agent_id,
74            args.agent_id
75        )
76    })?;
77
78    let scope = tau_pkg::Scope::resolve(&cwd).context("resolving package scope")?;
79
80    crate::cmd::resolve_helpers::resolve_and_install_for_agent(
81        entry,
82        &scope,
83        args.no_install,
84        output,
85    )?;
86
87    let (agent_def, manifest) = crate::config::build_agent_definition(entry, &cwd, &scope)
88        .with_context(|| format!("resolving agent {:?}", args.agent_id))?;
89
90    let mut options = RunOptions::default();
91    if let Some(n) = args.max_turns {
92        options.max_turns = n;
93    }
94    options.project_override = entry.capability_overrides.clone();
95
96    if args.dry_run {
97        emit_dry_run(
98            entry,
99            &agent_def,
100            &manifest,
101            &options,
102            args.prompt.as_deref(),
103            output,
104        )?;
105        return Ok(());
106    }
107
108    // ---- Plugin loading -----------------------------------------------------
109    //
110    // Now that test-mock is gone, every `tau run` invocation spawns the
111    // real LLM-backend + tool plugin processes recorded in the lockfile.
112    // Failures (plugin not installed / handshake timeout / wrong port)
113    // surface as `RuntimeError`s mapped to ExitCode::Error.
114
115    // run_id: per-invocation identifier so plugin-side traces can group
116    // events back to this run. Plain UUID — no need to coordinate with
117    // the kernel's own internal id (the kernel mints its own
118    // AgentInstanceId for the loop; the plugin's run_id is host-side).
119    //
120    // We intentionally avoid a uuid dep here and use a timestamp-based
121    // string: the only contract is uniqueness within a host process,
122    // and humantime-style strings are easier to read in logs.
123    let run_id = format!(
124        "tau-run-{}",
125        std::time::SystemTime::now()
126            .duration_since(std::time::UNIX_EPOCH)
127            .map(|d| d.as_nanos())
128            .unwrap_or(0)
129    );
130    let trace_context = TraceContext::new(run_id, args.agent_id.clone(), "root".to_string());
131    let (host_options, _ledger) = plugin_loader::build_host_options(
132        record_protocol.as_deref(),
133        force_passthrough,
134        force_adapter_kind,
135    );
136
137    let loaded = plugin_loader::load_plugins(entry, &scope, trace_context, host_options).await?;
138    let recorder_ledger = loaded.recorder_ledger.clone();
139
140    let runtime = loaded
141        .builder
142        .build()
143        .context("failed to build runtime from spawned plugins")?;
144
145    // Build the initial user message from --prompt or stdin.
146    let prompt_text = match &args.prompt {
147        Some(s) => s.clone(),
148        None => {
149            let mut buf = String::new();
150            std::io::stdin()
151                .read_to_string(&mut buf)
152                .context("reading initial prompt from stdin")?;
153            buf
154        }
155    };
156
157    let initial = Message::new(
158        Address::User,
159        // The kernel mints its own AgentInstanceId per run; this
160        // recipient placeholder gets replaced once the loop assigns its
161        // own. Using a fresh id keeps the typed Address::Agent variant
162        // happy without leaking implementation detail.
163        Address::Agent(AgentInstanceId::new()),
164        MessagePayload::Text {
165            content: prompt_text,
166        },
167    );
168
169    // ---- Multi-agent dispatch -----------------------------------------------
170    //
171    // Trigger: the agent's package manifest declares either an
172    // `Agent::Spawn` capability or a `TaskList::*` capability. When
173    // either is present, route through `Runtime::spawn_root_agent` instead
174    // of the single-agent flow. The single-agent flow is preserved as the
175    // default for backward compatibility.
176    let is_multi_agent = manifest.capabilities().iter().any(|c| {
177        matches!(
178            c,
179            tau_domain::Capability::Agent(tau_domain::AgentCapability::Spawn { .. })
180                | tau_domain::Capability::TaskList { .. }
181        )
182    });
183
184    if is_multi_agent {
185        use crate::cmd::output_orchestration::{print_summary, AgentStats};
186
187        let scope_root = scope.path().to_path_buf();
188        // v1.1: spawn_root_agent takes `Arc<Runtime>` so the in-stream
189        // agent.<kind>.spawn intercept can recursively invoke run_with_history
190        // on the same kernel. Single-agent runs below stay on the non-arc path.
191        let runtime_arc = std::sync::Arc::new(runtime);
192        let snapshot = runtime_arc
193            .clone()
194            .spawn_root_agent(
195                agent_def,
196                manifest,
197                initial,
198                tau_ports::RunBudget::default(), // TODO: thread from CLI flags
199                scope_root,
200            )
201            .await
202            .with_context(|| format!("multi-agent run for agent {:?}", args.agent_id))?;
203
204        drop(runtime_arc);
205        plugin_loader::flush_recorders(recorder_ledger).await;
206
207        // Print snapshot summary. Live trace rendering is deferred;
208        // v1 reads the JSONL log post-hoc. Stats are empty here because
209        // the printer was not wired to the trace stream — that wiring is
210        // a follow-up (requires subscribe-handle plumbing through
211        // spawn_root_agent).
212        let stats: std::collections::BTreeMap<String, AgentStats> =
213            std::collections::BTreeMap::new();
214        print_summary(&snapshot, &stats);
215
216        return if matches!(snapshot.status, tau_ports::RunStatus::Completed) {
217            Ok(())
218        } else {
219            Err(anyhow::anyhow!(
220                "multi-agent run failed (status: {:?})",
221                snapshot.status
222            ))
223        };
224    }
225
226    if args.stream {
227        let result =
228            run_streaming_path(&runtime, agent_def, manifest, initial, options, output).await;
229        drop(runtime);
230        plugin_loader::flush_recorders(recorder_ledger).await;
231        result
232    } else {
233        let run_outcome = runtime.run(agent_def, manifest, initial, options).await;
234
235        // Drop the runtime explicitly before flushing recorders. This
236        // ensures every plugin process is reaped and every host-side write
237        // task has completed, so `flush_recorders` drains a quiescent set
238        // of file buffers rather than racing with in-flight `record(...)`.
239        drop(runtime);
240        plugin_loader::flush_recorders(recorder_ledger).await;
241
242        let outcome = run_outcome.context("running agent")?;
243
244        match outcome {
245            RunOutcome::Completed {
246                ref final_message,
247                total_turns,
248                ref token_usage,
249                ..
250            } => {
251                if output.is_json() {
252                    let payload = serde_json::json!({
253                        "outcome": "completed",
254                        "final_message": format_message_text(&final_message.payload),
255                        "total_turns": total_turns,
256                        "token_usage": {
257                            "input_tokens": token_usage.input_tokens,
258                            "output_tokens": token_usage.output_tokens,
259                        },
260                    });
261                    output.json(&payload)?;
262                } else {
263                    let text = format_message_text(&final_message.payload);
264                    output.human(&text)?;
265                }
266                Ok(())
267            }
268            RunOutcome::Failed {
269                ref status,
270                total_turns,
271                ref token_usage,
272                ..
273            } => {
274                if output.is_json() {
275                    let payload = serde_json::json!({
276                        "outcome": "failed",
277                        "status": format!("{status:?}"),
278                        "total_turns": total_turns,
279                        "token_usage": {
280                            "input_tokens": token_usage.input_tokens,
281                            "output_tokens": token_usage.output_tokens,
282                        },
283                    });
284                    output.json(&payload)?;
285                } else {
286                    output.error(format!("agent failed: {status:?}"))?;
287                }
288                // Marker error → ExitCode::AgentFailed (1) via downcast in
289                // crate::run_main. Kernel/CLI errors map to ExitCode::Error
290                // (2); this case is the explicit Outcome::Failed bucket.
291                Err(AgentFailed.into())
292            }
293            // RunOutcome is `#[non_exhaustive]`; cross-crate match needs a
294            // wildcard. Any future variant should be classified explicitly
295            // by an ADR amendment; for now, treat unknown outcomes as a
296            // kernel error.
297            _ => Err(anyhow::anyhow!("unknown RunOutcome variant")),
298        }
299    }
300}

302/// Consume the `run_streaming` stream and render events as they arrive.
303///
304/// Human mode: text deltas appear inline on stdout (raw `print!` / flush
305/// for typewriter UX); tool annotations go to stderr so stdout stays the
306/// agent's text. A closing newline is printed on `RunCompleted`.
307///
308/// JSON mode (`--json`): one JSON object per event line via
309/// `Output::json`. Shape: `{"event":"<discriminator>", ...}`.
310///
311/// The caller is responsible for `drop(runtime)` and
312/// `flush_recorders` after this function returns (streaming path or
313/// batch path — the drop + flush sequence is the same).
314async fn run_streaming_path(
315    runtime: &tau_runtime::Runtime,
316    agent_def: tau_domain::AgentDefinition,
317    manifest: tau_domain::PackageManifest,
318    initial: tau_domain::Message,
319    options: tau_runtime::RunOptions,
320    output: &mut Output,
321) -> anyhow::Result<()> {
322    use futures_core::stream::Stream as _;
323    use std::io::Write as _;
324    use tau_runtime::{RunEvent, RunOutcome};
325
326    let stream = runtime
327        .run_streaming(agent_def, manifest, initial, options)
328        .await
329        .context("running agent (streaming)")?;
330    let mut stream = Box::pin(stream);
331
332    let mut final_outcome: Option<RunOutcome> = None;
333    let mut fatal: Option<(String, String)> = None;
334
335    loop {
336        let next = std::future::poll_fn(|cx| stream.as_mut().poll_next(cx)).await;
337        match next {
338            Some(RunEvent::TextDelta { delta }) => {
339                if output.is_json() {
340                    output.json(&serde_json::json!({
341                        "event": "text_delta",
342                        "delta": delta,
343                    }))?;
344                } else {
345                    print!("{delta}");
346                    std::io::stdout().flush()?;
347                }
348            }
349            Some(RunEvent::ToolCallStarted { id, name, args }) => {
350                if output.is_json() {
351                    output.json(&serde_json::json!({
352                        "event": "tool_call_started",
353                        "id": id,
354                        "name": name,
355                        "args": args,
356                    }))?;
357                } else {
358                    eprintln!("→ calling {name}...");
359                }
360            }
361            Some(RunEvent::ToolCallCompleted { id, name, result }) => {
362                if output.is_json() {
363                    let result_json = match &result {
364                        Ok(_) => serde_json::json!({"ok": true}),
365                        Err(reason) => serde_json::json!({"ok": false, "error": reason}),
366                    };
367                    output.json(&serde_json::json!({
368                        "event": "tool_call_completed",
369                        "id": id,
370                        "name": name,
371                        "result": result_json,
372                    }))?;
373                } else {
374                    match &result {
375                        Ok(_) => eprintln!("✓ {name} completed"),
376                        Err(reason) => eprintln!("✗ {name} failed: {reason}"),
377                    }
378                }
379            }
380            Some(RunEvent::TurnCompleted {
381                stop_reason,
382                usage,
383                turn,
384            }) => {
385                if output.is_json() {
386                    let usage_json = match usage {
387                        Some(u) => serde_json::json!({
388                            "input_tokens": u.input_tokens,
389                            "output_tokens": u.output_tokens,
390                        }),
391                        None => serde_json::Value::Null,
392                    };
393                    output.json(&serde_json::json!({
394                        "event": "turn_completed",
395                        "stop_reason": format!("{stop_reason:?}"),
396                        "usage": usage_json,
397                        "turn": turn,
398                    }))?;
399                }
400                // Human mode: no-op (text deltas already rendered inline).
401            }
402            Some(RunEvent::RunCompleted { outcome }) => {
403                final_outcome = Some(outcome);
404                break;
405            }
406            Some(RunEvent::FatalError { kind, detail, .. }) => {
407                fatal = Some((kind, detail));
408                break;
409            }
410            Some(_) => {
411                // Forward-compatibility for #[non_exhaustive] RunEvent.
412            }
413            None => {
414                return Err(anyhow::anyhow!(
415                    "stream ended without RunCompleted or FatalError"
416                ));
417            }
418        }
419    }
420
421    if let Some((kind, detail)) = fatal {
422        return Err(anyhow::anyhow!("kernel error ({kind}): {detail}"));
423    }
424    let outcome = final_outcome.expect("RunCompleted yielded final_outcome");
425
426    // After streaming completes, emit the same closing summary the batch path emits.
427    match outcome {
428        RunOutcome::Completed {
429            ref final_message,
430            total_turns,
431            ref token_usage,
432            ..
433        } => {
434            if output.is_json() {
435                output.json(&serde_json::json!({
436                    "event": "run_completed",
437                    "outcome": {
438                        "outcome": "completed",
439                        "final_message": format_message_text(&final_message.payload),
440                        "total_turns": total_turns,
441                        "token_usage": {
442                            "input_tokens": token_usage.input_tokens,
443                            "output_tokens": token_usage.output_tokens,
444                        },
445                    },
446                }))?;
447            } else {
448                // Human mode: text already streamed; print closing newline.
449                println!();
450            }
451            Ok(())
452        }
453        RunOutcome::Failed {
454            ref status,
455            total_turns,
456            ref token_usage,
457            ..
458        } => {
459            if output.is_json() {
460                output.json(&serde_json::json!({
461                    "event": "run_completed",
462                    "outcome": {
463                        "outcome": "failed",
464                        "status": format!("{status:?}"),
465                        "total_turns": total_turns,
466                        "token_usage": {
467                            "input_tokens": token_usage.input_tokens,
468                            "output_tokens": token_usage.output_tokens,
469                        },
470                    },
471                }))?;
472            } else {
473                output.error(format!("agent failed: {status:?}"))?;
474            }
475            Err(AgentFailed.into())
476        }
477        _ => Err(anyhow::anyhow!("unknown RunOutcome variant")),
478    }
479}

481/// Project a [`MessagePayload`] to a single text string for display.
482/// Non-text payloads degrade to a `Debug`-formatted preview.
483fn format_message_text(payload: &MessagePayload) -> String {
484    match payload {
485        MessagePayload::Text { content } => content.clone(),
486        other => format!("{other:?}"),
487    }
488}

490/// Render the dry-run preview to stderr per spec §3.9.
491fn emit_dry_run(
492    entry: &crate::config::AgentEntry,
493    agent_def: &tau_domain::AgentDefinition,
494    manifest: &tau_domain::PackageManifest,
495    options: &RunOptions,
496    prompt: Option<&str>,
497    output: &mut Output,
498) -> anyhow::Result<()> {
499    output.dry_run(format!(
500        "agent:           {} ({})",
501        entry.id, entry.display_name
502    ))?;
503    output.dry_run(format!(
504        "package:         {} {}",
505        manifest.name(),
506        manifest.version()
507    ))?;
508    output.dry_run(format!("llm backend:     {}", entry.llm_backend))?;
509    if let Some(sp) = &agent_def.system_prompt {
510        let preview: String = sp.chars().take(80).collect();
511        let suffix = if sp.chars().count() > 80 { "..." } else { "" };
512        output.dry_run(format!("system prompt:   {preview}{suffix}"))?;
513    } else {
514        output.dry_run("system prompt:   (none)")?;
515    }
516    output.dry_run(format!(
517        "granted caps:    {}",
518        manifest.capabilities().len()
519    ))?;
520    output.dry_run(format!("max_turns:       {}", options.max_turns))?;
521    let preview = prompt.unwrap_or("(stdin)");
522    let len = preview.len();
523    let trimmed: String = preview.chars().take(80).collect();
524    let suffix = if preview.chars().count() > 80 {
525        "..."
526    } else {
527        ""
528    };
529    output.dry_run(format!(
530        "initial message: {:?}{} ({len} bytes)",
531        trimmed, suffix
532    ))?;
533    output.dry_run("no LLM call made.")?;
534    Ok(())
535}

537#[cfg(test)]
538mod tests {

540    use tau_domain::Value;
541
542    #[test]
543    fn format_message_text_returns_text_content_directly() {
544        let payload = MessagePayload::Text {
545            content: "hello".into(),
546        };
547        assert_eq!(format_message_text(&payload), "hello");
548    }
549
550    #[test]
551    fn format_message_text_falls_back_to_debug_for_non_text() {
552        let payload = MessagePayload::ToolResult { body: Value::Null };
553        let s = format_message_text(&payload);
554        assert!(s.contains("ToolResult"), "got: {s}");
555    }
556
557    #[test]
558    fn agent_failed_renders_error_message() {
559        let err: anyhow::Error = AgentFailed.into();
560        let s = format!("{err}");
561        assert!(s.contains("agent failed"), "got: {s}");
562    }
