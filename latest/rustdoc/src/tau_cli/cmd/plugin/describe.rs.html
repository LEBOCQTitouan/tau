1//! `tau plugin describe <name>` — print a plugin's handshake metadata.
2//!
3//! Per spec §9 (debug tier): resolves the package by name in the
4//! current scope's lockfile, spawns the plugin binary, drives one
5//! `meta.handshake`, prints the advertised name / version / port /
6//! method list / schemas / source, then sends `meta.shutdown` for a
7//! clean exit. The plugin process does not stay alive after this
8//! command returns.
9
10use std::path::PathBuf;
11
12use anyhow::Context;
13use tau_pkg::{LockFile, Scope};
14use tau_plugin_protocol::handshake::TraceContext;
15use tau_runtime::plugin_host::{self, PluginHostOptions, RecordingSink};
16
17use crate::cli::PluginDescribeArgs;
18use crate::output::Output;
19
20/// Run `tau plugin describe`.
21pub async fn run(
22    args: &PluginDescribeArgs,
23    record_protocol: Option<PathBuf>,
24    output: &mut Output,
25) -> anyhow::Result<()> {
26    let cwd = std::env::current_dir()?;
27    let scope = Scope::resolve(&cwd).context("resolving package scope")?;
28
29    let lockfile = LockFile::load(&scope.lockfile_path())
30        .with_context(|| format!("loading lockfile {}", scope.lockfile_path().display()))?;
31
32    let pkg_name: tau_domain::PackageName = args.name.parse().with_context(|| {
33        format!(
34            "invalid package name {:?} (must be lowercase ASCII kebab-case)",
35            args.name
36        )
37    })?;
38
39    let pkg = lockfile.find(&pkg_name).ok_or_else(|| {
40        anyhow::anyhow!(
41            "package {:?} not installed in scope (run `tau install <url>`)",
42            args.name
43        )
44    })?;
45
46    let plugin = pkg.plugin.as_ref().ok_or_else(|| {
47        anyhow::anyhow!(
48            "package {:?} has no [plugin] table in its tau.toml \
49             (it is a data-only package; nothing to describe)",
50            args.name
51        )
52    })?;
53
54    // Build host options. `--record-protocol` is honored here so a
55    // describe invocation is itself capturable in the same JSONL
56    // format an operator inspects via `tau plugin protocol decode`.
57    let mut host_options = PluginHostOptions::default();
58    if let Some(path) = record_protocol {
59        host_options.recording = Some(RecordingSink::JsonlFile { path });
60    }
61
62    let trace_context = TraceContext::new(
63        format!(
64            "tau-describe-{}",
65            std::time::SystemTime::now()
66                .duration_since(std::time::UNIX_EPOCH)
67                .map(|d| d.as_nanos())
68                .unwrap_or(0)
69        ),
70        "describe".to_string(),
71        "root".to_string(),
72    );
73
74    let response = plugin_host::describe_plugin(plugin, trace_context, host_options)
75        .await
76        .with_context(|| format!("describing plugin {:?}", args.name))?;
77
78    if output.is_json() {
79        let payload = serde_json::json!({
80            "package": args.name,
81            "package_version": pkg.active_version.to_string(),
82            "source": pkg.source.to_string(),
83            "binary_path": plugin.binary_path,
84            "manifest": {
85                "provides": format!("{:?}", plugin.manifest.provides),
86                "kind": format!("{:?}", plugin.manifest.kind),
87                "bin": plugin.manifest.bin,
88            },
89            "handshake": {
90                "protocol_version": response.protocol_version,
91                "provides": format!("{:?}", response.provides),
92                "plugin_name": response.plugin_name,
93                "plugin_version": response.plugin_version,
94                "methods": response.methods,
95                "schemas": response.schemas,
96            },
97        });
98        output.json(&payload)?;
99    } else {
100        output.human(&format!("plugin:           {}", args.name))?;
101        output.human(&format!("package version:  {}", pkg.active_version))?;
102        output.human(&format!("source:           {}", pkg.source))?;
103        output.human(&format!(
104            "binary:           {}",
105            plugin.binary_path.display()
106        ))?;
107        output.human(&format!("provides:         {:?}", response.provides))?;
108        output.human(&format!("protocol version: {}", response.protocol_version))?;
109        output.human(&format!(
110            "plugin name:      {} {}",
111            response.plugin_name, response.plugin_version
112        ))?;
113        if response.methods.is_empty() {
114            output.human("methods:          (none beyond meta.*)")?;
115        } else {
116            output.human("methods:")?;
117            for method in &response.methods {
118                output.human(&format!("  - {method}"))?;
119            }
120        }
121        if response.schemas.is_empty() {
122            output.human("schemas:          (none advertised)")?;
123        } else {
124            output.human("schemas:")?;
125            for (method, schema) in &response.schemas {
126                output.human(&format!("  {method}:"))?;
127                let params = serde_json::to_string_pretty(&schema.params)
128                    .unwrap_or_else(|_| "<unrenderable>".into());
129                let result = serde_json::to_string_pretty(&schema.result)
130                    .unwrap_or_else(|_| "<unrenderable>".into());
131                for line in params.lines() {
132                    output.human(&format!("    params: {line}"))?;
133                }
134                for line in result.lines() {
135                    output.human(&format!("    result: {line}"))?;
136                }
137            }
138        }
139    }
140
141    Ok(())
142}
