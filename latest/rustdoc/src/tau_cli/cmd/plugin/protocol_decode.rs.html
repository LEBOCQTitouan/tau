1//! `tau plugin protocol decode <path>` — render a JSONL recording.
2//!
3//! Per spec §9 (debug tier): each line of the input file is one frame
4//! captured by `--record-protocol` (see `tau-runtime::plugin_host::recording`).
5//! The decoder:
6//!
7//! 1. Parses the line as JSON.
8//! 2. Base64-decodes the `frame` field.
9//! 3. MessagePack-decodes the resulting bytes via [`Frame::decode`] to
10//!    surface the inner request / response / notification body as
11//!    JSON-shaped MessagePack content.
12//! 4. Emits the decoded shape — either a human-readable transcript or
13//!    one machine-readable JSON object per line (`--json`).
14//!
15//! Filters (`--filter key=value`), time range (`--from` / `--to`), and
16//! `--json` are honored before emission.
17

28/// Run `tau plugin protocol decode`.
29pub async fn run(args: &PluginProtocolDecodeArgs, output: &mut Output) -> anyhow::Result<()> {
30    let filters = parse_filters(&args.filter)?;
31
32    let contents = tokio::fs::read_to_string(&args.path)
33        .await
34        .with_context(|| format!("reading recording {}", args.path.display()))?;
35
36    // The base timestamp anchors the human transcript's `[+1.234s]`
37    // column to the first frame in range, so transcripts are
38    // self-contained and don't drag wall-clock noise into review.
39    let mut base_ts: Option<f64> = None;
40
41    for (lineno, line) in contents.lines().enumerate() {
42        let line = line.trim();
43        if line.is_empty() {
44            continue;
45        }
46        let raw: serde_json::Value = serde_json::from_str(line).with_context(|| {
47            format!(
48                "parsing line {} of {} as JSON",
49                lineno + 1,
50                args.path.display()
51            )
52        })?;
53
54        let ts = raw.get("ts").and_then(serde_json::Value::as_f64);
55        let plugin = raw
56            .get("plugin")
57            .and_then(serde_json::Value::as_str)
58            .unwrap_or("?");
59        let dir = raw
60            .get("dir")
61            .and_then(serde_json::Value::as_str)
62            .unwrap_or("?");
63        let msgid = raw.get("msgid").and_then(serde_json::Value::as_u64);
64        let method = raw.get("method").and_then(serde_json::Value::as_str);
65        let frame_b64 = raw.get("frame").and_then(serde_json::Value::as_str);
66
67        // ---- Time range ----
68        if let Some(t) = ts {
69            if let Some(from) = args.from {
70                if t < from {
71                    continue;
72                }
73            }
74            if let Some(to) = args.to {
75                if t > to {
76                    continue;
77                }
78            }
79        }
80
81        // ---- Predicate filters ----
82        if !matches_filters(&filters, plugin, dir, method) {
83            continue;
84        }
85
86        // ---- Frame decode ----
87        let decoded = match frame_b64 {
88            Some(s) => match B64.decode(s) {
89                Ok(bytes) => decode_inner(&bytes),
90                Err(e) => Err(anyhow!("base64 decode: {e}")),
91            },
92            None => Err(anyhow!("missing `frame` field")),
93        };
94
95        // ---- Emission ----
96        let rel_ts = ts.map(|t| {
97            let base = *base_ts.get_or_insert(t);
98            t - base
99        });
100
101        if args.json {
102            let payload = serde_json::json!({
103                "ts": ts,
104                "rel_ts": rel_ts,
105                "plugin": plugin,
106                "dir": dir,
107                "msgid": msgid,
108                "method": method,
109                "decoded": match &decoded {
110                    Ok(v) => v.clone(),
111                    Err(e) => serde_json::json!({ "decode_error": format!("{e}") }),
112                },
113            });
114            output.json(&payload)?;
115        } else {
116            let ts_col = match rel_ts {
117                Some(t) => format!("[+{t:>6.3}s]"),
118                None => "[ ?     ]".to_string(),
119            };
120            let header = match (msgid, method) {
121                (Some(id), Some(m)) => format!("{ts_col} {plugin:<16} {dir} msgid={id} {m}"),
122                (Some(id), None) => format!("{ts_col} {plugin:<16} {dir} msgid={id}"),
123                (None, Some(m)) => format!("{ts_col} {plugin:<16} {dir} {m}"),
124                (None, None) => format!("{ts_col} {plugin:<16} {dir}"),
125            };
126            output.human(&header)?;
127            match decoded {
128                Ok(v) => {
129                    let pretty = serde_json::to_string_pretty(&v).unwrap_or_default();
130                    for ln in pretty.lines() {
131                        output.human(&format!("  {ln}"))?;
132                    }
133                }
134                Err(e) => {
135                    output.human(&format!("  <decode error: {e}>"))?;
136                }
137            }
138        }
139    }
140
141    Ok(())
142}

144/// Parse `--filter k=v` arguments into a map. Multiple values for the
145/// same key are not currently supported; the last write wins (kept
146/// simple for v0.1; only one value per key is meaningful for the
147/// supported keys `plugin`, `method`, `dir`).
148fn parse_filters(raw: &[String]) -> anyhow::Result<BTreeMap<String, String>> {
149    let mut out = BTreeMap::new();
150    for entry in raw {
151        let (k, v) = entry
152            .split_once('=')
153            .ok_or_else(|| anyhow!("invalid --filter {entry:?}: expected `key=value`"))?;
154        out.insert(k.to_string(), v.to_string());
155    }
156    Ok(out)
157}

159/// Test a frame's metadata against the user-supplied predicates.
160/// Supported keys: `plugin`, `dir`, `method`.
161fn matches_filters(
162    filters: &BTreeMap<String, String>,
163    plugin: &str,
164    dir: &str,
165    method: Option<&str>,
166) -> bool {
167    for (k, v) in filters {
168        let actual = match k.as_str() {
169            "plugin" => plugin,
170            "dir" => dir,
171            "method" => method.unwrap_or(""),
172            // Unknown keys are not a hard error: log-and-include keeps
173            // the decoder permissive against future recording fields.
174            _ => continue,
175        };
176        if actual != v {
177            return false;
178        }
179    }
180    true
181}

183/// Decode the inner [`Frame`] from raw MessagePack bytes and project
184/// it to a JSON-friendly summary.
185fn decode_inner(bytes: &[u8]) -> anyhow::Result<serde_json::Value> {
186    let frame = Frame::decode(bytes).context("decoding inner Frame")?;
187    let json = match frame {
188        Frame::Request { id, method, params } => serde_json::json!({
189            "kind": "request",
190            "id": id,
191            "method": method,
192            "params": decode_msgpack_to_json(&params),
193        }),
194        Frame::Response { id, error, result } => {
195            let err_json = error.map(|e| {
196                serde_json::json!({
197                    "code": e.code,
198                    "message": e.message,
199                })
200            });
201            let result_json = result.as_deref().map(decode_msgpack_to_json);
202            serde_json::json!({
203                "kind": "response",
204                "id": id,
205                "error": err_json,
206                "result": result_json,
207            })
208        }
209        Frame::Notification { method, params } => serde_json::json!({
210            "kind": "notification",
211            "method": method,
212            "params": decode_msgpack_to_json(&params),
213        }),
214        // Frame is `#[non_exhaustive]`; future variants summarise
215        // generically rather than panicking.
216        _ => serde_json::json!({ "kind": "unknown" }),
217    };
218    Ok(json)
219}

221/// Decode a MessagePack-encoded byte slice to a `serde_json::Value`,
222/// degrading on decode failure to a placeholder string. Used for the
223/// human-friendly `params` / `result` projection only — the lossless
224/// representation is the original base64 in the JSONL line.
225fn decode_msgpack_to_json(bytes: &[u8]) -> serde_json::Value {
226    let value: rmpv::Value = match rmpv::decode::read_value(&mut &bytes[..]) {
227        Ok(v) => v,
228        Err(_) => {
229            return serde_json::json!(format!("<msgpack decode error: {} bytes>", bytes.len()))
230        }
231    };
232    msgpack_value_to_json(&value)
233}
234
235fn msgpack_value_to_json(value: &rmpv::Value) -> serde_json::Value {
236    match value {
237        rmpv::Value::Nil => serde_json::Value::Null,
238        rmpv::Value::Boolean(b) => serde_json::Value::Bool(*b),
239        rmpv::Value::Integer(i) => {
240            if let Some(n) = i.as_i64() {
241                serde_json::Value::Number(n.into())
242            } else if let Some(n) = i.as_u64() {
243                serde_json::Value::Number(n.into())
244            } else {
245                serde_json::Value::String(format!("{i:?}"))
246            }
247        }
248        rmpv::Value::F32(f) => serde_json::Number::from_f64((*f).into())
249            .map(serde_json::Value::Number)
250            .unwrap_or(serde_json::Value::Null),
251        rmpv::Value::F64(f) => serde_json::Number::from_f64(*f)
252            .map(serde_json::Value::Number)
253            .unwrap_or(serde_json::Value::Null),
254        rmpv::Value::String(s) => match s.as_str() {
255            Some(s) => serde_json::Value::String(s.to_string()),
256            None => serde_json::Value::String(format!("{s:?}")),
257        },
258        rmpv::Value::Binary(b) => serde_json::Value::String(format!("<{}b binary>", b.len())),
259        rmpv::Value::Array(a) => {
260            serde_json::Value::Array(a.iter().map(msgpack_value_to_json).collect())
261        }
262        rmpv::Value::Map(m) => {
263            let mut obj = serde_json::Map::with_capacity(m.len());
264            for (k, v) in m {
265                let key = match k {
266                    rmpv::Value::String(s) => s.as_str().unwrap_or("<binary>").to_string(),
267                    other => format!("{other:?}"),
268                };
269                obj.insert(key, msgpack_value_to_json(v));
270            }
271            serde_json::Value::Object(obj)
272        }
273        rmpv::Value::Ext(tag, _bytes) => serde_json::Value::String(format!("<ext tag={tag}>")),
274    }
275}

277#[cfg(test)]
278mod tests {
279    use super::*;
280
281    #[test]
282    fn parse_filters_accepts_kv_pairs() {
283        let f = parse_filters(&["plugin=echo".into(), "dir=h2p".into()]).unwrap();
284        assert_eq!(f.get("plugin").map(String::as_str), Some("echo"));
285        assert_eq!(f.get("dir").map(String::as_str), Some("h2p"));
286    }
287
288    #[test]
289    fn parse_filters_rejects_missing_equals() {
290        let err = parse_filters(&["plugin".into()]).unwrap_err();
291        assert!(format!("{err}").contains("expected `key=value`"));
292    }
293
294    #[test]
295    fn matches_filters_passes_when_all_match() {
296        let mut f = BTreeMap::new();
297        f.insert("plugin".into(), "echo".into());
298        assert!(matches_filters(&f, "echo", "h2p", Some("llm.complete")));
299    }
300
301    #[test]
302    fn matches_filters_rejects_when_any_differs() {
303        let mut f = BTreeMap::new();
304        f.insert("plugin".into(), "echo".into());
305        f.insert("dir".into(), "p2h".into());
306        assert!(!matches_filters(&f, "echo", "h2p", Some("llm.complete")));
307    }
308
309    #[test]
310    fn matches_filters_unknown_key_is_ignored_not_rejected() {
311        let mut f = BTreeMap::new();
312        f.insert("nonsense".into(), "value".into());
313        assert!(matches_filters(&f, "echo", "h2p", None));
314    }
315
316    #[test]
317    fn decode_inner_round_trips_request() {
318        let frame = Frame::Request {
319            id: 7,
320            method: "llm.complete".into(),
321            params: rmp_serde::to_vec(&serde_json::json!({"prompt": "hi"})).unwrap(),
322        };
323        let bytes = frame.encode().unwrap();
324        let json = decode_inner(&bytes).unwrap();
325        assert_eq!(json["kind"], "request");
326        assert_eq!(json["id"], 7);
327        assert_eq!(json["method"], "llm.complete");
328        assert_eq!(json["params"]["prompt"], "hi");
329    }
330
331    #[test]
332    fn decode_inner_handles_response_with_error() {
333        let frame = Frame::Response {
334            id: 1,
335            error: Some(tau_plugin_protocol::RpcErrorEnvelope::new(
336                -32601,
337                "method not found".to_string(),
338                None,
339            )),
340            result: None,
341        };
342        let bytes = frame.encode().unwrap();
343        let json = decode_inner(&bytes).unwrap();
344        assert_eq!(json["kind"], "response");
345        assert_eq!(json["error"]["code"], -32601);
346        assert_eq!(json["error"]["message"], "method not found");
347    }
348}
