1//! `tau chat` — interactive REPL with an agent.
2//!
3//! Per spec §3.13 (chat row) and brainstorm Section 8.
4//!
5//! Mirrors `cmd::run`'s setup (project tau.toml → resolved
6//! `(AgentDefinition, PackageManifest)` → spawned plugin processes →
7//! [`tau_runtime::Runtime`]), then enters a rustyline-driven REPL. Each user prompt
8//! is sent through [`tau_runtime::Runtime::run_with_history`] so the conversation
9//! accumulates across turns; tool dispatch and capability filtering
10//! are inherited from the kernel, not re-implemented here.
11//!
12//! # Plugin lifecycle
13//!
14//! Per spec §11: plugin processes spawn once at REPL entry and stay
15//! alive for the duration of the session (multiplexed long-lived
16//! lifecycle). On `/exit` (or EOF / Ctrl-D), the [`tau_runtime::Runtime`] is
17//! dropped, which drops the `Arc<dyn DynLlmBackend>` and per-tool
18//! `Arc<dyn DynTool>`, which drops the underlying `PluginProcess` —
19//! and `kill_on_drop` ensures the plugin subprocess exits cleanly.
20//!
21//! # Slash commands
22//!
23//! See [`SlashCommand`] for the recognised set. Parsing is intentionally
24//! strict: only the four documented commands are slash-handled, and
25//! anything starting with `/` that isn't a known command is forwarded
26//! to the LLM as a normal prompt. This keeps the surface predictable
27//! and lets users send messages that happen to start with `/`.
28//!
29//! # Failure handling
30//!
31//! - `Ok(RunOutcome::Failed { .. })`: the kernel reported a typed
32//!   agent-level failure (e.g. `OutOfResources`). The REPL renders the
33//!   detail in red, accumulates partial history + tokens, and continues
34//!   — the user can ask a follow-up. (No process exit on agent failure
35//!   here, unlike `tau run`; chat is interactive.)
36//! - `Err(RuntimeError)`: kernel/operational error. The REPL prints the
37//!   error and aborts with the marker bubbling up to `run_main`, which
38//!   maps it to [`crate::ExitCode::Error`] (2).
39//!
40//! # JSON
41//!
42//! `--json` is rejected at handler entry: a streaming, terminal-driven
43//! REPL has no useful JSON projection. (See spec §3.6.)

62/// What a single line of REPL input represents.
63#[derive(Debug, Clone, PartialEq, Eq)]
64pub enum ReplInput {
65    /// A recognised slash command.
66    Slash(SlashCommand),
67    /// Free-form text to forward to the agent. Whitespace-preserved.
68    Prompt(String),
69}
70
71/// Slash commands recognised in the REPL.
72#[derive(Debug, Clone, Copy, PartialEq, Eq)]
73pub enum SlashCommand {
74    /// End the session and print the token-usage summary.
75    Exit,
76    /// Print the slash-command cheat sheet.
77    Help,
78    /// Deprecated: previously dropped in-memory conversation history.
79    Clear,
80    /// Render the current conversation history.
81    History,
82    /// Print information about the current session.
83    Info,
84}
85
86/// Parse a single line of REPL input.
87///
88/// Trims the line for the slash-command match (so `"  /exit "` still
89/// counts) but, for `Prompt`, returns the original line unchanged —
90/// trailing whitespace can be load-bearing for some prompts.
91///
92/// Unknown slash forms (`/foo`) fall through to `Prompt`, intentionally:
93/// see the module docs for the rationale.
94pub fn parse_input(line: &str) -> ReplInput {
95    let trimmed = line.trim();
96    match trimmed {
97        "/exit" => ReplInput::Slash(SlashCommand::Exit),
98        "/help" => ReplInput::Slash(SlashCommand::Help),
99        "/clear" => ReplInput::Slash(SlashCommand::Clear),
100        "/history" => ReplInput::Slash(SlashCommand::History),
101        "/info" => ReplInput::Slash(SlashCommand::Info),
102        _ => ReplInput::Prompt(line.to_string()),
103    }
104}

106/// Run `tau chat`.
107///
108/// `record_protocol` is the optional `--record-protocol <path>` global
109/// flag (Task 20 / spec §9 debug tier). When `Some`, every plugin
110/// frame in either direction is mirrored to the JSONL file at `path`;
111/// the recorders are flushed after the runtime drops to ensure
112/// pending writes reach disk.
113///
114/// `force_passthrough` and `force_adapter_kind` come from the global
115/// `--no-sandbox` / `--sandbox <kind>` flags (Task 7).
116pub async fn run(
117    args: &ChatArgs,
118    record_protocol: Option<PathBuf>,
119    force_passthrough: bool,
120    force_adapter_kind: Option<tau_runtime::sandbox::registry::RegistryKind>,
121    output: &mut Output,
122) -> anyhow::Result<()> {
123    if output.is_json() {
124        anyhow::bail!("tau chat does not support --json (REPL is interactive)");
125    }
126
127    let cwd = std::env::current_dir()?;
128    let project_path = cwd.join("tau.toml");
129    let project = ProjectConfig::from_path(&project_path)
130        .with_context(|| format!("project tau.toml required at {project_path:?}"))?;
131
132    let entry = project.agents.get(&args.agent_id).ok_or_else(|| {
133        anyhow::anyhow!(
134            "agent id {:?} not found in project tau.toml (declare it under [agents.{}])",
135            args.agent_id,
136            args.agent_id
137        )
138    })?;
139
140    let scope = tau_pkg::Scope::resolve(&cwd).context("resolving package scope")?;
141
142    crate::cmd::resolve_helpers::resolve_and_install_for_agent(
143        entry,
144        &scope,
145        args.no_install,
146        output,
147    )?;
148
149    let (agent_def, manifest) = crate::config::build_agent_definition(entry, &cwd, &scope)
150        .with_context(|| format!("resolving agent {:?}", args.agent_id))?;
151
152    let mut options = RunOptions::default();
153    if let Some(n) = args.max_turns {
154        options.max_turns = n;
155    }
156    options.project_override = entry.capability_overrides.clone();
157
158    if args.dry_run {
159        // Dry-run skips plugin spawn entirely (per spec §3.9: no
160        // process side-effects for a preview).
161        emit_dry_run_preview(entry, &agent_def, &manifest, &options, output)?;
162        return Ok(());
163    }
164
165    // ---- Plugin loading -----------------------------------------------------
166    //
167    // Spawn the LLM backend + tool plugins once for the entire REPL.
168    // The runtime owns the Arc<dyn Dyn*> shims; on /exit (or EOF), the
169    // runtime is dropped, which drops the shims, which drops the
170    // PluginProcess (kill_on_drop ensures the child exits).
171
172    let run_id = format!(
173        "tau-chat-{}",
174        std::time::SystemTime::now()
175            .duration_since(std::time::UNIX_EPOCH)
176            .map(|d| d.as_nanos())
177            .unwrap_or(0)
178    );
179    let trace_context = TraceContext::new(run_id, args.agent_id.clone(), "root".to_string());
180    let (host_options, _ledger) = plugin_loader::build_host_options(
181        record_protocol.as_deref(),
182        force_passthrough,
183        force_adapter_kind,
184    );
185
186    let loaded = plugin_loader::load_plugins(entry, &scope, trace_context, host_options).await?;
187    let recorder_ledger = loaded.recorder_ledger.clone();
188
189    let runtime = loaded
190        .builder
191        .build()
192        .context("failed to build runtime from spawned plugins")?;
193
194    // ---- Resume vs fresh-session branch ------------------------------------
195    let result = if let Some(id_or_prefix) = args.resume.as_deref() {
196        let sessions_dir = scope.state_path().join("sessions");
197        let id = crate::session::resolve_id_prefix(&sessions_dir, id_or_prefix)
198            .map_err(|e| anyhow::anyhow!("{e}"))?;
199        let path = sessions_dir.join(format!("{}.jsonl", id.as_str()));
200        let (header, entries) =
201            crate::session::SessionReader::read(&path).map_err(|e| anyhow::anyhow!("{e}"))?;
202
203        // Drift validation (skipped if --force).
204        if !args.force {
205            check_session_drift(&id, &header, &agent_def, &manifest)?;
206        } else {
207            warn_drift_if_any(&header, &agent_def, &manifest, output);
208        }
209
210        // Seed history from entries.
211        let history: Vec<Message> = entries
212            .iter()
213            .filter_map(|e| match e {
214                crate::session::SessionEntry::Message(m) => Some(m.clone()),
215                _ => None,
216            })
217            .collect();
218
219        // Count turns from TurnSummary entries.
220        let turn_count = entries
221            .iter()
222            .filter(|e| matches!(e, crate::session::SessionEntry::TurnSummary { .. }))
223            .count();
224
225        // Reopen the file in append mode.
226        let writer = crate::session::SessionWriter::open_append(&path)
227            .map_err(|e| anyhow::anyhow!("{e}"))?;
228
229        // Print the resume banner.
230        output.status(format!(
231            "Resumed session {} ({} turns)",
232            id.short(),
233            turn_count
234        ))?;
235
236        run_repl(
237            entry,
238            &agent_def,
239            &manifest,
240            &options,
241            args.no_stream,
242            id,
243            Some(writer),
244            history,
245            turn_count as u32,
246            &runtime,
247            output,
248        )
249        .await
250    } else {
251        // Fresh session: mint an id and create the writer now (before
252        // entering the REPL) so the call site mirrors the resume path.
253        let fresh_id = crate::session::mint();
254        let fresh_writer: Option<crate::session::SessionWriter> = if args.ephemeral {
255            output.status("Ephemeral session — not saved to disk")?;
256            None
257        } else {
258            let sessions_dir = scope.state_path().join("sessions");
259            let header = crate::session::SessionHeader::new(
260                &fresh_id,
261                entry.id.clone(),
262                crate::session::SessionPackage {
263                    name: manifest.name().to_string(),
264                    version: manifest.version().to_string(),
265                    resolved_commit: String::new(),
266                },
267                agent_def.llm_backend.to_string(),
268            );
269            let writer = crate::session::SessionWriter::create(&sessions_dir, &fresh_id, &header)?;
270            output.status(format!("Session: {} (will be saved)", fresh_id.short()))?;
271            Some(writer)
272        };
273
274        run_repl(
275            entry,
276            &agent_def,
277            &manifest,
278            &options,
279            args.no_stream,
280            fresh_id,
281            fresh_writer,
282            Vec::new(),
283            0,
284            &runtime,
285            output,
286        )
287        .await
288    };
289
290    // Drop the runtime before flushing recorders so every plugin
291    // process is reaped and the host-side write task is quiescent.
292    drop(runtime);
293    plugin_loader::flush_recorders(recorder_ledger).await;
294
295    result
296}

298/// REPL inner loop, factored out of [`run`] so the caller can wrap it
299/// with `--record-protocol` flush coordination after the runtime is
300/// dropped. Borrows the runtime so the chat session and the recording
301/// flush share a single ownership chain in `run`.
302///
303/// `session_id` is either freshly-minted (new session) or the resolved
304/// id from `--resume`. `session_writer` is `Some(writer)` when the
305/// session should be persisted (pre-created or opened-for-append by the
306/// caller) and `None` for ephemeral sessions. `initial_history` is empty
307/// for fresh sessions and pre-populated on resume. `initial_turn_counter`
308/// is 0 for fresh sessions and the turn count read from the session file
309/// on resume.
310#[allow(clippy::too_many_arguments)]
311async fn run_repl(
312    entry: &crate::config::AgentEntry,
313    agent_def: &tau_domain::AgentDefinition,
314    manifest: &tau_domain::PackageManifest,
315    options: &RunOptions,
316    no_stream: bool,
317    session_id: crate::session::SessionId,
318    session_writer_arg: Option<crate::session::SessionWriter>,
319    initial_history: Vec<Message>,
320    initial_turn_counter: u32,
321    runtime: &tau_runtime::Runtime,
322    output: &mut Output,
323) -> anyhow::Result<()> {
324    let is_resumed = !initial_history.is_empty();
325
326    // Welcome banner.
327    if is_resumed {
328        output.status(format!(
329            "Welcome back to tau chat with agent '{}' ({}@{}). Type /help for commands, /exit or Ctrl-D to quit.",
330            entry.id,
331            manifest.name(),
332            manifest.version()
333        ))?;
334    } else {
335        output.status(format!(
336            "Welcome to tau chat with agent '{}' ({}@{}). Type /help for commands, /exit or Ctrl-D to quit.",
337            entry.id,
338            manifest.name(),
339            manifest.version()
340        ))?;
341    }
342
343    // The caller is responsible for writer creation/opening and
344    // printing the appropriate status message (Session: ... or Ephemeral).
345    let mut session_writer: Option<crate::session::SessionWriter> = session_writer_arg;
346
347    let session_started_at = std::time::SystemTime::now();
348    let mut turn_counter: u32 = initial_turn_counter;
349
350    let mut editor = DefaultEditor::new().context("initialising rustyline editor")?;
351    let mut history: Vec<Message> = initial_history;
352    let mut total_tokens = TokenUsage::default();
353
354    loop {
355        let line = match editor.readline("> ") {
356            Ok(line) => line,
357            Err(ReadlineError::Eof) | Err(ReadlineError::Interrupted) => {
358                close_session_and_summary(
359                    session_writer.take(),
360                    &session_id,
361                    &entry.id,
362                    total_tokens,
363                    output,
364                )?;
365                return Ok(());
366            }
367            Err(e) => {
368                return Err(anyhow::Error::from(e).context("readline failure"));
369            }
370        };
371
372        // Best-effort: rustyline returns Err if the history is full. We
373        // don't surface that — it's pure input ergonomics, not state.
374        let _ = editor.add_history_entry(line.as_str());
375
376        match parse_input(&line) {
377            ReplInput::Slash(SlashCommand::Exit) => {
378                close_session_and_summary(
379                    session_writer.take(),
380                    &session_id,
381                    &entry.id,
382                    total_tokens,
383                    output,
384                )?;
385                return Ok(());
386            }
387            ReplInput::Slash(SlashCommand::Help) => {
388                print_help(output)?;
389            }
390            ReplInput::Slash(SlashCommand::Clear) => {
391                // /clear was removed — in-memory history clear is incoherent
392                // with persistence (clearing in-memory leaves the file intact,
393                // so resume would surface the cleared messages). Print a
394                // deprecation message and continue.
395                output.status(
396                    "/clear was removed. Exit (/exit) and re-run `tau chat <agent>` for a fresh session.",
397                )?;
398            }
399            ReplInput::Slash(SlashCommand::History) => {
400                render_history(&history, output)?;
401            }
402            ReplInput::Slash(SlashCommand::Info) => {
403                let path_label = session_writer
404                    .as_ref()
405                    .map(|w| w.path().display().to_string())
406                    .unwrap_or_else(|| "(ephemeral; not saved)".to_string());
407                let started = humantime::format_rfc3339_seconds(session_started_at).to_string();
408                output.status(format!(
409                    "Session: {}\nFile: {}\nTurns: {}\nStarted: {}\nAgent: {} ({}@{})",
410                    session_id.as_str(),
411                    path_label,
412                    turn_counter,
413                    started,
414                    entry.id,
415                    manifest.name(),
416                    manifest.version()
417                ))?;
418            }
419            ReplInput::Prompt(text) if text.trim().is_empty() => {
420                // Empty / whitespace-only input — no-op. Skip the round
421                // trip rather than send an empty prompt.
422                continue;
423            }
424            ReplInput::Prompt(text) => {
425                let prev_history_len = history.len();
426                let initial = Message::new(
427                    Address::User,
428                    // The kernel mints its own per-run AgentInstanceId;
429                    // this recipient placeholder is overwritten by the
430                    // loop. Fresh id keeps the typed Address::Agent
431                    // variant honest.
432                    Address::Agent(AgentInstanceId::new()),
433                    MessagePayload::Text { content: text },
434                );
435
436                if no_stream {
437                    // ---- Batch (non-streaming) path ----
438                    let outcome = runtime
439                        .run_with_history(
440                            agent_def.clone(),
441                            manifest.clone(),
442                            history.clone(),
443                            initial,
444                            options.clone(),
445                        )
446                        .await;
447
448                    match outcome {
449                        Ok(RunOutcome::Completed {
450                            final_message,
451                            all_messages,
452                            token_usage,
453                            ..
454                        }) => {
455                            render_final_message(&final_message, output)?;
456                            accumulate_tokens(&mut total_tokens, &token_usage);
457                            // Persist new messages before updating history.
458                            if let Some(writer) = session_writer.as_mut() {
459                                let new_messages = &all_messages[prev_history_len..];
460                                if let Err(e) = writer.append_messages(new_messages) {
461                                    tracing::warn!(
462                                        name = "session.write_failed",
463                                        error = %e,
464                                        "session write failed; continuing"
465                                    );
466                                } else {
467                                    turn_counter += 1;
468                                    let stop_str = "Completed".to_string();
469                                    let _ = writer.append_turn_summary(
470                                        turn_counter,
471                                        &stop_str,
472                                        Some(token_usage.input_tokens),
473                                        Some(token_usage.output_tokens),
474                                    );
475                                }
476                            }
477                            history = all_messages;
478                        }
479                        Ok(RunOutcome::Failed {
480                            status,
481                            all_messages,
482                            token_usage,
483                            ..
484                        }) => {
485                            // Agent-level failure: render in red and let the
486                            // user retry. (Distinct from a kernel error,
487                            // which aborts the REPL below.)
488                            output.error(format!("agent failed: {status:?}"))?;
489                            accumulate_tokens(&mut total_tokens, &token_usage);
490                            // Persist partial messages even on failure.
491                            if let Some(writer) = session_writer.as_mut() {
492                                let new_messages = &all_messages[prev_history_len..];
493                                if let Err(e) = writer.append_messages(new_messages) {
494                                    tracing::warn!(
495                                        name = "session.write_failed",
496                                        error = %e,
497                                        "session write failed; continuing"
498                                    );
499                                } else {
500                                    turn_counter += 1;
501                                    let stop_str = format!("Failed({status:?})");
502                                    let _ = writer.append_turn_summary(
503                                        turn_counter,
504                                        &stop_str,
505                                        Some(token_usage.input_tokens),
506                                        Some(token_usage.output_tokens),
507                                    );
508                                }
509                            }
510                            history = all_messages;
511                        }
512                        Ok(_) => {
513                            return Err(anyhow::anyhow!("unknown RunOutcome variant"));
514                        }
515                        Err(e) => {
516                            // Kernel/operational error: print and abort the
517                            // REPL with a kernel-error exit. Token usage so
518                            // far is still surfaced.
519                            output.error(format!("kernel error: {e}"))?;
520                            return Err(anyhow::Error::from(e));
521                        }
522                    }
523                } else {
524                    // ---- Streaming path (default) ----
525                    //
526                    // Two-pass rendering per spec §4.5:
527                    //   Pass 1 (during streaming): text deltas via raw
528                    //     print!/flush (typewriter UX); tool annotations
529                    //     to stderr so stdout stays the agent's text.
530                    //   Pass 2 (on RunCompleted): re-render via termimad
531                    //     (render_final_message) so the user sees final
532                    //     formatted markdown after the rough-draft pass.
533                    let stream_result = runtime
534                        .run_streaming_with_history(
535                            agent_def.clone(),
536                            manifest.clone(),
537                            history.clone(),
538                            initial,
539                            options.clone(),
540                        )
541                        .await;
542
543                    let stream = match stream_result {
544                        Ok(s) => s,
545                        Err(e) => {
546                            output.error(format!("kernel error: {e}"))?;
547                            return Err(anyhow::Error::from(e));
548                        }
549                    };
550                    let mut stream = Box::pin(stream);
551                    let mut stdout = io::stdout();
552
553                    loop {
554                        let next = std::future::poll_fn(|cx| stream.as_mut().poll_next(cx)).await;
555                        match next {
556                            Some(RunEvent::TextDelta { delta }) => {
557                                // Pass 1: typewriter output — raw print to
558                                // stdout, bypassing Output's writer, so the
559                                // text appears inline without a trailing newline
560                                // until the full message is assembled.
561                                print!("{delta}");
562                                let _ = stdout.flush();
563                            }
564                            Some(RunEvent::ToolCallStarted { name, .. }) => {
565                                // Route tool annotations to stderr so stdout
566                                // remains the agent's clean text stream.
567                                eprintln!("-> calling {name}...");
568                            }
569                            Some(RunEvent::ToolCallCompleted { name, result, .. }) => {
570                                match result {
571                                    Ok(_) => eprintln!("✓ {name} completed"),
572                                    Err(reason) => eprintln!("✗ {name} failed: {reason}"),
573                                }
574                            }
575                            Some(RunEvent::TurnCompleted { .. }) => {
576                                // Nothing during streaming; re-render on
577                                // RunCompleted below.
578                            }
579                            Some(RunEvent::FatalError { kind, detail, .. }) => {
580                                // Kernel-level fatal error: mirror the batch
581                                // path's Err(RuntimeError) handling — print
582                                // and abort the REPL.
583                                println!(); // end the typewriter line
584                                output.error(format!("kernel error ({kind}): {detail}"))?;
585                                return Err(anyhow::anyhow!(
586                                    "streaming fatal error ({kind}): {detail}"
587                                ));
588                            }
589                            Some(RunEvent::RunCompleted { outcome }) => {
590                                // Pass 2: close typewriter line, then
591                                // re-render the final message with termimad
592                                // markdown formatting.
593                                println!();
594                                match outcome {
595                                    RunOutcome::Completed {
596                                        final_message,
597                                        all_messages,
598                                        token_usage,
599                                        ..
600                                    } => {
601                                        render_final_message(&final_message, output)?;
602                                        accumulate_tokens(&mut total_tokens, &token_usage);
603                                        // Persist new messages before updating history.
604                                        if let Some(writer) = session_writer.as_mut() {
605                                            let new_messages = &all_messages[prev_history_len..];
606                                            if let Err(e) = writer.append_messages(new_messages) {
607                                                tracing::warn!(
608                                                    name = "session.write_failed",
609                                                    error = %e,
610                                                    "session write failed; continuing"
611                                                );
612                                            } else {
613                                                turn_counter += 1;
614                                                let stop_str = "Completed".to_string();
615                                                let _ = writer.append_turn_summary(
616                                                    turn_counter,
617                                                    &stop_str,
618                                                    Some(token_usage.input_tokens),
619                                                    Some(token_usage.output_tokens),
620                                                );
621                                            }
622                                        }
623                                        history = all_messages;
624                                    }
625                                    RunOutcome::Failed {
626                                        status,
627                                        all_messages,
628                                        token_usage,
629                                        ..
630                                    } => {
631                                        output.error(format!("agent failed: {status:?}"))?;
632                                        accumulate_tokens(&mut total_tokens, &token_usage);
633                                        // Persist partial messages even on failure.
634                                        if let Some(writer) = session_writer.as_mut() {
635                                            let new_messages = &all_messages[prev_history_len..];
636                                            if let Err(e) = writer.append_messages(new_messages) {
637                                                tracing::warn!(
638                                                    name = "session.write_failed",
639                                                    error = %e,
640                                                    "session write failed; continuing"
641                                                );
642                                            } else {
643                                                turn_counter += 1;
644                                                let stop_str = format!("Failed({status:?})");
645                                                let _ = writer.append_turn_summary(
646                                                    turn_counter,
647                                                    &stop_str,
648                                                    Some(token_usage.input_tokens),
649                                                    Some(token_usage.output_tokens),
650                                                );
651                                            }
652                                        }
653                                        history = all_messages;
654                                    }
655                                    _ => {
656                                        return Err(anyhow::anyhow!(
657                                            "unknown RunOutcome variant in streaming path"
658                                        ));
659                                    }
660                                }
661                                break;
662                            }
663                            None => {
664                                // Stream exhausted without RunCompleted — should
665                                // not happen per stream invariants, but handle
666                                // defensively.
667                                println!();
668                                break;
669                            }
670                            // RunEvent is #[non_exhaustive]; ignore unknown
671                            // future variants so forward-compatibility is
672                            // preserved.
673                            Some(_) => {}
674                        }
675                    }
676                }
677            }
678        }
679    }
680}

682/// Close the session writer (if any) and emit the end-of-session summary.
683///
684/// Prints a resume hint for non-ephemeral sessions or "Session discarded."
685/// for ephemeral ones. Called from both exit paths (`/exit` and EOF/Ctrl-D).
686fn close_session_and_summary(
687    session_writer: Option<crate::session::SessionWriter>,
688    session_id: &crate::session::SessionId,
689    agent_id: &str,
690    total_tokens: TokenUsage,
691    output: &mut Output,
692) -> anyhow::Result<()> {
693    emit_session_summary(total_tokens, output)?;
694    if let Some(writer) = session_writer {
695        writer.close()?;
696        output.status(format!(
697            "Session saved. Resume with: tau chat {} --resume {}",
698            agent_id,
699            session_id.short()
700        ))?;
701    } else {
702        output.status("Session discarded.")?;
703    }
704    Ok(())
705}

707/// Render the final message of a completed turn. Text payloads go
708/// through termimad for markdown rendering; non-text payloads degrade
709/// to a `Debug` preview through the standard human channel.
710fn render_final_message(msg: &Message, output: &mut Output) -> anyhow::Result<()> {
711    match &msg.payload {
712        MessagePayload::Text { content } => {
713            // termimad writes directly to stdout via crossterm; this
714            // bypasses `Output`'s writer abstraction, but the REPL only
715            // runs in non-JSON mode (rejected at entry) and stdout is
716            // exactly where the agent's text response belongs.
717            let skin = termimad::MadSkin::default_dark();
718            skin.print_text(content);
719            Ok(())
720        }
721        other => output
722            .human(&format!("{other:?}"))
723            .context("writing non-text final message"),
724    }
725}

727/// Accumulate per-turn token usage into the session-wide running total.
728fn accumulate_tokens(total: &mut TokenUsage, delta: &TokenUsage) {
729    total.input_tokens = total.input_tokens.saturating_add(delta.input_tokens);
730    total.output_tokens = total.output_tokens.saturating_add(delta.output_tokens);
731    // `total_tokens` is the backend-reported unified count when present;
732    // sum it when both sides are Some, otherwise leave the existing
733    // value. Most backends report only input/output, so this is largely
734    // a courtesy.
735    total.total_tokens = match (total.total_tokens, delta.total_tokens) {
736        (Some(a), Some(b)) => Some(a.saturating_add(b)),
737        (Some(a), None) => Some(a),
738        (None, Some(b)) => Some(b),
739        (None, None) => None,
740    };
741}

743/// Render a one-line summary of every message in the session history.
744fn render_history(history: &[Message], output: &mut Output) -> anyhow::Result<()> {
745    if history.is_empty() {
746        output.human("(no history yet)")?;
747        return Ok(());
748    }
749    for (i, msg) in history.iter().enumerate() {
750        let preview = match &msg.payload {
751            MessagePayload::Text { content } => {
752                let p: String = content.chars().take(80).collect();
753                if content.chars().count() > 80 {
754                    format!("{p}...")
755                } else {
756                    p
757                }
758            }
759            other => format!("{other:?}"),
760        };
761        output.human(&format!(
762            "[{i}] {:?} -> {:?}: {preview}",
763            msg.sender, msg.recipient
764        ))?;
765    }
766    Ok(())
767}

789/// Render the dry-run preview to stderr. Mirrors `cmd::run`'s preview,
790/// minus the `initial message:` row (no prompt yet — we'd be entering
791/// the REPL).
792fn emit_dry_run_preview(
793    entry: &crate::config::AgentEntry,
794    agent_def: &tau_domain::AgentDefinition,
795    manifest: &tau_domain::PackageManifest,
796    options: &RunOptions,
797    output: &mut Output,
798) -> anyhow::Result<()> {
799    output.dry_run(format!(
800        "agent:           {} ({})",
801        entry.id, entry.display_name
802    ))?;
803    output.dry_run(format!(
804        "package:         {} {}",
805        manifest.name(),
806        manifest.version()
807    ))?;
808    output.dry_run(format!("llm backend:     {}", entry.llm_backend))?;
809    if let Some(sp) = &agent_def.system_prompt {
810        let preview: String = sp.chars().take(80).collect();
811        let suffix = if sp.chars().count() > 80 { "..." } else { "" };
812        output.dry_run(format!("system prompt:   {preview}{suffix}"))?;
813    } else {
814        output.dry_run("system prompt:   (none)")?;
815    }
816    output.dry_run(format!(
817        "granted caps:    {}",
818        manifest.capabilities().len()
819    ))?;
820    output.dry_run(format!("max_turns:       {}", options.max_turns))?;
821    output.dry_run("REPL would start with the above setup.")?;
822    output.dry_run("no session opened.")?;
823    Ok(())
824}

826/// Validate that the session header's recorded metadata still matches the
827/// current project state. Returns `Err` describing the first mismatch found.
828/// Called on `--resume` when `--force` is not set.
829fn check_session_drift(
830    session_id: &crate::session::SessionId,
831    header: &crate::session::SessionHeader,
832    agent_def: &tau_domain::AgentDefinition,
833    manifest: &tau_domain::PackageManifest,
834) -> anyhow::Result<()> {
835    let current_agent_id = agent_def.id.to_string();
836    let current_pkg_name = manifest.name().to_string();
837    let current_pkg_version = manifest.version().to_string();
838    let current_llm = agent_def.llm_backend.to_string();
839
840    if header.agent_id != current_agent_id {
841        return Err(anyhow::anyhow!(
842            "session {} drift: agent_id was {:?}, now {:?}. Use --force to resume anyway.",
843            session_id.short(),
844            header.agent_id,
845            current_agent_id
846        ));
847    }
848    if header.package.name != current_pkg_name {
849        return Err(anyhow::anyhow!(
850            "session {} drift: package.name was {:?}, now {:?}. Use --force to resume anyway.",
851            session_id.short(),
852            header.package.name,
853            current_pkg_name
854        ));
855    }
856    if header.package.version != current_pkg_version {
857        return Err(anyhow::anyhow!(
858            "session {} drift: package.version was {:?}, now {:?}. Use --force to resume anyway.",
859            session_id.short(),
860            header.package.version,
861            current_pkg_version
862        ));
863    }
864    if header.llm_backend != current_llm {
865        return Err(anyhow::anyhow!(
866            "session {} drift: llm_backend was {:?}, now {:?}. Use --force to resume anyway.",
867            session_id.short(),
868            header.llm_backend,
869            current_llm
870        ));
871    }
872    Ok(())
873}

875/// Emit warnings for any drift detected but do not abort. Called on
876/// `--resume --force`.
877fn warn_drift_if_any(
878    header: &crate::session::SessionHeader,
879    agent_def: &tau_domain::AgentDefinition,
880    manifest: &tau_domain::PackageManifest,
881    output: &mut Output,
882) {
883    let current_pkg_name = manifest.name().to_string();
884    let current_pkg_version = manifest.version().to_string();
885    let current_llm = agent_def.llm_backend.to_string();
886    let current_agent = agent_def.id.to_string();
887
888    if header.package.version != current_pkg_version || header.package.name != current_pkg_name {
889        let _ = output.status(format!(
890            "warning: resuming with {}@{} (session was {}@{})",
891            current_pkg_name, current_pkg_version, header.package.name, header.package.version
892        ));
893    }
894    if header.llm_backend != current_llm {
895        let _ = output.status(format!(
896            "warning: resuming with llm_backend {} (session was {})",
897            current_llm, header.llm_backend
898        ));
899    }
900    if header.agent_id != current_agent {
901        let _ = output.status(format!(
902            "warning: resuming with agent_id {} (session was {})",
903            current_agent, header.agent_id
904        ));
905    }
906}

974        // is blocked from this crate. Use default() + field mutation.
975        let mut total = TokenUsage::default();

989        assert_eq!(total.input_tokens, 13);
