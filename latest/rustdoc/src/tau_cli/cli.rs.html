1//! Command-line argument parser. clap v4 derive.
2//!
3//! [`Cli`] is the top-level parser; [`Command`] enumerates the v0.1
4//! subcommands plus the debug-tier `plugin` subcommand group. Each
5//! subcommand has its own `*Args` struct.
6//!
7//! Global flags (verbose, quiet, debug, color, json, record-protocol)
8//! are accessible from any subcommand via the parsed [`Cli`].
9
10use std::path::PathBuf;
11
12use clap::{Args, Parser, Subcommand, ValueEnum};
13
14/// Top-level parser for the `tau` binary.
15#[derive(Parser, Debug)]
16#[command(name = "tau", version, about = "tau runtime CLI")]
17pub struct Cli {
18    /// The subcommand to dispatch.
19    #[command(subcommand)]
20    pub command: Command,
21
22    /// Verbosity level. Repeat for more (-v: DEBUG, -vv: TRACE).
23    #[arg(short, long, action = clap::ArgAction::Count, global = true)]
24    pub verbose: u8,
25
26    /// Suppress non-error output (sets tracing to WARN).
27    #[arg(short, long, global = true, conflicts_with = "verbose")]
28    pub quiet: bool,
29
30    /// Show full error chain + DEBUG-level tracing on failures.
31    #[arg(long, global = true)]
32    pub debug: bool,
33
34    /// Color mode for terminal output.
35    #[arg(long, global = true, value_enum, default_value = "auto")]
36    pub color: ColorMode,
37
38    /// Emit structured JSON output instead of human-readable.
39    /// Only supported on `install`, `list`, `run`, `init`.
40    #[arg(long, global = true)]
41    pub json: bool,
42
43    /// Mirror every plugin protocol frame to a JSONL file at this path.
44    ///
45    /// Useful for debugging plugin behavior and replaying via
46    /// `tau plugin protocol decode <path>`. The recording is best-effort
47    /// per spec §7.8: write/open failures are logged but never abort the
48    /// invocation.
49    #[arg(long, global = true, value_name = "PATH")]
50    pub record_protocol: Option<PathBuf>,
51
52    /// Disable sandboxing entirely for this invocation (shorthand for
53    /// `--sandbox passthrough`). For development and one-off debugging.
54    /// Auditable in shell history.
55    #[arg(long, global = true)]
56    pub no_sandbox: bool,
57
58    /// Force a specific sandbox adapter kind, overriding the resolver.
59    /// `--no-sandbox` is shorthand for `--sandbox passthrough`. Conflicts
60    /// with `--no-sandbox` if both set.
61    #[arg(long, global = true, value_enum, conflicts_with = "no_sandbox")]
62    pub sandbox: Option<SandboxKindArg>,
63}
64
65/// CLI value for `--sandbox <kind>`. Maps to the resolver's adapter
66/// kinds. `passthrough` is exposed so `--sandbox passthrough` is a
67/// fully-spelled equivalent of `--no-sandbox`.
68#[derive(Debug, Clone, Copy, PartialEq, Eq, ValueEnum)]
69#[value(rename_all = "kebab-case")]
70pub enum SandboxKindArg {
71    /// Linux landlock + seccomp + namespaces.
72    Native,
73    /// Docker / Podman shell-out.
74    Container,
75    /// No isolation (explicit opt-out).
76    Passthrough,
77}

79/// All v0.1 subcommands.
80#[derive(Subcommand, Debug)]
81pub enum Command {
82    /// Scaffold a project `tau.toml`.
83    Init(InitArgs),
84    /// Install a package from a Git URL.
85    Install(InstallArgs),
86    /// List installed packages or available agents.
87    List(ListArgs),
88    /// Invoke an agent one-shot.
89    Run(RunArgs),
90    /// Open a REPL chat session with an agent.
91    Chat(ChatArgs),
92    /// Install missing requires.tools dependencies for all agents in
93    /// the project tau.toml. Project-wide form of the lazy resolve
94    /// that `tau run` and `tau chat` perform per-agent.
95    Resolve(ResolveArgs),
96    /// Uninstall a package and remove its lockfile entry.
97    Uninstall(UninstallArgs),
98    /// Verify installed packages against the lockfile (spec §3).
99    Verify(VerifyArgs),
100    /// Update an installed package to a newer or specific version (spec §2).
101    Update(UpdateArgs),
102    /// Plugin debugging utilities (spec §9 debug tier).
103    Plugin {
104        /// Sub-action within the plugin group.
105        #[command(subcommand)]
106        action: PluginAction,
107    },
108    /// Session management (list, show, delete, export).
109    Session(SessionArgs),
110    /// Sandbox configuration and diagnostics.
111    Sandbox(SandboxArgs),
112    /// Workflow subcommand group (list, run, log, resume).
113    #[command(subcommand)]
114    Workflow(WorkflowSubcommand),
115    /// Inspect installed skill packages (list, show).
116    #[command(subcommand)]
117    Skill(SkillSubcommand),
118    /// Start serve mode: accept JSON-RPC requests over stdio.
119    Serve(ServeArgs),
120}
