1//! UUID v7 session ids with prefix resolution.
2//!
3//! v7 = timestamp-prefixed UUIDs (sortable lexicographically by
4//! creation time). Matches the `AgentInstanceId::new()` precedent in
5//! `tau_domain`. CLI accepts shortened prefixes (≥8 chars); resolution
6//! finds the longest unique match.
7
8use std::fs;
9use std::path::Path;
10
11use uuid::Uuid;
12
13use super::SessionError;
14
15/// Minimum prefix length the CLI will accept.
16pub const MIN_PREFIX_LEN: usize = 8;
17
18/// A session id wrapping a UUID v7.
19#[non_exhaustive]
20#[derive(Debug, Clone, PartialEq, Eq, Hash)]
21pub struct SessionId(Uuid);
22
23impl SessionId {
24    /// Wrap a raw `Uuid` (used by parsers/tests).
25    // Staged for Task 6 (resume parser).
26    #[allow(dead_code)]
27    pub fn from_uuid(u: Uuid) -> Self {
28        Self(u)
29    }
30
31    /// Underlying UUID.
32    // Staged for Task 6+ internals.
33    #[allow(dead_code)]
34    pub fn as_uuid(&self) -> &Uuid {
35        &self.0
36    }
37
38    /// Full 36-char canonical form.
39    pub fn as_str(&self) -> String {
40        self.0.to_string()
41    }
42
43    /// 8-char prefix used for displays and stem of the JSONL filename.
44    pub fn short(&self) -> String {
45        self.0.to_string()[..MIN_PREFIX_LEN].to_string()
46    }
47}
48
49impl std::fmt::Display for SessionId {
50    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
51        self.0.fmt(f)
52    }
53}

55impl std::str::FromStr for SessionId {
56    type Err = uuid::Error;
57    fn from_str(s: &str) -> Result<Self, Self::Err> {
58        Uuid::parse_str(s).map(Self)
59    }
60}

62/// Mint a new session id (UUID v7 — timestamp-prefixed, sortable).
63pub fn mint() -> SessionId {
64    SessionId(Uuid::now_v7())
65}
66
67/// Resolve a user-supplied id-or-prefix to an exact `SessionId` by
68/// scanning `<sessions_dir>/*.jsonl`.
69// Staged for Task 6 (--resume flag).
70#[allow(dead_code)]
71///
72/// - Exact 36-char match: short-circuits.
73/// - 8+ char prefix: matches against canonical filenames; one hit
74///   = success, multiple = `AmbiguousPrefix`, zero = `NotFound`.
75/// - Anything else: `NotFound`.
76///
77/// `sessions_dir` is `<scope.state_path()>/sessions`. If the dir does
78/// not exist, returns `NotFound`.
79pub fn resolve_id_prefix(
80    sessions_dir: &Path,
81    id_or_prefix: &str,
82) -> Result<SessionId, SessionError> {
83    // Exact UUID? Skip the directory walk.
84    if let Ok(uuid) = id_or_prefix.parse::<Uuid>() {
85        let target = sessions_dir.join(format!("{uuid}.jsonl"));
86        if target.exists() {
87            return Ok(SessionId(uuid));
88        }
89        return Err(SessionError::NotFound {
90            id_or_prefix: id_or_prefix.to_string(),
91            scope_path: sessions_dir.to_path_buf(),
92        });
93    }
94
95    if id_or_prefix.len() < MIN_PREFIX_LEN {
96        return Err(SessionError::NotFound {
97            id_or_prefix: id_or_prefix.to_string(),
98            scope_path: sessions_dir.to_path_buf(),
99        });
100    }
101
102    if !sessions_dir.exists() {
103        return Err(SessionError::NotFound {
104            id_or_prefix: id_or_prefix.to_string(),
105            scope_path: sessions_dir.to_path_buf(),
106        });
107    }
108
109    let entries = fs::read_dir(sessions_dir).map_err(|e| SessionError::Io {
110        path: sessions_dir.to_path_buf(),
111        message: format!("listing sessions dir: {e}"),
112    })?;
113
114    let mut matches: Vec<Uuid> = Vec::new();
115    for entry in entries {
116        let entry = entry.map_err(|e| SessionError::Io {
117            path: sessions_dir.to_path_buf(),
118            message: format!("reading dir entry: {e}"),
119        })?;
120        let Some(name) = entry.file_name().to_str().map(|s| s.to_string()) else {
121            continue;
122        };
123        let Some(stem) = name.strip_suffix(".jsonl") else {
124            continue;
125        };
126        if !stem.starts_with(id_or_prefix) {
127            continue;
128        }
129        if let Ok(uuid) = stem.parse::<Uuid>() {
130            matches.push(uuid);
131        }
132    }
133
134    match matches.len() {
135        0 => Err(SessionError::NotFound {
136            id_or_prefix: id_or_prefix.to_string(),
137            scope_path: sessions_dir.to_path_buf(),
138        }),
139        1 => Ok(SessionId(matches.into_iter().next().unwrap())),
140        _ => {
141            let candidates = matches
142                .iter()
143                .map(|u| u.to_string()[..MIN_PREFIX_LEN].to_string())
144                .collect();
145            Err(SessionError::AmbiguousPrefix {
146                prefix: id_or_prefix.to_string(),
147                candidates,
148            })
149        }
150    }
151}

153#[cfg(test)]
154mod tests {
155    use super::*;
156    use std::fs;
157    use tempfile::TempDir;
158
159    fn touch_session(dir: &Path, id: &str) {
160        fs::write(dir.join(format!("{id}.jsonl")), b"{}\n").unwrap();
161    }
162
163    #[test]
164    fn mint_returns_v7_uuid() {
165        let id1 = mint();
166        let id2 = mint();
167        // v7 → timestamp-prefixed; sortable; non-equal across calls.
168        assert_ne!(id1, id2);
169        assert_eq!(id1.as_str().len(), 36);
170        assert_eq!(id1.short().len(), 8);
171    }
172
173    #[test]
174    fn resolve_exact_match_succeeds() {
175        let td = TempDir::new().unwrap();
176        let id = mint();
177        touch_session(td.path(), &id.as_str());
178        let got = resolve_id_prefix(td.path(), &id.as_str()).unwrap();
179        assert_eq!(got, id);
180    }
181
182    #[test]
183    fn resolve_prefix_match_succeeds() {
184        let td = TempDir::new().unwrap();
185        let id = mint();
186        touch_session(td.path(), &id.as_str());
187        let prefix = &id.as_str()[..10];
188        let got = resolve_id_prefix(td.path(), prefix).unwrap();
189        assert_eq!(got, id);
190    }
191
192    #[test]
193    fn resolve_short_prefix_returns_not_found() {
194        let td = TempDir::new().unwrap();
195        let id = mint();
196        touch_session(td.path(), &id.as_str());
197        // Less than MIN_PREFIX_LEN chars.
198        let err = resolve_id_prefix(td.path(), "abc").unwrap_err();
199        assert!(matches!(err, SessionError::NotFound { .. }));
200    }
201
202    #[test]
203    fn resolve_unknown_id_returns_not_found() {
204        let td = TempDir::new().unwrap();
205        let id = mint();
206        touch_session(td.path(), &id.as_str());
207        let err = resolve_id_prefix(td.path(), "00000000").unwrap_err();
208        assert!(matches!(err, SessionError::NotFound { .. }));
209    }
210
211    #[test]
212    fn resolve_ambiguous_prefix_returns_candidates() {
213        let td = TempDir::new().unwrap();
214        // Two ids that share a known prefix. Use UUIDs constructed by
215        // hand so we can guarantee the shared first byte.
216        let a = "01234567-0000-7000-8000-000000000001";
217        let b = "01234567-0000-7000-8000-000000000002";
218        touch_session(td.path(), a);
219        touch_session(td.path(), b);
220        let err = resolve_id_prefix(td.path(), "01234567").unwrap_err();
221        let SessionError::AmbiguousPrefix { candidates, .. } = err else {
222            panic!("expected AmbiguousPrefix")
223        };
224        assert_eq!(candidates.len(), 2);
225    }
226}
