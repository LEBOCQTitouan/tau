26/// Top-level entry point used by `main` and integration tests.
27///
28/// Parses CLI arguments, dispatches to the appropriate `cmd::*::run`
29/// handler, and maps the result to a process exit code.
30///
31/// At v0.1 of Task 4, all subcommand handlers are stubs that return
32/// an "unimplemented" error. Tasks 10-14 land the real handlers.
33pub async fn run_main() -> std::process::ExitCode {
34    let cli = cli::Cli::parse();
35    tracing::install(&cli);
36    // Capture `cli.debug` before `dispatch` consumes the parsed `Cli`.
37    // When set, the error path renders the full `anyhow` chain via
38    // `{err:?}` instead of the single-line top-level message. This is
39    // the integration-level surface for `--debug` (the tracing module
40    // already promotes the filter to DEBUG independently).
41    let debug = cli.debug;
42    match dispatch(cli).await {
43        Ok(()) => ExitCode::Success.into(),
44        Err(err) => {
45            // The AgentFailed marker is emitted by `cmd::run` when the
46            // agent reaches `RunOutcome::Failed`. It must NOT print the
47            // generic "error:" prefix — the run handler has already
48            // emitted a structured failure to the user. All other
49            // errors are kernel/CLI failures; they get the prefix and
50            // map to `ExitCode::Error`.
51            if err.downcast_ref::<crate::cmd::run::AgentFailed>().is_some() {
52                ExitCode::AgentFailed.into()
53            } else {
54                if debug {
55                    eprintln!("error: {err:?}");
56                } else {
57                    eprintln!("error: {err}");
58                }
59                ExitCode::from(&err).into()
60            }
61        }
62    }
63}

65async fn dispatch(cli: cli::Cli) -> anyhow::Result<()> {
66    use cli::SandboxKindArg;
67    use tau_runtime::sandbox::registry::RegistryKind;
68
69    let mut output = Output::from_cli(&cli);
70    let record_protocol = cli.record_protocol.clone();
71
72    // Resolve sandbox override flags (--no-sandbox / --sandbox <kind>) once,
73    // so every plugin-loading subcommand gets the same derived values.
74    let force_passthrough =
75        cli.no_sandbox || matches!(cli.sandbox, Some(SandboxKindArg::Passthrough));
76    let force_adapter_kind: Option<RegistryKind> = match cli.sandbox {
77        Some(SandboxKindArg::Native) => Some(RegistryKind::Native),
78        Some(SandboxKindArg::Container) => Some(RegistryKind::Container),
79        Some(SandboxKindArg::Passthrough) | None => None,
80    };
81
82    match cli.command {
83        cli::Command::Init(args) => cmd::init::run(&args, &mut output).await,
84        cli::Command::Install(args) => cmd::install::run(&args, &mut output).await,
85        cli::Command::List(args) => cmd::list::run(&args, &mut output).await,
86        cli::Command::Run(args) => {
87            cmd::run::run(
88                &args,
89                record_protocol,
90                force_passthrough,
91                force_adapter_kind,
92                &mut output,
93            )
94            .await
95        }
96        cli::Command::Chat(args) => {
97            cmd::chat::run(
98                &args,
99                record_protocol,
100                force_passthrough,
101                force_adapter_kind,
102                &mut output,
103            )
104            .await
105        }
106        cli::Command::Resolve(args) => cmd::resolve::run(&args, &mut output).await,
107        cli::Command::Uninstall(args) => cmd::uninstall::run(&args, &mut output).await,
108        cli::Command::Update(args) => cmd::update::run(&args, &mut output).await,
109        cli::Command::Verify(args) => cmd::verify::run(&args, &mut output).await,
110        cli::Command::Plugin { action } => {
111            cmd::plugin::dispatch(action, record_protocol, &mut output).await
112        }
113        cli::Command::Session(args) => cmd::session::run(&args, &mut output).await,
114        cli::Command::Sandbox(args) => cmd::sandbox::run(&args, &mut output).await,
115        cli::Command::Skill(sub) => cmd::skill::dispatch(sub, &mut output).await,
116        cli::Command::Workflow(sub) => cmd::workflow::dispatch(sub, &mut output).await,
117        cli::Command::Serve(args) => cmd::serve::run(&args).await,
118    }
119}
