1//! Dispatch resolution helpers. Looks up plugin instances by name and
2//! converts [`tau_domain::Address`] to a tool name. Pure logic, no I/O.
3//!
4//! All helpers are kernel-internal (`pub(crate)`) — dispatch routing
5//! is not part of the public `tau-runtime` API surface.
6//!
7//! # Dead-code allow
8//!
9//! [`address_to_tool_name`] is reached only by the dispatcher (Task 10)
10//! and tests; the resolver methods on [`Runtime`] are exercised both by
11//! tests and the run loop. We keep the module-level `allow` so the
12//! v0.1 surface — small, with a few helpers that are used through
13//! transitive call sites — doesn't sprout one-off annotations.
14
15#![allow(dead_code)]
16
17use std::sync::Arc;
18
19use tau_domain::Address;
20
21use crate::builder::{DynLlmBackend, DynTool};
22use crate::error::RuntimeError;
23use crate::Runtime;
24
25impl Runtime {
26    /// Resolve an LLM backend by name. Returns
27    /// [`RuntimeError::LlmBackendNotRegistered`] if the agent's
28    /// requested backend is not in the registry.
29    pub(crate) fn resolve_llm_backend(
30        &self,
31        agent_id: &str,
32        backend_name: &str,
33    ) -> Result<&Arc<dyn DynLlmBackend>, RuntimeError> {
34        self.llm_backends()
35            .get(backend_name)
36            .ok_or_else(|| RuntimeError::LlmBackendNotRegistered {
37                agent_id: agent_id.to_owned(),
38                backend: backend_name.to_owned(),
39            })
40    }
41
42    /// Resolve a tool by name. On miss, returns
43    /// [`RuntimeError::ToolNotRegistered`] populated with the sorted
44    /// list of registered tool names for diagnostics.
45    pub(crate) fn resolve_tool(&self, tool_name: &str) -> Result<&Arc<dyn DynTool>, RuntimeError> {
46        self.tools().get(tool_name).ok_or_else(|| {
47            let mut registered: Vec<String> = self.tools().keys().cloned().collect();
48            registered.sort();
49            RuntimeError::ToolNotRegistered {
50                tool_name: tool_name.to_owned(),
51                registered,
52            }
53        })
54    }
55}
56
57/// Convert a recipient [`Address`] to a tool name. v0.1 only routes
58/// to tools (`Address::Tool`); other variants return `None`.
59pub(crate) fn address_to_tool_name(addr: &Address) -> Option<&str> {
60    match addr {
61        Address::Tool(name) => Some(name.as_str()),
62        _ => None,
63    }
64}

81    #[test]
82    fn resolve_llm_backend_present_returns_arc() {
83        let runtime = Runtime::builder()
84            .with_llm_backend(MockLlmBackend::new("gpt-4"))
85            .build()
86            .expect("build runtime");
87
88        let backend = runtime
89            .resolve_llm_backend("agent-x", "gpt-4")
90            .expect("backend present");
91        assert_eq!(backend.name(), "gpt-4");
92    }
93
94    #[test]
95    fn resolve_llm_backend_absent_returns_error() {
96        let runtime = Runtime::builder()
97            .with_llm_backend(MockLlmBackend::new("gpt-4"))
98            .build()
99            .expect("build runtime");
100
101        let result = runtime.resolve_llm_backend("agent-x", "missing-backend");
102        // `Ok` side is `&Arc<dyn DynLlmBackend>` which is not `Debug`,
103        // so we can't `{result:?}` the whole `Result` — discriminate
104        // first, then debug-format only the `Err`.
105        let Err(err) = result else {
106            panic!("expected LlmBackendNotRegistered, got Ok")
107        };
108        let RuntimeError::LlmBackendNotRegistered {
109            agent_id, backend, ..
110        } = err
111        else {
112            panic!("expected LlmBackendNotRegistered: {err:?}")
113        };
114        assert_eq!(agent_id, "agent-x");
115        assert_eq!(backend, "missing-backend");
116    }
117
118    #[test]
119    fn resolve_tool_present_returns_arc() {
120        let runtime = Runtime::builder()
121            .with_llm_backend(MockLlmBackend::new("gpt-4"))
122            .with_tool(MockTool::new("echo", empty_tool_spec("echo")))
123            .build()
124            .expect("build runtime");
125
126        let tool = runtime.resolve_tool("echo").expect("tool present");
127        assert_eq!(tool.name(), "echo");
128    }

130    #[test]
131    fn resolve_tool_absent_returns_error_with_registered_list() {
132        let runtime = Runtime::builder()
133            .with_llm_backend(MockLlmBackend::new("gpt-4"))
134            .with_tool(MockTool::new("echo", empty_tool_spec("echo")))
135            .with_tool(MockTool::new("reverse", empty_tool_spec("reverse")))
136            .build()
137            .expect("build runtime");
138
139        let result = runtime.resolve_tool("missing");
140        // `Ok` side is `&Arc<dyn DynTool>` which is not `Debug`, so we
141        // can't `{result:?}` the whole `Result` — discriminate first,
142        // then debug-format only the `Err`.
143        let Err(err) = result else {
144            panic!("expected ToolNotRegistered, got Ok")
145        };
146        let RuntimeError::ToolNotRegistered {
147            tool_name,
148            registered,
149            ..
150        } = err
151        else {
152            panic!("expected ToolNotRegistered: {err:?}")
153        };
154        assert_eq!(tool_name, "missing");
155        assert_eq!(
156            registered,
157            vec!["echo".to_string(), "reverse".to_string()],
158            "registered list should contain both tools, sorted"
159        );
160    }

162    #[test]
163    fn address_to_tool_name_routes_only_tool_addresses() {
164        // Tool variant -> Some(name).
165        let tool_addr = Address::Tool("foo".into());
166        assert_eq!(address_to_tool_name(&tool_addr), Some("foo"));
167
168        // User -> None.
169        assert_eq!(address_to_tool_name(&Address::User), None);
170
171        // System -> None.
172        assert_eq!(address_to_tool_name(&Address::System), None);
173
174        // Agent -> None.
175        let agent_addr = Address::Agent(AgentInstanceId::new());
176        assert_eq!(address_to_tool_name(&agent_addr), None);
177    }
