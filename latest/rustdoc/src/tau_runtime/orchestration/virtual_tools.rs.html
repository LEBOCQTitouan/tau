1//! Virtual-tool resolver intercepted before plugin dispatch.
2//!
3//! When an agent calls a tool whose name starts with `task.` / `run.` /
4//! `agent.<kind>.spawn`, the runtime resolves it here instead of forwarding
5//! to a plugin host. Result is returned synchronously as a tool_result.

16/// Returns true iff `tool_name` is handled by the virtual-tool resolver.
17pub fn is_virtual(tool_name: &str) -> bool {
18    matches!(
19        tool_name,
20        "task.create"
21            | "task.claim"
22            | "task.heartbeat"
23            | "task.release"
24            | "task.update"
25            | "task.complete"
26            | "task.fail"
27            | "task.discard"
28            | "task.list"
29            | "task.get"
30            | "run.note"
31            | "run.plan"
32    ) || (tool_name.starts_with("agent.") && tool_name.ends_with(".spawn"))
33        || tool_name
34            .strip_prefix("skill.")
35            .and_then(|s| s.strip_suffix(".spawn"))
36            .map(|name| !name.is_empty())
37            .unwrap_or(false)
38}

40/// Capability requirement for a given virtual tool. Used by the dispatch
41/// path to gate the call before invoking the handler.
42pub fn required_capability(tool_name: &str) -> Capability {
43    match tool_name {
44        "task.list" | "task.get" => Capability::TaskList {
45            mode: "read".into(),
46        },
47        "task.create" | "task.claim" | "task.heartbeat" | "task.release" | "task.update"
48        | "task.complete" | "task.fail" => Capability::TaskList {
49            mode: "write".into(),
50        },
51        "task.discard" => Capability::TaskList {
52            mode: "manage".into(),
53        },
54        "run.note" => Capability::Plan {
55            mode: "write".into(),
56        },
57        "run.plan" => Capability::Plan {
58            mode: "read".into(),
59        },
60        s if s.starts_with("agent.") && s.ends_with(".spawn") => {
61            // The Spawn capability's allowed_kinds list is checked in
62            // validate_agent_spawn (Task 9), not here.
63            // Use serde round-trip to construct because AgentCapability::Spawn is #[non_exhaustive].
64            serde_json::from_value(serde_json::json!({"kind": "agent.spawn", "allowed_kinds": []}))
65                .unwrap_or(Capability::Custom {
66                    name: "agent.spawn".into(),
67                    params: Default::default(),
68                })
69        }
70        s if s.starts_with("skill.") && s.ends_with(".spawn") => {
71            // The Spawn capability's allowed_skills list is checked in
72            // validate_skill_spawn, not here.
73            // Use serde round-trip to construct because SkillCapability::Spawn is #[non_exhaustive].
74            serde_json::from_value(serde_json::json!({"kind": "skill.spawn", "allowed_skills": []}))
75                .unwrap_or(Capability::Custom {
76                    name: "skill.spawn".into(),
77                    params: Default::default(),
78                })
79        }
80        _ => Capability::Custom {
81            name: tool_name.into(),
82            params: Default::default(),
83        },
84    }
85}

87/// Dispatch a virtual tool call. Returns the JSON result body (the caller
88/// wraps it in a normal tool_result envelope).
89///
90/// Note: `agent.<kind>.spawn` is NOT dispatched here — it requires recursive
91/// Runtime::run invocation at the kernel layer (Task 12). Callers must use
92/// `validate_agent_spawn` instead for those tools.
93pub fn dispatch(
94    tool_name: &str,
95    args: Value,
96    agent_id: &AgentId,
97    state: &mut RunState,
98) -> Result<Value, OrchestrationError> {
99    match tool_name {
100        "task.create" => handle_task_create(args, agent_id, state),
101        "task.claim" => handle_task_claim(args, agent_id, state),
102        "task.heartbeat" => handle_task_heartbeat(args, agent_id, state),
103        "task.release" => handle_task_release(args, agent_id, state),
104        "task.update" => handle_task_update(args, agent_id, state),
105        "task.complete" => handle_task_complete(args, agent_id, state),
106        "task.fail" => handle_task_fail(args, agent_id, state),
107        "task.discard" => handle_task_discard(args, agent_id, state),
108        "task.list" => handle_task_list(args, state),
109        "task.get" => handle_task_get(args, state),
110        "run.note" => handle_run_note(args, agent_id, state),
111        "run.plan" => handle_run_plan(state),
112        s if s.starts_with("agent.") && s.ends_with(".spawn") => {
113            Err(OrchestrationError::ArgError {
114                tool: tool_name.into(),
115                detail: "agent.spawn dispatched via virtual_tools::validate_agent_spawn + kernel; not via dispatch()".into(),
116            })
117        }
118        _ => Err(OrchestrationError::ArgError {
119            tool: tool_name.into(),
120            detail: format!("unknown virtual tool: {tool_name}"),
121        }),
122    }
123}

338/// Check the capability subset law: every capability in `child_grant` must
339/// be present in `parent_grant`. v1 uses strict equality on the JSON-serialized
340/// form; future versions may relax to allow narrowing.
341pub fn check_capability_subset(
342    parent_grant: &[Capability],
343    child_grant: &[Capability],
344) -> Result<(), OrchestrationError> {
345    let parent_keys: std::collections::BTreeSet<String> = parent_grant
346        .iter()
347        .map(|c| serde_json::to_string(c).unwrap_or_default())
348        .collect();
349    let extras: Vec<String> = child_grant
350        .iter()
351        .map(|c| serde_json::to_string(c).unwrap_or_default())
352        .filter(|k| !parent_keys.contains(k))
353        .collect();
354    if extras.is_empty() {
355        Ok(())
356    } else {
357        Err(OrchestrationError::GrantNotSubset { extras })
358    }
359}

361#[derive(Deserialize)]
362struct AgentSpawnArgs {
363    /// Capabilities to grant the child (must be ⊆ parent's grant).
364    #[serde(default)]
365    grant: Vec<Capability>,
366    /// Initial user message for the child.
367    message: String,
368    /// Optional system prompt for the child. When `None`, the child
369    /// inherits the parent's system_prompt. When `Some`, replaces it.
370    /// This is the v1.2 foundation for per-kind agent definitions
371    /// (a "skill" = `(system_prompt, grant, optional tools)`).
372    #[serde(default)]
373    system_prompt: Option<String>,
374    // `kind` is in the tool name, not here — ignore any `kind` field in args.
375}

377/// A validated agent.spawn request, ready for the runtime kernel to
378/// transform into a recursive `Runtime::run` invocation.
379#[derive(Debug, Clone)]
380pub struct AgentSpawnRequest {
381    /// The agent kind parsed from the tool name.
382    pub kind: String,
383    /// Capabilities granted to the child (verified ⊆ parent's grant).
384    pub grant: Vec<Capability>,
385    /// Initial user message for the child.
386    pub message: String,
387    /// Optional system prompt override for the child. `None` ⇒ inherit
388    /// parent's. v1.2: foundation for per-kind agent definitions.
389    pub system_prompt: Option<String>,
390}
391
392/// Validate an `agent.<kind>.spawn` virtual tool call.
393///
394/// 1. Parses `kind` from the tool name.
395/// 2. Parses args.
396/// 3. Checks spawn authorization (parent's `Agent::Spawn { allowed_kinds }` must include `kind`).
397/// 4. Checks capability subset law (`child.grant ⊆ parent.grant`).
398///
399/// Returns a validated [`AgentSpawnRequest`] the kernel uses to invoke a
400/// recursive `Runtime::run`.
401pub fn validate_agent_spawn(
402    tool_name: &str,
403    args: &Value,
404    parent: &AgentId,
405    parent_grant: &[Capability],
406) -> Result<AgentSpawnRequest, OrchestrationError> {
407    let kind = tool_name
408        .strip_prefix("agent.")
409        .and_then(|s| s.strip_suffix(".spawn"))
410        .ok_or_else(|| OrchestrationError::ArgError {
411            tool: tool_name.into(),
412            detail: "malformed virtual tool name".into(),
413        })?;
414
415    let a: AgentSpawnArgs =
416        serde_json::from_value(args.clone()).map_err(|e| OrchestrationError::ArgError {
417            tool: tool_name.into(),
418            detail: format!("arg parse error: {e}"),
419        })?;
420
421    // Spawn-authorization check: parent must have Agent(Spawn) granting `kind`.
422    let allowed = parent_grant.iter().any(|c| match c {
423        Capability::Agent(tau_domain::AgentCapability::Spawn { allowed_kinds, .. }) => {
424            allowed_kinds.iter().any(|k| k == kind)
425        }
426        _ => false,
427    });
428    if !allowed {
429        return Err(OrchestrationError::SpawnNotAuthorized {
430            parent: parent.clone(),
431            kind: kind.into(),
432        });
433    }
434
435    // Capability subset law: child.grant ⊆ parent.grant.
436    check_capability_subset(parent_grant, &a.grant)?;
437
438    Ok(AgentSpawnRequest {
439        kind: kind.into(),
440        grant: a.grant,
441        message: a.message,
442        system_prompt: a.system_prompt,
443    })
444}

448/// Validate a `skill.<name>.spawn` virtual tool call.
449///
450/// 1. Parses `name` from the tool name.
451/// 2. Parses args (`message`, optional `system_prompt`, optional `scope_paths`).
452/// 3. Checks spawn authorization (parent's
453///    `Capability::Skill(SkillCapability::Spawn { allowed_skills })` must
454///    include `name`).
455/// 4. Looks up the installed skill + substitutes `${SKILL_DIR}` +
456///    applies `scope_paths` + verifies capability subset law via
457///    [`crate::orchestration::resolve_skill_for_spawn`].
458///
459/// Returns a fully validated [`crate::orchestration::SkillSpawnRequest`]
460/// the kernel uses to invoke a recursive `Runtime::run`.
461pub fn validate_skill_spawn(
462    tool_name: &str,
463    args: &Value,
464    parent: &AgentId,
465    parent_grant: &[Capability],
466    scope: &tau_pkg::Scope,
467) -> Result<crate::orchestration::SkillSpawnRequest, OrchestrationError> {
468    let name = tool_name
469        .strip_prefix("skill.")
470        .and_then(|s| s.strip_suffix(".spawn"))
471        .ok_or_else(|| OrchestrationError::ArgError {
472            tool: tool_name.into(),
473            detail: "malformed skill.<name>.spawn tool name".into(),
474        })?;
475
476    #[derive(serde::Deserialize)]
477    struct SkillSpawnArgsRaw {
478        message: String,
479        #[serde(default)]
480        system_prompt: Option<String>,
481        #[serde(default)]
482        scope_paths: Option<Vec<String>>,
483    }
484    let a: SkillSpawnArgsRaw =
485        serde_json::from_value(args.clone()).map_err(|e| OrchestrationError::ArgError {
486            tool: tool_name.into(),
487            detail: format!("skill.<name>.spawn args: {e}"),
488        })?;
489
490    // Spawn authorization: parent must have Skill(Spawn) granting `name`.
491    let allowed = parent_grant.iter().any(|c| match c {
492        Capability::Skill(tau_domain::SkillCapability::Spawn { allowed_skills, .. }) => {
493            allowed_skills.iter().any(|s| s == name)
494        }
495        _ => false,
496    });
497    if !allowed {
498        return Err(OrchestrationError::SkillSpawnNotAuthorized {
499            parent: parent.clone(),
500            name: name.into(),
501        });
502    }
503
504    let spawn_args = crate::orchestration::SkillSpawnArgs {
505        message: a.message,
506        system_prompt: a.system_prompt,
507        scope_paths: a.scope_paths,
508    };
509    crate::orchestration::resolve_skill_for_spawn(name, &spawn_args, parent_grant, scope)
510}
