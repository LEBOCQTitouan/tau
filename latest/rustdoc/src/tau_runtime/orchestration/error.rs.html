1//! Typed errors raised by multi-agent orchestration operations.
2
3use tau_ports::{AgentId, TaskId};
4
5/// Errors surfaced by virtual-tool dispatch + TaskList state transitions.
6#[derive(Debug, thiserror::Error)]
7#[non_exhaustive]
8pub enum OrchestrationError {
9    /// `agent.<kind>.spawn`: the parent agent's grant does not include
10    /// the requested child kind in its `Agent::Spawn { allowed_kinds }`.
11    #[error("agent {parent:?} not authorized to spawn kind {kind:?}")]
12    SpawnNotAuthorized {
13        /// Parent agent id.
14        parent: AgentId,
15        /// Requested child kind.
16        kind: String,
17    },
18
19    /// Generic capability check failure for a virtual tool call.
20    #[error("agent {agent:?} lacks capability {needed}")]
21    CapabilityMissing {
22        /// The agent.
23        agent: AgentId,
24        /// Description of the missing capability.
25        needed: String,
26    },
27
28    /// `task.claim`: task is currently owned by another agent and its
29    /// lease has not expired.
30    #[error("task {task:?} already locked by {by:?} until {until}")]
31    TaskLocked {
32        /// Task id.
33        task: TaskId,
34        /// Current lock holder.
35        by: AgentId,
36        /// Lease expiry (RFC 3339).
37        until: String,
38    },
39
40    /// `task.update` / `task.complete` / `task.heartbeat`: the caller is
41    /// not the current owner.
42    #[error("task {task:?}: agent {agent:?} is not the owner")]
43    NotTaskOwner {
44        /// Task id.
45        task: TaskId,
46        /// Calling agent.
47        agent: AgentId,
48    },
49
50    /// `task.get` / `task.update`: id doesn't exist.
51    #[error("task {task:?} not found")]
52    TaskNotFound {
53        /// Task id.
54        task: TaskId,
55    },
56
57    /// Invalid state transition (e.g. claim a task that's already Done).
58    #[error("task {task:?}: cannot transition to {target}")]
59    InvalidTaskTransition {
60        /// Task id.
61        task: TaskId,
62        /// Target status.
63        target: String,
64    },
65
66    /// Budget exceeded; aborting.
67    #[error("budget {budget} exceeded: {value} / {limit}")]
68    BudgetExceeded {
69        /// Budget name.
70        budget: String,
71        /// Final value.
72        value: u64,
73        /// Limit.
74        limit: u64,
75    },
76
77    /// Capability subset law violated at agent.spawn time: child grant
78    /// contains capabilities not present in the parent's grant.
79    #[error("child grant exceeds parent grant: extra = {extras:?}")]
80    GrantNotSubset {
81        /// Capabilities in child but not in parent (JSON-serialized form).
82        extras: Vec<String>,
83    },
84
85    /// Persistence I/O failed.
86    #[error("orchestration persistence error: {0}")]
87    PersistenceError(#[from] std::io::Error),
88
89    /// Argument parsing failure (malformed virtual-tool args).
90    #[error("virtual tool {tool}: {detail}")]
91    ArgError {
92        /// Tool name.
93        tool: String,
94        /// Detail.
95        detail: String,
96    },
97
98    /// Skills-4: `skill.<name>.spawn` — no installed skill matches `name`.
99    #[error("skill not installed: {name:?}")]
100    SkillNotInstalled {
101        /// The unresolved skill name.
102        name: String,
103    },
104
105    /// Skills-4: skill's lockfile entry exists but install path is
106    /// missing on disk.
107    #[error("skill {name:?} install path missing at {expected_path:?}")]
108    SkillInstallPathMissing {
109        /// Skill name.
110        name: String,
111        /// The expected install path (the manifest location).
112        expected_path: std::path::PathBuf,
113    },
114
115    /// Skills-4: SKILL.md couldn't be read or parsed.
116    #[error("skill {name:?} content invalid: {detail}")]
117    SkillContentInvalid {
118        /// Skill name.
119        name: String,
120        /// Reason (read error, YAML parse failure, missing required field).
121        detail: String,
122    },
123
124    /// Skills-4: caller's `scope_paths` includes a path not covered
125    /// by any declared fs.* path. Typo detection.
126    #[error("skill scope_path {path:?} is not covered by any declared fs.* path")]
127    SkillScopePathNotCovered {
128        /// The offending scope_path entry.
129        path: String,
130    },
131
132    /// Skills-4: parent's `Capability::Skill(SkillCapability::Spawn)`
133    /// doesn't include the requested skill in `allowed_skills`.
134    #[error("agent {parent:?} not authorized to spawn skill {name:?}")]
135    SkillSpawnNotAuthorized {
136        /// Parent agent id.
137        parent: AgentId,
138        /// The requested skill name.
139        name: String,
140    },
141}
