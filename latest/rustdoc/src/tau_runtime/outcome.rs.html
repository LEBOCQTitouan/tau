1//! Outcome of a `Runtime::run` call.
2//!
3//! Distinguishes successful completion from agent-level failures
4//! (which are NOT errors — see [`crate::RuntimeError`] docs for the
5//! ADR-0006 dichotomy).
6
7use tau_domain::{AgentStatus, Message};
8
9use crate::options::TokenUsage;
10
11/// Outcome of a `Runtime::run` call (added in Task 10).
12///
13/// - [`RunOutcome::Completed`]: agent finished successfully and produced
14///   a final response message.
15/// - [`RunOutcome::Failed`]: agent ran but failed via a typed
16///   [`AgentStatus::Failed { kind, detail }`]. Partial conversation
17///   preserved in `all_messages` for inspection.
18///
19/// Kernel-level operational failures (plugin errors, dispatch errors)
20/// are reported via `Err(RuntimeError)`, NOT this enum.
21///
22/// # Example
23///
24/// ```ignore
25/// // `RunOutcome` is `#[non_exhaustive]`; constructed by `Runtime::run`.
26/// use tau_runtime::RunOutcome;
27///
28/// fn handle(outcome: RunOutcome) {
29///     match outcome {
30///         RunOutcome::Completed { final_message, total_turns, .. } => {
31///             println!("done in {total_turns} turns");
32///         }
33///         RunOutcome::Failed { status, .. } => {
34///             eprintln!("agent failed: {status:?}");
35///         }
36///         _ => {}
37///     }
38/// }
39/// ```
40#[non_exhaustive]
41#[derive(Debug, Clone, PartialEq)]
42pub enum RunOutcome {
43    /// Agent completed and produced a final response.
44    Completed {
45        /// The final assistant message (LLM response with no tool_uses).
46        final_message: Message,
47        /// Full conversation history: initial message + every LLM
48        /// response + every tool_use / tool_result.
49        all_messages: Vec<Message>,
50        /// Number of agent loop iterations performed.
51        total_turns: u32,
52        /// Token usage summed across the run.
53        token_usage: TokenUsage,
54    },
55    /// Agent ran but failed via a typed `FailureKind`. Partial
56    /// conversation preserved.
57    Failed {
58        /// Always `AgentStatus::Failed { kind, detail }`. The `kind`
59        /// distinguishes `PolicyDenied` (capability denial),
60        /// `OutOfResources` (max turns reached), `BackendError`,
61        /// `Crashed`, `InternalError`.
62        status: AgentStatus,
63        /// Full conversation history up to the failure point.
64        all_messages: Vec<Message>,
65        /// Number of agent loop iterations performed.
66        total_turns: u32,
67        /// Token usage summed across the run.
68        token_usage: TokenUsage,
69    },
70}

72#[cfg(test)]
73mod tests {
74    use super::*;
75    use tau_domain::fixtures;
76
77    // AgentStatus::Failed is variant-level #[non_exhaustive] (E0639), so
78    // struct-literal construction of that variant is blocked from outside
79    // tau-domain. We use AgentStatus::Stopped (a unit variant) to test
80    // RunOutcome::Failed structure, and note that AgentStatus::Failed
81    // itself will be constructed only by tau-runtime internals (Task 10).
82    #[test]
83    fn run_outcome_failed_carries_status_and_messages() {
84        let outcome = RunOutcome::Failed {
85            status: AgentStatus::Stopped,
86            all_messages: vec![],
87            total_turns: 3,
88            token_usage: TokenUsage::default(),
89        };
90        let RunOutcome::Failed {
91            status,
92            total_turns,
93            ..
94        } = outcome
95        else {
96            panic!("expected Failed");
97        };
98        assert_eq!(status, AgentStatus::Stopped);
99        assert_eq!(total_turns, 3);
100    }
101
102    #[test]
103    fn run_outcome_failed_preserves_token_usage() {
104        let usage = TokenUsage {
105            input_tokens: 10,
106            output_tokens: 5,
107            total_tokens: Some(15),
108        };
109        let outcome = RunOutcome::Failed {
110            status: AgentStatus::Stopped,
111            all_messages: vec![],
112            total_turns: 16,
113            token_usage: usage,
114        };
115        let RunOutcome::Failed {
116            total_turns,
117            token_usage,
118            ..
119        } = outcome
120        else {
121            panic!("expected Failed");
122        };
123        assert_eq!(total_turns, 16);
124        assert_eq!(token_usage.input_tokens, 10);
125        assert_eq!(token_usage.output_tokens, 5);
126        assert_eq!(token_usage.total_tokens, Some(15));
127    }
128
129    #[test]
130    fn run_outcome_completed_construction() {
131        // Uses tau_domain::fixtures::any_message() (behind the
132        // "test-fixtures" feature) because Message is #[non_exhaustive]
133        // and has no public constructor outside tau-domain.
134        let msg = fixtures::any_message();
135        let outcome = RunOutcome::Completed {
136            final_message: msg.clone(),
137            all_messages: vec![msg.clone()],
138            total_turns: 2,
139            token_usage: TokenUsage::default(),
140        };
141        let RunOutcome::Completed {
142            total_turns,
143            all_messages,
144            ..
145        } = outcome
146        else {
147            panic!("expected Completed");
148        };
149        assert_eq!(total_turns, 2);
150        assert_eq!(all_messages.len(), 1);
151    }
152}
