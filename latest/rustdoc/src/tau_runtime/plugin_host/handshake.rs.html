1//! Host-side handshake driver.
2//!
3//! Drives the host end of the `meta.handshake` exchange:
4//!
5//! 1. Build a [`HandshakeRequest`] from the host's protocol version,
6//!    expected port, trace context, and per-plugin config.
7//! 2. Encode it as the `params[0]` of a `meta.handshake` request frame
8//!    (msgid 1) and write it.
9//! 3. Await the response with a timeout.
10//! 4. Validate `protocol_version`, `provides`, and `required_methods`.
11//!
12//! Each failure mode maps to a specific
13//! [`HandshakeFailureReason`] so the calling code surfaces typed
14//! errors instead of stringly-typed wire failures.
15//!
16//! See `docs/superpowers/specs/2026-04-28-plugin-loading-design.md`
17//! §7.3 (the host side of the handshake) and §4.4 (the wire-level
18//! protocol the plugin SDK implements).
19
20use std::time::Duration;
21
22use tau_domain::PortKind;
23use tau_plugin_protocol::{
24    handshake::{meta, HandshakeRequest, HandshakeResponse, TraceContext, PROTOCOL_VERSION},
25    Frame, FramedReader, FramedWriter,
26};
27use tokio::io::{AsyncRead, AsyncWrite};
28
29use crate::error::{HandshakeFailureReason, RuntimeError};
30
31/// Msgid used for the handshake request. Subsequent calls (Task 15+)
32/// allocate from `PluginProcess::next_msgid` starting at 2.
33const HANDSHAKE_MSGID: u32 = 1;
34
35/// Drive the host end of the `meta.handshake` exchange.
36///
37/// Returns the validated [`HandshakeResponse`] on success; otherwise
38/// returns a [`RuntimeError::PluginHandshakeFailed`] carrying the
39/// typed [`HandshakeFailureReason`].
40///
41/// # Errors
42///
43/// * [`HandshakeFailureReason::Timeout`] if the plugin doesn't reply
44///   within `handshake_timeout`.
45/// * [`HandshakeFailureReason::ProtocolVersionMismatch`] if the plugin
46///   advertises a different `protocol_version`.
47/// * [`HandshakeFailureReason::ProvidesMismatch`] if the plugin
48///   advertises a different port than the host expected.
49/// * [`HandshakeFailureReason::MissingRequiredMethod`] if the plugin's
50///   advertised `methods` doesn't include one of `required_methods`.
51/// * [`HandshakeFailureReason::Malformed`] for any structural failure
52///   (frame decode, msgid mismatch, missing result field, plugin
53///   returned an error envelope, etc.).
54#[allow(clippy::too_many_arguments)]
55#[doc(hidden)]
56pub async fn drive_handshake<R, W>(
57    reader: &mut FramedReader<R>,
58    writer: &mut FramedWriter<W>,
59    plugin_name: &str,
60    expected_port: PortKind,
61    required_methods: &[&str],
62    config: serde_json::Value,
63    trace_context: TraceContext,
64    handshake_timeout: Duration,
65) -> Result<HandshakeResponse, RuntimeError>
66where
67    R: AsyncRead + Unpin,
68    W: AsyncWrite + Unpin,
69{
70    // 1. Build the handshake request.
71    let request = HandshakeRequest::new(
72        PROTOCOL_VERSION.to_string(),
73        expected_port,
74        trace_context,
75        config,
76    );
77
78    // The wire shape for `meta.handshake` params is a 1-element array
79    // (`Vec<HandshakeRequest>`) per the SDK's matching decoder; see
80    // `tau_plugin_sdk::handshake::drive_handshake`.
81    let params_bytes = rmp_serde::to_vec(&vec![&request])
82        .map_err(|e| handshake_malformed(plugin_name, format!("encode handshake params: {e}")))?;
83    let request_frame = Frame::Request {
84        id: HANDSHAKE_MSGID,
85        method: meta::HANDSHAKE_METHOD.to_string(),
86        params: params_bytes,
87    };
88    let request_body = request_frame
89        .encode()
90        .map_err(|e| handshake_malformed(plugin_name, format!("encode handshake frame: {e}")))?;
91    writer
92        .write_frame(&request_body)
93        .await
94        .map_err(|e| handshake_malformed(plugin_name, format!("write handshake frame: {e}")))?;
95
96    // 2. Await the response with a timeout. The timeout *only* covers
97    //    the next frame read — once the response arrives, decode is
98    //    fast and not subject to the same SLO.
99    let read_outcome = tokio::time::timeout(handshake_timeout, reader.next_frame()).await;
100    let response_body = match read_outcome {
101        Err(_elapsed) => {
102            return Err(RuntimeError::PluginHandshakeFailed {
103                plugin: plugin_name.to_string(),
104                reason: HandshakeFailureReason::Timeout,
105            });
106        }
107        Ok(Err(e)) => {
108            return Err(handshake_malformed(
109                plugin_name,
110                format!("read handshake response: {e}"),
111            ));
112        }
113        Ok(Ok(None)) => {
114            return Err(handshake_malformed(
115                plugin_name,
116                "EOF before handshake response".to_string(),
117            ));
118        }
119        Ok(Ok(Some(body))) => body,
120    };
121
122    // 3. Decode the response frame.
123    let response_frame = Frame::decode(&response_body)
124        .map_err(|e| handshake_malformed(plugin_name, format!("decode handshake response: {e}")))?;
125
126    let (id, error, result) = match response_frame {
127        Frame::Response { id, error, result } => (id, error, result),
128        other => {
129            return Err(handshake_malformed(
130                plugin_name,
131                format!("expected Response frame, got {other:?}"),
132            ));
133        }
134    };
135    if id != HANDSHAKE_MSGID {
136        return Err(handshake_malformed(
137            plugin_name,
138            format!("expected msgid {HANDSHAKE_MSGID}, got {id}"),
139        ));
140    }
141    if let Some(envelope) = error {
142        return Err(handshake_malformed(
143            plugin_name,
144            format!(
145                "plugin returned error envelope: code={} message={}",
146                envelope.code, envelope.message
147            ),
148        ));
149    }
150    let result_bytes = result.ok_or_else(|| {
151        handshake_malformed(plugin_name, "handshake response missing result".to_string())
152    })?;
153
154    let response: HandshakeResponse = rmp_serde::from_slice(&result_bytes).map_err(|e| {
155        handshake_malformed(plugin_name, format!("decode handshake response body: {e}"))
156    })?;
157
158    // 4. Validate protocol_version.
159    if response.protocol_version != PROTOCOL_VERSION {
160        return Err(RuntimeError::PluginHandshakeFailed {
161            plugin: plugin_name.to_string(),
162            reason: HandshakeFailureReason::ProtocolVersionMismatch {
163                host: PROTOCOL_VERSION.to_string(),
164                plugin: response.protocol_version,
165            },
166        });
167    }
168
169    // 5. Validate provides matches expected_port.
170    if response.provides != expected_port {
171        return Err(RuntimeError::PluginHandshakeFailed {
172            plugin: plugin_name.to_string(),
173            reason: HandshakeFailureReason::ProvidesMismatch {
174                manifest: expected_port,
175                plugin_advertised: response.provides,
176            },
177        });
178    }
179
180    // 6. Validate required methods.
181    for required in required_methods {
182        if !response.methods.iter().any(|m| m == required) {
183            return Err(RuntimeError::PluginHandshakeFailed {
184                plugin: plugin_name.to_string(),
185                reason: HandshakeFailureReason::MissingRequiredMethod {
186                    method: (*required).to_string(),
187                },
188            });
189        }
190    }
191
192    tracing::info!(
193        target: "tau_runtime::plugin_host",
194        plugin = plugin_name,
195        methods = ?response.methods,
196        "plugin.handshake.completed"
197    );
198
199    Ok(response)
200}
201
202/// Helper: build a `PluginHandshakeFailed { reason: Malformed }`.
203fn handshake_malformed(plugin: &str, detail: String) -> RuntimeError {
204    RuntimeError::PluginHandshakeFailed {
205        plugin: plugin.to_string(),
206        reason: HandshakeFailureReason::Malformed { detail },
207    }
