1//! Host-side [`crate::builder::DynLlmBackend`] implementation backed by
2//! a spawned plugin subprocess.
3//!
4//! Thin RPC dispatcher: each `complete()` call allocates a msgid,
5//! registers a `oneshot::Sender` in the shared `in_flight_responses`
6//! map, sends a single `Frame::Request` over the writer, and awaits the
7//! response.
8//!
9//! See spec §7.4 (per-port adapters) and §7.3 (PluginProcess + read
10//! loop). Streaming (`llm.stream`) is wired alongside `llm.complete`:
11//! [`IpcLlmBackend::stream`] dispatches an `llm.stream` request,
12//! registers both the response oneshot and a chunk mpsc keyed on the
13//! same msgid, and hands the assembled stream off to
14//! [`crate::plugin_host::stream_router`].
15
16use std::pin::Pin;
17use std::sync::atomic::Ordering;
18use std::sync::Arc;
19
20use tau_plugin_protocol::Frame;
21use tau_ports::{CompletionRequest, CompletionResponse, CompletionStream, LlmError};
22use tokio::sync::{mpsc, oneshot};
23
24use crate::builder::DynLlmBackend;
25
26use super::process::{PluginProcess, RpcResult};
27
28/// Wire method name for the non-streaming LLM completion.
29const LLM_COMPLETE_METHOD: &str = "llm.complete";
30
31/// Wire method name for the streaming LLM completion. Per the plan-
32/// erratum carryover from Task 9, the SDK runner dispatches
33/// `llm.stream` (NOT `llm.complete_streaming`); both ends of the wire
34/// must agree.
35const LLM_STREAM_METHOD: &str = "llm.stream";
36
37/// Bound on the per-stream chunk channel. Sized to absorb a short burst
38/// of provider-side chunks without forcing the read loop to await an
39/// idle consumer (which would block the loop from servicing other
40/// in-flight calls on the same plugin process). 64 is a starting
41/// guess — adjust if production traces show backpressure.
42const STREAM_CHUNK_CHANNEL_CAPACITY: usize = 64;
43
44/// IPC-backed [`DynLlmBackend`] adapter. Holds an `Arc<PluginProcess>`
45/// shared with the read loop and any other adapters bound to the same
46/// plugin instance.
47///
48/// Constructed by [`crate::plugin_host::load_llm_backend`] after the
49/// handshake validates that the plugin advertises `llm.complete`. The
50/// type is `pub` so the integration-test `__internals` re-exports
51/// (gated by `feature = "test-support"`) can build instances without
52/// spawning a real subprocess; production code only sees an
53/// `Arc<dyn DynLlmBackend>` and never the concrete type.
54pub struct IpcLlmBackend {
55    pub(crate) name: String,
56    pub(crate) process: Arc<PluginProcess>,
57}
58
59impl IpcLlmBackend {
60    /// Construct an adapter from a plugin name (host-visible identity)
61    /// and a shared [`PluginProcess`]. The name is what the kernel uses
62    /// to resolve `agent_def.llm_backend` against the registry, and is
63    /// expected to match the plugin's [`tau_pkg::LockedPlugin`] manifest
64    /// name.
65    pub fn new(name: String, process: Arc<PluginProcess>) -> Self {
66        Self { name, process }
67    }
68}
69
70impl DynLlmBackend for IpcLlmBackend {
71    fn name(&self) -> &str {
72        &self.name
73    }
74
75    fn complete<'a>(
76        &'a self,
77        req: CompletionRequest,
78    ) -> Pin<Box<dyn std::future::Future<Output = Result<CompletionResponse, LlmError>> + 'a>> {
79        let process = self.process.clone();
80        Box::pin(async move {
81            // Dispatch via the shared msgid+oneshot pattern (spec §7.4).
82            let id = process.next_msgid.fetch_add(1, Ordering::Relaxed);
83            let (tx, rx) = oneshot::channel::<RpcResult>();
84            {
85                let mut map = process.in_flight_responses.lock().await;
86                map.insert(id, tx);
87            }
88            // Wire shape: params is a 1-element array `[CompletionRequest]`,
89            // matching `tau_plugin_sdk::runners::llm_backend::dispatch_llm`.
90            let params_bytes = rmp_serde::to_vec(&vec![&req]).map_err(|e| LlmError::Internal {
91                message: format!("rmp encode CompletionRequest: {e}"),
92            })?;
93            let frame = Frame::Request {
94                id,
95                method: LLM_COMPLETE_METHOD.to_string(),
96                params: params_bytes,
97            };
98            let frame_bytes = frame.encode().map_err(|e| LlmError::Internal {
99                message: format!("frame encode: {e}"),
100            })?;
101            process
102                .send_frame(&frame_bytes)
103                .await
104                .map_err(|e| LlmError::Internal {
105                    message: format!("write frame: {e}"),
106                })?;
107            let result = rx.await.map_err(|_| LlmError::Internal {
108                message: "in-flight response sender dropped (plugin crashed?)".to_string(),
109            })?;
110            match result {
111                Ok(bytes) => rmp_serde::from_slice::<CompletionResponse>(&bytes).map_err(|e| {
112                    LlmError::Internal {
113                        message: format!("rmp decode CompletionResponse: {e}"),
114                    }
115                }),
116                Err(envelope) => Err(LlmError::Internal {
117                    message: format!(
118                        "plugin error code {} message {}",
119                        envelope.code, envelope.message
120                    ),
121                }),
122            }
123        })
124    }
125
126    fn stream<'a>(
127        &'a self,
128        req: CompletionRequest,
129    ) -> Pin<Box<dyn std::future::Future<Output = Result<CompletionStream, LlmError>> + 'a>> {
130        let process = self.process.clone();
131        Box::pin(async move {
132            // Allocate the msgid that ties together the request, the
133            // response oneshot, and the per-stream chunk channel. The
134            // read loop in `process::read_loop` keys both maps on this
135            // id when routing inbound `Frame::Response` and
136            // `stream.chunk` notifications.
137            let id = process.next_msgid.fetch_add(1, Ordering::Relaxed);
138
139            // Set up both delivery channels *before* sending the
140            // request: chunks could land before the host's `await`
141            // resumes, so the receivers must already be installed in
142            // the in-flight maps. Same registration ordering as the
143            // non-streaming path.
144            let (chunk_tx, chunk_rx) = mpsc::channel(STREAM_CHUNK_CHANNEL_CAPACITY);
145            let (final_tx, final_rx) = oneshot::channel::<RpcResult>();
146            {
147                let mut streams = process.in_flight_streams.lock().await;
148                streams.insert(id, chunk_tx);
149            }
150            {
151                let mut responses = process.in_flight_responses.lock().await;
152                responses.insert(id, final_tx);
153            }
154
155            // Wire shape: params is a 1-element array `[CompletionRequest]`,
156            // matching `tau_plugin_sdk::runners::llm_backend`'s
157            // `llm.stream` dispatch.
158            let params_bytes = rmp_serde::to_vec(&vec![&req]).map_err(|e| LlmError::Internal {
159                message: format!("rmp encode CompletionRequest (stream): {e}"),
160            })?;
161            let frame = Frame::Request {
162                id,
163                method: LLM_STREAM_METHOD.to_string(),
164                params: params_bytes,
165            };
166            let frame_bytes = frame.encode().map_err(|e| LlmError::Internal {
167                message: format!("frame encode (stream): {e}"),
168            })?;
169            process
170                .send_frame(&frame_bytes)
171                .await
172                .map_err(|e| LlmError::Internal {
173                    message: format!("write frame (stream): {e}"),
174                })?;
175
176            Ok(crate::plugin_host::stream_router::assemble(
177                chunk_rx, final_rx,
178            ))
179        })
180    }
181}
