1//! Agent multi-turn run loop. The kernel surface — see spec §3.7.
2//!
3//! [`Runtime::run`] receives an initial [`Message`], drives the LLM
4//! backend and tool plugins through a turn loop bounded by
5//! [`RunOptions::max_turns`], applies capability checks before each
6//! tool call, and returns a [`RunOutcome`].
7//!
8//! # Error vs failure dichotomy (ADR-0006)
9//!
10//! - Plugin/dispatch failures (LLM error, tool error, missing backend)
11//!   bubble up as `Err(RuntimeError)`. The agent terminates abnormally.
12//! - Agent-level failures — capability denied, max turns reached —
13//!   are reported as `Ok(RunOutcome::Failed { status, .. })` with
14//!   `status = AgentStatus::Failed { kind, .. }`. The conversation
15//!   history is preserved for inspection.
16//!
17//! # Tracing
18//!
19//! Per spec §3.9 the run loop emits a fixed vocabulary of events
20//! (`runtime.run_started`, `runtime.turn_started`, `llm.request_built`,
21//! …) under named spans (`runtime.agent_run`, `runtime.turn`,
22//! `llm.complete`, `dispatch.tool`, `capability.check`,
23//! `tool.session_open`, `tool.invoke`, `tool.session_close`).
24//! Sensitive-data discipline: arguments and message content never
25//! travel above DEBUG; full content is TRACE-only and otherwise
26//! truncated to a 256-char preview.
27

46    /// Run an agent with a pre-existing conversation history.
47    ///
48    /// Identical to [`Runtime::run`] except `history` is pre-loaded into
49    /// the messages buffer for turn 1: the kernel projects `history`
50    /// (plus the new `initial_message`) onto the
51    /// [`tau_ports::CompletionRequest`]'s `messages` list before the first LLM call.
52    /// Subsequent turns evolve the buffer normally — assistant text,
53    /// tool calls, tool results — exactly as in
54    /// [`Runtime::run`].
55    ///
56    /// This is the entry point used by `tau chat` (sub-project 5) to
57    /// thread REPL conversation history across user turns. Per NG6
58    /// ("no persistent agent memory in core"), the kernel itself does
59    /// not retain history between calls — the CLI accumulates a
60    /// `Vec<Message>` and passes it back in on each turn.
61    ///
62    /// The loop is otherwise identical to [`Runtime::run`]:
63    ///
64    /// 1. Build a [`tau_ports::CompletionRequest`] from `history + initial_message`
65    ///    on turn 1; from the evolving buffer on later turns.
66    /// 2. Call the LLM backend.
67    /// 3. Append the assistant text to the history.
68    /// 4. If no tool_uses were emitted, return
69    ///    [`RunOutcome::Completed`].
70    /// 5. Otherwise, for each tool_use: capability-check, dispatch
71    ///    through the registered tool plugin, append the result to the
72    ///    history, and resume from step 1.
73    /// 6. After [`RunOptions::max_turns`] iterations without
74    ///    completion, return [`RunOutcome::Failed`] with
75    ///    [`FailureKind::OutOfResources`].
76    ///
77    /// `Ok(RunOutcome::Failed)` is returned in two cases: capability
78    /// denial ([`FailureKind::PolicyDenied`]) and max turns reached
79    /// ([`FailureKind::OutOfResources`]). Plugin and dispatch errors
80    /// bubble up as `Err(RuntimeError)` per ADR-0006.
81    ///
82    /// # Example
83    ///
84    /// ```rust,ignore
85    /// // RunOutcome and AgentDefinition are #[non_exhaustive]; doctests
86    /// // can't construct them via struct-literal syntax. Example illustrative.
87    /// use tau_runtime::{Runtime, RunOptions};
88    ///
89    /// let outcome = runtime
90    ///     .run_with_history(agent_def, manifest, history, initial_message, RunOptions::default())
91    ///     .await?;
92    /// ```
93    #[instrument(
94        name = "runtime.agent_run",
95        skip_all,
96        fields(
97            agent_id = %agent_def.id,
98            display_name = %agent_def.display_name,
99            package_id = %agent_def.package.name,
100            llm_backend_name = %agent_def.llm_backend,
101            max_turns = options.max_turns,
102            history_len = history.len(),
103        ),
104    )]
105    pub async fn run_with_history(
106        &self,
107        agent_def: AgentDefinition,
108        package_manifest: PackageManifest,
109        history: Vec<Message>,
110        initial_message: Message,
111        options: RunOptions,
112    ) -> Result<RunOutcome, RuntimeError> {
113        use crate::stream::RunEvent;
114        use futures_core::Stream as _;
115        use tau_ports::{LlmError, ToolError};
116
117        // Delegate all agent-loop logic to run_streaming_with_history.
118        // This function is now a thin stream-drainer: it consumes the
119        // stream and returns the terminal RunCompleted.outcome.
120        // The streaming pump (stream.rs::run_streaming_inner) is the
121        // single source of truth for the agent loop.
122        let stream = self
123            .run_streaming_with_history(
124                agent_def,
125                package_manifest,
126                history,
127                initial_message,
128                options,
129            )
130            .await?;
131        let mut stream = Box::pin(stream);
132        loop {
133            let next = std::future::poll_fn(|cx| stream.as_mut().poll_next(cx)).await;
134            match next {
135                Some(RunEvent::RunCompleted { outcome }) => return Ok(outcome),
136                Some(RunEvent::FatalError {
137                    kind,
138                    detail,
139                    context_json,
140                    tool_error_variant,
141                }) => {
142                    // ADR-0006 error/failure dichotomy: the streaming pump
143                    // emits FatalError for plugin/dispatch failures that
144                    // must propagate as Err(RuntimeError) in the batch path.
145                    return Err(match kind.as_str() {
146                        "ToolNotRegistered" => {
147                            // Parse context_json to extract tool_name and
148                            // registered list (emitted by make_tool_not_registered_error).
149                            let (tool_name, registered) = context_json
150                                .as_deref()
151                                .and_then(|s| serde_json::from_str::<serde_json::Value>(s).ok())
152                                .and_then(|v| {
153                                    let tn = v["tool_name"].as_str()?.to_owned();
154                                    let reg: Vec<String> = v["registered"]
155                                        .as_array()?
156                                        .iter()
157                                        .filter_map(|x| x.as_str().map(String::from))
158                                        .collect();
159                                    Some((tn, reg))
160                                })
161                                .unwrap_or_else(|| (detail.clone(), vec![]));
162                            RuntimeError::ToolNotRegistered {
163                                tool_name,
164                                registered,
165                            }
166                        }
167                        "Llm" => RuntimeError::Llm(LlmError::Internal { message: detail }),
168                        // Reconstruct the typed ToolError variant using
169                        // `tool_error_variant` recorded by make_tool_fatal_error.
170                        // This preserves the BadArgs/SessionDead/etc. variant
171                        // through the FatalError round-trip (Approach A fix).
172                        "Tool" => {
173                            let tool_err = match tool_error_variant.as_deref() {
174                                Some("BadArgs") => ToolError::BadArgs { reason: detail },
175                                Some("SessionDead") => ToolError::SessionDead { reason: detail },
176                                Some("DeadlineExceeded") => ToolError::DeadlineExceeded,
177                                Some("CapabilityDenied") => {
178                                    ToolError::CapabilityDenied { capability: detail }
179                                }
180                                // Llm/Storage/Internal and unknown future
181                                // variants all map to Internal — the detail
182                                // string carries the Display output.
183                                _ => ToolError::Internal { message: detail },
184                            };
185                            RuntimeError::Tool(tool_err)
186                        }
187                        _ => RuntimeError::Internal { message: detail },
188                    });
189                }
190                Some(_) => continue,
191                None => unreachable!(
192                    "run_streaming_inner must yield exactly one RunCompleted before stream end"
193                ),
194            }
195        }
196    }

239    /// Invoke a single tool by name without engaging the LLM loop.
240    ///
241    /// Bypasses the multi-turn agent driver — useful for callers that
242    /// want to compose tools directly (e.g., `tau-workflow`'s
243    /// `tool.call` step kind). The tool's capability requirements are
244    /// still checked against the `agent_def`'s package grant set, so
245    /// the caller must pass the workflow's default-agent definition.
246    ///
247    /// Follows the same sequence as the run loop's tool-dispatch arm:
248    /// `resolve_tool → capability check → init → invoke → teardown`.
249    ///
250    /// # Errors
251    ///
252    /// - [`RuntimeError::ToolNotRegistered`] — the tool name is unknown.
253    /// - [`RuntimeError::Internal`] — the agent's package does not grant
254    ///   a capability required by the tool (capability-denied path; the
255    ///   run loop surfaces this as `Ok(RunOutcome::Failed)` instead, but
256    ///   the direct-dispatch caller has no `RunOutcome` envelope).
257    /// - [`RuntimeError::Tool`] — the tool's `init`, `invoke`, or
258    ///   `teardown` returned a [`tau_ports::ToolError`].
259    pub async fn invoke_tool(
260        &self,
261        agent_def: &AgentDefinition,
262        package_manifest: &PackageManifest,
263        tool_name: &str,
264        args: tau_domain::Value,
265    ) -> Result<tau_ports::ToolResult, RuntimeError> {
266        use tau_domain::AgentInstanceId;
267        use tau_ports::SessionContext;
268
269        let tool = self.resolve_tool(tool_name)?.clone();
270
271        // Capability check: mirror the run loop's structural check.
272        // If the agent's package grants are insufficient, surface as
273        // RuntimeError::Internal (no CapabilityDenied variant exists on
274        // RuntimeError; the run loop returns Ok(RunOutcome::Failed) in
275        // the same situation, but invoke_tool has no RunOutcome envelope).
276        let granted: Vec<tau_domain::Capability> = package_manifest.capabilities().to_vec();
277        let required: &[tau_domain::Capability] = tool.capabilities();
278        if let Some(missing) = crate::capability::check_capabilities(&granted, required) {
279            let denial = crate::error::CapabilityDenial {
280                agent_id: agent_def.id.to_string(),
281                package_id: agent_def.package.name.to_string(),
282                tool_name: tool_name.to_owned(),
283                required_kind: crate::run::capability_kind_str(missing),
284                required_detail: format!("{missing:?}"),
285            };
286            return Err(RuntimeError::Internal {
287                message: format!("capability denied: {denial}"),
288            });
289        }
290
291        // Build a minimal SessionContext (no deadline, no deny entries).
292        // invoke_tool is a direct-dispatch path — callers that need
293        // capability narrowing or deny carve-outs should use run().
294        let ctx = SessionContext::new(AgentInstanceId::new(), uuid::Uuid::new_v4(), None)
295            .with_granted_capabilities(granted);
296
297        tool.init(ctx.clone()).await?;
298        let result = tool.invoke(&ctx, &mut (), args).await;
299        // teardown best-effort: don't mask invoke's error if both fail.
300        let _ = tool.teardown(()).await;
301        Ok(result?)
302    }

304    /// Multi-agent orchestrated run entry point (ROADMAP §9, v1).
305    ///
306    /// Builds a shared [`crate::orchestration::run_state::RunState`] (TaskList,
307    /// trace stream, budget counters), spawns a JSONL persister subscribed to
308    /// the trace stream, then runs the root agent via [`Runtime::run_with_history`]
309    /// with `orchestration_state` set in `RunOptions`. Virtual tool calls
310    /// (`task.*` / `run.*`) inside the agent loop are intercepted at the
311    /// kernel-dispatch boundary (see `crate::stream::run_streaming_inner`).
312    ///
313    /// On completion: marks the run terminal status (Completed iff agent
314    /// finished AND no orphan tasks), emits a final
315    /// `TraceEventKind::OrphanedTasksAtTermination` if orphans remain, and
316    /// returns a read-only [`tau_ports::RunSnapshot`].
317    ///
318    /// As of v1.1 (ADR-0025), `agent.<kind>.spawn` recursively invokes
319    /// [`Runtime::run_with_history`] for the child via the `Arc<Self>`
320    /// recursion handle threaded through `RunOptions::orchestration_runtime`.
321    /// The child run inherits the parent's `PackageManifest`, gets a
322    /// fresh `AgentId`, and runs with the validated narrowed grant from
323    /// [`crate::orchestration::AgentSpawnRequest`].
324    pub async fn spawn_root_agent(
325        self: std::sync::Arc<Self>,
326        root_agent_def: tau_domain::AgentDefinition,
327        root_manifest: tau_domain::PackageManifest,
328        initial_message: tau_domain::Message,
329        budget: tau_ports::RunBudget,
330        scope_root: std::path::PathBuf,
331    ) -> Result<tau_ports::RunSnapshot, RuntimeError> {
332        use std::sync::Arc;
333        use tokio::sync::Mutex;
334
335        let run_id = ulid::Ulid::new().to_string();
336        let root_agent_id = root_agent_def.id.to_string();
337        let now = chrono::Utc::now();
338
339        let mut state = crate::orchestration::run_state::RunState::new(
340            run_id.clone(),
341            root_agent_id,
342            budget,
343            now,
344        );
345
346        // Subscribe a JSONL writer before wrapping state in Arc<Mutex<>>.
347        let log_path = crate::orchestration::persistence::run_log_path(&scope_root, &run_id);
348        let writer_rx = state.trace.subscribe();
349        let _writer_handle = crate::orchestration::persistence::spawn_writer(log_path, writer_rx);
350
351        let state_arc = Arc::new(Mutex::new(state));
352
353        let opts = crate::RunOptions {
354            orchestration_state: Some(state_arc.clone()),
355            orchestration_runtime: Some(self.clone()),
356            ..crate::RunOptions::default()
357        };
358
359        let outcome = self
360            .run_with_history(
361                root_agent_def,
362                root_manifest,
363                Vec::new(),
364                initial_message,
365                opts,
366            )
367            .await?;
368
369        let now_end = chrono::Utc::now();
370        {
371            let mut s = state_arc.lock().await;
372            s.ended_at = Some(now_end);
373            let success = matches!(outcome, crate::RunOutcome::Completed { .. });
374            let orphans_present = !s.task_list.all_terminal();
375            s.status = if success && !orphans_present {
376                tau_ports::RunStatus::Completed
377            } else {
378                tau_ports::RunStatus::Failed
379            };
380            if orphans_present {
381                let orphan_ids: Vec<_> = s
382                    .task_list
383                    .all()
384                    .into_iter()
385                    .filter(|t| {
386                        !matches!(
387                            t.status,
388                            tau_ports::TaskStatus::Done
389                                | tau_ports::TaskStatus::Failed
390                                | tau_ports::TaskStatus::Discarded
391                        )
392                    })
393                    .map(|t| t.id)
394                    .collect();
395                s.trace.emit(tau_ports::TraceEvent {
396                    id: ulid::Ulid::new().to_string(),
397                    ts: now_end,
398                    run_id: run_id.clone(),
399                    agent_id: None,
400                    kind: tau_ports::TraceEventKind::OrphanedTasksAtTermination {
401                        task_ids: orphan_ids,
402                    },
403                });
404            }
405        }
406
407        let snapshot = {
408            let s = state_arc.lock().await;
409            s.snapshot(now_end)
410        };
411        Ok(snapshot)
412    }
413}

440/// Project the agent's [`Message`] history onto the LLM-call shape.
441///
442/// Per `tau_ports::llm` module-level docs, `tau_domain::Message`
443/// (universal envelope) and [`LlmProviderMessage`] (provider call
444/// shape) are intentionally distinct. This function is the single
445/// projection point in the kernel.
446///
447/// v0.1 mapping:
448///
449/// | Sender / payload                 | Provider role | Block(s)                          |
450/// | -------------------------------- | ------------- | --------------------------------- |
451/// | `User` / `Text`                  | `User`        | `Text`                            |
452/// | `Agent` / `Text`                 | `Assistant`   | `Text`                            |
453/// | `Agent` / `ToolCall`             | `Assistant`   | `ToolUse` (id derived from name)  |
454/// | `Tool` / `ToolResult` or Error   | `ToolResult`  | `Text` (flattened body)           |
455/// | other                            | (skipped)     | —                                 |
456///
457/// Lifecycle/Custom payloads are skipped — they are not part of the
458/// agent↔LLM dialogue at v0.1.
459pub(crate) fn agent_messages_to_provider_messages(history: &[Message]) -> Vec<LlmProviderMessage> {
460    let mut out = Vec::with_capacity(history.len());
461    for m in history {
462        match (&m.sender, &m.payload) {
463            (tau_domain::Address::User, MessagePayload::Text { content }) => {
464                out.push(LlmProviderMessage::user(vec![ContentBlock::Text(
465                    content.clone(),
466                )]));
467            }
468            (tau_domain::Address::Agent(_), MessagePayload::Text { content }) => {
469                out.push(LlmProviderMessage::assistant(vec![ContentBlock::Text(
470                    content.clone(),
471                )]));
472            }
473            (tau_domain::Address::Agent(_), MessagePayload::ToolCall { args }) => {
474                // `tool_use.id` round-trips into the provider's
475                // `tool_use_id`; v0.1 derives it from the message id so
476                // a follow-up `ToolResult` can be paired by the backend.
477                let tool_name = match &m.recipient {
478                    tau_domain::Address::Tool(name) => name.clone(),
479                    _ => String::new(),
480                };
481                out.push(LlmProviderMessage::assistant(vec![ContentBlock::ToolUse(
482                    ToolUse::new(format!("toolu_{}", m.id), tool_name, args.clone()),
483                )]));
484            }
485            (tau_domain::Address::Tool(_), MessagePayload::ToolResult { body }) => {
486                out.push(LlmProviderMessage::tool_result(
487                    format!("toolu_{}", m.id),
488                    vec![ContentBlock::Text(value_to_preview_string(body))],
489                    false,
490                ));
491            }
492            (
493                tau_domain::Address::Tool(_),
494                MessagePayload::ToolError {
495                    kind: _,
496                    message,
497                    details: _,
498                },
499            ) => {
500                out.push(LlmProviderMessage::tool_result(
501                    format!("toolu_{}", m.id),
502                    vec![ContentBlock::Text(message.clone())],
503                    true,
504                ));
505            }
506            // Lifecycle / Custom / cross-talk patterns: not part of the
507            // v0.1 agent↔LLM dialogue. Skip silently — the integration
508            // tests assert the projected shape, and additive semantics
509            // are non-breaking.
510            _ => {}
511        }
512    }
513    out
514}

516/// Append the assistant's response (text + tool_uses) to the history.
517///
518/// Mints one `Message` per logical assistant action:
519///
520/// - One [`MessagePayload::Text`] message (only if `text` is non-empty).
521/// - The tool_uses themselves are NOT pushed here — they become
522///   `MessagePayload::ToolCall` messages later in the dispatch loop,
523///   one per `tool_use`. This keeps the history's cause-effect order
524///   consistent: text before any tool call, then the tool call paired
525///   immediately with its result.
526///
527/// Now only called from unit tests (the agent loop moved to stream.rs).
528#[cfg(test)]
529pub(crate) fn append_assistant_response(
530    history: &mut Vec<Message>,
531    text: &str,
532    tool_uses: &[ToolUse],
533    agent_id: &AgentInstanceId,
534) {
535    if !text.is_empty() {
536        history.push(Message::new(
537            tau_domain::Address::Agent(*agent_id),
538            tau_domain::Address::User,
539            MessagePayload::Text {
540                content: text.to_owned(),
541            },
542        ));
543    } else if tool_uses.is_empty() {
544        // Defensive: the LLM returned neither text nor tool_uses.
545        // Spec says this shouldn't happen, but we still need a
546        // recognisable assistant turn in the history so callers'
547        // `final_message` is the assistant's empty response (not the
548        // initial user prompt). Push an empty-text message rather than
549        // synthesising a richer placeholder.
550        history.push(Message::new(
551            tau_domain::Address::Agent(*agent_id),
552            tau_domain::Address::User,
553            MessagePayload::Text {
554                content: String::new(),
555            },
556        ));
557    }
558}

574/// Flatten a tool's content blocks into a single human-readable string,
575/// for use as the `message` field of [`MessagePayload::ToolError`].
576pub(crate) fn flatten_content_to_string(blocks: &[ToolContent]) -> String {
577    let mut out = String::new();
578    for block in blocks {
579        match block {
580            ToolContent::Text { text } => {
581                if !out.is_empty() {
582                    out.push('\n');
583                }
584                out.push_str(text);
585            }
586            ToolContent::Json { data } => {
587                if !out.is_empty() {
588                    out.push('\n');
589                }
590                out.push_str(&value_to_preview_string(data));
591            }
592            // `ToolContent` is `#[non_exhaustive]`; future variants
593            // contribute nothing to the preview at v0.1.
594            _ => {}
595        }
596    }
597    out
598}

600/// Build a [`Value`] from a tool's content blocks. v0.1 rule:
601///
602/// - exactly one [`ToolContent::Json { data }`] → return `data` directly;
603/// - everything else → wrap into an `Object { content: Array of strings/values }`.
604pub(crate) fn content_to_value(blocks: &[ToolContent]) -> Value {
605    if blocks.len() == 1 {
606        if let ToolContent::Json { data } = &blocks[0] {
607            return data.clone();
608        }
609    }
610    let arr: Vec<Value> = blocks
611        .iter()
612        .map(|b| match b {
613            ToolContent::Text { text } => Value::String(text.clone()),
614            ToolContent::Json { data } => data.clone(),
615            _ => Value::Null,
616        })
617        .collect();
618    let mut obj = BTreeMap::new();
619    obj.insert("content".to_string(), Value::Array(arr));
620    Value::Object(obj)
621}

623/// Compact preview string for a [`Value`]. Used for `LlmProviderMessage`
624/// projection of a tool result body (the LLM-call shape carries text,
625/// not structured Value, in v0.1's content-block surface).
626fn value_to_preview_string(v: &Value) -> String {
627    match v {
628        Value::String(s) => s.clone(),
629        Value::Null => "null".to_string(),
630        Value::Bool(b) => b.to_string(),
631        Value::Integer(i) => i.to_string(),
632        Value::Float(f) => f.to_string(),
633        Value::Bytes(b) => format!("<{} bytes>", b.len()),
634        Value::Array(_) | Value::Object(_) => format!("{v:?}"),
635        // `Value` is `#[non_exhaustive]`; future variants degrade to a
636        // generic Debug-format string rather than mis-classifying.
637        _ => format!("{v:?}"),
638    }
639}

641/// Build the post-narrow `Capability` view that flows to plugins via
642/// `SessionContext.granted_capabilities`. Capability inner variants are
643/// `#[non_exhaustive]` and can't be constructed cross-crate; we serialize
644/// the source, splice in the narrowed allow-list / max_bytes, and
645/// deserialize back.
646///
647/// Failure-safe: any serialization failure falls back to `eff.source.clone()`.
648/// The kernel's structural cap check still applies — narrowing is best-effort
649/// at this layer; panicking on a security-enforcement path would be the
650/// wrong failure mode.
651pub(crate) fn narrowed_capability_for_session(eff: &EffectiveCapability) -> Capability {
652    use serde_json::{json, Value as Jv};
653
654    let source_json = match serde_json::to_value(&eff.source) {
655        Ok(v) => v,
656        Err(_) => return eff.source.clone(),
657    };
658    let mut obj = match source_json.as_object() {
659        Some(m) => m.clone(),
660        None => return eff.source.clone(),
661    };
662    if let Some(allow) = &eff.allow_override {
663        // Replace the kind-appropriate field. For unknown kinds (e.g. Custom),
664        // bail and return source unchanged — narrowing is unsupported.
665        let field = match obj.get("kind").and_then(Jv::as_str) {
666            Some("fs.read") | Some("fs.write") | Some("fs.exec") => "paths",
667            Some("net.http") => "hosts",
668            Some("process.spawn") => "commands",
669            _ => return eff.source.clone(),
670        };
671        obj.insert(field.to_string(), json!(allow));
672    }
673    if let Some(mb) = eff.max_bytes_override {
674        obj.insert("max_bytes".to_string(), json!(mb));
675    }
676    serde_json::from_value(Jv::Object(obj)).unwrap_or_else(|_| eff.source.clone())
677}

679/// Top-level capability kind string used in
680/// [`CapabilityDenial::required_kind`] and the `capability.deny` event.
681pub(crate) fn capability_kind_str(cap: &Capability) -> String {
682    use tau_domain::{
683        AgentCapability, FsCapability, NetCapability, ProcessCapability, SkillCapability,
684    };
685    match cap {
686        Capability::Filesystem(FsCapability::Read { .. }) => "fs.read".into(),
687        Capability::Filesystem(FsCapability::Write { .. }) => "fs.write".into(),
688        Capability::Filesystem(FsCapability::Exec { .. }) => "fs.exec".into(),
689        Capability::Network(NetCapability::Http { .. }) => "net.http".into(),
690        Capability::Process(ProcessCapability::Spawn { .. }) => "process.spawn".into(),
691        Capability::Agent(AgentCapability::Spawn { .. }) => "agent.spawn".into(),
692        Capability::TaskList { .. } => "task_list".into(),
693        Capability::Plan { .. } => "plan".into(),
694        Capability::Skill(SkillCapability::Spawn { .. }) => "skill.spawn".into(),
695        Capability::Custom { name, .. } => name.clone(),
696        // `Capability` is `#[non_exhaustive]`; future variants degrade
697        // to a generic tag — additive evolution must not silently
698        // mis-classify them.
699        _ => "unknown".into(),
700    }
701}
