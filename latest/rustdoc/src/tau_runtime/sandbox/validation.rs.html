1//! Layer 3 pre-flight plan validation.
2//!
3//! [`validate_plan_against_adapter`] cross-checks every capability in a
4//! [`tau_ports::SandboxPlan`] against the adapter's declared
5//! `supported_shapes`. All validation errors are collected and returned at
6//! once so callers see the complete picture in a single pass.
7
8use tau_domain::Capability;
9use tau_ports::{Sandbox, SandboxPlan};
10
11/// A single capability-shape rejection produced by
12/// [`validate_plan_against_adapter`].
13///
14/// Carries enough context for callers to format a user-facing message
15/// without further synthesis.
16#[non_exhaustive]
17#[derive(Debug, Clone)]
18pub struct SandboxValidationError {
19    /// Identifier the caller passed in (typically a plugin id).
20    pub plan_id: String,
21    /// The capability that produced the error.
22    pub capability: Capability,
23    /// Human-readable reason — typically the shape rejected and what
24    /// the adapter supports. Concrete enough to render directly to the
25    /// user without further synthesis.
26    pub reason: String,
27}
28
29impl SandboxValidationError {
30    /// Construct a [`SandboxValidationError`].
31    ///
32    /// `#[non_exhaustive]` blocks struct-literal construction outside this
33    /// crate; use this constructor instead.
34    pub fn new(
35        plan_id: impl Into<String>,
36        capability: Capability,
37        reason: impl Into<String>,
38    ) -> Self {
39        Self {
40            plan_id: plan_id.into(),
41            capability,
42            reason: reason.into(),
43        }
44    }
45}
46
47impl std::fmt::Display for SandboxValidationError {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        write!(f, "plan {}: {}", self.plan_id, self.reason)
50    }
51}
52
53impl std::error::Error for SandboxValidationError {}
54
55/// Cross-check every capability in `plan` against `adapter.supported_shapes()`.
56///
57/// **All** unsupported capabilities are collected before returning — the error
58/// `Vec` may contain more than one entry. This is the key value of Layer 3:
59/// a single `tau check` run surfaces every problem at once.
60///
61/// # Arguments
62///
63/// * `plan_id` — free-form identifier carried into each [`SandboxValidationError`]
64///   so callers can format messages like `"plugin foo: capability fs.read shape
65///   unsupported by adapter X"`. Typically a plugin id.
66/// * `plan` — the [`SandboxPlan`] to validate.
67/// * `adapter` — any [`Sandbox`] implementor (e.g. `SandboxAdapter` from the
68///   chain, `MockSandbox` from fixtures, or another test double).
69///
70/// # Returns
71///
72/// `Ok(())` if every capability shape in `plan` is in `adapter.supported_shapes()`.
73/// `Err(errors)` with the complete list of failures otherwise.
74pub fn validate_plan_against_adapter<S: Sandbox>(
75    plan_id: &str,
76    plan: &SandboxPlan,
77    adapter: &S,
78) -> Result<(), Vec<SandboxValidationError>> {
79    let supported = adapter.supported_shapes();
80
81    let sup_list: String = supported
82        .iter()
83        .map(|s| format!("{s:?}"))
84        .collect::<Vec<_>>()
85        .join(", ");
86
87    let mut errors: Vec<SandboxValidationError> = Vec::new();
88
89    for cap in &plan.capabilities {
90        let required = cap.required_shape();
91        if !supported.contains(&required) {
92            let reason =
93                format!("adapter does not support shape {required:?} (supported: {sup_list})");
94            errors.push(SandboxValidationError::new(plan_id, cap.clone(), reason));
95        }
96    }
97
98    if errors.is_empty() {
99        Ok(())
100    } else {
101        Err(errors)
102    }
103}

105#[cfg(test)]
106mod tests {
107    use super::*;
108    use tau_domain::CapabilityShape;
109    use tau_ports::fixtures::MockSandbox;
110
111    fn cap(json: &str) -> Capability {
112        serde_json::from_str(json).expect("test capability JSON must be valid")
113    }
114
115    fn plan_with(caps: Vec<Capability>) -> SandboxPlan {
116        SandboxPlan::new(caps, None, None)
117    }
118
119    #[test]
120    fn validation_passes_when_all_shapes_supported() {
121        // MockSandbox supports the 5 standard shapes (not Custom).
122        let adapter = MockSandbox::new("mock");
123        let plan = plan_with(vec![cap(r#"{"kind":"fs.read","paths":["/tmp/**"]}"#)]);
124        assert!(
125            validate_plan_against_adapter("test-plugin", &plan, &adapter).is_ok(),
126            "expected Ok for fs.read on MockSandbox"
127        );
128    }
129
130    #[test]
131    fn validation_returns_all_errors_not_just_first() {
132        // Two Custom capabilities — MockSandbox does not support CapabilityShape::Custom.
133        let adapter = MockSandbox::new("mock");
134        let plan = plan_with(vec![
135            cap(r#"{"kind":"mcp.tool.use","tool":"x"}"#),
136            cap(r#"{"kind":"mcp.tool.other","tool":"y"}"#),
137        ]);
138        let errors = validate_plan_against_adapter("test-plugin", &plan, &adapter)
139            .expect_err("expected validation errors for Custom capabilities");
140        assert_eq!(
141            errors.len(),
142            2,
143            "both custom capabilities should produce errors, got {:?}",
144            errors.len()
145        );
146        for e in &errors {
147            assert_eq!(e.plan_id, "test-plugin");
148        }
149    }
150
151    #[test]
152    fn validation_includes_plan_id_in_each_error() {
153        let adapter = MockSandbox::new("mock");
154        let plan = plan_with(vec![cap(r#"{"kind":"mcp.tool.use","tool":"x"}"#)]);
155        let errors = validate_plan_against_adapter("test-plugin", &plan, &adapter)
156            .expect_err("expected error for Custom capability");
157        assert_eq!(errors.len(), 1);
158        assert_eq!(errors[0].plan_id, "test-plugin");
159    }
160
161    #[test]
162    fn validation_empty_plan_is_ok() {
163        let adapter = MockSandbox::new("mock");
164        let plan = plan_with(vec![]);
165        assert!(
166            validate_plan_against_adapter("test-plugin", &plan, &adapter).is_ok(),
167            "empty plan should always pass"
168        );
169    }
170
171    #[test]
172    fn sandbox_validation_error_display_includes_id_and_reason() {
173        let cap_val = cap(r#"{"kind":"mcp.tool.use","tool":"x"}"#);
174        let e = SandboxValidationError::new(
175            "my-plugin",
176            cap_val,
177            "adapter does not support shape Custom { name: \"mcp.tool.use\" } (supported: FilesystemRead)",
178        );
179        let display = format!("{e}");
180        assert!(
181            display.contains("my-plugin"),
182            "display must contain plan_id; got: {display}"
183        );
184        assert!(
185            display.contains("adapter does not support shape"),
186            "display must contain the reason text; got: {display}"
187        );
188        // Verify shape info is in the output too.
189        assert!(
190            display.contains("Custom"),
191            "display must include shape info; got: {display}"
192        );
193        // Also verify the capability's required_shape is Custom (sanity check on the fixture).
194        let shape = e.capability.required_shape();
195        assert!(
196            matches!(shape, CapabilityShape::Custom { .. }),
197            "shape must be Custom"
198        );
199    }
