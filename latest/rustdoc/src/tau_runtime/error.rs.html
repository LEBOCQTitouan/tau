1//! Per-operation typed errors for `tau-runtime`.
2//!
3//! All public errors are `#[non_exhaustive]` so additive variants are
4//! non-breaking. [`BuildError`] derives
5//! `Debug + Clone + PartialEq + Eq + Error`; [`RuntimeError`] derives
6//! `Debug + Error` only — plugin-host variants carry
7//! [`std::io::Error`] and [`std::process::ExitStatus`], neither of
8//! which is `Clone`/`Eq`, so we can't keep the richer set of derives
9//! at the top level. Tests with free-form `String` fields use
10//! `matches!()` to avoid brittle wording comparisons.
11//!
12//! The error taxonomy splits into two layers:
13//!
14//! - [`BuildError`] — failures during `RuntimeBuilder::build()`. The
15//!   runtime never gets constructed.
16//! - `RuntimeError` (added in Task 4) — kernel-level operational
17//!   failures during `Runtime::run`. Composes `tau_ports` plugin errors
18//!   via `#[from]`. Agent-level failures (capability denied, max turns
19//!   reached) are reported via `Ok(RunOutcome::Failed { status:
20//!   AgentStatus::Failed })`, NOT `Err(RuntimeError)`.
21//!
22//! [`CapabilityDenial`] is a helper type embedded as the `detail`
23//! string of `AgentStatus::Failed { kind: PolicyDenied }` when
24//! capability enforcement rejects a tool call. It is NOT a variant
25//! of `RuntimeError`.
26
27use thiserror::Error;
28
29/// Tag identifying a plugin kind in error messages and tracing fields.
30#[non_exhaustive]
31#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
32pub enum PluginKind {
33    /// LLM backend plugin (`kind = "llm-backend"`).
34    LlmBackend,
35    /// Tool plugin (`kind = "tool"`).
36    Tool,
37    /// Storage plugin (`kind = "storage"`).
38    Storage,
39    /// Sandbox plugin (`kind = "sandbox"`); reserved for forward compat.
40    Sandbox,
41}
42
43impl std::fmt::Display for PluginKind {
44    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
45        match self {
46            PluginKind::LlmBackend => f.write_str("llm-backend"),
47            PluginKind::Tool => f.write_str("tool"),
48            PluginKind::Storage => f.write_str("storage"),
49            PluginKind::Sandbox => f.write_str("sandbox"),
50        }
51    }
52}

54/// Errors from `RuntimeBuilder::build()` (added in Task 7).
55///
56/// # Example
57///
58/// ```ignore
59/// // `BuildError` is `#[non_exhaustive]`; constructed by `build()`.
60/// // Construction example deferred to Task 7 when the builder lands.
61/// ```
62#[non_exhaustive]
63#[derive(Debug, Clone, PartialEq, Eq, Error)]
64pub enum BuildError {
65    /// At least one LLM backend must be registered before `build()`.
66    #[error("no LLM backends registered; at least one is required")]
67    NoLlmBackend,
68
69    /// Two plugins of the same kind registered with the same `name()`.
70    #[error("name collision: two {kind}s registered as {name:?}")]
71    NameCollision {
72        /// Which plugin kind collided.
73        kind: PluginKind,
74        /// The colliding name.
75        name: String,
76    },
77
78    /// A registered tool's declared `input_schema` is not a valid
79    /// Draft 7 JSON Schema. Surfaced at `RuntimeBuilder::build()`
80    /// via `crate::tool_args::ToolArgsValidator::compile`. Realizes
81    /// ADR-0010 (Tier 2 priority 6).
82    #[error("tool {tool_name:?}: input_schema is not a valid Draft 7 schema: {detail}")]
83    ToolSchemaInvalid {
84        /// The tool name whose schema failed to compile.
85        tool_name: String,
86        /// Diagnostic from the schema compiler.
87        detail: String,
88    },
89
90    /// Catch-all for invariant violations during build.
91    /// See: [escape-hatches.md#builderror-internal](../docs/explanation/escape-hatches.md#builderror-internal).
92    #[error("internal: {message}")]
93    Internal {
94        /// Human-readable message describing the internal failure.
95        message: String,
96    },
97}

99/// Capability-denial detail. Embedded as the `detail` string of
100/// `AgentStatus::Failed { kind: PolicyDenied, .. }` when capability
101/// enforcement rejects a tool call.
102///
103/// NOT a variant of `RuntimeError` (added in Task 4) — capability
104/// denial is an agent-level failure (`Ok(RunOutcome::Failed)`), not
105/// a kernel-level error (`Err(RuntimeError)`). See ADR-0006 for the
106/// dichotomy.
107#[non_exhaustive]
108#[derive(Debug, Clone, PartialEq, Eq)]
109pub struct CapabilityDenial {
110    /// `AgentDefinition::id` formatted via `Display`.
111    pub agent_id: String,
112    /// `AgentDefinition::package` formatted via `Display`.
113    pub package_id: String,
114    /// The tool the agent attempted to call.
115    pub tool_name: String,
116    /// Top-level kind of the missing capability ("filesystem.read",
117    /// "network.http", "tool.echo" — convention).
118    pub required_kind: String,
119    /// Human-readable description of the capability that wasn't satisfied.
120    pub required_detail: String,
121}
122
123impl std::fmt::Display for CapabilityDenial {
124    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
125        write!(
126            f,
127            "agent {} (package {}) lacks capability `{}` ({}) required to call tool `{}`",
128            self.agent_id,
129            self.package_id,
130            self.required_kind,
131            self.required_detail,
132            self.tool_name,
133        )
134    }
135}

137/// Errors from `Runtime::run` (added in Task 10) — kernel-level
138/// operational failures.
139///
140/// Agent-level failures (capability denied, max turns reached) flow
141/// through `Ok(RunOutcome::Failed { status: AgentStatus::Failed{..} })`
142/// instead. See [`crate::error`] module-level docs for the dichotomy.
143///
144/// Plugin errors (`LlmError`, `ToolError`, `StorageError`, `SandboxError`)
145/// compose via `#[from]` for ergonomic `?`-propagation throughout the
146/// agent loop.
147///
148/// `RuntimeError` does **not** derive `Clone + PartialEq + Eq` — the
149/// plugin-host variants ([`RuntimeError::PluginSpawnFailed`],
150/// [`RuntimeError::PluginCrashed`]) carry [`std::io::Error`] and
151/// [`std::process::ExitStatus`], neither of which is `Clone` or `Eq`.
152/// Existing tests destructure with `let RuntimeError::X { .. } = ...`
153/// rather than relying on equality.
154#[non_exhaustive]
155#[derive(Debug, Error)]
156pub enum RuntimeError {
157    /// Agent's `llm_backend` references a backend that wasn't registered.
158    #[error("LLM backend `{backend}` not registered (agent {agent_id} requested it)")]
159    LlmBackendNotRegistered {
160        /// The agent's `id` formatted via `Display`.
161        agent_id: String,
162        /// The backend name the agent requested.
163        backend: String,
164    },
165
166    /// LLM emitted a tool_use targeting a tool not in the registry.
167    #[error("tool `{tool_name}` not registered; registered: {registered:?}")]
168    ToolNotRegistered {
169        /// The tool name the LLM requested.
170        tool_name: String,
171        /// Names of registered tools (for diagnostics).
172        registered: Vec<String>,
173    },
174
175    /// Plugin returned successfully but its output violates the contract
176    /// (malformed JSON args from LLM, undeserializable response from a
177    /// loaded plugin, etc.). Surfaced both by the run loop's
178    /// argument-validation pass and by the `plugin_host` IPC adapters
179    /// (Tasks 14-17) when a plugin's response is structurally invalid.
180    #[error("plugin {plugin} contract violation: {detail}")]
181    PluginContractViolation {
182        /// The plugin name (its `name()` value, or the manifest name
183        /// for plugin-host violations).
184        plugin: String,
185        /// Human-readable detail describing what was malformed.
186        detail: String,
187    },
188
189    /// `tokio::process::Command::spawn` failed for the plugin binary
190    /// (binary missing, not executable, sandbox policy denied, etc.).
191    /// Surfaced by [`crate::plugin_host::load_llm_backend`] and friends
192    /// (Tasks 14+).
193    #[error("failed to spawn plugin {plugin}: {source}")]
194    PluginSpawnFailed {
195        /// The plugin name (from `LockedPlugin::manifest.name`).
196        plugin: String,
197        /// The underlying `std::io::Error` from `spawn`.
198        #[source]
199        source: std::io::Error,
200    },
201
202    /// Plugin spawned successfully but the `meta.handshake` exchange
203    /// failed for one of the reasons enumerated in
204    /// [`HandshakeFailureReason`]. Surfaced by
205    /// [`crate::plugin_host`] (Tasks 14+).
206    #[error("plugin {plugin} handshake failed: {reason}")]
207    PluginHandshakeFailed {
208        /// The plugin name (from `LockedPlugin::manifest.name`).
209        plugin: String,
210        /// Specific reason the handshake failed.
211        reason: HandshakeFailureReason,
212    },
213
214    /// Plugin process exited unexpectedly during a request. The
215    /// captured `stderr_tail` (last N bytes of stderr) aids triage.
216    /// Surfaced by [`crate::plugin_host`] (Tasks 14+).
217    #[error("plugin {plugin} crashed: exit {exit_status}")]
218    PluginCrashed {
219        /// The plugin name (from `LockedPlugin::manifest.name`).
220        plugin: String,
221        /// Exit status reported by the OS.
222        exit_status: std::process::ExitStatus,
223        /// Tail of the plugin's stderr, captured for diagnostics.
224        stderr_tail: String,
225    },
226
227    /// LLM backend plugin returned an error.
228    #[error("llm: {0}")]
229    Llm(#[from] tau_ports::LlmError),
230
231    /// Tool plugin returned an error.
232    #[error("tool: {0}")]
233    Tool(#[from] tau_ports::ToolError),
234
235    /// Storage plugin returned an error.
236    #[error("storage: {0}")]
237    Storage(#[from] tau_ports::StorageError),
238
239    /// Sandbox adapter returned an error. Reserved for future use;
240    /// v0.1 in-tree adapters (`tau-sandbox-native`, etc.) populate this.
241    #[error("sandbox: {0}")]
242    Sandbox(#[from] tau_ports::SandboxError),
243
244    /// Sandbox plan validation failed before plugin spawn.
245    ///
246    /// Raised by [`crate::plugin_host`] when
247    /// [`crate::sandbox::validate_plan_against_adapter`] returns one or
248    /// more errors (Layer 3 enforcement at spawn time).
249    #[non_exhaustive]
250    #[error("plugin {plugin}: sandbox validation failed:\n{}",
251        errors.iter().map(|e| format!("  - {e}")).collect::<Vec<_>>().join("\n"))]
252    SandboxValidationFailed {
253        /// Plugin name (from `LockedPlugin::manifest.bin`).
254        plugin: String,
255        /// All validation errors collected in a single pass.
256        errors: Vec<crate::sandbox::SandboxValidationError>,
257    },
258
259    /// Sandbox adapter failed to wrap the spawn command.
260    ///
261    /// Raised by [`crate::plugin_host`] when
262    /// [`crate::sandbox::SandboxAdapter::wrap_spawn`] returns `Err`.
263    #[non_exhaustive]
264    #[error("plugin {plugin}: sandbox wrap-spawn failed: {source}")]
265    SandboxWrapFailed {
266        /// Plugin name (from `LockedPlugin::manifest.bin`).
267        plugin: String,
268        /// Underlying [`tau_ports::SandboxError`].
269        #[source]
270        source: tau_ports::SandboxError,
271    },
272
273    /// Manifest validation failed (caller-supplied manifest invalid).
274    #[error("manifest validation: {0}")]
275    Manifest(#[from] tau_domain::PackageManifestError),
276
277    /// Project capability override expanded the package's grant.
278    /// Raised at runtime as a defense-in-depth check (tau-cli also
279    /// rejects at parse time via its own ProjectConfigError variant).
280    #[error("capability override on {kind:?} expands package grant: {reason}")]
281    CapabilityOverrideExpands {
282        /// The capability kind that expanded.
283        kind: String,
284        /// Human-readable reason from `compute_effective`.
285        reason: String,
286    },
287
288    /// Catch-all for invariant violations / unexpected states.
289    /// See: [escape-hatches.md#runtimeerror-internal](../docs/explanation/escape-hatches.md#runtimeerror-internal).
290    #[error("internal: {message}")]
291    Internal {
292        /// Human-readable message describing the internal failure.
293        message: String,
294    },
295}
296
297/// Specific reason a plugin handshake (`meta.handshake` exchange)
298/// failed. Carried inside [`RuntimeError::PluginHandshakeFailed`].
299///
300/// `#[non_exhaustive]`: the Phase-1 protocol surface may grow
301/// additional handshake-failure modes as the schema-introspection
302/// surface evolves; additive variants must remain non-breaking.
303#[non_exhaustive]
304#[derive(Debug, thiserror::Error, Clone, PartialEq)]
305pub enum HandshakeFailureReason {
306    /// Plugin did not respond to `meta.handshake` within
307    /// [`crate::plugin_host::PluginHostOptions::handshake_timeout`].
308    #[error("timeout")]
309    Timeout,
310
311    /// Plugin's advertised `protocol_version` differs from the host's.
312    #[error("protocol version mismatch: host {host}, plugin {plugin}")]
313    ProtocolVersionMismatch {
314        /// Host's protocol version.
315        host: String,
316        /// Plugin's advertised protocol version.
317        plugin: String,
318    },
319
320    /// Plugin's advertised `provides` port differs from the
321    /// `LockedPlugin.manifest.provides` claim. Catches manifest
322    /// drift and binary swaps.
323    #[error("provides mismatch: manifest says {manifest}, plugin advertises {plugin_advertised}")]
324    ProvidesMismatch {
325        /// What the install-time manifest declared.
326        manifest: tau_domain::PortKind,
327        /// What the plugin advertised in its handshake response.
328        plugin_advertised: tau_domain::PortKind,
329    },
330
331    /// Plugin doesn't advertise a method the host requires for the
332    /// declared port (e.g. an LLM backend without `llm.complete`).
333    #[error("missing required method: {method}")]
334    MissingRequiredMethod {
335        /// The wire-method name that was missing
336        /// (`"llm.complete"`, `"tool.call"`, etc.).
337        method: String,
338    },
339
340    /// Plugin's handshake response was structurally malformed (failed
341    /// to deserialize, missing required fields, etc.).
342    #[error("malformed response: {detail}")]
343    Malformed {
344        /// Human-readable detail describing the malformation.
345        detail: String,
346    },
347}
