1//! Tool-args schema validation at the kernel boundary.
2//!
3//! Validates every tool-call's arguments against the tool's declared
4//! `ToolSpec.input_schema` (JSON Schema Draft 7) before invoking
5//! `Tool::invoke`. Validators are pre-compiled once at
6//! `RuntimeBuilder::build()` so per-invoke cost stays in the tens of
7//! microseconds.
8//!
9//! Two failure paths:
10//!   - Schema malformed at registration → `BuildError::ToolSchemaInvalid`
11//!     (returned by `compile`; surfaced by the builder).
12//!   - Args mismatch at invoke → `ToolError::BadArgs` carrying a
13//!     self-instructive reason that includes the full declared
14//!     `input_schema` plus the original args plus the specific issue.
15//!
16//! See `docs/superpowers/specs/2026-04-30-tool-args-schema-design.md`
17//! and ADR-0010.
18
19use std::sync::Arc;
20
21use jsonschema::{Draft, ValidationOptions, Validator};
22use tau_domain::Value;
23use tau_ports::ToolError;

25/// Compiled validator for a single tool's `input_schema`.
26///
27/// Built once at `RuntimeBuilder::build()` per registered tool.
28/// `compile()` rejects malformed schemas; `validate()` rejects
29/// non-conforming runtime args with a self-instructive error string.
30///
31/// # Cloneability
32///
33/// `ToolArgsValidator` is `Clone`: the compiled `jsonschema::Validator`
34/// is wrapped in `Arc` so cloning is a reference-count bump rather than
35/// a schema recompile. This is required by `Runtime::run_streaming_with_history`,
36/// which snapshots the `tool_validators` registry into an owned HashMap
37/// for the `'static` stream.
38#[non_exhaustive]
39#[derive(Debug, Clone)]
40pub(crate) struct ToolArgsValidator {
41    /// Compiled jsonschema instance. `None` = tool opted out (empty
42    /// schema or `Value::Null`); validation is a no-op.
43    ///
44    /// Wrapped in `Arc` so that `ToolArgsValidator` can be `Clone`
45    /// without recompiling the schema — clone is an Arc reference-count
46    /// bump.
47    compiled: Option<Arc<Validator>>,
48    /// The original input_schema as declared, kept for inclusion in
49    /// BadArgs error messages (the MANDATORY rule).
50    declared_schema_json: String,
51}
52
53impl ToolArgsValidator {
54    /// Compile from a tool's declared `input_schema`. Returns
55    /// `Err(SchemaCompileError)` if the schema is not a valid Draft 7
56    /// schema.
57    ///
58    /// Empty (`Value::Null` or empty `Value::Object`) opts out — the
59    /// returned validator passes everything.
60    #[allow(dead_code)] // wired up by Task 3 (builder integration)
61    pub(crate) fn compile(input_schema: &Value) -> Result<Self, SchemaCompileError> {
62        let schema_json = serde_json::to_value(input_schema).map_err(|e| SchemaCompileError {
63            kind: format!("internal: schema serialization failed: {e}"),
64            schema_excerpt: String::new(),
65        })?;
66        let declared_schema_json = serde_json::to_string(&schema_json).unwrap_or_default();
67
68        // Opt-out: null or empty object schema accepts everything.
69        let is_opt_out = match &schema_json {
70            serde_json::Value::Null => true,
71            serde_json::Value::Object(map) => map.is_empty(),
72            _ => false,
73        };
74        if is_opt_out {
75            return Ok(Self {
76                compiled: None,
77                declared_schema_json,
78            });
79        }
80
81        let compiled = draft7_options()
82            .build(&schema_json)
83            .map_err(|err| SchemaCompileError {
84                kind: format!("schema compile failed: {err}"),
85                schema_excerpt: declared_schema_json.chars().take(200).collect(),
86            })?;
87        Ok(Self {
88            compiled: Some(Arc::new(compiled)),
89            declared_schema_json,
90        })
91    }
92
93    /// Validate runtime args against the compiled schema. Returns
94    /// `Err(reason)` on failure where `reason` follows the MANDATORY
95    /// template from ADR-0010 §4: every error contains `"You sent:"`,
96    /// `"Expected (input_schema):"`, and `"Specific issue"` substrings.
97    ///
98    /// On opt-out (empty/null schema), validation is a no-op and
99    /// returns `Ok(())`.
100    pub(crate) fn validate(&self, args: &Value, tool_name: &str) -> Result<(), String> {
101        let Some(compiled) = &self.compiled else {
102            return Ok(());
103        };
104        let args_json = serde_json::to_value(args)
105            .map_err(|e| format!("internal: args serialization failed: {e}"))?;
106        let issues: Vec<String> = compiled
107            .iter_errors(&args_json)
108            .map(|e| format!("  {}: {}", e.instance_path(), e))
109            .collect();
110        if !issues.is_empty() {
111            let args_repr =
112                serde_json::to_string(&args_json).unwrap_or_else(|_| "<unprintable>".into());
113            return Err(format!(
114                "{tool_name}: args validation failed.\n\n\
115                 You sent: {args_repr}\n\n\
116                 Expected (input_schema):\n{schema}\n\n\
117                 Specific issue(s):\n{issues}",
118                schema = self.declared_schema_json,
119                issues = issues.join("\n"),
120            ));
121        }
122        Ok(())
123    }
124}

126/// Construct a `ValidationOptions` builder configured for JSON Schema Draft 7.
127#[allow(dead_code)] // called only via compile(), which is wired up by Task 3
128fn draft7_options() -> ValidationOptions<'static> {
129    jsonschema::options().with_draft(Draft::Draft7)
130}
131
132/// Error returned when a tool's `input_schema` fails to compile as a
133/// valid Draft 7 schema. Surfaced by `RuntimeBuilder::build()` as
134/// `BuildError::ToolSchemaInvalid`.
135#[allow(dead_code)] // wired up by Task 3 (builder integration)
136#[non_exhaustive]
137#[derive(Debug, Clone)]
138pub(crate) struct SchemaCompileError {
139    /// jsonschema's diagnostic.
140    pub kind: String,
141    /// First 200 chars of the schema for context.
142    pub schema_excerpt: String,
143}
144
145/// Validate runtime args via the kernel's pre-compiled validator.
146/// Replaces the v0.1 `deserialize_tool_args` passthrough at `run.rs:417`.
147///
148/// Returns `Err(ToolError::BadArgs { reason })` on validation failure.
149/// The caller in `run.rs` is responsible for routing this through the
150/// validation-failure short-circuit (write MessagePayload::ToolError
151/// to the conversation; `continue` the inner loop) — DO NOT propagate
152/// via `?`, which would terminate the run.
153#[allow(dead_code)] // wired up by Task 4 (run.rs call-site integration)
154pub(crate) fn validate_tool_args<'a>(
155    value: &'a Value,
156    tool_name: &str,
157    validator: &ToolArgsValidator,
158) -> Result<&'a Value, ToolError> {
159    match validator.validate(value, tool_name) {
160        Ok(()) => Ok(value),
161        Err(reason) => Err(ToolError::BadArgs { reason }),
162    }
163}

170    /// Build a tau_domain::Value JSON Schema from a `serde_json::json!` literal.
171    fn schema(json: serde_json::Value) -> Value {
172        let s = serde_json::to_string(&json).expect("schema serializes");
173        serde_json::from_str(&s).expect("schema round-trips through tau_domain::Value")
174    }
175
176    /// Build a tau_domain::Value args object from a `serde_json::json!` literal.
177    fn args(json: serde_json::Value) -> Value {
178        let s = serde_json::to_string(&json).expect("args serialize");
179        serde_json::from_str(&s).expect("args round-trip through tau_domain::Value")
180    }
