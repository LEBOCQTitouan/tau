1//! Run-time options for `Runtime::run` (added in Task 10) and the
2//! token-usage report carried in `RunOutcome` (added in Task 6).
3
4/// Token usage reported by the LLM backend, summed across the run.
5///
6/// Some backends report `total_tokens`; some report only input/output.
7/// `Default` returns all zeros (useful when no backend was called yet).
8///
9/// # Example
10///
11/// ```
12/// use tau_runtime::TokenUsage;
13///
14/// let usage = TokenUsage::default();
15/// assert_eq!(usage.input_tokens, 0);
16/// assert_eq!(usage.output_tokens, 0);
17/// assert_eq!(usage.total_tokens, None);
18/// ```
19#[non_exhaustive]
20#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
21pub struct TokenUsage {
22    /// Total input (prompt) tokens used across the run.
23    pub input_tokens: u64,
24    /// Total output (completion) tokens emitted across the run.
25    pub output_tokens: u64,
26    /// `Some(n)` when the backend reports a unified count; `None` otherwise.
27    pub total_tokens: Option<u64>,
28}

30/// Options for `Runtime::run`.
31///
32/// Constructed via `RunOptions::default()` (then mutated via public
33/// fields if needed). `#[non_exhaustive]` to allow additive options
34/// later — Phase-1+ may add `soft_fail_tool_errors`, `llm_retry_policy`,
35/// `overall_timeout`, etc.
36///
37/// # Example
38///
39/// ```ignore
40/// // `RunOptions` is `#[non_exhaustive]`; default + field mutation
41/// // is the only construction pattern.
42/// use tau_runtime::RunOptions;
43///
44/// let mut opts = RunOptions::default();
45/// opts.max_turns = 8;
46/// opts.trace_label = Some("session-abc".into());
47/// ```
48#[non_exhaustive]
49#[derive(Clone)]
50pub struct RunOptions {
51    /// Hard cap on agent loop iterations. Hitting this returns
52    /// `Ok(RunOutcome::Failed { kind: OutOfResources, .. })`.
53    /// Default: 16.
54    pub max_turns: u32,
55
56    /// Optional caller-supplied label included in tracing spans for
57    /// log correlation (e.g. session UUID from a TUI).
58    pub trace_label: Option<String>,
59
60    /// Project tau.toml capability override; default empty. Validated
61    /// at runtime via `compute_effective` (defense-in-depth — tau-cli
62    /// also validates at parse time). When non-empty, narrows the
63    /// agent's effective grant from its package manifest.
64    pub project_override: Vec<crate::capability_override::CapabilityOverride>,
65
66    /// Set by `Runtime::spawn_root_agent` when running inside a
67    /// multi-agent orchestrated run. When present, virtual tool calls
68    /// (`task.*`, `run.*`, `agent.<kind>.spawn`) are intercepted before
69    /// plugin dispatch and routed through `crate::orchestration`.
70    /// Callers using single-agent `Runtime::run` should leave this `None`.
71    pub orchestration_state:
72        Option<std::sync::Arc<tokio::sync::Mutex<crate::orchestration::run_state::RunState>>>,
73
74    /// Set by `Runtime::spawn_root_agent` (v1.1+). Carries the `Arc<Runtime>`
75    /// so the in-stream `agent.<kind>.spawn` intercept can recursively
76    /// invoke a child run via `run_with_history` without re-resolving
77    /// the kernel. None for single-agent runs.
78    pub orchestration_runtime: Option<std::sync::Arc<crate::builder::Runtime>>,
79
80    /// Set by the orchestration recursion path (v1.1+). When `Some`,
81    /// short-circuits the `compute_effective(manifest + project_override)`
82    /// calculation and uses this list directly as the agent's effective
83    /// grant. The capability subset law (`check_capability_subset`) is the
84    /// authoritative gate before this is set, so the kernel trusts it as a
85    /// pre-validated narrowing of the parent's grant.
86    pub granted_capabilities_override: Option<Vec<tau_domain::Capability>>,
87}
88
89impl std::fmt::Debug for RunOptions {
90    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
91        f.debug_struct("RunOptions")
92            .field("max_turns", &self.max_turns)
93            .field("trace_label", &self.trace_label)
94            .field("project_override", &self.project_override)
95            .field(
96                "orchestration_state",
97                &self.orchestration_state.as_ref().map(|_| "<RunState>"),
98            )
99            .field(
100                "orchestration_runtime",
101                &self.orchestration_runtime.as_ref().map(|_| "<Runtime>"),
102            )
103            .field(
104                "granted_capabilities_override",
105                &self.granted_capabilities_override,
106            )
107            .finish()
108    }
109}
110
111impl Default for RunOptions {
112    fn default() -> Self {
113        Self {
114            max_turns: 16,
115            trace_label: None,
116            project_override: Vec::new(),
117            orchestration_state: None,
118            orchestration_runtime: None,
119            granted_capabilities_override: None,
120        }
121    }
122}

124#[cfg(test)]
125mod tests {
126    use super::*;
127
128    #[test]
129    fn run_options_default_max_turns_is_16() {
130        let opts = RunOptions::default();
131        assert_eq!(opts.max_turns, 16);
132        assert_eq!(opts.trace_label, None);
133    }
134
135    #[test]
136    #[allow(clippy::field_reassign_with_default)]
137    fn run_options_can_override_max_turns_and_trace_label() {
138        // RunOptions is #[non_exhaustive]: from outside the crate,
139        // struct-literal construction is blocked.  This test intentionally
140        // exercises the default() + field-mutation pattern callers must use.
141        let mut opts = RunOptions::default();
142        opts.max_turns = 100;
143        opts.trace_label = Some("session-abc".into());
144        assert_eq!(opts.max_turns, 100);
145        assert_eq!(opts.trace_label.as_deref(), Some("session-abc"));
146    }
147
148    #[test]
149    fn run_options_default_project_override_is_empty() {
150        let opts = RunOptions::default();
151        assert!(opts.project_override.is_empty());
152    }
153
154    #[test]
155    fn token_usage_default_is_all_zero() {
156        let usage = TokenUsage::default();
157        assert_eq!(usage.input_tokens, 0);
158        assert_eq!(usage.output_tokens, 0);
159        assert_eq!(usage.total_tokens, None);
160    }
161
162    #[test]
163    fn token_usage_is_copy() {
164        let a = TokenUsage {
165            input_tokens: 1,
166            output_tokens: 2,
167            total_tokens: Some(3),
168        };
169        let b = a; // requires Copy
170        assert_eq!(a, b);
171    }
