1//! Port-aware test driver for sub-project D's Layer 4 plugin compat
2//! tests.
3//!
4//! Wraps the public `tau_runtime::plugin_host::load_{tool,llm_backend,storage}`
5//! functions. Tests use this to spawn a real plugin under the resolved
6//! sandbox adapter and invoke the high-level `DynTool` / `DynLlmBackend` /
7//! `DynStorage` traits directly — no manual `Frame::Request` construction.
8//!
9//! # Why this exists
10//!
11//! Sub-project B's `tau plugin run --script` driver hardcoded the
12//! handshake port to `LlmBackend`, breaking tool-port plugin tests
13//! (commit a449c10 marked them `#[ignore]`'d). This module is the
14//! port-aware replacement: the port is determined by which `spawn_*`
15//! function the test calls (`spawn_tool_under_sandbox` for tool plugins,
16//! etc.).
17//!
18//! Internally this calls into `tau_runtime::plugin_host::load_*`
19//! which themselves call `PluginProcess::spawn_and_handshake` (private).
20//! Tests don't need raw `Frame::Request` access.
21

30/// Errors raised by driver helpers.
31#[non_exhaustive]
32#[derive(Debug, thiserror::Error)]
33pub enum DriveError {
34    /// `tau_runtime::plugin_host::load_*` returned a `RuntimeError`.
35    #[error("plugin load failed: {0}")]
36    LoadFailed(String),
37    /// The plugin's port doesn't match what the caller expected.
38    #[error("port mismatch: caller expected {expected:?}, plugin provides {actual:?}")]
39    PortMismatch { expected: String, actual: String },
40    /// A tool invocation returned an error.
41    #[error("tool invocation failed: {0}")]
42    ToolFailed(String),
43    /// An LLM completion returned an error.
44    #[error("llm completion failed: {0}")]
45    LlmFailed(String),
46    /// A storage call returned an error.
47    #[error("storage call failed: {0}")]
48    StorageFailed(String),
49}

51/// Construct test trace context. Each test call gets a fresh context;
52/// use a synthetic run/agent/span ID stable enough for log correlation
53/// but unique enough that parallel test runs don't conflate.
54pub fn test_trace_context() -> tau_plugin_protocol::handshake::TraceContext {
55    use std::sync::atomic::{AtomicU64, Ordering};
56    static COUNTER: AtomicU64 = AtomicU64::new(0);
57    let nanos = std::time::SystemTime::now()
58        .duration_since(std::time::UNIX_EPOCH)
59        .unwrap_or_default()
60        .as_nanos();
61    // Atomic counter guarantees uniqueness across parallel test runs even
62    // when nanosecond clock resolution can't distinguish two consecutive
63    // calls (observed on macOS CI where two identical nanos timestamps
64    // surfaced from back-to-back invocations).
65    let seq = COUNTER.fetch_add(1, Ordering::Relaxed);
66    tau_plugin_protocol::handshake::TraceContext::new(
67        format!("test-driver-run-{nanos}-{seq}"),
68        format!("test-agent-{nanos}-{seq}"),
69        format!("test-span-{nanos}-{seq}"),
70    )
71}

73/// Construct `PluginHostOptions` for a test, with the supplied sandbox
74/// adapter and standard test timeouts.
75///
76/// `PluginHostOptions` is `#[non_exhaustive]`, so struct-literal
77/// initialisation is not allowed from outside the defining crate. We
78/// call `default()` and mutate the one field we care about.
79pub fn test_plugin_host_options(adapter: Option<Arc<SandboxAdapter>>) -> PluginHostOptions {
80    let mut opts = PluginHostOptions::default();
81    opts.sandbox_adapter = adapter;
82    opts
83}

85/// Spawn a tool plugin under the given sandbox adapter and return the
86/// `DynTool` handle for direct method invocation.
87pub async fn spawn_tool_under_sandbox(
88    plugin: &LockedPlugin,
89    config: serde_json::Value,
90    adapter: Option<Arc<SandboxAdapter>>,
91    sandbox_plan: Option<&SandboxPlan>,
92) -> Result<Arc<dyn DynTool>, DriveError> {
93    let trace = test_trace_context();
94    let options = test_plugin_host_options(adapter);
95    plugin_host::load_tool(plugin, config, trace, options, sandbox_plan)
96        .await
97        .map_err(|e| DriveError::LoadFailed(format!("{e:?}")))
98}

130#[cfg(test)]
131mod tests {
132    use super::*;
133
134    #[test]
135    fn drive_error_display_includes_detail() {
136        let err = DriveError::LoadFailed("plugin handshake timed out".to_string());
137        let msg = format!("{err}");
138        assert!(msg.contains("plugin handshake timed out"));
139    }
140
141    #[test]
142    fn drive_error_is_non_exhaustive_via_match() {
143        let err = DriveError::LoadFailed("e".to_string());
144        match err {
145            DriveError::LoadFailed(_)
146            | DriveError::PortMismatch { .. }
147            | DriveError::ToolFailed(_)
148            | DriveError::LlmFailed(_)
149            | DriveError::StorageFailed(_) => {}
150        }
151    }
152
153    #[test]
154    fn test_trace_context_unique_across_calls() {
155        let a = test_trace_context();
156        let b = test_trace_context();
157        // Fields are pub per tau-plugin-protocol handshake design.
158        assert!(
159            a.run_id != b.run_id,
160            "expected unique run_ids; got a={:?}, b={:?}",
161            a.run_id,
162            b.run_id
163        );
164    }
165
166    #[test]
167    fn test_plugin_host_options_carries_no_adapter_by_default() {
168        let opts = test_plugin_host_options(None);
169        assert!(opts.sandbox_adapter.is_none());
170    }
171
172    #[test]
173    fn port_mismatch_error_displays_clearly() {
174        let err = DriveError::PortMismatch {
175            expected: "Tool".to_string(),
176            actual: "LlmBackend".to_string(),
177        };
178        let msg = format!("{err}");
179        assert!(msg.contains("Tool"));
180        assert!(msg.contains("LlmBackend"));
181    }
