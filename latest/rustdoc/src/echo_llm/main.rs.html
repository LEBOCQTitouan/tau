1//! Toy `LlmBackend` plugin that replays canned responses from config.
2//!
3//! Used by tau-cli integration tests to exercise the plugin loading
4//! mechanism end-to-end without depending on a real LLM provider.
5//!
6//! # Configuration
7//!
8//! Configurable via the handshake `config` field (set in
9//! `[agents.<id>.config]` of the project tau.toml):
10//!
11//! - `canned_text: String` — single canned text returned by every
12//!   `llm.complete` call. Default: empty string.
13//! - `script: Vec<String>` — multi-turn script. Indexed by an internal
14//!   atomic counter that increments on each `complete` call. If the
15//!   counter exceeds the script length, falls back to `canned_text`.
16//! - `crash_after_handshake: bool` — if `true`, panic at the start of
17//!   any `complete`/`stream` call. The handshake itself completes; the
18//!   panic surfaces at first dispatch. Used by failure-path tests.
19//! - `delay_response_ms: Option<u64>` — sleep this many milliseconds
20//!   before responding. Used by handshake/timeout tests.
21//! - `error_on_method: Option<String>` — return `Err(LlmError::Internal)`
22//!   when this method is called (e.g. `"llm.complete"` or
23//!   `"llm.stream"`).
24

28use std::sync::atomic::{AtomicUsize, Ordering};
29use std::time::Duration;

31use serde::Deserialize;
32use tau_plugin_sdk::{run_llm_backend_with_config, ConfigError, Configure, SdkError};
33use tau_ports::{

35    CompletionStream, LlmBackend, LlmError, StopReason,
36};

38/// Static configuration consumed from the handshake `config` field.
39#[derive(Debug, Default, Deserialize)]
40struct EchoConfig {
41    /// Single canned text returned by every `llm.complete` call.
42    #[serde(default)]
43    canned_text: String,
44    /// Multi-turn script; indexed by an atomic turn counter.
45    #[serde(default)]
46    script: Vec<String>,
47    /// If `true`, panic at the start of any `complete`/`stream` call.
48    #[serde(default)]
49    crash_after_handshake: bool,
50    /// Sleep this many milliseconds before responding. Used by
51    /// handshake/timeout tests.
52    #[serde(default)]
53    delay_response_ms: Option<u64>,
54    /// Return `Err(LlmError::Internal)` when this method is called.
55    #[serde(default)]
56    error_on_method: Option<String>,
57}
58
59/// Toy `LlmBackend` plugin.
60struct EchoLlm {
61    config: EchoConfig,
62    turn: AtomicUsize,
63}
64
65impl Configure for EchoLlm {
66    type Config = EchoConfig;
67
68    fn from_config(config: Self::Config) -> Result<Self, ConfigError> {
69        Ok(EchoLlm {
70            config,
71            turn: AtomicUsize::new(0),
72        })
73    }
74}
75
76impl EchoLlm {
77    /// Apply the test-only side effects (`crash_after_handshake`,
78    /// `error_on_method`, `delay_response_ms`) and produce the next
79    /// canned text. Shared by `complete` and `stream`.
80    async fn next_text(&self, method: &str) -> Result<String, LlmError> {
81        if self.config.crash_after_handshake {
82            panic!("echo-llm crash_after_handshake = true (test-only mode)");
83        }
84        if self.config.error_on_method.as_deref() == Some(method) {
85            return Err(LlmError::Internal {
86                message: format!("echo-llm error_on_method test mode tripped on {method}"),
87            });
88        }
89        if let Some(ms) = self.config.delay_response_ms {
90            tokio::time::sleep(Duration::from_millis(ms)).await;
91        }
92        let i = self.turn.fetch_add(1, Ordering::Relaxed);
93        let text = self
94            .config
95            .script
96            .get(i)
97            .cloned()
98            .unwrap_or_else(|| self.config.canned_text.clone());
99        Ok(text)
100    }
101}
102
103impl LlmBackend for EchoLlm {
104    fn name(&self) -> &str {
105        "echo-llm"
106    }
107
108    async fn complete(&self, _req: CompletionRequest) -> Result<CompletionResponse, LlmError> {
109        let text = self.next_text("llm.complete").await?;
110        Ok(make_completion_response(
111            text,
112            Vec::new(),
113            StopReason::EndTurn,
114            None,
115        ))
116    }
117
118    async fn stream(&self, _req: CompletionRequest) -> Result<CompletionStream, LlmError> {
119        let text = self.next_text("llm.stream").await?;
120        let resp = make_completion_response(text, Vec::new(), StopReason::EndTurn, None);
121        Ok(batch_to_stream(resp))
122    }
123}

125#[tokio::main]
126async fn main() -> Result<(), SdkError> {
127    run_llm_backend_with_config::<EchoLlm>(env!("CARGO_PKG_NAME"), env!("CARGO_PKG_VERSION")).await
