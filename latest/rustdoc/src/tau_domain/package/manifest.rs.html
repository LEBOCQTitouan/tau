13/// A dependency declaration: a package name plus a SemVer requirement.
14///
15/// # Example
16///
17/// ```ignore
18/// // Construction is performed inside `tau-domain` (e.g. by manifest
19/// // validation in tau-pkg). External crates receive `PackageDep` values;
20/// // they cannot be built via struct expression because the type is
21/// // `#[non_exhaustive]`.
22/// use tau_domain::{PackageDep, PackageName, VersionReq};
23/// use std::str::FromStr;
24///
25/// let dep = PackageDep {
26///     name: PackageName::from_str("fs-tools").unwrap(),
27///     version_req: VersionReq::parse("^0.3").unwrap(),
28/// };
29/// assert_eq!(dep.name.as_str(), "fs-tools");
30/// ```
31#[non_exhaustive]
32#[derive(Debug, Clone, PartialEq)]
33#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
34pub struct PackageDep {
35    /// Dependency package name.
36    pub name: PackageName,
37    /// SemVer version requirement.
38    pub version_req: VersionReq,
39}

41/// A canonical package identity: `(name, version)`.
42///
43/// # Example
44///
45/// ```ignore
46/// // Like `PackageDep`, `PackageId` is `#[non_exhaustive]` and cannot be
47/// // built via struct expression from outside `tau-domain`. External
48/// // crates receive instances from manifest validation in tau-pkg.
49/// use tau_domain::{PackageId, PackageName, Version};
50/// use std::str::FromStr;
51///
52/// let id = PackageId {
53///     name: PackageName::from_str("fs-tools").unwrap(),
54///     version: Version::parse("0.3.0").unwrap(),
55/// };
56/// assert_eq!(id.name.as_str(), "fs-tools");
57/// ```
58#[non_exhaustive]
59#[derive(Debug, Clone, PartialEq, Eq, Hash)]
60#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
61pub struct PackageId {
62    /// Package name.
63    pub name: PackageName,
64    /// Package version.
65    pub version: Version,
66}
67
68impl PackageId {
69    /// Construct a [`PackageId`] from a validated name and version.
70    ///
71    /// `PackageId` is `#[non_exhaustive]`: external crates cannot use
72    /// struct-literal construction. Callers (notably tau-runtime
73    /// integration tests, which assemble `AgentDefinition`s by hand)
74    /// use this constructor.
75    ///
76    /// # Example
77    ///
78    /// ```
79    /// use tau_domain::{PackageId, PackageName, Version};
80    /// use std::str::FromStr;
81    ///
82    /// let id = PackageId::new(
83    ///     PackageName::from_str("fs-tools").unwrap(),
84    ///     Version::parse("0.3.0").unwrap(),
85    /// );
86    /// assert_eq!(id.name.as_str(), "fs-tools");
87    /// ```
88    pub fn new(name: PackageName, version: Version) -> Self {
89        Self { name, version }
90    }
91}

93/// Package kind. Structural at v0.1: every kind goes through `Custom`.
94/// Typed variants land additively as tau-runtime gains plugin trait
95/// awareness for each kind.
96///
97/// See: [escape-hatches.md#packagekind-custom](../../../../../docs/explanation/escape-hatches.md#packagekind-custom).
98///
99/// # Example
100///
101/// ```ignore
102/// // `PackageKind` is `#[non_exhaustive]` and cannot be built via struct
103/// // expression from outside `tau-domain`. Construction is performed
104/// // inside the crate (e.g. by manifest validation in tau-pkg).
105/// use tau_domain::PackageKind;
106/// let k = PackageKind::Custom { kind: "tool".into() };
107/// ```
108#[non_exhaustive]
109#[derive(Debug, Clone, PartialEq, Eq)]
110pub enum PackageKind {
111    /// A package kind not yet typed in core.
112    /// See: [escape-hatches.md#packagekind-custom](../../../../../docs/explanation/escape-hatches.md#packagekind-custom).
113    Custom {
114        /// The kind name. By convention one of [`crate::package::kinds`]'s
115        /// constants (e.g. `"llm-backend"`, `"tool"`).
116        kind: String,
117    },
118}
119
120#[cfg(feature = "serde")]
121impl serde::Serialize for PackageKind {
122    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
123    where
124        S: serde::Serializer,
125    {
126        match self {
127            PackageKind::Custom { kind } => serializer.serialize_str(kind),
128        }
129    }
130}
131
132#[cfg(feature = "serde")]
133impl<'de> serde::Deserialize<'de> for PackageKind {
134    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
135    where
136        D: serde::Deserializer<'de>,
137    {
138        struct Visitor;
139        impl<'de> serde::de::Visitor<'de> for Visitor {
140            type Value = PackageKind;
141            fn expecting(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
142                f.write_str("a package kind string (e.g. \"tool\", \"llm-backend\")")
143            }
144            fn visit_str<E>(self, v: &str) -> Result<PackageKind, E>
145            where
146                E: serde::de::Error,
147            {
148                if v.is_empty() {
149                    return Err(E::custom("package kind cannot be empty"));
150                }
151                Ok(PackageKind::Custom { kind: v.to_owned() })
152            }
153            fn visit_string<E>(self, v: String) -> Result<PackageKind, E>
154            where
155                E: serde::de::Error,
156            {
157                if v.is_empty() {
158                    return Err(E::custom("package kind cannot be empty"));
159                }
160                Ok(PackageKind::Custom { kind: v })
161            }
162        }
163        deserializer.deserialize_str(Visitor)
164    }
165}

188/// Raw manifest as it appears on disk or on the wire. Deserializes from
189/// TOML/JSON directly. May contain field combinations that violate
190/// cross-field invariants — call [`UncheckedManifest::validate`] to
191/// obtain a verified [`PackageManifest`].
192///
193/// # Example
194///
195/// ```no_run
196/// use tau_domain::UncheckedManifest;
197/// // toml::from_str::<UncheckedManifest>(&raw)?.validate()?;
198/// # let _ = std::any::type_name::<UncheckedManifest>();
199/// ```
200#[non_exhaustive]
201#[derive(Debug, Clone, PartialEq)]
202#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
203pub struct UncheckedManifest {
204    /// Package name.
205    pub name: PackageName,
206    /// Package version.
207    pub version: Version,
208    /// Free-form description.
209    pub description: String,
210    /// Authors (free-form, e.g. `"Acme Inc <support@acme.dev>"`).
211    pub authors: Vec<String>,
212    /// SPDX license expression as opaque text. `None` for unlicensed.
213    pub license: Option<String>,
214    /// Where the package lives.
215    pub source: PackageSource,
216    /// What the package provides.
217    pub kind: PackageKind,
218    /// Required dependencies.
219    pub dependencies: Vec<PackageDep>,
220    /// Capability declarations (G14).
221    pub capabilities: Vec<Capability>,
222    /// Plugin manifest declared via the `[plugin]` table.
223    ///
224    /// `None` for data-only packages (no plugin table). `Some` for
225    /// plugin packages — `tau-pkg` uses this to gate the build step
226    /// during install (see plugin-loading spec §6.1, §6.3).
227    #[cfg_attr(feature = "serde", serde(default))]
228    pub plugin: Option<PluginManifest>,
229    /// Plugin-side sandbox requirements declared via `[sandbox]` table.
230    ///
231    /// Optional. Default = `PluginSandboxRequirements::default()` (no
232    /// tier floor; auto-derived shapes). See
233    /// [`PluginSandboxRequirements`](crate::package::sandbox::PluginSandboxRequirements).
234    #[cfg_attr(feature = "serde", serde(default))]
235    pub sandbox: crate::package::sandbox::PluginSandboxRequirements,
236    /// Skill manifest declared via the `[skill]` table.
237    ///
238    /// `None` for non-skill packages (no skill table). `Some` for skill
239    /// packages — `tau-pkg::skill_check` (Skills-2) uses this to gate
240    /// SKILL.md validation during install. See ROADMAP §16 and
241    /// `docs/decisions/0025-skills-foundation.md`.
242    #[cfg_attr(feature = "serde", serde(default))]
243    pub skill: Option<crate::package::skill::SkillManifest>,
244}

246/// Validated package manifest. By construction, satisfies all cross-field
247/// invariants enforced by [`UncheckedManifest::validate`]. Cannot be
248/// constructed directly — must go through validation.
249///
250/// To mutate a `PackageManifest`, downgrade via `Into<UncheckedManifest>`,
251/// edit, then call [`UncheckedManifest::validate`] again.
252///
253/// # Example
254///
255/// ```no_run
256/// use tau_domain::{UncheckedManifest, PackageManifest};
257/// // let manifest: PackageManifest = unchecked.validate()?;
258/// # let _ = std::any::type_name::<PackageManifest>();
259/// ```
260#[non_exhaustive]
261#[derive(Debug, Clone, PartialEq)]
262pub struct PackageManifest(UncheckedManifest);
263
264impl PackageManifest {
265    /// Package name.
266    pub fn name(&self) -> &PackageName {
267        &self.0.name
268    }
269    /// Package version.
270    pub fn version(&self) -> &Version {
271        &self.0.version
272    }
273    /// Free-form description.
274    pub fn description(&self) -> &str {
275        &self.0.description
276    }
277    /// Authors.
278    pub fn authors(&self) -> &[String] {
279        &self.0.authors
280    }
281    /// SPDX license expression, if present.
282    pub fn license(&self) -> Option<&str> {
283        self.0.license.as_deref()
284    }
285    /// Source location.
286    pub fn source(&self) -> &PackageSource {
287        &self.0.source
288    }
289    /// Package kind.
290    pub fn kind(&self) -> &PackageKind {
291        &self.0.kind
292    }
293    /// Required dependencies.
294    pub fn dependencies(&self) -> &[PackageDep] {
295        &self.0.dependencies
296    }
297    /// Capability declarations.
298    pub fn capabilities(&self) -> &[Capability] {
299        &self.0.capabilities
300    }
301    /// Plugin manifest from the `[plugin]` table, if any.
302    ///
303    /// `None` for data-only packages; `Some` for plugin packages.
304    /// Surfaced verbatim from the `[plugin]` TOML table; structurally
305    /// validated by `PluginManifest`'s typed fields.
306    pub fn plugin(&self) -> Option<&PluginManifest> {
307        self.0.plugin.as_ref()
308    }
309
310    /// Plugin-side sandbox requirements (from `[sandbox]` table).
311    pub fn sandbox(&self) -> &crate::package::sandbox::PluginSandboxRequirements {
312        &self.0.sandbox
313    }
314
315    /// Skill manifest from the `[skill]` table, if any.
316    ///
317    /// `None` for non-skill packages; `Some` for `kind = "skill"`
318    /// packages. Surfaced verbatim from the `[skill]` TOML table.
319    pub fn skill(&self) -> Option<&crate::package::skill::SkillManifest> {
320        self.0.skill.as_ref()
321    }
322
323    /// Wrap a checked `UncheckedManifest` without re-running validation.
324    /// Internal use only — public API must go through
325    /// [`UncheckedManifest::validate`].
326    pub(crate) fn from_checked(u: UncheckedManifest) -> Self {
327        Self(u)
328    }
329}

492use crate::error::PackageManifestError;
493use crate::package::capability::Capability as Cap;
494
495impl UncheckedManifest {
496    /// Run cross-field validation. Returns the validated manifest on
497    /// success.
498    ///
499    /// Field types are already validated at construction (`PackageName`,
500    /// `PackageSource`, etc.); this checks invariants those types
501    /// can't enforce alone (non-empty description, non-empty Custom
502    /// capability names, etc.).
503    ///
504    /// # Example
505    ///
506    /// ```ignore
507    /// // `UncheckedManifest` is `#[non_exhaustive]`, so it cannot be
508    /// // built via struct expression from outside `tau-domain`. In
509    /// // practice, callers obtain one by deserializing a manifest file
510    /// // (in tau-pkg) and then call `.validate()`.
511    /// use tau_domain::{UncheckedManifest, PackageManifestError};
512    /// // let err = unchecked.validate().unwrap_err();
513    /// // assert_eq!(err, PackageManifestError::EmptyDescription);
514    /// # let _ = std::any::type_name::<UncheckedManifest>();
515    /// # let _ = std::any::type_name::<PackageManifestError>();
516    /// ```
517    pub fn validate(self) -> Result<PackageManifest, PackageManifestError> {
518        if self.description.is_empty() {
519            return Err(PackageManifestError::EmptyDescription);
520        }
521        // dependency names are already PackageName values (pre-validated),
522        // but the loop is here as a hook for future per-dep invariants
523        // (e.g. duplicate-name detection, version-range cross-checks).
524        // The `index` is kept so it can be threaded into
525        // `PackageManifestError::DependencyName { index, source }`.
526        #[allow(clippy::unused_enumerate_index)]
527        for (_index, _dep) in self.dependencies.iter().enumerate() {
528            // no-op at v0.1
529        }
530        for (i, cap) in self.capabilities.iter().enumerate() {
531            if let Cap::Custom { name, .. } = cap {
532                if name.is_empty() {
533                    return Err(PackageManifestError::CapabilityEmptyName { index: i });
534                }
535            }
536        }
537        // Skills-2: kind = "skill" rejects [plugin] block.
538        if matches!(&self.kind, PackageKind::Custom { kind } if kind == kinds::SKILL)
539            && self.plugin.is_some()
540        {
541            return Err(PackageManifestError::SkillCannotHavePluginBlock);
542        }
543        Ok(PackageManifest::from_checked(self))
544    }
545}
