1//! Linux kernel feature probe.
2//!
3//! Detects which sandbox tier the running kernel can support:
4//! - **Strict**: landlock V1 + seccomp BPF + unprivileged user namespaces.
5//! - **Light**: landlock V1 only.
6//! - **Unavailable**: landlock V1 missing (kernel < 5.13).
7//!
8//! The probe does NOT call `restrict_self()` and does NOT install any filter,
9//! so it has no side-effects on the calling process.
10//!
11//! # Simplification note (v0.1)
12//! - seccomp BPF is assumed available on any Linux kernel ≥ 3.5 (all current
13//!   production kernels qualify). No explicit `prctl(PR_GET_SECCOMP)` probe is
14//!   performed.
15//! - Unprivileged user namespaces are assumed available unless the file
16//!   `/proc/sys/kernel/unprivileged_userns_clone` exists and contains `0`.
17//!   On kernels that do not have this sysctl (i.e. vanilla upstream), unprivileged
18//!   user namespaces are enabled by default. A future task can add a fork-based
19//!   probe for more precise detection.
20
21use tau_ports::{SandboxProbe, SandboxTier};
22
23/// Probe the host kernel for sandbox features.
24///
25/// Returns `Available { tier }` where `tier` is the strongest tier the
26/// adapter can support, capped at the caller's requested tier.
27pub(crate) async fn probe(requested: SandboxTier) -> SandboxProbe {
28    decide_probe(requested, landlock_v1_supported(), user_ns_supported())
29}
30
31/// Pure tier-decision function: given a requested tier and the kernel feature
32/// flags, return the best available [`SandboxProbe`].
33///
34/// Extracted from [`probe`] so the decision matrix can be unit-tested without
35/// invoking real kernel syscalls. The two boolean inputs mirror the runtime
36/// probes: `landlock_ok` corresponds to [`landlock_v1_supported`], and
37/// `user_ns_ok` corresponds to [`user_ns_supported`].
38fn decide_probe(requested: SandboxTier, landlock_ok: bool, user_ns_ok: bool) -> SandboxProbe {
39    if !landlock_ok {
40        return SandboxProbe::Unavailable {
41            reason: "landlock V1 unsupported (kernel < 5.13)".into(),
42        };
43    }
44    let effective = match requested {
45        SandboxTier::None => SandboxTier::None,
46        // Light needs landlock only — already verified above.
47        SandboxTier::Light => SandboxTier::Light,
48        // Strict needs landlock + seccomp + user namespaces.
49        // seccomp is assumed available (Linux 3.5+, see module-level note).
50        // User namespaces are checked via unprivileged_userns_clone sysctl.
51        SandboxTier::Strict => {
52            if user_ns_ok {
53                SandboxTier::Strict
54            } else {
55                tracing::info!("unprivileged user namespaces disabled; capping Strict -> Light");
56                SandboxTier::Light
57            }
58        }
59        // Non-exhaustive arm: unknown future tier — warn and report unavailable.
60        other => {
61            tracing::warn!(
62                ?other,
63                "unknown SandboxTier in probe — returning Unavailable"
64            );
65            return SandboxProbe::Unavailable {
66                reason: format!("tier {other:?} not implemented"),
67            };
68        }
69    };
70    SandboxProbe::Available {
71        tier: effective,
72        details: format!("landlock V1 ok, effective tier: {effective:?}"),
73    }
74}

76/// Check whether this kernel supports Landlock V1 by attempting to create a
77/// V1 ruleset with `HardRequirement` compat level. Creating a ruleset (without
78/// calling `restrict_self()`) is a side-effect-free probe: the kernel FD is
79/// opened and immediately dropped, applying no restrictions to the process.
80fn landlock_v1_supported() -> bool {
81    use landlock::{Access, AccessFs, CompatLevel, Compatible, Ruleset, RulesetAttr, ABI};
82
83    Ruleset::default()
84        .set_compatibility(CompatLevel::HardRequirement)
85        .handle_access(AccessFs::from_all(ABI::V1))
86        .and_then(|r| r.create())
87        .is_ok()
88}
89
90/// Check whether unprivileged user namespaces are available.
91///
92/// # Simplification (v0.1)
93/// Reads `/proc/sys/kernel/unprivileged_userns_clone`. If the file exists and
94/// contains `0`, user namespaces are disabled. On kernels that do not expose
95/// this sysctl (mainline upstream), the file is absent and we conservatively
96/// assume user namespaces are enabled. A full probe would `fork()` a child and
97/// attempt `unshare(CLONE_NEWUSER)`, but that adds latency and complexity.
98fn user_ns_supported() -> bool {
99    match std::fs::read_to_string("/proc/sys/kernel/unprivileged_userns_clone") {
100        Ok(contents) => contents.trim() != "0",
101        // File absent: sysctl not present → assume enabled (mainline default).
102        Err(_) => true,
103    }
104}

110    // ---------- decide_probe: landlock unavailable ----------

112    #[test]
113    fn unavailable_when_landlock_missing_regardless_of_request() {
114        for tier in [SandboxTier::None, SandboxTier::Light, SandboxTier::Strict] {
115            let p = decide_probe(
116                tier, /* landlock_ok */ false, /* user_ns_ok */ true,
117            );
118            assert!(
119                matches!(p, SandboxProbe::Unavailable { ref reason } if reason.contains("landlock V1")),
120                "tier {tier:?} should report Unavailable when landlock is missing — got {p:?}"
121            );
122        }
123    }
124
125    #[test]
126    fn unavailable_when_landlock_missing_even_if_user_ns_present() {
127        let p = decide_probe(SandboxTier::Strict, false, true);
128        assert!(matches!(p, SandboxProbe::Unavailable { .. }));
129    }

138            other => panic!("expected Available(None), got {other:?}"),

152    fn light_request_does_not_depend_on_user_ns() {

168                ..
169            }
170        ));
171    }

213    #[test]
214    fn effective_tier_never_exceeds_requested() {
215        // Property: for any (requested, landlock_ok, user_ns_ok), the
216        // effective tier (when Available) is ≤ requested.
217        for landlock_ok in [true, false] {
218            for user_ns_ok in [true, false] {
219                for requested in [SandboxTier::None, SandboxTier::Light, SandboxTier::Strict] {
220                    let p = decide_probe(requested, landlock_ok, user_ns_ok);
221                    if let SandboxProbe::Available { tier, .. } = p {
222                        assert!(
223                            tier <= requested,
224                            "effective tier {tier:?} exceeds requested {requested:?} \
225                             (landlock_ok={landlock_ok}, user_ns_ok={user_ns_ok})"
226                        );
227                    }
228                }
229            }
230        }
231    }
