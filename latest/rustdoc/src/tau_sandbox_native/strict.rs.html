1//! Strict-tier enforcement: landlock + user/network namespace unshare + seccomp BPF.
2//!
3//! The pre_exec hook runs three operations in this exact order:
4//! 1. **Landlock** (`install_landlock`) — uses `landlock_*` syscalls that seccomp
5//!    would block if installed first.
6//! 2. **`unshare(flags)`** — drops the child into a fresh user namespace (gaining
7//!    all capabilities within it) and a new network namespace. `CLONE_NEWNET` is
8//!    always included (sub-project F; see `net::unshare_flags_for_plan`).
9//!    Must run before seccomp blocks `unshare(2)`.
10//! 3. **seccomp BPF filter** (`apply_filter`) — installed last; once active it
11//!    blocks `unshare`, `landlock_*`, and any other syscall not in the allow-list.
12//!    The allow-list is the baseline extended by capability-conditional rules:
13//!    `exec::extend_with_exec_rules` (v0.1 no-op) and
14//!    `net::extend_with_network_rules` (adds socket-family syscalls for `Network(Http)`).
15//!
16//! The BPF program is **compiled in the parent** (cheap, one-time) and the
17//! compiled byte-slice is moved into the pre_exec closure by value. The child
18//! only calls `prctl(PR_SET_NO_NEW_PRIVS)` + `seccomp(SET_MODE_FILTER)`.
19//!
20//! # Known limitation (async-signal-safety)
21//! This closure inherits the async-signal-safety hazard documented in `light.rs`:
22//! it runs between fork and exec in a multi-threaded (tokio) process.
23//! `nix::sched::unshare` is a thin syscall wrapper (signal-safe).
24//! `seccompiler::apply_filter` calls `prctl` then `seccomp` (both signal-safe).
25//! The main remaining risk is from `install_landlock` (see light.rs for details).
26//! A future task should consider a fork-server pattern to eliminate the window.
27

38/// Build the baseline syscall allow-list as a rules map.
39///
40/// Exposed for unit tests that need to introspect the real baseline rather than
41/// constructing their own ad-hoc copies. Each entry maps a syscall number to an
42/// empty rules vec (meaning "unconditionally allow" — no argument matching).
43///
44/// # Syscall set rationale
45/// The allow-list covers the baseline needs of a plugin communicating over stdio/socketpair
46/// without network access. `SYS_socket`, `SYS_connect`, `SYS_bind`, and `SYS_listen` are
47/// intentionally **excluded** — Task 5 will add them conditionally per `NetworkHttp` capability.
48///
49/// # Architecture note
50/// Some syscall numbers differ between x86_64 and aarch64. This function uses
51/// `#[cfg(target_arch)]` guards for arch-specific constants. The seccompiler crate handles
52/// endianness; only the syscall numbers need to be arch-correct.
53pub(crate) fn baseline_syscall_map() -> std::collections::BTreeMap<i64, Vec<SeccompRule>> {
54    let mut rules: std::collections::BTreeMap<i64, Vec<SeccompRule>> =
55        std::collections::BTreeMap::new();
56
57    macro_rules! allow {
58        ($($nr:expr),+ $(,)?) => {
59            $(rules.entry($nr).or_default();)+
60        };
61    }
62
63    // ---- File I/O ----
64    allow!(
65        libc::SYS_read,
66        libc::SYS_write,
67        libc::SYS_readv,
68        libc::SYS_writev,
69        libc::SYS_preadv,
70        libc::SYS_pwritev,
71        libc::SYS_pread64,
72        libc::SYS_pwrite64,
73        libc::SYS_openat,
74        libc::SYS_close,
75        libc::SYS_fstat,
76        libc::SYS_lseek,
77        libc::SYS_readlinkat,
78        libc::SYS_getdents64,
79        libc::SYS_fcntl,
80        libc::SYS_dup,
81        libc::SYS_dup3,
82        libc::SYS_pipe2,
83        libc::SYS_mkdirat,
84        libc::SYS_unlinkat,
85        libc::SYS_linkat,
86        libc::SYS_renameat,
87        libc::SYS_renameat2,
88        libc::SYS_symlinkat,
89        libc::SYS_chdir,
90        libc::SYS_fchdir,
91        libc::SYS_getcwd,
92        libc::SYS_umask,
93        libc::SYS_faccessat,
94        libc::SYS_truncate,
95        libc::SYS_ftruncate,
96    );
97
98    // Arch-specific file I/O constants that only exist on x86_64.
99    // dup2 is legacy (aarch64 uses dup3 exclusively).
100    #[cfg(target_arch = "x86_64")]
101    allow!(
102        libc::SYS_stat,
103        libc::SYS_lstat,
104        libc::SYS_access,
105        libc::SYS_pipe,
106        libc::SYS_open,
107        libc::SYS_creat,
108        libc::SYS_dup2,
109    );
110
111    // openat2, newfstatat, statx, faccessat2 via raw syscall numbers.
112    // Using raw numbers avoids libc version skew; values are stable in the kernel ABI.
113    // x86_64: openat2=437, newfstatat=262, statx=332, faccessat2=439
114    // aarch64: openat2=437, newfstatat=79,  statx=291, faccessat2=439
115    #[cfg(target_arch = "x86_64")]
116    allow!(
117        262_i64, // newfstatat / fstatat64
118        332_i64, // statx
119        437_i64, // openat2
120        439_i64, // faccessat2
121    );
122
123    #[cfg(target_arch = "aarch64")]
124    allow!(
125        79_i64,  // newfstatat
126        291_i64, // statx
127        437_i64, // openat2
128        439_i64, // faccessat2
129    );
130
131    // ---- Memory ----
132    allow!(
133        libc::SYS_mmap,
134        libc::SYS_munmap,
135        libc::SYS_mprotect,
136        libc::SYS_mremap,
137        libc::SYS_madvise,
138        libc::SYS_brk,
139    );
140
141    // mmap2 is 32-bit-only; not present on x86_64 or aarch64.
142
143    // ---- Process / thread ----
144    allow!(
145        libc::SYS_clone,
146        libc::SYS_wait4,
147        libc::SYS_waitid,
148        libc::SYS_getpid,
149        libc::SYS_gettid,
150        libc::SYS_getppid,
151        libc::SYS_set_tid_address,
152        libc::SYS_set_robust_list,
153        libc::SYS_prlimit64,
154        libc::SYS_getrusage,
155        libc::SYS_getuid,
156        libc::SYS_geteuid,
157        libc::SYS_getgid,
158        libc::SYS_getegid,
159        libc::SYS_setresuid,
160        libc::SYS_setresgid,
161        libc::SYS_setpgid,
162        libc::SYS_getpgid,
163        libc::SYS_getsid,
164        libc::SYS_setsid,
165        libc::SYS_getgroups,
166        libc::SYS_prctl,
167        libc::SYS_sched_yield,
168        // T1 (2026-05-09): tokio's Builder::new_multi_thread sizes the
169        // worker pool by calling sched_getaffinity. Without this, the
170        // KillProcess mismatch action SIGSYSes the child before handshake.
171        libc::SYS_sched_getaffinity,
172        // Defensive: tokio doesn't currently call sched_setaffinity, but
173        // some runtime configurations do. Allowed alongside getaffinity.
174        libc::SYS_sched_setaffinity,
175        libc::SYS_nanosleep,
176        libc::SYS_clock_nanosleep,
177        libc::SYS_clock_gettime,
178        libc::SYS_clock_getres,
179        libc::SYS_gettimeofday,
180    );

182    // arch_prctl is x86_64-only.
183    #[cfg(target_arch = "x86_64")]
184    allow!(libc::SYS_arch_prctl);
185

187    allow!(435_i64);

193    allow!(434_i64);

206    // ppoll, epoll_create1, epoll_ctl, epoll_pwait — exist on both x86_64 and aarch64.
207    // poll, epoll_wait — x86_64 only (aarch64 uses ppoll and epoll_pwait exclusively).
208    allow!(
209        libc::SYS_ppoll,
210        libc::SYS_epoll_create1,
211        libc::SYS_epoll_ctl,
212        libc::SYS_epoll_pwait,
213        libc::SYS_eventfd2,
214        libc::SYS_timerfd_create,

216        libc::SYS_futex,
217    );
218
219    // poll, epoll_create (legacy), epoll_wait, select, pselect6, eventfd — x86_64 only.
220    // aarch64 uses only the newer variants (ppoll, epoll_create1, epoll_pwait).
221    #[cfg(target_arch = "x86_64")]
222    allow!(
223        libc::SYS_poll,
224        libc::SYS_epoll_create,
225        libc::SYS_epoll_wait,
226        libc::SYS_select,
227        libc::SYS_pselect6,
228        libc::SYS_eventfd,
229    );
230
231    // ---- Misc ----
232    allow!(
233        libc::SYS_exit,
234        libc::SYS_exit_group,
235        libc::SYS_restart_syscall,
236        libc::SYS_sysinfo,
237        libc::SYS_uname,
238        libc::SYS_getrandom,
239        libc::SYS_membarrier,
240        libc::SYS_rseq,
241    );
242
243    // ---- IPC for plugin protocol (socketpair + send/recv; NOT socket/connect/bind/listen) ----
244    allow!(
245        libc::SYS_socketpair,
246        libc::SYS_sendmsg,
247        libc::SYS_recvmsg,
248        libc::SYS_shutdown,
249        libc::SYS_setsockopt,
250        libc::SYS_getsockopt,
251        libc::SYS_sendto,
252        libc::SYS_recvfrom,
253    );
254
255    // execve / execveat — allowed at baseline; Task 5 tightens via argument filtering.
256    allow!(libc::SYS_execve);
257
258    // execveat: x86_64=322, aarch64=281
259    #[cfg(target_arch = "x86_64")]
260    allow!(322_i64);
261    #[cfg(target_arch = "aarch64")]
262    allow!(281_i64);
263
264    rules
265}

267/// Build the baseline plugin allow-list seccomp BPF program.
268///
269/// Compiled in the **parent** process once; the resulting `BpfProgram` (a `Vec<sock_filter>`)
270/// is moved into the pre_exec closure by value. The child only calls `prctl` + `seccomp`.
271///
272/// Used by unit tests (verifying baseline filter compiles); production callers
273/// go through `baseline_syscall_map` + per-plan extensions + `compile_filter`.
274#[allow(dead_code)]
275pub(crate) fn build_baseline_filter() -> Result<BpfProgram, SandboxError> {
276    let rules = baseline_syscall_map();
277    compile_filter(rules)
278}
279
280/// Compile a rules map into a BPF program.
281///
282/// Shared by `build_baseline_filter` (for tests that use the raw baseline) and
283/// `apply_strict` (which extends the baseline before compiling).
284fn compile_filter(
285    rules: std::collections::BTreeMap<i64, Vec<SeccompRule>>,
286) -> Result<BpfProgram, SandboxError> {
287    let arch: seccompiler::TargetArch =
288        std::env::consts::ARCH
289            .try_into()
290            .map_err(|_| SandboxError::WrapFailed {
291                message: format!(
292                    "seccompiler does not support arch '{}'; cannot build strict filter",
293                    std::env::consts::ARCH
294                ),
295            })?;
296
297    let filter = SeccompFilter::new(
298        rules,
299        // mismatch_action: kill the whole process for unlisted syscalls.
300        SeccompAction::KillProcess,
301        // match_action: allow listed syscalls.
302        SeccompAction::Allow,
303        arch,
304    )
305    .map_err(|e| SandboxError::WrapFailed {
306        message: format!("seccomp filter build error: {e}"),
307    })?;
308
309    filter
310        .try_into()
311        .map_err(|e: seccompiler::BackendError| SandboxError::WrapFailed {
312            message: format!("seccomp BPF compile error: {e}"),
313        })
314}

316/// Apply Strict-tier isolation to `cmd`:
317/// 1. Landlock filesystem rules (from plan).
318/// 2. `unshare` with `CLONE_NEWUSER | CLONE_NEWNET` (always both — sub-project F
319///    simplified `net::unshare_flags_for_plan` to return both flags unconditionally).
320/// 3. seccomp BPF allow-list filter (baseline extended by exec + network capability rules).
321///
322/// All preparation (path collection, rules extension, BPF compilation) happens in the
323/// **parent** before `fork`. The `pre_exec` closure in the child only calls three
324/// signal-safe operations: landlock, unshare, seccomp.
325///
326/// # Ordering within pre_exec
327///
328/// The order is fixed and must not be changed:
329/// 1. Landlock — uses `landlock_*` syscalls that seccomp would block if installed first.
330/// 2. unshare — must precede seccomp because `unshare(2)` is not in the allow-list.
331/// 3. seccomp — installed last; once active it blocks everything not in the allow-list.
332///
333/// For plans with `Network(Http)`: spawns a userspace proxy task (T3-T4) in the parent
334/// and wraps the child command with `tau-net-bridge` (T5). The proxy guard is nested
335/// inside the returned `SandboxHandle` for LIFO cleanup.
336pub(crate) fn apply_strict(
337    plan: &SandboxPlan,
338    cmd: &mut Command,
339) -> Result<SandboxHandle, SandboxError> {
340    // Collect landlock paths from the plan (same logic as light tier).
341    // Made mutable so Network(Http) can append the proxy socket path.
342    let (mut read_paths, mut write_paths) = crate::light::collect_landlock_paths(plan, cmd)?;
343
344    // Collect exec-gated paths from Filesystem(Exec) and Process(Spawn) capabilities.
345    // Resolve symlinks so landlock path matching covers both the link and its target.
346    let exec_paths: Vec<std::path::PathBuf> = crate::light::collect_exec_paths(plan)
347        .into_iter()
348        .filter_map(|p| {
349            match std::fs::canonicalize(&p) {
350                Ok(canonical) if canonical == p => Some(vec![p]),
351                Ok(canonical) => Some(vec![p, canonical]),
352                Err(_) => None, // Skip unresolvable exec paths silently.
353            }
354        })
355        .flatten()
356        .collect();
357
358    // Build the extended rules map: baseline → exec extension → network extension.
359    let mut rules = baseline_syscall_map();
360    crate::exec::extend_with_exec_rules(&mut rules, plan);
361    crate::net::extend_with_network_rules(&mut rules, plan);
362
363    // Compile the BPF program in the parent — cheap, deterministic.
364    let bpf: BpfProgram = compile_filter(rules)?;
365
366    // Determine unshare flags: always CLONE_NEWUSER | CLONE_NEWNET.
367    // Sub-project F simplified unshare_flags_for_plan — see net.rs and
368    // net_filter/INTEGRATION.md for the post-spawn hook plan (F task 6.5).
369    let unshare_flags = crate::net::unshare_flags_for_plan(plan);
370
371    // Determine if the plan requests outbound HTTP.
372    let has_network_http = plan.capabilities.iter().any(|c| {
373        matches!(
374            c,
375            tau_domain::Capability::Network(tau_domain::NetCapability::Http { .. })
376        )
377    });
378
379    // Capture host uid/gid in the parent so the child can write
380    // /proc/self/uid_map / gid_map after unshare(CLONE_NEWUSER). The
381    // writes gate CAP_NET_ADMIN inside the new user_ns, which the bridge
382    // needs to bring `lo` up via rtnetlink (so the plugin's reqwest can
383    // dial 127.0.0.1:8443). Only relevant for HTTP plugins — non-HTTP
384    // plugins don't use the bridge and don't need loopback up.
385    let host_uid = nix::unistd::getuid().as_raw();
386    let host_gid = nix::unistd::getgid().as_raw();
387    let write_id_maps = has_network_http;
388
389    // For Network(Http): spawn the userspace proxy, extend landlock paths,
390    // and wrap cmd with tau-net-bridge so the child dials through the proxy.
391    // The proxy guard is returned inside the SandboxHandle for LIFO cleanup.
392    let proxy_handle = if has_network_http {
393        // Collect allowed hosts from all Http capabilities.
394        let mut allowed_hosts: Vec<String> = Vec::new();
395        for cap in &plan.capabilities {
396            if let tau_domain::Capability::Network(tau_domain::NetCapability::Http {
397                hosts, ..
398            }) = cap
399            {
400                allowed_hosts.extend(hosts.iter().cloned());
401            }
402        }
403
404        // Validate hosts: rejects wildcards + non-loopback IP literals.
405        tau_sandbox_proxy::validate_hosts(&allowed_hosts).map_err(|e| SandboxError::Proxy {
406            message: format!("host validation: {e}"),
407        })?;
408
409        // Spawn the proxy task in the parent's tokio runtime.
410        let handle =
411            tau_sandbox_proxy::spawn_proxy(allowed_hosts).map_err(|e| SandboxError::Proxy {
412                message: format!("spawn_proxy: {e}"),
413            })?;
414        let proxy_sock_path = handle.sock_path().to_path_buf();
415
416        // Grant the child read+write access to the proxy socket via landlock.
417        read_paths.push(proxy_sock_path.clone());
418        write_paths.push(proxy_sock_path.clone());
419
420        // Snapshot the original program + args so we can wrap them.
421        let original_program = cmd.get_program().to_os_string();
422        let original_args: Vec<std::ffi::OsString> =
423            cmd.get_args().map(|a| a.to_os_string()).collect();
424        // Snapshot existing envs so they survive the cmd replacement.
425        let original_envs: Vec<(std::ffi::OsString, Option<std::ffi::OsString>)> = cmd
426            .get_envs()
427            .map(|(k, v)| (k.to_os_string(), v.map(|v| v.to_os_string())))
428            .collect();
429
430        // Replace the command: tau-net-bridge --proxy-sock=<path>
431        //   --listen=127.0.0.1:8443 -- <original> <args>
432        // std::process::Command has no "set program" method, so we rebuild it.
433        // Bridge binary path: runtime env var, default to PATH lookup. Tests
434        // set TAU_NET_BRIDGE_PATH to env!("CARGO_BIN_EXE_tau-net-bridge")
435        // (that env var IS set in test contexts that depend on the bin target).
436        let bridge_path = std::env::var_os("TAU_NET_BRIDGE_PATH")
437            .unwrap_or_else(|| std::ffi::OsString::from("tau-net-bridge"));
438        *cmd = std::process::Command::new(bridge_path);
439        cmd.arg(format!("--proxy-sock={}", proxy_sock_path.display()))
440            .arg("--listen=127.0.0.1:8443")
441            .arg("--")
442            .arg(&original_program)
443            .args(&original_args);
444        // Restore stdio piping. `std::process::Command` has no getter for
445        // stdin/stdout/stderr so we cannot snapshot the caller's choice the
446        // way we do for envs. Pipe all three unconditionally — this matches
447        // tau-runtime's `plugin_host::process::spawn_and_handshake` (which
448        // always sets `.stdin(piped()).stdout(piped()).stderr(piped())`)
449        // and the controlled-env tests that route stdout/stderr through
450        // `Command::output()`. The bridge process inherits these pipes
451        // across its fork+exec to the plugin (see bin/tau-net-bridge.rs),
452        // so the host-side Child handles produced after `spawn()` are the
453        // pipes the plugin actually reads/writes through.
454        //
455        // Without this, the rebuild above silently drops the caller's
456        // stdio settings and the spawn returns a Child whose
457        // `stdin`/`stdout`/`stderr` are all `None`, panicking the
458        // host-side handshake driver at `child.stdin.take().expect(...)`.
459        cmd.stdin(std::process::Stdio::piped())
460            .stdout(std::process::Stdio::piped())
461            .stderr(std::process::Stdio::piped());
462        // Restore the original environment, then append HTTPS_PROXY.
463        for (k, v) in original_envs {
464            match v {
465                Some(val) => {
466                    cmd.env(k, val);
467                }
468                None => {
469                    cmd.env_remove(k);
470                }
471            }
472        }
473        // Set BOTH HTTPS_PROXY and HTTP_PROXY: reqwest's auto-detection
474        // gates each by scheme — a cassette test that returns an http://
475        // base_url is only proxied if HTTP_PROXY is set.
476        cmd.env("HTTPS_PROXY", "http://127.0.0.1:8443");
477        cmd.env("HTTP_PROXY", "http://127.0.0.1:8443");
478
479        Some(handle)
480    } else {
481        None
482    };
483
484    // KNOWN-LIMITATION: async-signal-safety — see light.rs for the full note.
485    // Additional operations added here:
486    // - `nix::sched::unshare` is a thin syscall wrapper; signal-safe.
487    // - `seccompiler::apply_filter` calls `prctl` + `seccomp`; signal-safe.
488    // The remaining allocation risk is from `install_landlock` (step 1).
489    //
490    // SAFETY: pre_exec runs in the child after fork() but before exec().
491    // All operations (landlock, unshare, seccomp) are child-local and do
492    // not affect the parent process.
493    unsafe {
494        cmd.pre_exec(move || {
495            // Step 1: drop into new user namespace + isolated network namespace
496            // BEFORE landlock so that writes to /proc/self/{setgroups,uid_map,
497            // gid_map} succeed. After landlock, /proc/self is read-only and
498            // those writes return EACCES.
499            unshare(unshare_flags).map_err(|e| std::io::Error::other(e.to_string()))?;
500
501            // Step 1a: write setgroups/uid_map/gid_map so the child has real
502            // CAP_* inside the new user_ns. Without this, the bridge runs as
503            // nobody (uid 65534) and lacks CAP_NET_ADMIN to bring `lo` up,
504            // which silently fails — manifests as the plugin's reqwest
505            // failing to connect to the bridge proxy listener.
506            //
507            // Only gated by HTTP capability: non-HTTP plugins don't run the
508            // bridge and don't need lo up, so they don't need uid_map.
509            // (Skipping for non-HTTP also preserves pre-existing behavior
510            // for those plugins on kernels/runners where the writes fail.)
511            //
512            // Order: setgroups "deny" must be written BEFORE gid_map for
513            // unprivileged user_ns (kernel ≥ 3.19). Tolerate any setgroups
514            // error — the parent user_ns may have already locked us into
515            // deny (e.g. nested under Podman), or the kernel may refuse the
516            // write for state-specific reasons. If setgroups isn't actually
517            // deny, gid_map will fail with EPERM and we'll surface that.
518            if write_id_maps {
519                // All three writes are best-effort: success grants CAP_NET_ADMIN
520                // inside the new user_ns, which the bridge needs to bring `lo`
521                // up. If the kernel/host refuses (GitHub Actions runner ns
522                // layout, some hardened distros), the bridge's rtnetlink call
523                // will fail and lo-up degrades to best-effort — matching the
524                // bridge's existing `bridge: bring lo up failed (continuing)`
525                // path. The plugin's HTTP request then either succeeds (lo
526                // already up by inheritance) or fails like before.
527                //
528                // Critically: we MUST NOT propagate these errors out of
529                // pre_exec. Returning Err here causes std::process to abort
530                // the spawn entirely, which breaks every Network(Http) plan.
531                let _ = std::fs::write("/proc/self/setgroups", "deny\n");
532                let _ = std::fs::write("/proc/self/uid_map", format!("0 {host_uid} 1\n"));
533                let _ = std::fs::write("/proc/self/gid_map", format!("0 {host_gid} 1\n"));
534            }
535
536            // Step 2: landlock filesystem isolation (read/write) + exec gating.
537            // Must come AFTER the uid_map writes above (which need /proc/self
538            // writable) but BEFORE seccomp (which blocks the landlock syscalls).
539            install_landlock_from_plan(&read_paths, &write_paths, &exec_paths)
540                .map_err(|e| std::io::Error::other(e.to_string()))?;
541
542            // Step 3: install seccomp BPF allow-list (blocks unshare/landlock after this).
543            seccompiler::apply_filter(bpf.as_slice())
544                .map_err(|e| std::io::Error::other(e.to_string()))?;
545
546            Ok(())
547        });
548    }
549
550    // Nest the proxy guard inside the SandboxHandle so it is dropped (LIFO)
551    // when the handle is dropped.
552    let mut handle = SandboxHandle::noop();
553    if let Some(p) = proxy_handle {
554        handle.nest_handle(Box::new(p));
555    }
556
557    Ok(handle)
558}
