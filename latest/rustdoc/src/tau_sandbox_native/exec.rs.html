1//! Per-command exec gating helpers for the Strict sandbox tier.
2//!
3//! # Architecture — landlock-based exec gating
4//!
5//! True per-command exec gating is implemented by granting `AccessFs::Execute`
6//! on the **listed** paths only via landlock's per-path filesystem rules, and
7//! deliberately omitting `Execute` from all other paths in the ruleset.
8//!
9//! This works because:
10//! 1. `AccessFs::Execute` is part of landlock **V1** (kernel ≥ 5.13) — no
11//!    V2/V3 upgrade is needed.
12//! 2. landlock V1 covers every filesystem object by its path at `restrict_self()`
13//!    time; any path not covered by an `AccessFs::Execute`-granting rule will
14//!    return `EACCES` on `execve(2)`.
15//!
16//! ## What this module provides
17//!
18//! [`collect_exec_paths`] — extracts the allow-listed executable paths from
19//! the plan's capabilities:
20//!
21//! - `Capability::Filesystem(Exec { paths })` — each entry is used verbatim
22//!   after trimming trailing glob suffixes (same convention as read/write paths
23//!   in `light.rs::resolve_anchors`).
24//! - `Capability::Process(Spawn { commands })` — each entry is either:
25//!   - a full path (starts with `/`) → used directly.
26//!   - a bare command name (`git`, `ls`) → resolved via `PATH` environment
27//!     variable; unresolvable names are silently skipped.
28//!
29//! The resolved paths are returned as `Vec<PathBuf>`. `light.rs` passes them
30//! to `install_landlock` which grants each one `AccessFs::Execute`.
31//!
32//! ## seccomp (unchanged)
33//!
34//! [`extend_with_exec_rules`] remains a **no-op** on the seccomp BPF rule
35//! map. `execve` and `execveat` must remain unconditionally allowed at the
36//! seccomp layer so the initial plugin exec succeeds. The actual per-path
37//! restriction is landlock's job (see above). See the in-function doc for
38//! the full seccomp rationale.
39

47/// Collect the paths that should receive `AccessFs::Execute`-only landlock
48/// rules from the plan's exec-related capabilities.
49///
50/// Two capability variants contribute:
51/// - `Capability::Filesystem(Exec { paths })` — paths are trimmed of trailing
52///   glob suffixes and used directly.
53/// - `Capability::Process(Spawn { commands })` — full-path commands are used
54///   directly; bare command names are resolved through the `PATH` environment
55///   variable at rule-collection time (mirroring shell-resolution at spawn
56///   time). Commands that cannot be resolved are silently skipped.
57///
58/// The caller (`light.rs::install_landlock`) is responsible for adding each
59/// returned path to the landlock ruleset with `AccessFs::Execute` access.
60pub(crate) fn collect_exec_paths(plan: &SandboxPlan) -> Vec<PathBuf> {
61    let mut paths: Vec<PathBuf> = Vec::new();
62
63    for cap in &plan.capabilities {
64        match cap {
65            Capability::Filesystem(FsCapability::Exec {
66                paths: exec_paths, ..
67            }) => {
68                for p in exec_paths {
69                    // Trim trailing glob suffix — landlock's PathFd works on
70                    // directory roots or exact file paths, not glob patterns.
71                    let trimmed = p.trim_end_matches("/**").trim_end_matches("/*").to_string();
72                    paths.push(PathBuf::from(trimmed));
73                }
74            }
75            Capability::Process(ProcessCapability::Spawn { commands, .. }) => {
76                for cmd in commands {
77                    if cmd.starts_with('/') {
78                        // Full path — use directly.
79                        paths.push(PathBuf::from(cmd));
80                    } else {
81                        // Bare command name — resolve via PATH.
82                        if let Some(resolved) = resolve_command_in_path(cmd) {
83                            paths.push(resolved);
84                        }
85                        // Unresolvable commands are silently skipped; the
86                        // plugin simply cannot exec them under landlock, which
87                        // is the correct secure behavior.
88                    }
89                }
90            }
91            _ => {}
92        }
93    }
94
95    paths
96}

98/// Resolve a bare command name (e.g. `"git"`) to a full path by searching
99/// the `PATH` environment variable, mirroring standard shell resolution.
100///
101/// Returns the first matching executable path, or `None` if the command
102/// cannot be found. Entries in `PATH` that don't exist or aren't
103/// accessible are silently skipped.
104fn resolve_command_in_path(cmd: &str) -> Option<PathBuf> {
105    let path_var = std::env::var_os("PATH")?;
106    for dir in std::env::split_paths(&path_var) {
107        let candidate = dir.join(cmd);
108        if candidate.is_file() {
109            return Some(candidate);
110        }
111    }
112    None
113}

115/// Extend the seccomp rules map with exec-gating rules for the given plan.
116///
117/// # Seccomp cannot gate exec by path — this is intentionally a no-op
118///
119/// `execve` and `execveat` remain in the allow-list unconditionally (as set
120/// by `baseline_syscall_map`). The reasons are:
121///
122/// 1. **Plugin-startup chicken-and-egg.** The seccomp filter is installed in
123///    `pre_exec` — before the kernel runs `execve(<plugin binary>)`. If we
124///    deny `execve` at the seccomp layer, the plugin can never start.
125///
126/// 2. **Pointer-vs-content.** `seccomp SeccompCondition::Arg(0, ...)` compares
127///    the *pointer value* of `execve`'s `path` argument, not the string
128///    contents the pointer addresses. A per-path allow-list is impossible via
129///    BPF arg matching.
130///
131/// The per-path exec restriction is enforced via landlock in
132/// `light.rs::install_landlock` using the paths from [`collect_exec_paths`].
133/// This function exists only to satisfy the call site in `strict.rs` and as
134/// the placeholder for any future seccomp-level exec-adjacent rules.
135#[allow(unused_variables)]
136pub(crate) fn extend_with_exec_rules(
137    rules: &mut BTreeMap<i64, Vec<SeccompRule>>,
138    plan: &SandboxPlan,
139) {
140    // No-op: exec path gating is enforced by landlock (collect_exec_paths +
141    // install_landlock), not by seccomp arg matching. See function doc.
142}
