5//! Parameterized conformance test suite for tau `LlmBackend` plugins.
6//!
7//! Charter (per ADR-0009): tests **mechanical correctness** (request
8//! shape, response shape, error typing, stream chunk ordering) NOT
9//! response quality (NG7). The suite is conservative: 6 baseline tests
10//! at v0.1; extension requires a follow-up.
11//!
12//! # Usage
13//!
14//! From a plugin's `tests/conformance.rs`:
15//!
16//! ```ignore
17//! use std::path::Path;
18//! use tau_plugin_conformance::ConformanceSuite;
19//!
20//! #[tokio::test]
21//! async fn run_conformance_suite() {
22//!     ConformanceSuite::default()
23//!         .run(
24//!             |base_url| {
25//!                 // Build a plugin pointing at the cassette server's URL.
26//!                 my_plugin::Plugin::new(base_url)
27//!             },
28//!             Path::new("tests/conformance-cassettes"),
29//!         )
30//!         .await;
31//! }
32//! ```
33//!
34//! Each test reads `cassettes_dir/<test_name>.yaml`:
35//! - `batch_happy_path.yaml`
36//! - `batch_with_tools.yaml`
37//! - `streaming_text.yaml`
38//! - `streaming_tool_use.yaml`
39//! - `error_rate_limited.yaml`
40//! - `error_auth.yaml`
41//!
42//! See `docs/superpowers/specs/2026-04-29-openai-plugin-design.md`
43//! §8.3.
44
45use std::path::Path;
46
47use tau_ports::LlmBackend;
48
49mod cases;
50
51/// Parameterized conformance suite. The catalog at v0.1 is fixed
52/// (6 tests); future versions may gain configurable knobs (skip
53/// individual cases, extend the catalog) — hence `#[non_exhaustive]`.
54#[non_exhaustive]
55#[derive(Debug, Default)]
56pub struct ConformanceSuite {}
57
58impl ConformanceSuite {
59    /// Run the full battery against a plugin built by `build_plugin`.
60    ///
61    /// `build_plugin` is a closure that takes the per-test cassette
62    /// server's URL and returns a fresh plugin instance pointed at it.
63    /// The closure is called once per test (a fresh server is spawned
64    /// for each test from the cassette file at
65    /// `cassettes_dir/<test_name>.yaml`).
66    ///
67    /// Panics on the first assertion failure with a descriptive
68    /// message including the test name. The caller's `#[tokio::test]`
69    /// surface fails accordingly.
70    pub async fn run<B, F>(&self, build_plugin: F, cassettes_dir: &Path)
71    where
72        B: LlmBackend,
73        F: Fn(String) -> B + Send + Sync,
74    {
75        cases::batch_happy_path::run(&build_plugin, cassettes_dir).await;
76        cases::batch_with_tools::run(&build_plugin, cassettes_dir).await;
77        cases::streaming_text::run(&build_plugin, cassettes_dir).await;
78        cases::streaming_tool_use::run(&build_plugin, cassettes_dir).await;
79        cases::error_rate_limited::run(&build_plugin, cassettes_dir).await;
80        cases::error_auth::run(&build_plugin, cassettes_dir).await;
81    }
82}
