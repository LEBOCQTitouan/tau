1//! Conformance case: streaming tool-use round-trip.
2//!
3//! Asserts:
4//! - `stream()` returns Ok.
5//! - At least one `ToolUse` chunk arrives BEFORE `Finish`.
6//! - `tu.input` is a JSON object.
7
8use std::path::Path;
9
10use futures_util::StreamExt;
11use tau_plugin_test_support::cassette;
12use tau_ports::{CompletionChunk, CompletionRequest, ContentBlock, LlmBackend, LlmProviderMessage};
13
14pub(crate) async fn run<B, F>(build_plugin: &F, cassettes_dir: &Path)
15where
16    B: LlmBackend,
17    F: Fn(String) -> B + Send + Sync,
18{
19    let server = cassette::replay(cassettes_dir.join("streaming_tool_use.yaml")).await;
20    let plugin = build_plugin(server.uri().into());
21
22    let mut req = CompletionRequest::new("conformance-model".into());
23    req.messages
24        .push(LlmProviderMessage::user(vec![ContentBlock::Text(
25            "use echo".into(),
26        )]));
27    req.max_tokens = Some(20);
28    req.tools.push(tau_ports::fixtures::make_tool_spec(
29        "echo".into(),
30        "echo input".into(),
31        tau_domain::Value::Object(Default::default()),
32    ));
33
34    let mut stream = match plugin.stream(req).await {
35        Ok(s) => s,
36        Err(e) => panic!("[conformance streaming_tool_use] stream() failed: {e:?}"),
37    };
38
39    let mut chunks = Vec::new();
40    while let Some(item) = stream.next().await {
41        chunks.push(item);
42    }
43
44    let mut tool_uses_before_finish: Vec<&tau_ports::ToolUse> = Vec::new();
45    let mut saw_finish = false;
46    for c in &chunks {
47        match c {
48            Ok(CompletionChunk::ToolUse(tu)) => {
49                if !saw_finish {
50                    tool_uses_before_finish.push(tu);
51                }
52            }
53            Ok(CompletionChunk::Finish { .. }) => {
54                saw_finish = true;
55            }
56            Ok(_) => {}
57            Err(e) => panic!("[conformance streaming_tool_use] chunk was Err: {e:?}"),
58        }
59    }
60    assert!(
61        !tool_uses_before_finish.is_empty(),
62        "[conformance streaming_tool_use] expected >= 1 ToolUse chunk before Finish",
63    );
64    let tu = tool_uses_before_finish[0];
65    assert!(
66        matches!(tu.input, tau_domain::Value::Object(_)),
67        "[conformance streaming_tool_use] expected tool_use.input to be Object, got {:?}",
68        tu.input,
69    );
70}
