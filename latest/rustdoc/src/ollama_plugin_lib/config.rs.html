1//! Ollama plugin configuration.
2//!
3//! Deserialized from the handshake `config` field by
4//! [`tau_plugin_sdk::run_llm_backend_with_config`]. Three concerns:
5//! base URL, optional bearer-token auth, and retry tuning.
6//!
7//! See `docs/superpowers/specs/2026-04-29-ollama-plugin-design.md` §6.
8
9use serde::Deserialize;
10use std::time::Duration;
11use tau_plugin_sdk::ConfigError;
12
13/// Top-level config for the Ollama plugin.
14///
15/// Deserialized from the handshake `config: serde_json::Value`. All
16/// fields have defaults so a project tau.toml `[agents.<id>.config]`
17/// section can be empty.
18///
19/// `#[non_exhaustive]`: additive fields are non-breaking.
20///
21/// # Example
22///
23/// ```ignore
24/// // `OllamaConfig` is `#[non_exhaustive]`; external callers
25/// // construct via serde or Default.
26/// use ollama_plugin_lib::config::OllamaConfig;
27/// let cfg = OllamaConfig::default();
28/// assert_eq!(cfg.base_url, "http://localhost:11434");
29/// ```
30#[non_exhaustive]
31#[derive(Debug, Clone, Deserialize)]
32#[serde(deny_unknown_fields)]
33pub struct OllamaConfig {
34    /// Override base URL. Default: <http://localhost:11434>.
35    /// Tests use this to point at the cassette replayer.
36    #[serde(default = "default_base_url")]
37    pub base_url: String,
38
39    /// Override env var name for an optional bearer token.
40    /// Default: `OLLAMA_BEARER_TOKEN`. Unset env var → no
41    /// `Authorization` header sent (correct for local Ollama).
42    #[serde(default = "default_bearer_token_env")]
43    pub bearer_token_env: String,
44
45    /// Direct bearer-token override. **Test-only** — never put a real
46    /// token in project tau.toml. If both `bearer_token` and
47    /// `bearer_token_env` are present, `bearer_token` wins and a
48    /// `tracing::warn!` is emitted.
49    #[serde(default)]
50    pub bearer_token: Option<String>,
51
52    /// Per-request HTTP timeout in seconds. Default: 900 (15 min).
53    /// Local Ollama can take 30–60s to load a model on first call.
54    #[serde(default = "default_request_timeout_secs")]
55    pub request_timeout_secs: u64,
56
57    /// Retry behavior. Defaults match the Anthropic plugin.
58    #[serde(default)]
59    pub retry: RetryConfig,
60}
61
62impl Default for OllamaConfig {
63    fn default() -> Self {
64        Self {
65            base_url: default_base_url(),
66            bearer_token_env: default_bearer_token_env(),
67            bearer_token: None,
68            request_timeout_secs: default_request_timeout_secs(),
69            retry: RetryConfig::default(),
70        }
71    }
72}
73
74impl OllamaConfig {
75    /// Per-request HTTP timeout as a `Duration`.
76    pub fn request_timeout(&self) -> Duration {
77        Duration::from_secs(self.request_timeout_secs)
78    }
79}

81/// Retry behavior for transient errors (429, 503-on-model-load,
82/// network timeouts).
83///
84/// 503 is the load-bearing case for Ollama: returned during model
85/// load, which can take 10–60s. Standard exponential backoff
86/// (1s, 2s, 4s) handles short loads.
87///
88/// `#[non_exhaustive]`: additive fields are non-breaking.
89#[non_exhaustive]
90#[derive(Debug, Clone, Deserialize)]
91#[serde(deny_unknown_fields)]
92pub struct RetryConfig {
93    /// Maximum total attempts including the initial request. `1`
94    /// disables retry (one-shot). Default: 3.
95    #[serde(default = "default_max_attempts")]
96    pub max_attempts: u32,
97
98    /// Base delay in milliseconds for exponential backoff.
99    /// Effective delay = `base_delay_ms * 2^(attempt-1)`, capped at 60s.
100    /// Default: 1000.
101    #[serde(default = "default_base_delay_ms")]
102    pub base_delay_ms: u64,
103
104    /// Honor the `Retry-After` response header when present (parsed
105    /// as integer seconds). Default: true.
106    #[serde(default = "default_respect_retry_after")]
107    pub respect_retry_after: bool,
108}
109
110impl Default for RetryConfig {
111    fn default() -> Self {
112        Self {
113            max_attempts: default_max_attempts(),
114            base_delay_ms: default_base_delay_ms(),
115            respect_retry_after: default_respect_retry_after(),
116        }
117    }
118}

139/// Resolve an optional bearer token from config or env.
140///
141/// Returns `Ok(None)` when neither is set — the common case for local
142/// Ollama. **Distinct from the Anthropic plugin's `resolve_api_key`,
143/// which errors on missing env var because Anthropic auth is required.**
144///
145/// Wired into `Configure::from_config` in Task 8.
146pub(crate) fn resolve_bearer_token(cfg: &OllamaConfig) -> Result<Option<String>, ConfigError> {
147    if let Some(direct) = cfg.bearer_token.as_ref() {
148        tracing::warn!(
149            target: "ollama_plugin::config",
150            "config.bearer_token set directly — recommended only for tests",
151        );
152        return Ok(Some(direct.clone()));
153    }
154    match std::env::var(&cfg.bearer_token_env) {
155        Ok(v) if v.is_empty() => Ok(None),
156        Ok(v) => Ok(Some(v)),
157        Err(_) => Ok(None),
158    }
159}
160
161/// Validate retry-config invariants beyond what serde catches.
162///
163/// Wired into `Configure::from_config` in Task 8.
164pub(crate) fn validate_retry(retry: &RetryConfig) -> Result<(), ConfigError> {
165    if retry.max_attempts == 0 {
166        return Err(ConfigError::InvalidValue {
167            field: "retry.max_attempts",
168            detail: "must be >= 1 (use 1 for no-retry semantics)".into(),
169        });
170    }
171    Ok(())
172}

178    #[test]
179    fn defaults_are_production_ready() {
180        let cfg = OllamaConfig::default();
181        assert_eq!(cfg.base_url, "http://localhost:11434");
182        assert_eq!(cfg.bearer_token_env, "OLLAMA_BEARER_TOKEN");
183        assert!(cfg.bearer_token.is_none());
184        assert_eq!(cfg.request_timeout_secs, 900);
185        assert_eq!(cfg.retry.max_attempts, 3);
186        assert_eq!(cfg.retry.base_delay_ms, 1000);
187        assert!(cfg.retry.respect_retry_after);
188    }
189
190    #[test]
191    fn deserializes_empty_object_as_defaults() {
192        let cfg: OllamaConfig = serde_json::from_str("{}").unwrap();
193        assert_eq!(cfg.base_url, "http://localhost:11434");
194        assert_eq!(cfg.retry.max_attempts, 3);
195    }

197    #[test]
198    fn rejects_unknown_fields() {
199        let result: Result<OllamaConfig, _> = serde_json::from_str(r#"{"unknown_key": "value"}"#);
200        assert!(result.is_err());

203    #[test]
204    fn resolve_bearer_token_uses_config_override() {
205        let cfg = OllamaConfig {
206            bearer_token: Some("hosted-token-xyz".into()),
207            ..OllamaConfig::default()
208        };
209        let token = resolve_bearer_token(&cfg).unwrap();
210        assert_eq!(token.as_deref(), Some("hosted-token-xyz"));
211    }

213    #[test]
214    fn resolve_bearer_token_reads_env_var() {

216        std::env::set_var(env_name, "envtoken123");
217        let cfg = OllamaConfig {
218            bearer_token_env: env_name.into(),
219            ..OllamaConfig::default()
220        };
221        let token = resolve_bearer_token(&cfg).unwrap();
222        assert_eq!(token.as_deref(), Some("envtoken123"));
223        std::env::remove_var(env_name);
224    }

226    #[test]
227    fn resolve_bearer_token_missing_env_returns_none() {
228        // Distinct from Anthropic: Ollama auth is OPTIONAL. A missing
229        // env var is not an error; it means "no auth header sent".
230        let cfg = OllamaConfig {
231            bearer_token_env: "DEFINITELY_NOT_SET_OLLAMA_TOK_QXZ".into(),
232            ..OllamaConfig::default()
233        };
234        let token = resolve_bearer_token(&cfg).unwrap();
235        assert!(token.is_none());
236    }

240        // Defensive: an empty-string env var is treated the same as

242        let env_name = "TEST_OLLAMA_EMPTY_TOKEN";
243        std::env::set_var(env_name, "");
244        let cfg = OllamaConfig {
