1//! [`OllamaPlugin`] — top-level type implementing
2//! [`tau_ports::LlmBackend`] for Ollama's native `/api/chat` API.
3//!
4//! Per spec §6.2, §6.3:
5//! - [`Configure::from_config`] validates inputs
6//!   (`resolve_bearer_token`, `validate_retry`), constructs the
7//!   `reqwest::Client` with the configured timeout + a tau-branded
8//!   user-agent, and assembles an `OllamaPlugin { client: OllamaClient }`.
9//! - [`LlmBackend::name`] returns `"ollama"`.
10//! - [`LlmBackend::complete`] builds a non-streaming body via
11//!   `build_chat_body`, posts it via `OllamaClient::post_chat`,
12//!   and parses the response with `parse_chat_response`.
13//! - [`LlmBackend::stream`] builds a streaming body, posts it, and
14//!   hands the `reqwest::Response` to `parse_ndjson` to produce a
15//!   [`tau_ports::CompletionStream`].
16
17use secrecy::SecretString;
18use tau_plugin_sdk::{ConfigError, Configure};
19use tau_ports::{CompletionRequest, CompletionResponse, CompletionStream, LlmBackend, LlmError};
20
21use crate::client::OllamaClient;
22use crate::config::{resolve_bearer_token, validate_retry, OllamaConfig};
23use crate::error::{map_client_error, map_response_error};
24use crate::request::build_chat_body;
25use crate::response::parse_chat_response;
26use crate::stream::parse_ndjson;

28/// Ollama (local LLM runner) plugin.
29///
30/// Constructed via [`Configure::from_config`] from an
31/// [`OllamaConfig`]. Holds an `OllamaClient` preconfigured with the
32/// optional bearer token (None for local Ollama at
33/// `http://localhost:11434`), base URL, retry policy, and a
34/// `reqwest::Client` carrying the per-request HTTP timeout.
35pub struct OllamaPlugin {
36    client: OllamaClient,
37}
38
39impl Configure for OllamaPlugin {
40    type Config = OllamaConfig;
41
42    fn from_config(cfg: Self::Config) -> Result<Self, ConfigError> {
43        let bearer_token = resolve_bearer_token(&cfg)?;
44        validate_retry(&cfg.retry)?;
45
46        let inner = reqwest::Client::builder()
47            .timeout(cfg.request_timeout())
48            .user_agent(format!("tau-ollama-plugin/{}", env!("CARGO_PKG_VERSION"),))
49            .build()
50            .map_err(|e| ConfigError::InvalidValue {
51                field: "request_timeout",
52                detail: format!("could not build HTTP client: {e}"),
53            })?;
54
55        let client = OllamaClient::new(
56            inner,
57            cfg.base_url,
58            bearer_token.map(|t| SecretString::new(t.into())),
59            cfg.retry,
60        );
61        Ok(OllamaPlugin { client })
62    }
63}
64
65impl LlmBackend for OllamaPlugin {
66    fn name(&self) -> &str {
67        "ollama"
68    }
69
70    async fn complete(&self, req: CompletionRequest) -> Result<CompletionResponse, LlmError> {
71        let body = build_chat_body(&req, false).map_err(|e| LlmError::Internal {
72            message: format!("ollama: build request body: {e}"),
73        })?;
74        let resp = self
75            .client
76            .post_chat(&body, false)
77            .await
78            .map_err(map_client_error)?;
79
80        let status = resp.status();
81        if !status.is_success() {
82            let headers = resp.headers().clone();
83            let body = resp.text().await.unwrap_or_default();
84            return Err(map_response_error(status, &headers, &body));
85        }
86
87        let body = resp.text().await.map_err(|e| LlmError::Internal {
88            message: format!("ollama: read response body: {e}"),
89        })?;
90        parse_chat_response(&body).map_err(|e| LlmError::Internal {
91            message: format!("ollama: parse response: {e}"),
92        })
93    }
94
95    async fn stream(&self, req: CompletionRequest) -> Result<CompletionStream, LlmError> {
96        let body = build_chat_body(&req, true).map_err(|e| LlmError::Internal {
97            message: format!("ollama: build request body: {e}"),
98        })?;
99        let resp = self
100            .client
101            .post_chat(&body, true)
102            .await
103            .map_err(map_client_error)?;
104
105        let status = resp.status();
106        if !status.is_success() {
107            let headers = resp.headers().clone();
108            let body = resp.text().await.unwrap_or_default();
109            return Err(map_response_error(status, &headers, &body));
110        }
111
112        parse_ndjson(resp).await
113    }
114}

116#[cfg(test)]
117mod tests {
118    use super::*;
119
120    #[test]
121    fn from_config_default_succeeds_with_no_bearer_token() {
122        // Local Ollama needs no auth. Default config + no env var set
123        // should construct cleanly.
124        let cfg = OllamaConfig {
125            // Use a definitely-unset env var name to ensure no accidental
126            // leakage from the test environment.
127            bearer_token_env: "DEFINITELY_NOT_SET_OLLAMA_LOCAL_TOKEN".into(),
128            ..OllamaConfig::default()
129        };
130        let plugin = OllamaPlugin::from_config(cfg).expect("plugin should build");
131        assert_eq!(plugin.name(), "ollama");
132    }
133
134    #[test]
135    fn from_config_invalid_retry_max_attempts_zero_returns_invalid_value() {
136        let mut cfg = OllamaConfig::default();
137        cfg.retry.max_attempts = 0;
138        match OllamaPlugin::from_config(cfg) {
139            Ok(_) => panic!("expected ConfigError::InvalidValue"),
140            Err(err) => assert!(matches!(
141                err,
142                ConfigError::InvalidValue {
143                    field: "retry.max_attempts",
144                    ..
145                }
146            )),
147        }
148    }
149
150    #[test]
151    fn from_config_with_bearer_token_succeeds() {
152        let cfg = OllamaConfig {
153            bearer_token: Some("hosted-token-xyz".into()),
154            ..OllamaConfig::default()
155        };
156        let plugin = OllamaPlugin::from_config(cfg).expect("plugin should build");
157        assert_eq!(plugin.name(), "ollama");
158    }
