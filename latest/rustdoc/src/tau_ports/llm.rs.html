1//! LLM-backend port — `kind = "llm-backend"` plugin contracts.
2//!
3//! This module defines the [`LlmBackend`] trait, the [`CompletionStream`]
4//! type alias, and the data types exchanged between tau-runtime and
5//! plugin adapters. The `batch_to_stream` / `stream_to_batch` /
6//! `ToolUseAccumulator` helpers land in T7.
7//!
8//! # Layer separation: `LlmProviderMessage` vs `tau_domain::Message`
9//!
10//! tau distinguishes the agent's **universal message envelope**
11//! (`tau_domain::Message`, used for inbox/outbox routing between agents,
12//! tools, and the runtime) from the **LLM-call shape**
13//! ([`LlmProviderMessage`], used to construct a single completion
14//! request to a provider).
15//!
16//! - `tau_domain::Message` carries an [`AgentInstanceId`] sender, an
17//!   [`Address`] recipient, a payload, a timestamp, and a message id.
18//!   It is the universal envelope that tau-runtime routes.
19//! - [`LlmProviderMessage`] carries only the role
20//!   (`User` / `Assistant` / `ToolResult`) and a list of
21//!   [`ContentBlock`]s. It is the over-the-wire shape an `LlmBackend`
22//!   plugin serialises into a provider-specific completion call.
23//!
24//! tau-runtime mediates between the two: it consumes
25//! `tau_domain::Message`s from agent inboxes, projects them into a
26//! `Vec<LlmProviderMessage>`, and hands the result to the
27//! `LlmBackend`. Plugins never see `tau_domain::Message` directly.
28//!
29//! [`AgentInstanceId`]: tau_domain::AgentInstanceId
30//! [`Address`]: tau_domain::Address
31

40/// Parameters for a single completion request to an `LlmBackend`.
41///
42/// `CompletionRequest` is `#[non_exhaustive]`: external callers cannot
43/// construct it via struct-literal syntax. Construction is gated through
44/// a builder (added alongside the trait in T6); fields are `pub` so
45/// in-tree code and tests can pattern-match on the parameters.
46///
47/// # Example
48///
49/// ```text
50/// // Struct-literal construction is forbidden externally because
51/// // `CompletionRequest` is `#[non_exhaustive]`. The example here is
52/// // illustrative; real callers use the builder added in T6.
53/// use tau_ports::llm::{CompletionRequest, ToolChoice};
54///
55/// let req = CompletionRequest {
56///     model: "claude-3-5-sonnet".into(),
57///     system: Some("You are helpful.".into()),
58///     messages: vec![],
59///     tools: vec![],
60///     max_tokens: Some(1024),
61///     temperature: Some(0.7),
62///     top_p: None,
63///     seed: None,
64///     tool_choice: ToolChoice::Auto,
65///     stop_sequences: vec![],
66///     provider_specific: Default::default(),
67/// };
68/// ```
69#[non_exhaustive]
70#[derive(Debug, Clone)]
71#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
72pub struct CompletionRequest {
73    /// Model identifier (provider-specific; e.g. `"claude-3-5-sonnet"`).
74    pub model: String,
75    /// Optional system prompt prepended before `messages`.
76    pub system: Option<String>,
77    /// Conversation history as a sequence of provider-shaped messages.
78    pub messages: Vec<LlmProviderMessage>,
79    /// Tool specifications advertised to the model.
80    pub tools: Vec<ToolSpec>,
81    /// Maximum tokens to generate. `None` defers to the provider default.
82    pub max_tokens: Option<u32>,
83    /// Sampling temperature. `None` defers to the provider default.
84    pub temperature: Option<f32>,
85    /// Nucleus-sampling cutoff. `None` defers to the provider default.
86    pub top_p: Option<f32>,
87    /// Deterministic-sampling seed. `None` defers to the provider default.
88    pub seed: Option<u64>,
89    /// Tool-selection policy. Defaults to [`ToolChoice::Auto`].
90    pub tool_choice: ToolChoice,
91    /// Custom stop sequences. Empty defers to the provider default.
92    pub stop_sequences: Vec<String>,
93    /// Provider-specific parameters not yet typed in core (e.g. `top_k`,
94    /// `presence_penalty`, `response_format`).
95    ///
96    /// This is a registered escape hatch. See:
97    /// [escape-hatches.md#completionrequest-provider-specific](../docs/explanation/escape-hatches.md#completionrequest-provider-specific).
98    pub provider_specific: BTreeMap<String, Value>,
99}

101impl CompletionRequest {
102    /// Construct a [`CompletionRequest`] with all optional fields at
103    /// their defaults: `system = None`, `messages` and `tools` empty,
104    /// no sampling overrides, [`ToolChoice::Auto`], no stop sequences,
105    /// no provider-specific overrides.
106    ///
107    /// `CompletionRequest` is `#[non_exhaustive]`: external crates
108    /// (notably tau-runtime, which mints one of these per turn) cannot
109    /// use struct-literal construction. Callers populate the optional
110    /// fields by mutating the returned value via the public fields.
111    ///
112    /// # Example
113    ///
114    /// ```
115    /// use tau_ports::CompletionRequest;
116    ///
117    /// let req = CompletionRequest::new("claude-3-5-sonnet".into());
118    /// assert_eq!(req.model, "claude-3-5-sonnet");
119    /// assert!(req.messages.is_empty());
120    /// ```
121    pub fn new(model: String) -> Self {
122        Self {
123            model,
124            system: None,
125            messages: Vec::new(),
126            tools: Vec::new(),
127            max_tokens: None,
128            temperature: None,
129            top_p: None,
130            seed: None,
131            tool_choice: ToolChoice::Auto,
132            stop_sequences: Vec::new(),
133            provider_specific: BTreeMap::new(),
134        }
135    }
136}

185/// Tool-selection policy for a [`CompletionRequest`].
186///
187/// Defaults to [`ToolChoice::Auto`]: the model decides whether to call
188/// a tool.

191#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
192pub enum ToolChoice {
193    /// Model decides whether to call a tool (default).
194    #[default]
195    Auto,
196    /// Model must call at least one tool.
197    Required,
198    /// Model must not call any tool.
199    None,
200    /// Model must call the named tool.

202        /// Name of the required tool. Must match a [`ToolSpec::name`]

204        name: String,
205    },

271/// One streamed event from a `CompletionStream`.
272///
273/// Plugin authors are responsible for buffering provider-specific
274/// streaming representations into the shape below: `Text` deltas are
275/// forwarded incrementally, `ToolUse` blocks are emitted only when
276/// fully assembled, and exactly one terminal `Finish` is emitted at end
277/// of stream. See `ToolUseAccumulator` (added in T7) for the
278/// JSON-delta-buffering helper.
279#[non_exhaustive]
280#[derive(Debug, Clone)]
281#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
282pub enum CompletionChunk {
283    /// Streamed text delta to append to the assistant response.
284    Text {
285        /// Text fragment to append.
286        delta: String,
287    },
288    /// A tool-use block emitted once fully assembled by the plugin.
289    ToolUse(ToolUse),
290    /// Final marker. Emitted exactly once at end of stream.
291    Finish {
292        /// Why the stream ended.
293        stop_reason: StopReason,
294        /// Token-usage report. `None` if the provider did not return one.
295        usage: Option<TokenUsage>,
296    },
297}

329/// Reason a completion stopped.
330///
331/// `StopReason::Error` indicates the stream completed but reported an
332/// error mid-flight; this is distinct from [`crate::LlmError`], which
333/// indicates the trait method itself failed.
334#[non_exhaustive]
335#[derive(Debug, Clone, Copy, PartialEq, Eq)]
336#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
337pub enum StopReason {
338    /// Model finished naturally.
339    EndTurn,
340    /// Hit the `max_tokens` cap.
341    MaxTokens,
342    /// Matched one of the configured stop sequences.
343    StopSequence,
344    /// Model emitted a tool-use block and is awaiting its result.
345    ToolUse,
346    /// Stream completed but reported an error mid-flight (distinct
347    /// from [`crate::LlmError`], which is a trait-method failure).
348    Error,
349}

362/// Boxed dyn-stream type at the runtime registry boundary. Returned from
363/// [`LlmBackend::stream`].
364pub type CompletionStream = Pin<Box<dyn Stream<Item = Result<CompletionChunk, LlmError>> + Send>>;

366/// Trait implemented by `kind = "llm-backend"` plugins.
367///
368/// Native `async fn in trait` (Rust 1.75+; tau MSRV is 1.91). Both
369/// `complete` and `stream` are required to avoid mutual-recursion
370/// footguns; helpers in [the helpers section] make the inverse
371/// implementation a one-liner for plugin authors who only have one
372/// path natively.
373///
374/// `Send + Sync` so the runtime can store impls in a multi-task plugin
375/// registry.
376///
377/// The `async_fn_in_trait` lint is suppressed: tau-ports intentionally
378/// uses native `async fn in trait` (no `async-trait` macro, no boxed
379/// future per call). tau-runtime boxes once at the dyn-cast boundary
380/// where it stores `Arc<dyn LlmBackend>` in the plugin registry. See
381/// spec §3.1 design call "Native `async fn in trait`" and ADR-0003.
382#[allow(async_fn_in_trait)]
383pub trait LlmBackend: Send + Sync {
384    /// Plugin-visible name (matches the package name; for diagnostics).
385    fn name(&self) -> &str;
386
387    /// Make a batch completion request.
388    ///
389    /// Plugin authors with batch-only SDKs implement natively.
390    /// Plugin authors with streaming SDKs call
391    /// `stream_to_batch(self.stream(req).await?)` (helper in Task 7).
392    async fn complete(&self, req: CompletionRequest) -> Result<CompletionResponse, LlmError>;
393
394    /// Make a streaming completion request.
395    ///
396    /// Plugin authors with streaming SDKs implement natively.
397    /// Plugin authors with batch-only SDKs call
398    /// `Ok(batch_to_stream(self.complete(req).await?))` (helper in Task 7).
399    async fn stream(&self, req: CompletionRequest) -> Result<CompletionStream, LlmError>;
400}

406/// Convert a batch [`CompletionResponse`] into a [`CompletionStream`] that
407/// yields the equivalent chunks: zero-or-one [`CompletionChunk::Text`]
408/// (only when `resp.text` is non-empty), one [`CompletionChunk::ToolUse`]
409/// per entry in `resp.tool_uses` (in order), and one terminal
410/// [`CompletionChunk::Finish`] carrying `resp.stop_reason` and `resp.usage`.
411///
412/// Plugin authors with batch-only SDKs use this in their
413/// [`LlmBackend::stream`] impl: `Ok(batch_to_stream(self.complete(req).await?))`.
414///
415/// # Example
416///
417/// ```text
418/// // Illustrative; `CompletionResponse` is `#[non_exhaustive]` so external
419/// // callers must build it via the data-types builder added in T5.
420/// let resp = /* CompletionResponse */;
421/// let stream = tau_ports::batch_to_stream(resp);
422/// ```
423pub fn batch_to_stream(resp: CompletionResponse) -> CompletionStream {
424    let CompletionResponse {
425        text,
426        tool_uses,
427        stop_reason,
428        usage,
429    } = resp;
430
431    let mut chunks: Vec<Result<CompletionChunk, LlmError>> = Vec::new();
432    if !text.is_empty() {
433        chunks.push(Ok(CompletionChunk::Text { delta: text }));
434    }
435    for tu in tool_uses {
436        chunks.push(Ok(CompletionChunk::ToolUse(tu)));
437    }
438    chunks.push(Ok(CompletionChunk::Finish { stop_reason, usage }));
439
440    Box::pin(VecStream {
441        items: chunks.into_iter(),
442    })
443}

445/// Adapter from a `Vec` of pre-computed items into a `Stream`. Used by
446/// [`batch_to_stream`] to avoid pulling in `futures` as a runtime dep.
447struct VecStream<T> {
448    items: std::vec::IntoIter<T>,
449}
450
451impl<T> Stream for VecStream<T>
452where
453    T: Unpin,
454{
455    type Item = T;
456
457    fn poll_next(
458        self: Pin<&mut Self>,
459        _cx: &mut std::task::Context<'_>,
460    ) -> std::task::Poll<Option<Self::Item>> {
461        std::task::Poll::Ready(self.get_mut().items.next())
462    }
463}

465/// Consume a [`CompletionStream`] and reassemble it into a
466/// [`CompletionResponse`].
467///
468/// Concatenates [`CompletionChunk::Text`] deltas in order, collects
469/// [`CompletionChunk::ToolUse`] blocks in order, and captures the final
470/// [`CompletionChunk::Finish`]'s `stop_reason` and `usage`.
471///
472/// Returns [`LlmError::Stream`] if the stream ends without emitting a
473/// `Finish` chunk. Mid-stream errors propagate as-is.
474///
475/// Plugin authors with streaming-only SDKs use this in their
476/// [`LlmBackend::complete`] impl: `stream_to_batch(self.stream(req).await?).await`.
477///
478/// # Example
479///
480/// ```text
481/// // Illustrative; building a `CompletionStream` requires constructing
482/// // `#[non_exhaustive]` types via the data-types builder added in T5.
483/// let stream: tau_ports::CompletionStream = /* ... */;
484/// let resp = tau_ports::stream_to_batch(stream).await?;
485/// ```
486pub async fn stream_to_batch(mut stream: CompletionStream) -> Result<CompletionResponse, LlmError> {
487    let mut text = String::new();
488    let mut tool_uses: Vec<ToolUse> = Vec::new();
489    let mut finish: Option<(StopReason, Option<TokenUsage>)> = None;
490
491    loop {
492        let next = std::future::poll_fn(|cx| stream.as_mut().poll_next(cx)).await;
493        match next {
494            None => break,
495            Some(Err(e)) => return Err(e),
496            Some(Ok(CompletionChunk::Text { delta })) => text.push_str(&delta),
497            Some(Ok(CompletionChunk::ToolUse(tu))) => tool_uses.push(tu),
498            Some(Ok(CompletionChunk::Finish { stop_reason, usage })) => {
499                finish = Some((stop_reason, usage));
500                // Per spec: Finish is the terminal marker; stop here so any
501                // post-Finish items would be a stream-shape bug we don't mask.
502                break;
503            }
504        }
505    }
506
507    let (stop_reason, usage) = finish.ok_or_else(|| LlmError::Stream {
508        message: "stream ended without Finish chunk".into(),
509    })?;
510
511    Ok(CompletionResponse {
512        text,
513        tool_uses,
514        stop_reason,
515        usage,
516    })
517}

519/// Helper for plugin authors whose streaming SDKs emit JSON tool-use input
520/// deltas (e.g. Anthropic's `input_json_delta` events).
521///
522/// Call [`ToolUseAccumulator::append`] for each delta event, then
523/// [`ToolUseAccumulator::finalize_with`] when the tool-use block closes
524/// to obtain a fully-assembled [`ToolUse`].
525///
526/// `finalize_with` takes a parse callback so plugin authors plug in their
527/// preferred JSON parser. tau-ports has no runtime `serde_json` dependency;
528/// the callback returns a `Result<Value, String>` so any parser crate can
529/// feed in. On `Err(message)` the accumulator returns
530/// [`LlmError::Stream`] wrapping the parse-error message.
531///
532/// # Example
533///
534/// ```
535/// use tau_ports::ToolUseAccumulator;
536///
537/// let mut acc = ToolUseAccumulator::new("toolu_01".into(), "search".into());
538/// acc.append(r#"{"q":"#);
539/// acc.append(r#""hello"}"#);
540///
541/// let tool_use = acc
542///     .finalize_with(|s| {
543///         serde_json::from_str::<tau_domain::Value>(s).map_err(|e| e.to_string())
544///     })
545///     .unwrap();
546/// ```
547#[derive(Debug, Clone)]
548pub struct ToolUseAccumulator {
549    id: String,
550    name: String,
551    input_buffer: String,
552}
553
554impl ToolUseAccumulator {
555    /// Create a fresh accumulator for the tool-use block identified by
556    /// `id` (provider-supplied; round-trips to
557    /// [`LlmProviderMessage::ToolResult::tool_use_id`]) and named `name`
558    /// (must match a [`ToolSpec::name`] in the originating request).
559    pub fn new(id: String, name: String) -> Self {
560        Self {
561            id,
562            name,
563            input_buffer: String::new(),
564        }
565    }
566
567    /// Append a raw JSON delta fragment to the input buffer. Plugin
568    /// authors call this once per provider-specific delta event.
569    pub fn append(&mut self, json_delta: &str) {
570        self.input_buffer.push_str(json_delta);
571    }
572
573    /// Borrow the current input buffer (for diagnostics or partial-parse
574    /// inspection).
575    pub fn input_buffer(&self) -> &str {
576        &self.input_buffer
577    }
578
579    /// Finalize into a [`ToolUse`] using the supplied JSON parser.
580    ///
581    /// `parse` is called once with the accumulated buffer; it returns
582    /// `Ok(Value)` for a successful parse or `Err(message)` to surface a
583    /// parse failure as [`LlmError::Stream`].
584    pub fn finalize_with<F>(self, parse: F) -> Result<ToolUse, LlmError>
585    where
586        F: FnOnce(&str) -> Result<Value, String>,
587    {
588        let input = parse(&self.input_buffer).map_err(|message| LlmError::Stream {
589            message: format!("tool_use input JSON parse failed: {message}"),
590        })?;
591        Ok(ToolUse {
592            id: self.id,
593            name: self.name,
594            input,
595        })
596    }
597}

607    /// Drain a `CompletionStream` into a `Vec<CompletionChunk>` for
608    /// assertion. Errors short-circuit.
609    async fn drain(mut stream: CompletionStream) -> Result<Vec<CompletionChunk>, LlmError> {
610        let mut out = Vec::new();
611        loop {
612            let next: Option<Result<CompletionChunk, LlmError>> =
613                std::future::poll_fn(|cx| Pin::new(&mut stream).poll_next(cx)).await;
614            match next {
615                None => break,
616                Some(Ok(c)) => out.push(c),
617                Some(Err(e)) => return Err(e),
618            }
619        }
620        Ok(out)
621    }

623    #[tokio::test]
624    async fn batch_to_stream_to_batch_empty_round_trip() {
625        let resp = CompletionResponse {
626            text: String::new(),
627            tool_uses: vec![],
628            stop_reason: StopReason::EndTurn,
629            usage: None,
630        };
631
632        let stream = batch_to_stream(resp);
633        let chunks = drain(stream).await.expect("drain");
634        // Empty text => no Text chunk; no tool_uses => no ToolUse chunks;
635        // exactly one Finish.
636        assert_eq!(chunks.len(), 1);
637        let CompletionChunk::Finish { stop_reason, usage } = &chunks[0] else {
