1//! Sandbox port — the `tau_ports::Sandbox` trait + supporting types.
2//!
3//! Hexagonal port: `tau-runtime` consumes this trait; `tau-sandbox-native`,
4//! `tau-sandbox-container`, and `MockSandbox` (in [`crate::fixtures`])
5//! implement it. The runtime selects an adapter via a probe-based chain
6//! configured in `<scope>/.tau/config.toml`.
7//!
8//! Stable as of v0.1 of the sandboxing sub-project. Variant evolution is
9//! handled by `#[non_exhaustive]` on every public type.
10

19/// Plan provided to [`Sandbox::wrap_spawn`].
20#[non_exhaustive]
21#[derive(Debug, Clone)]
22#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
23pub struct SandboxPlan {
24    /// Capabilities the sandboxed code is allowed to exercise. The runtime
25    /// composes this from the package's `compute_effective` capability set
26    /// before calling `wrap_spawn`.
27    pub capabilities: Vec<Capability>,
28    /// Optional working-context hint (working dir + env).
29    pub context: Option<WorkingContext>,
30    /// Optional resource limits.
31    pub limits: Option<ResourceLimits>,
32}

34impl SandboxPlan {
35    /// Construct a [`SandboxPlan`].
36    ///
37    /// `#[non_exhaustive]` blocks struct-literal construction outside

40        capabilities: Vec<Capability>,
41        context: Option<WorkingContext>,
42        limits: Option<ResourceLimits>,
43    ) -> Self {
44        Self {
45            capabilities,
46            context,
47            limits,
48        }
49    }

78/// Probe result describing an adapter's runtime availability.
79#[non_exhaustive]
80#[derive(Debug, Clone)]
81#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
82pub enum SandboxProbe {
83    /// Adapter is usable on this host with the indicated tier.
84    Available {
85        /// Best tier the adapter can guarantee right now.
86        tier: SandboxTier,
87        /// Free-form diagnostic ("landlock V1; seccomp BPF; user_ns ok").
88        details: String,
89    },
90    /// Adapter is not usable on this host.
91    Unavailable {
92        /// Human-readable reason ("kernel < 5.13", "no docker on PATH").
93        reason: String,
94    },
95}

97/// Enforcement tier an adapter can deliver. Forms a total order: `None` <
98/// `Light` < `Strict`. Higher tiers are stricter; project config can RAISE
99/// but never WEAKEN the tier the adapter advertises.
100#[non_exhaustive]
101#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
102#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
103pub enum SandboxTier {
104    /// No enforcement (only valid for the mock adapter).
105    None,
106    /// Filesystem isolation only (e.g. landlock without seccomp).
107    Light,
108    /// Filesystem + syscall + namespace isolation (full Strict tier).
109    Strict,
110}

112/// Opaque handle returned by [`Sandbox::wrap_spawn`]. Drops automatically
113/// release any resources the adapter holds (e.g. cgroup, namespace fd).
114///
115/// `nested`: drop guards that run LIFO before the main cleanup closure.
116/// NativeSandbox uses this to nest a proxy task guard whose Drop signals
117/// the proxy to shut down.
118#[non_exhaustive]
119pub struct SandboxHandle {
120    cleanup: Option<Box<dyn FnOnce() + Send + 'static>>,
121    nested: Vec<Box<dyn Send>>,
122}
123
124impl SandboxHandle {
125    /// Construct a handle from an adapter-defined cleanup closure.
126    /// The closure runs exactly once when the handle is dropped.
127    pub fn new<F: FnOnce() + Send + 'static>(cleanup: F) -> Self {
128        Self {
129            cleanup: Some(Box::new(cleanup)),
130            nested: Vec::new(),
131        }
132    }
133
134    /// A handle that releases nothing (mock / no-op).
135    pub fn noop() -> Self {
136        Self {
137            cleanup: None,
138            nested: Vec::new(),
139        }
140    }
141
142    /// Add a drop guard nested inside this handle's lifetime.
143    ///
144    /// Drop order: nested guards drop LIFO (latest-attached drops first)
145    /// before the main cleanup closure. NativeSandbox uses this to nest
146    /// a proxy task guard whose Drop signals the proxy to shut down.
147    pub fn nest_handle(&mut self, guard: Box<dyn Send>) {
148        self.nested.push(guard);
149    }
150}
151
152impl Drop for SandboxHandle {
153    fn drop(&mut self) {
154        // Drop nested guards LIFO (latest-attached drops first).
155        for guard in self.nested.drain(..).rev() {
156            drop(guard);
157        }
158
159        // Run main cleanup closure.
160        if let Some(cleanup) = self.cleanup.take() {
161            cleanup();
162        }
163    }
164}
165
166impl std::fmt::Debug for SandboxHandle {
167    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
168        f.debug_struct("SandboxHandle").finish_non_exhaustive()
169    }
170}

172/// Trait implemented by sandbox adapters. The runtime calls these methods
173/// in this order:
174///
175/// 1. [`Sandbox::probe`] at startup (cached) — discover what the adapter can do.
176/// 2. [`Sandbox::supported_shapes`] for static cross-checks.
177/// 3. [`Sandbox::validate_plan`] before spawning a plugin process.
178/// 4. [`Sandbox::wrap_spawn`] applies sandbox enforcement to a `Command`.
179#[allow(async_fn_in_trait)]
180pub trait Sandbox: Send + Sync {
181    /// Plugin-visible name (matches the package name; for diagnostics).
182    fn name(&self) -> &str;
183
184    /// Probe the host for adapter availability. Cached by the runtime.
185    async fn probe(&self) -> SandboxProbe;
186
187    /// Capability shapes this adapter can enforce. Used at install time
188    /// (Layer 2) and at `tau check` time (Layer 3) to refuse plans this
189    /// adapter cannot honor.
190    fn supported_shapes(&self) -> CapabilityShapeSet;
191
192    /// Validate that this plan can be executed by this adapter.
193    /// Returns `Err(SandboxError::ShapeUnsupported)` if any required shape
194    /// is not in [`Sandbox::supported_shapes`].
195    fn validate_plan(&self, plan: &SandboxPlan) -> Result<(), SandboxError>;
196
197    /// Apply sandbox enforcement to a [`Command`] in preparation for spawn.
198    /// On Linux native, this typically registers `pre_exec` hooks. The
199    /// returned [`SandboxHandle`] holds any ambient resources (cgroup,
200    /// namespace fd) and releases them on drop.
201    async fn wrap_spawn(
202        &self,
203        plan: &SandboxPlan,
204        cmd: &mut Command,
205    ) -> Result<SandboxHandle, SandboxError>;
206
207    /// Adapter-specific post-spawn setup. Called by the runtime after
208    /// `cmd.spawn()` succeeds and the child PID is known.
209    ///
210    /// Default: no-op. Mock + Container adapters use the default.
211    /// NativeSandbox (Linux) may use this to start a proxy task and nest
212    /// a proxy task guard into `handle` when the plan has
213    /// `Capability::Network(Http)`.
214    async fn apply_post_spawn(
215        &self,
216        plan: &SandboxPlan,
217        child_pid: i32,
218        handle: &mut SandboxHandle,
219    ) -> Result<(), SandboxError> {
220        let _ = (plan, child_pid, handle);
221        Ok(())
222    }
223}

225#[cfg(test)]
226mod tests {
227    use super::*;
228
229    #[test]
230    fn nest_handle_drops_in_lifo_order() {
231        use std::sync::{Arc, Mutex};
232
233        let order: Arc<Mutex<Vec<&'static str>>> = Arc::new(Mutex::new(Vec::new()));
234        let order_main = Arc::clone(&order);
235
236        let mut handle = SandboxHandle::new(move || {
237            order_main.lock().unwrap().push("main_cleanup");
238        });
239
240        // Add 2 nested guards. Each pushes its label on Drop.
241        struct Guard(Arc<Mutex<Vec<&'static str>>>, &'static str);
242        impl Drop for Guard {
243            fn drop(&mut self) {
244                self.0.lock().unwrap().push(self.1);
245            }
246        }
247        handle.nest_handle(Box::new(Guard(Arc::clone(&order), "first_nested")));
248        handle.nest_handle(Box::new(Guard(Arc::clone(&order), "second_nested")));
249
250        drop(handle);
251
252        // Expected order: LIFO of nested, then main cleanup.
253        assert_eq!(
254            *order.lock().unwrap(),
255            vec!["second_nested", "first_nested", "main_cleanup"]
256        );
257    }
