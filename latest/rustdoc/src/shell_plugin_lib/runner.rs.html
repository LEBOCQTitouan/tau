1//! Subprocess runner with wall-clock timeout + 1 MiB output cap.
2//!
3//! Implements the kill+drain pattern per spec §5.3, §5.4, §5.5:
4//! - Spawn with no stdin, piped stdout/stderr, no env inheritance.
5//! - Read stdout + stderr concurrently into Vec<u8> buffers.
6//! - tokio::select! between child.wait() and tokio::time::sleep(timeout).
7//! - On timeout: child.kill() + child.wait() to reap; partial buffers
8//!   are returned with timed_out: true.
9//! - Each buffer capped at MAX_OUTPUT_BYTES (1 MiB); excess truncated
10//!   and the `*_truncated: bool` flag set.
11
12use std::process::Stdio;
13use std::time::Duration;
14
15use tokio::io::AsyncReadExt;
16use tokio::process::Command;
17
18/// Maximum bytes captured per stream (stdout / stderr).
19pub(crate) const MAX_OUTPUT_BYTES: usize = 1024 * 1024;
20
21/// Result of a subprocess invocation.
22#[derive(Debug)]
23pub(crate) struct RunResult {
24    /// Captured stdout bytes (possibly truncated).
25    pub stdout: Vec<u8>,
26    /// Captured stderr bytes (possibly truncated).
27    pub stderr: Vec<u8>,
28    /// Process exit code; -1 when the wall-clock timeout fired.
29    pub exit_code: i32,
30    /// True iff the wall-clock timeout fired (process was SIGKILLed).
31    pub timed_out: bool,
32    /// True iff stdout exceeded MAX_OUTPUT_BYTES and was truncated.
33    pub stdout_truncated: bool,
34    /// True iff stderr exceeded MAX_OUTPUT_BYTES and was truncated.
35    pub stderr_truncated: bool,
36}
37
38/// Run a subprocess with wall-clock timeout + output capping.
39///
40/// `command` is the program name (looked up via PATH or absolute);
41/// `args` are positional arguments. `cwd` is an optional working
42/// directory. `timeout_secs` is the wall-clock timeout in seconds.
43///
44/// On timeout, the child is SIGKILLed and partial buffers are
45/// returned with `timed_out: true, exit_code: -1`.
46pub(crate) async fn run_subprocess(
47    command: &str,
48    args: &[String],
49    timeout_secs: u64,
50    cwd: Option<&str>,
51) -> std::io::Result<RunResult> {
52    let mut cmd = Command::new(command);
53    cmd.args(args)
54        .env_clear() // no env inheritance
55        .stdin(Stdio::null())
56        .stdout(Stdio::piped())
57        .stderr(Stdio::piped());
58    if let Some(dir) = cwd {
59        cmd.current_dir(dir);
60    }
61    let mut child = cmd.spawn()?;
62
63    // Take stdout/stderr handles before await so we can read them
64    // concurrently with child.wait().
65    let mut stdout_handle = child.stdout.take().expect("stdout was piped");
66    let mut stderr_handle = child.stderr.take().expect("stderr was piped");
67
68    let stdout_task = tokio::spawn(async move {
69        let mut buf = Vec::new();
70        let _ = stdout_handle.read_to_end(&mut buf).await;
71        buf
72    });
73    let stderr_task = tokio::spawn(async move {
74        let mut buf = Vec::new();
75        let _ = stderr_handle.read_to_end(&mut buf).await;
76        buf
77    });
78
79    // tokio::select! between child completion and the timeout.
80    let timed_out = tokio::select! {
81        _ = child.wait() => false,
82        _ = tokio::time::sleep(Duration::from_secs(timeout_secs)) => {
83            // Best-effort SIGKILL + reap.
84            let _ = child.kill().await;
85            let _ = child.wait().await;
86            true
87        }
88    };
89
90    // Join the buffer-collecting tasks. They finish naturally because
91    // the child has exited (either normally or via SIGKILL).
92    let stdout_raw = stdout_task.await.unwrap_or_default();
93    let stderr_raw = stderr_task.await.unwrap_or_default();
94
95    let (stdout, stdout_truncated) = cap_and_flag(stdout_raw);
96    let (stderr, stderr_truncated) = cap_and_flag(stderr_raw);
97
98    let exit_code = if timed_out {
99        -1
100    } else {
101        // After child.wait() completed, the child's status was set;
102        // try_wait gives us the ExitStatus without re-awaiting.
103        match child.try_wait() {
104            Ok(Some(status)) => status.code().unwrap_or(-1),
105            _ => -1,
106        }
107    };
108
109    Ok(RunResult {
110        stdout,
111        stderr,
112        exit_code,
113        timed_out,
114        stdout_truncated,
115        stderr_truncated,
116    })
117}
118
119/// Cap a buffer at MAX_OUTPUT_BYTES; return (capped_buffer, truncated_flag).
120pub(crate) fn cap_and_flag(buf: Vec<u8>) -> (Vec<u8>, bool) {
121    if buf.len() > MAX_OUTPUT_BYTES {
122        (buf[..MAX_OUTPUT_BYTES].to_vec(), true)
123    } else {
124        (buf, false)
125    }
126}

128#[cfg(test)]
129#[cfg(unix)]
130mod tests {
131    use super::*;
132
133    #[tokio::test]
134    async fn run_echo_returns_stdout() {
135        let result = run_subprocess("/bin/echo", &["hello".to_string()], 5, None)
136            .await
137            .expect("echo runs");
138        assert_eq!(result.exit_code, 0);
139        assert!(!result.timed_out);
140        assert_eq!(result.stdout, b"hello\n");
141        assert!(result.stderr.is_empty());
142        assert!(!result.stdout_truncated);
143        assert!(!result.stderr_truncated);
144    }
145
146    #[tokio::test]
147    async fn run_nonzero_exit_returns_exit_code() {
148        let result = run_subprocess(
149            "/bin/sh",
150            &["-c".to_string(), "exit 7".to_string()],
151            5,
152            None,
153        )
154        .await
155        .expect("sh runs");
156        assert_eq!(result.exit_code, 7);
157        assert!(!result.timed_out);
158    }
159
160    #[tokio::test]
161    async fn run_timeout_kills_and_flags_timed_out() {
162        let result = run_subprocess(
163            "/bin/sh",
164            &["-c".to_string(), "sleep 5".to_string()],
165            1,
166            None,
167        )
168        .await
169        .expect("sh runs");
170        assert_eq!(result.exit_code, -1);
171        assert!(result.timed_out);
172    }
173
174    #[tokio::test]
175    async fn run_command_not_found_returns_io_err() {
176        let result = run_subprocess("definitely-not-a-real-command-xyz-9999", &[], 5, None).await;
177        assert!(result.is_err());
178    }
179
180    #[tokio::test]
181    async fn run_with_cwd_runs_in_directory() {
182        let dir = tempfile::tempdir().unwrap();
183        let path = dir.path().to_str().unwrap();
184        let result = run_subprocess(
185            "/bin/sh",
186            &["-c".to_string(), "pwd".to_string()],
187            5,
188            Some(path),
189        )
190        .await
191        .expect("sh runs");
192        assert_eq!(result.exit_code, 0);
193        // pwd outputs the cwd + newline; canonicalize-tolerant compare.
194        let actual = String::from_utf8(result.stdout).unwrap();
195        let actual_trimmed = actual.trim();
196        let expected_canonical = std::fs::canonicalize(path).unwrap();
197        let actual_canonical = std::fs::canonicalize(actual_trimmed).unwrap();
198        assert_eq!(actual_canonical, expected_canonical);
199    }
200
201    #[test]
202    fn cap_and_flag_under_limit_no_flag() {
203        let buf = vec![b'a'; 100];
204        let (capped, truncated) = cap_and_flag(buf);
205        assert_eq!(capped.len(), 100);
206        assert!(!truncated);
207    }
208
209    #[test]
210    fn cap_and_flag_at_exact_limit_no_flag() {
211        let buf = vec![b'a'; MAX_OUTPUT_BYTES];
212        let (capped, truncated) = cap_and_flag(buf);
213        assert_eq!(capped.len(), MAX_OUTPUT_BYTES);
214        assert!(!truncated);
215    }
216
217    #[test]
218    fn cap_and_flag_over_limit_truncates() {
219        let buf = vec![b'a'; MAX_OUTPUT_BYTES + 1024];
220        let (capped, truncated) = cap_and_flag(buf);
221        assert_eq!(capped.len(), MAX_OUTPUT_BYTES);
222        assert!(truncated);
