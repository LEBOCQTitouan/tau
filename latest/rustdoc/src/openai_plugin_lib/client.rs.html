1//! OpenAI HTTP client with retry + required Bearer auth.
2//!
3//! `post_chat_completions` is the only outbound entrypoint. Owns a
4//! `reqwest::Client` with TLS + timeout configured by the plugin's
5//! `Configure::from_config`.
6//!
7//! Retry policy per spec §4.1:
8//! - 2xx → return immediately.
9//! - 429, 503 → retry with exponential backoff; honor `Retry-After`
10//!   when `retry.respect_retry_after` is true. Exhausted →
11//!   `ClientError::Exhausted`.
12//! - 4xx other than 429 → return immediately (caller maps to LlmError).
13//! - 5xx other than 503 → retry (treated as transient).
14//! - Network timeout → retry (synthesized as 408).
15//! - Other transport error (DNS, TLS, connection refused) →
16//!   `ClientError::Transport`, no retry.
17

26/// HTTP client for OpenAI's `/v1/chat/completions` endpoint.
27pub(crate) struct OpenAIClient {
28    inner: reqwest::Client,
29    base_url: String,
30    api_key: SecretString,
31    organization: Option<String>,
32    retry: RetryConfig,

54    /// `POST /v1/chat/completions` with retry. The body is a
55    /// `serde_json::Value` produced by `request::build_chat_completions_body`.
56    /// `stream == true` adds the `accept: text/event-stream` header.
57    pub(crate) async fn post_chat_completions(
58        &self,
59        body: &serde_json::Value,
60        stream: bool,
61    ) -> Result<Response, ClientError> {
62        let url = format!(
63            "{}/v1/chat/completions",
64            self.base_url.trim_end_matches('/'),
65        );
66        let mut attempt: u32 = 0;
67
68        loop {
69            attempt += 1;
70
71            let mut req = self
72                .inner
73                .post(&url)
74                .header(
75                    "authorization",
76                    format!("Bearer {}", self.api_key.expose_secret()),
77                )
78                .header("content-type", "application/json")
79                .json(body);
80            if stream {
81                req = req.header("accept", "text/event-stream");
82            }
83            if let Some(org) = &self.organization {
84                req = req.header("openai-organization", org);
85            }
86
87            let send_result = req.send().await;
88            match self.classify(send_result, attempt) {
89                Decision::Return(resp) => return Ok(resp),
90                Decision::Error(err) => return Err(err),
91                Decision::Retry { delay_ms, status } => {
92                    if attempt >= self.retry.max_attempts {
93                        return Err(ClientError::Exhausted {
94                            status,
95                            attempts: attempt,
96                        });
97                    }
98                    tracing::warn!(
99                        target: "openai_plugin::retry",
100                        attempt,
101                        max = self.retry.max_attempts,
102                        delay_ms,
103                        status = status.as_u16(),
104                        "retrying transient error",
105                    );
106                    sleep(Duration::from_millis(delay_ms)).await;
107                }
108            }
109        }
110    }
111
112    fn classify(&self, res: reqwest::Result<Response>, attempt: u32) -> Decision {
113        match res {
114            Ok(resp) => {
115                let status = resp.status();
116                if status.is_success() {
117                    Decision::Return(resp)
118                } else if is_retryable_status(status) {
119                    let delay_ms = self.retry_delay(&resp, attempt);
120                    Decision::Retry { delay_ms, status }
121                } else {
122                    Decision::Return(resp)
123                }
124            }
125            Err(e) if e.is_timeout() => {
126                let delay_ms = self.backoff_only(attempt);
127                Decision::Retry {
128                    delay_ms,
129                    status: StatusCode::REQUEST_TIMEOUT,
130                }
131            }
132            Err(e) => Decision::Error(ClientError::Transport(e)),
133        }
134    }
135
136    fn retry_delay(&self, resp: &Response, attempt: u32) -> u64 {
137        if self.retry.respect_retry_after {
138            if let Some(secs) = resp
139                .headers()
140                .get("retry-after")
141                .and_then(|h| h.to_str().ok())
142                .and_then(|s| s.parse::<u64>().ok())
143            {
144                return secs * 1000;
145            }
146        }
147        self.backoff_only(attempt)
148    }
149
150    fn backoff_only(&self, attempt: u32) -> u64 {
151        let shift = (attempt - 1).min(6);
152        let delay = self.retry.base_delay_ms.saturating_mul(1u64 << shift);
153        delay.min(60_000)
154    }
155}
156
157fn is_retryable_status(status: StatusCode) -> bool {
158    matches!(status.as_u16(), 429 | 503) || (status.as_u16() >= 500 && status.as_u16() != 501)
159}
160
161enum Decision {
162    Return(Response),
163    Retry { delay_ms: u64, status: StatusCode },
164    Error(ClientError),
165}

175    /// Spawn a tiny HTTP/1.1 server on 127.0.0.1:0 that replies with
176    /// pre-canned responses for the first N connections.
177    /// Returns the listener URL + a counter of accepted connections +
178    /// captured request bytes (for header assertions).
179    async fn spawn_canned_server(
180        responses: Vec<&'static str>,
181    ) -> (
182        String,
183        Arc<AtomicUsize>,
184        Arc<tokio::sync::Mutex<Vec<Vec<u8>>>>,
185    ) {
186        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
187        let port = listener.local_addr().unwrap().port();
188        let url = format!("http://127.0.0.1:{port}");
189        let counter = Arc::new(AtomicUsize::new(0));
190        let received = Arc::new(tokio::sync::Mutex::new(Vec::<Vec<u8>>::new()));
191        let counter_clone = counter.clone();
192        let received_clone = received.clone();
193
194        tokio::spawn(async move {
195            for canned in responses {
196                let (mut sock, _) = listener.accept().await.unwrap();
197                counter_clone.fetch_add(1, Ordering::Relaxed);
198                let mut buf = [0u8; 4096];
199                let n = sock.read(&mut buf).await.unwrap_or(0);
200                received_clone.lock().await.push(buf[..n].to_vec());
201                sock.write_all(canned.as_bytes()).await.unwrap();
202                sock.flush().await.unwrap();
203                drop(sock);
204            }
205        });
206
207        (url, counter, received)
208    }

219    fn rate_limited_body(retry_after: Option<&str>) -> String {
220        let mut s = String::from("HTTP/1.1 429 Too Many Requests\r\n");
221        if let Some(ra) = retry_after {
222            s.push_str(&format!("retry-after: {ra}\r\n"));
223        }
224        s.push_str(
225            "content-type: application/json\r\n\
226             content-length: 0\r\n\
227             connection: close\r\n\
228             \r\n",
229        );
230        s
231    }

255    #[tokio::test]
256    async fn post_chat_completions_happy_path_sends_authorization_header() {
257        let (url, counter, received) = spawn_canned_server(vec![ok_body()]).await;
258        let client = build_client(url, None, 3, 0);
259        let resp = client
260            .post_chat_completions(&serde_json::json!({"model": "m", "messages": []}), false)
261            .await
262            .unwrap();
263        assert!(resp.status().is_success());
264        assert_eq!(counter.load(Ordering::Relaxed), 1);
265
266        let raw = received.lock().await;
267        let req_str = std::str::from_utf8(&raw[0]).unwrap_or("");
268        assert!(
269            req_str
270                .to_ascii_lowercase()
271                .contains("authorization: bearer sk-test-1234"),
272            "expected Authorization: Bearer header; got: {req_str}",
273        );
274        // Organization header NOT sent when None.
275        assert!(
276            !req_str
277                .to_ascii_lowercase()
278                .contains("openai-organization:"),
279            "no org header expected; got: {req_str}",
280        );
281    }
