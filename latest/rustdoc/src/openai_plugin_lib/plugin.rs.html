1//! [`OpenAIPlugin`] — top-level type implementing
2//! [`tau_ports::LlmBackend`] for OpenAI's Chat Completions API.
3//!
4//! Per spec §6:
5//! - [`Configure::from_config`] validates inputs
6//!   (`resolve_api_key`, `validate_retry`), constructs the
7//!   `reqwest::Client` with the configured timeout + a tau-branded
8//!   user-agent, and assembles an `OpenAIPlugin { client: OpenAIClient }`.
9//! - [`LlmBackend::name`] returns `"openai"`.
10//! - [`LlmBackend::complete`] builds a non-streaming body via
11//!   `build_chat_completions_body`, posts it via
12//!   `OpenAIClient::post_chat_completions`, and parses the response
13//!   with `parse_chat_completions_response`.
14//! - [`LlmBackend::stream`] builds a streaming body, posts it, and
15//!   hands the `reqwest::Response` to `parse_sse` to produce a
16//!   [`tau_ports::CompletionStream`].
17//!
18//! Non-success responses extract `headers` BEFORE consuming the
19//! response via `text()` so 429 responses can populate
20//! `LlmError::RateLimited.retry_after_seconds`.
21
22use secrecy::SecretString;
23use tau_plugin_sdk::{ConfigError, Configure};
24use tau_ports::{CompletionRequest, CompletionResponse, CompletionStream, LlmBackend, LlmError};
25
26use crate::client::OpenAIClient;
27use crate::config::{resolve_api_key, validate_retry, OpenAIConfig};
28use crate::error::{map_client_error, map_response_error};
29use crate::request::build_chat_completions_body;
30use crate::response::parse_chat_completions_response;
31use crate::stream::parse_sse;

33/// OpenAI Chat Completions API plugin.
34///
35/// Constructed via [`Configure::from_config`] from an
36/// [`OpenAIConfig`]. Holds an `OpenAIClient` preconfigured with the
37/// resolved API key, base URL, optional organization id, retry policy,
38/// and a `reqwest::Client` carrying the per-request HTTP timeout.
39pub struct OpenAIPlugin {
40    client: OpenAIClient,
41}
42
43impl Configure for OpenAIPlugin {
44    type Config = OpenAIConfig;
45
46    fn from_config(cfg: Self::Config) -> Result<Self, ConfigError> {
47        let api_key = resolve_api_key(&cfg)?;
48        validate_retry(&cfg.retry)?;
49
50        let inner = reqwest::Client::builder()
51            .timeout(cfg.request_timeout())
52            .user_agent(format!("tau-openai-plugin/{}", env!("CARGO_PKG_VERSION")))
53            .build()
54            .map_err(|e| ConfigError::InvalidValue {
55                field: "request_timeout",
56                detail: format!("could not build HTTP client: {e}"),
57            })?;
58
59        let client = OpenAIClient::new(
60            inner,
61            cfg.base_url,
62            SecretString::new(api_key.into()),
63            cfg.organization,
64            cfg.retry,
65        );
66        Ok(OpenAIPlugin { client })
67    }
68}

70impl LlmBackend for OpenAIPlugin {
71    fn name(&self) -> &str {
72        "openai"
73    }
74
75    async fn complete(&self, req: CompletionRequest) -> Result<CompletionResponse, LlmError> {
76        let body = build_chat_completions_body(&req, false).map_err(|e| LlmError::Internal {
77            message: format!("openai: build request body: {e}"),
78        })?;
79        let resp = self
80            .client
81            .post_chat_completions(&body, false)
82            .await
83            .map_err(map_client_error)?;
84
85        let status = resp.status();
86        if !status.is_success() {
87            // Extract headers BEFORE consuming the response via text().
88            let headers = resp.headers().clone();
89            let body = resp.text().await.unwrap_or_default();
90            return Err(map_response_error(status, &headers, &body));
91        }
92
93        let body = resp.text().await.map_err(|e| LlmError::Internal {
94            message: format!("openai: read response body: {e}"),
95        })?;
96        parse_chat_completions_response(&body).map_err(|e| LlmError::Internal {
97            message: format!("openai: parse response: {e}"),
98        })
99    }
100
101    async fn stream(&self, req: CompletionRequest) -> Result<CompletionStream, LlmError> {
102        let body = build_chat_completions_body(&req, true).map_err(|e| LlmError::Internal {
103            message: format!("openai: build request body: {e}"),
104        })?;
105        let resp = self
106            .client
107            .post_chat_completions(&body, true)
108            .await
109            .map_err(map_client_error)?;
110
111        let status = resp.status();
112        if !status.is_success() {
113            let headers = resp.headers().clone();
114            let body = resp.text().await.unwrap_or_default();
115            return Err(map_response_error(status, &headers, &body));
116        }
117
118        parse_sse(resp).await
119    }
120}

122#[cfg(test)]
123mod tests {
124    use super::*;
125
126    #[test]
127    fn from_config_valid_api_key_constructs_plugin() {
128        let env_name = "TEST_OPENAI_FROM_CONFIG_OK";
129        std::env::set_var(env_name, "sk-proj-test-key-12345");
130        let cfg = OpenAIConfig {
131            api_key_env: env_name.into(),
132            ..OpenAIConfig::default()
133        };
134        let result = OpenAIPlugin::from_config(cfg);
135        assert!(result.is_ok(), "from_config should succeed");
136        std::env::remove_var(env_name);
137    }
138
139    #[test]
140    fn from_config_invalid_retry_max_attempts_zero_returns_invalid_value() {
141        let env_name = "TEST_OPENAI_INVALID_RETRY";
142        std::env::set_var(env_name, "sk-test");
143        let mut cfg = OpenAIConfig {
144            api_key_env: env_name.into(),
145            ..OpenAIConfig::default()
146        };
147        cfg.retry.max_attempts = 0;
148        let err = match OpenAIPlugin::from_config(cfg) {
149            Ok(_) => panic!("expected ConfigError::InvalidValue"),
150            Err(e) => e,
151        };
152        assert!(matches!(
153            err,
154            ConfigError::InvalidValue {
155                field: "retry.max_attempts",
156                ..
157            }
158        ));
159        std::env::remove_var(env_name);
160    }
161
162    #[test]
163    fn name_returns_openai() {
164        let env_name = "TEST_OPENAI_NAME";
165        std::env::set_var(env_name, "sk-test");
166        let cfg = OpenAIConfig {
167            api_key_env: env_name.into(),
168            ..OpenAIConfig::default()
169        };
170        let plugin = OpenAIPlugin::from_config(cfg).unwrap();
171        assert_eq!(plugin.name(), "openai");
172        std::env::remove_var(env_name);
173    }
