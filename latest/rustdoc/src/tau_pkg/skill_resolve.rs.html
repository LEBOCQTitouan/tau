1//! Skill resolution: lookup an installed skill by name.
2//!
3//! Skills-4 (ROADMAP §16). Reads the scope lockfile (one file open
4//! handled by `LockFile::load`) + the skill package's `tau.toml` (one
5//! additional file open). Returns a fully-built `InstalledSkill` with
6//! resolved install path + parsed manifest + cached frontmatter.
7
8use std::path::PathBuf;
9
10use tau_domain::{Capability, PackageManifest, PackageName, SkillManifest, Version};
11
12use crate::error::RegistryError;
13use crate::lockfile::{LockFile, SkillFrontmatterSnapshot};
14use crate::scope::Scope;
15
16/// A fully-resolved installed skill, ready for runtime invocation.
17#[derive(Debug, Clone)]
18pub struct InstalledSkill {
19    /// Package name (matches `tau.toml` `name` field).
20    pub name: PackageName,
21    /// Active version (from lockfile).
22    pub version: Version,
23    /// Absolute path to the installed package directory
24    /// (`<scope_state>/.tau/packages/<name>/<version>/`).
25    pub install_path: PathBuf,
26    /// Parsed manifest from `<install_path>/tau.toml`.
27    pub manifest: PackageManifest,
28    /// Cached SKILL.md frontmatter snapshot (from lockfile).
29    pub frontmatter: SkillFrontmatterSnapshot,
30    /// Skill-specific manifest block (from manifest).
31    pub skill: SkillManifest,
32    /// Declared capabilities (from manifest).
33    pub capabilities: Vec<Capability>,
34}
35
36/// Errors raised by [`find_installed_skill`].
37#[derive(Debug, thiserror::Error)]
38#[non_exhaustive]
39pub enum FindSkillError {
40    /// Failed to load the scope lockfile.
41    #[error("lockfile load: {0}")]
42    Lockfile(#[from] RegistryError),
43    /// Lockfile entry exists, but the install path is missing on disk.
44    #[error("skill {name:?} lockfile entry points at {path:?} but no tau.toml found there")]
45    InstallPathMissing {
46        /// Skill name.
47        name: String,
48        /// The expected install path.
49        path: PathBuf,
50    },
51    /// I/O error reading the manifest.
52    #[error("reading manifest at {path:?}: {source}")]
53    ReadManifest {
54        /// The manifest path.
55        path: PathBuf,
56        /// Underlying io error.
57        #[source]
58        source: std::io::Error,
59    },
60    /// TOML parse failure on the manifest.
61    #[error("parsing manifest at {path:?}: {detail}")]
62    ParseManifest {
63        /// The manifest path.
64        path: PathBuf,
65        /// Parser error detail.
66        detail: String,
67    },
68    /// Manifest validation failure.
69    #[error("validating manifest at {path:?}: {detail}")]
70    ValidateManifest {
71        /// The manifest path.
72        path: PathBuf,
73        /// Validation error detail.
74        detail: String,
75    },
76    /// Manifest declared a kind other than "skill" but was found in a
77    /// skill entry in the lockfile.
78    #[error(
79        "manifest at {path:?} has kind != \"skill\" but is recorded as a skill in the lockfile"
80    )]
81    NotASkillManifest {
82        /// The manifest path.
83        path: PathBuf,
84    },
85}

87/// Resolve an installed skill by name. Returns `Ok(None)` if no
88/// installed package matches `name` and has a skill block.
89///
90/// Reads the scope lockfile + one `tau.toml` for the matched skill.
91pub fn find_installed_skill(
92    scope: &Scope,
93    name: &str,
94) -> Result<Option<InstalledSkill>, FindSkillError> {
95    let lockfile_path = scope.lockfile_path();
96    if !lockfile_path.exists() {
97        return Ok(None);
98    }
99    let lockfile = LockFile::load(&lockfile_path)?;
100
101    let pkg = match lockfile
102        .packages
103        .iter()
104        .find(|p| p.name.as_str() == name && p.skill.is_some())
105    {
106        Some(p) => p,
107        None => return Ok(None),
108    };
109
110    let locked_skill = pkg.skill.as_ref().expect("filtered to Some(skill) above");
111
112    // Install path uses state_path (e.g. <project>/.tau/packages/<name>/<version>)
113    let install_path = scope.package_dir(&pkg.name, &pkg.active_version);
114
115    let toml_path = install_path.join("tau.toml");
116    if !toml_path.exists() {
117        return Err(FindSkillError::InstallPathMissing {
118            name: name.to_string(),
119            path: install_path,
120        });
121    }
122    let toml_text =
123        std::fs::read_to_string(&toml_path).map_err(|e| FindSkillError::ReadManifest {
124            path: toml_path.clone(),
125            source: e,
126        })?;
127    let unchecked: tau_domain::UncheckedManifest =
128        toml::from_str(&toml_text).map_err(|e| FindSkillError::ParseManifest {
129            path: toml_path.clone(),
130            detail: e.to_string(),
131        })?;
132    let manifest = unchecked
133        .validate()
134        .map_err(|e| FindSkillError::ValidateManifest {
135            path: toml_path.clone(),
136            detail: e.to_string(),
137        })?;
138
139    let skill = match manifest.skill() {
140        Some(s) => s.clone(),
141        None => {
142            return Err(FindSkillError::NotASkillManifest { path: toml_path });
143        }
144    };
145
146    let capabilities = manifest.capabilities().to_vec();
147
148    Ok(Some(InstalledSkill {
149        name: pkg.name.clone(),
150        version: pkg.active_version.clone(),
151        install_path,
152        manifest,
153        frontmatter: locked_skill.frontmatter.clone(),
154        skill,
155        capabilities,
156    }))
157}

159#[cfg(test)]
160mod tests {
161    use super::*;
162    use crate::lockfile::{
163        LockFile, LockedPackage, LockedSkill, LockedVersion, SkillFrontmatterSnapshot,
164    };
165    use std::fs;
166    use std::time::Duration;
167    use std::time::UNIX_EPOCH;

169
170    fn make_critic_scope(tmp: &std::path::Path) -> Scope {
171        // Scope::resolve walks up looking for .tau/; we create .tau/ here.
172        let tau_dir = tmp.join(".tau");
173        fs::create_dir_all(&tau_dir).unwrap();
174        fs::write(
175            tau_dir.join("config.toml"),
176            "schema_version = 3\nkind = \"project\"\ncreated_at = \"2026-05-14T00:00:00Z\"\ncreated_by_tau_version = \"0.0.0\"\n\n[sandbox]\nrequired_tier = \"none\"\n",
177        )
178        .unwrap();
179
180        // Install path uses state_path: <tmp>/.tau/packages/<name>/<version>
181        let install_dir = tau_dir.join("packages").join("critic").join("0.1.0");
182        fs::create_dir_all(&install_dir).unwrap();
183        fs::write(
184            install_dir.join("tau.toml"),
185            r#"name = "critic"
186version = "0.1.0"
187description = "Reviews drafts."
188authors = []
189source = "https://example.com/critic.git"
190kind = "skill"
191dependencies = []
192capabilities = []
193
194[skill]
195"#,
196        )
197        .unwrap();
198        fs::write(
199            install_dir.join("SKILL.md"),
200            "---\nname: critic\ndescription: x\n---\nbody\n",
201        )
202        .unwrap();
203
204        // Build LockedPackage directly (not via TOML deserialization of a
205        // nested struct literal — instead construct it in Rust).
206        let locked_pkg = LockedPackage {
207            name: "critic".parse().unwrap(),
208            active_version: "0.1.0".parse().unwrap(),
209            source: "https://example.com/critic.git".parse().unwrap(),
210            installed_versions: vec![LockedVersion {
211                version: "0.1.0".parse().unwrap(),
212                rev: None,
213                resolved_commit: "0000000000000000000000000000000000000000".into(),
214                sha256: String::new(),
215                installed_at: UNIX_EPOCH + Duration::from_secs(1_747_180_800),
216            }],
217            plugin: None,
218            skill: Some(LockedSkill::new(
219                "deadbeef".into(),
220                SkillFrontmatterSnapshot {
221                    name: "critic".into(),
222                    description: "x".into(),
223                },
224            )),
225            synthesized_from: None,
226        };
227
228        let mut lf = LockFile::default();
229        lf.packages.push(locked_pkg);
230        // Lockfile lives at scope.lockfile_path() = tmp/tau-lock.toml
231        lf.save(&tmp.join("tau-lock.toml")).unwrap();
232
233        Scope::resolve(tmp).unwrap()
234    }

236    #[test]
237    fn returns_none_when_skill_absent() {
238        let tmp = tempdir().unwrap();
239        let tau_dir = tmp.path().join(".tau");
240        fs::create_dir_all(&tau_dir).unwrap();
241        fs::write(
242            tau_dir.join("config.toml"),
243            "schema_version = 3\nkind = \"project\"\ncreated_at = \"2026-05-14T00:00:00Z\"\ncreated_by_tau_version = \"0.0.0\"\n\n[sandbox]\nrequired_tier = \"none\"\n",
244        )
245        .unwrap();
246        // No lockfile at all — find_installed_skill returns None.
247        let scope = Scope::resolve(tmp.path()).unwrap();
248        let result = find_installed_skill(&scope, "anything").unwrap();
249        assert!(result.is_none());
250    }
251
252    #[test]
253    fn returns_skill_when_found() {
254        let tmp = tempdir().unwrap();
255        let scope = make_critic_scope(tmp.path());
256        let skill = find_installed_skill(&scope, "critic").unwrap().unwrap();
257        assert_eq!(skill.name.as_str(), "critic");
258        assert_eq!(skill.frontmatter.description, "x");
259        // install_path is <tmp>/.tau/packages/critic/0.1.0
260        assert!(skill.install_path.ends_with("packages/critic/0.1.0"));
261    }
262
263    #[test]
264    fn returns_err_when_install_path_missing() {
265        let tmp = tempdir().unwrap();
266        let scope = make_critic_scope(tmp.path());
267        // Remove the install path tau.toml
268        let toml_path = scope
269            .package_dir(&"critic".parse().unwrap(), &"0.1.0".parse().unwrap())
270            .join("tau.toml");
271        fs::remove_file(&toml_path).unwrap();
272        let result = find_installed_skill(&scope, "critic");
273        assert!(matches!(
274            result,
275            Err(FindSkillError::InstallPathMissing { .. })
276        ));
277    }
