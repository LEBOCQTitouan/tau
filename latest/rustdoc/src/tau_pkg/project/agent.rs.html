1//! Resolution from a project `[agents.<id>]` entry to a kernel-ready
2//! [`tau_domain::AgentDefinition`] paired with the resolved package's
3//! [`tau_domain::PackageManifest`].
4//!
5//! Per spec §3.3 and §3.4: walks the per-scope lockfile to find the
6//! highest-version installed package matching the entry's `package`
7//! reference, reads its manifest, verifies the declared `llm_backend`
8//! is installed, and materializes the `system_prompt` (inline or file).
9//! `requires.tools` resolve+install happens at the CLI call sites
10//! (cmd/run.rs, cmd/chat.rs, cmd/resolve.rs) before this function runs.
11

20/// Errors from [`build_agent_definition`].
21///
22/// Per spec §3.4. All variants carry the offending `agent_id` so the
23/// CLI can emit "agent X: ..." style messages without further
24/// plumbing.
25#[non_exhaustive]
26#[derive(Debug, thiserror::Error)]
27pub enum AgentResolutionError {
28    /// The agent's `package` reference doesn't match any installed
29    /// package by name in the resolved scope.
30    #[error(
31        "agent {agent_id:?} package {package:?} not installed in scope \
32         (run `tau install <url>`)"
33    )]
34    PackageNotFound {
35        /// Agent id from `[agents.<id>]`.
36        agent_id: String,
37        /// Raw `package = "..."` reference from the entry.
38        package: String,
39    },
40
41    /// Installed packages with the requested name exist but none satisfy
42    /// the version requirement.
43    #[error(
44        "agent {agent_id:?} package {package:?} matches no installed \
45         version satisfying requirement {req:?}"
46    )]
47    PackageVersionUnsatisfied {
48        /// Agent id from `[agents.<id>]`.
49        agent_id: String,
50        /// Raw `package = "..."` reference from the entry.
51        package: String,
52        /// Stringified semver requirement.
53        req: String,
54    },
55
56    /// The agent's `llm_backend` reference doesn't match any installed
57    /// package in the resolved scope. v0.1 does NOT enforce that the
58    /// package's `kind` is `llm-backend` — that check lands when the
59    /// runtime grows kind-aware loading (Phase 1+).
60    #[error(
61        "agent {agent_id:?} llm backend {backend:?} not installed \
62         (run `tau install <backend-url>`)"
63    )]
64    LlmBackendNotFound {
65        /// Agent id from `[agents.<id>]`.
66        agent_id: String,
67        /// Raw `llm_backend = "..."` from the entry.
68        backend: String,
69    },
70
71    /// Failed to read the resolved package's `tau.toml` manifest.
72    #[error("agent {agent_id:?}: failed to read manifest: {source}")]
73    ManifestRead {
74        /// Agent id from `[agents.<id>]`.
75        agent_id: String,
76        /// Underlying manifest read / validation error.
77        #[source]
78        source: ManifestReadError,
79    },
80
81    /// Failed to read a `prompt.system_file`.
82    #[error("agent {agent_id:?}: prompt file {path:?} read failed: {source}")]
83    PromptFileRead {
84        /// Agent id from `[agents.<id>]`.
85        agent_id: String,
86        /// Resolved (possibly relative-joined-with-project-root) path.
87        path: PathBuf,
88        /// Underlying I/O error.
89        #[source]
90        source: std::io::Error,
91    },
92
93    /// The `package = "name@^0.1"` string failed to parse into name +
94    /// semver requirement.
95    #[error("agent {agent_id:?}: package reference {package:?} could not be parsed: {message}")]
96    PackageParse {
97        /// Agent id from `[agents.<id>]`.
98        agent_id: String,
99        /// Raw `package = "..."` reference.
100        package: String,
101        /// Human-readable parse failure message.
102        message: String,
103    },
104
105    /// Failed to call into tau-pkg's registry layer (filesystem error,
106    /// schema mismatch, etc.).
107    #[error("agent {agent_id:?}: registry error: {source}")]
108    Registry {
109        /// Agent id from `[agents.<id>]`.
110        agent_id: String,
111        /// Underlying tau-pkg registry error.
112        #[source]
113        source: RegistryError,
114    },
115
116    /// Failed to construct an `AgentId` or `PackageName` from a string
117    /// the project config supplied. Usually means the ASCII kebab-case
118    /// invariant is violated.
119    #[error("agent {agent_id:?}: invalid identifier: {message}")]
120    InvalidIdentifier {
121        /// Agent id from `[agents.<id>]` (may itself be the invalid id).
122        agent_id: String,
123        /// Human-readable message describing what was invalid.
124        message: String,
125    },
126}

128/// Resolve a project `[agents.<id>]` entry to the kernel-ready
129/// `(AgentDefinition, PackageManifest)` pair.
130///
131/// Per spec §3.3 steps 1–6:
132///
133/// 1. Parse `entry.package` into `(name, semver req)`.
134/// 2. List installed packages in `scope`; pick the highest version of
135///    `name` satisfying the requirement.
136/// 3. Read the resolved package's manifest from
137///    `scope.package_dir(name, version)/tau.toml`.
138/// 4. Verify `entry.llm_backend` resolves to an installed package.
139/// 5. (Removed) `requires.tools` is no longer verified here; resolve+install
140///    happens at the CLI call sites before `build_agent_definition` is called.
141/// 6. Read `prompt.system` / `prompt.system_file` (path relative to
142///    `project_root`) into `Option<String>`.
143/// 7. Construct `AgentDefinition::new(...)` chained with
144///    `.with_system_prompt(...)` and `.with_config(...)`.
145///
146/// Returns the manifest alongside the definition because the kernel
147/// needs the manifest's capability declarations and dependencies for
148/// `Runtime::run` (per the tau-runtime amendments in Tasks 2 & 3).
149pub fn build_agent_definition(
150    entry: &AgentEntry,
151    project_root: &Path,
152    scope: &Scope,
153) -> Result<(AgentDefinition, PackageManifest), AgentResolutionError> {
154    // Step 1: parse `package = "name@^0.1"` into name + semver req.
155    let (raw_name, version_req) = parse_package_ref(&entry.package).map_err(|message| {
156        AgentResolutionError::PackageParse {
157            agent_id: entry.id.clone(),
158            package: entry.package.clone(),
159            message,
160        }
161    })?;
162    let pkg_name =
163        PackageName::from_str(&raw_name).map_err(|e| AgentResolutionError::InvalidIdentifier {
164            agent_id: entry.id.clone(),
165            message: format!("package name {raw_name:?}: {e}"),
166        })?;
167
168    // Step 2: list installed packages.
169    let installed = crate::list(scope).map_err(|source| AgentResolutionError::Registry {
170        agent_id: entry.id.clone(),
171        source,
172    })?;
173
174    // Filter to packages with the requested name. If none match, the
175    // package isn't installed at all.
176    let matching: Vec<&crate::LockedPackage> = installed
177        .iter()
178        .filter(|pkg| pkg.name == pkg_name)
179        .collect();
180
181    if matching.is_empty() {
182        return Err(AgentResolutionError::PackageNotFound {
183            agent_id: entry.id.clone(),
184            package: entry.package.clone(),
185        });
186    }
187
188    // Pick the highest installed version satisfying the requirement,
189    // searching across every matching package's `installed_versions`.
190    // (In practice there's exactly one matching package, since
191    // `LockedPackage::name` is a primary key — but the iteration is
192    // robust to a future schema change.)
193    let resolved_version = matching
194        .iter()
195        .flat_map(|pkg| pkg.installed_versions.iter())
196        .filter(|v| version_req.matches(&v.version))
197        .map(|v| &v.version)
198        .max();
199
200    let resolved_version = resolved_version
201        .ok_or_else(|| AgentResolutionError::PackageVersionUnsatisfied {
202            agent_id: entry.id.clone(),
203            package: entry.package.clone(),
204            req: version_req.to_string(),
205        })?
206        .clone();
207
208    // Step 3: read the manifest from `<scope>/packages/<name>/<version>/tau.toml`.
209    let manifest_path = scope
210        .package_dir(&pkg_name, &resolved_version)
211        .join("tau.toml");
212    let manifest = crate::read_manifest(&manifest_path).map_err(|source| {
213        AgentResolutionError::ManifestRead {
214            agent_id: entry.id.clone(),
215            source,
216        }
217    })?;
218
219    // Step 4: verify `llm_backend` is installed.
220    let llm_name = PackageName::from_str(&entry.llm_backend).map_err(|e| {
221        AgentResolutionError::InvalidIdentifier {
222            agent_id: entry.id.clone(),
223            message: format!("llm_backend {:?}: {e}", entry.llm_backend),
224        }
225    })?;
226    if !installed.iter().any(|pkg| pkg.name == llm_name) {
227        return Err(AgentResolutionError::LlmBackendNotFound {
228            agent_id: entry.id.clone(),
229            backend: entry.llm_backend.clone(),
230        });
231    }
232
233    // Step 5 (verify requires.tools) was removed in Tier 2 priority 5.
234    // resolve+install for requires.tools now lives at the CLI call sites
235    // (cmd/run.rs, cmd/chat.rs, cmd/resolve.rs). By the time
236    // build_agent_definition runs, the lockfile is authoritative.
237
238    // Step 6: read system prompt (inline / file / none).
239    let system_prompt = match &entry.prompt {
240        PromptEntry::None => None,
241        PromptEntry::Inline(s) => Some(s.clone()),
242        PromptEntry::File(rel_or_abs) => {
243            let path = if rel_or_abs.is_absolute() {
244                rel_or_abs.clone()
245            } else {
246                project_root.join(rel_or_abs)
247            };
248            let contents = std::fs::read_to_string(&path).map_err(|source| {
249                AgentResolutionError::PromptFileRead {
250                    agent_id: entry.id.clone(),
251                    path: path.clone(),
252                    source,
253                }
254            })?;
255            Some(contents)
256        }
257    };
258
259    // Step 7: construct AgentDefinition.
260    let agent_id =
261        AgentId::from_str(&entry.id).map_err(|e| AgentResolutionError::InvalidIdentifier {
262            agent_id: entry.id.clone(),
263            message: format!("agent id: {e}"),
264        })?;
265
266    let pkg_id = PackageId::new(manifest.name().clone(), manifest.version().clone());
267
268    let mut def = AgentDefinition::new(agent_id, entry.display_name.clone(), pkg_id, llm_name);
269    if let Some(prompt) = system_prompt {
270        def = def.with_system_prompt(prompt);
271    }
272    if !entry.config.is_empty() {
273        let domain_config = entry
274            .config
275            .iter()
276            .map(|(k, v)| (k.clone(), toml_value_to_domain_value(v.clone())))
277            .collect();
278        def = def.with_config(domain_config);
279    }
280
281    Ok((def, manifest))
282}

284/// Split `<name>@<semver-req>` into `(name, VersionReq)`. Bare `<name>`
285/// with no `@` defaults to `*`.
286fn parse_package_ref(package: &str) -> Result<(String, semver::VersionReq), String> {
287    let (name, req_str) = match package.split_once('@') {
288        Some((n, r)) => (n.trim().to_string(), r.trim().to_string()),
289        None => (package.trim().to_string(), "*".to_string()),
290    };
291    if name.is_empty() {
292        return Err("package name half is empty".to_string());
293    }
294    let req = semver::VersionReq::parse(&req_str)
295        .map_err(|e| format!("bad semver requirement {req_str:?}: {e}"))?;
296    Ok((name, req))
297}

299/// Convert a `toml::Value` to a `tau_domain::Value` for the
300/// `[agents.<id>.config]` passthrough. Best-effort — at v0.1 the runtime
301/// only validates these values per-plugin, so structural fidelity is
302/// what matters.
303fn toml_value_to_domain_value(v: toml::Value) -> Value {
304    match v {
305        toml::Value::String(s) => Value::String(s),
306        toml::Value::Integer(i) => Value::Integer(i),
307        toml::Value::Float(f) => Value::Float(f),
308        toml::Value::Boolean(b) => Value::Bool(b),
309        toml::Value::Datetime(dt) => Value::String(dt.to_string()),
310        toml::Value::Array(arr) => {
311            Value::Array(arr.into_iter().map(toml_value_to_domain_value).collect())
312        }
313        toml::Value::Table(t) => Value::Object(
314            t.into_iter()
315                .map(|(k, v)| (k, toml_value_to_domain_value(v)))
316                .collect(),
317        ),
318    }

351    /// Materialize an installed package (lockfile entry + on-disk
352    /// `tau.toml`) at `scope.package_dir(name, version)`. Uses raw TOML
353    /// I/O because `LockedPackage` / `LockedVersion` are
354    /// `#[non_exhaustive]` and can't be struct-literal-constructed from
355    /// outside `tau_pkg`.
356    fn install_fixture(scope: &Scope, name: &str, version: &str, kind: &str, source_url: &str) {
357        // 1. Write the package's tau.toml to the canonical install path.
358        let pkg_name: PackageName = name.parse().unwrap();
359        let ver: tau_domain::Version = version.parse().unwrap();
360        let pkg_dir = scope.package_dir(&pkg_name, &ver);
361        fs::create_dir_all(&pkg_dir).unwrap();
362        let manifest = format!(
363            r#"name = "{name}"
364version = "{version}"
365description = "fixture"
366authors = ["tester <test@example.com>"]
367source = "{source_url}"
368kind = "{kind}"
369dependencies = []
370capabilities = []
371"#
372        );
373        fs::write(pkg_dir.join("tau.toml"), manifest).unwrap();
374
375        // 2. Append/upsert this version into the lockfile, preserving any
376        //    existing entries (so multiple install_fixture calls compose).
377        let lockfile_path = scope.lockfile_path();
378        let existing = if lockfile_path.exists() {
379            fs::read_to_string(&lockfile_path).unwrap()
380        } else {
381            String::new()
382        };
383
384        // Hand-rolled RFC3339 timestamp at second precision, anchored at
385        // a fixed instant. Tests don't care about the actual time;
386        // `humantime-serde` (used by tau_pkg's lockfile) accepts any
387        // RFC3339 string. Using a fixed value also makes test output
388        // deterministic in case anyone snapshots a lockfile.
389        let now_rfc3339 = "2026-04-28T00:00:00Z";
390
391        // We hand-author the TOML rather than going through `LockFile` /
392        // `LockedPackage` because both are `#[non_exhaustive]` outside
393        // tau_pkg. The schema is stable (Task 6) so this is safe.
394        let resolved_commit = "0".repeat(40);
395        let new_entry = format!(
396            r#"
397[[package]]
398name = "{name}"
399active_version = "{version}"
400source = "{source_url}"
401
402[[package.versions]]
403version = "{version}"
404resolved_commit = "{resolved_commit}"
405sha256 = ""
406installed_at = "{now_rfc3339}"
407"#
408        );
409
410        let new_lockfile = if existing.is_empty() {
411            format!(
412                r#"schema_version = 1
413generated_by_tau_version = "0.0.0"
414generated_at = "{now_rfc3339}"
415{new_entry}"#
416            )
417        } else {
418            // Naive append: each [[package]] / [[package.versions]] block is
419            // additive, so concatenating works for our test fixture needs.
420            // If we add the same name twice (different versions), both
421            // entries land in the lockfile — `tau_pkg::list` returns them
422            // in lockfile order, and `build_agent_definition` walks all of
423            // them when version-matching.
424            format!("{existing}\n{new_entry}")
425        };
426        fs::write(&lockfile_path, new_lockfile).unwrap();
427    }

609    #[test]
610    fn build_agent_definition_does_not_check_requires_tools() {
611        // Step 5 verify was removed in Tier 2 priority 5 — the resolve+install
612        // happens at cmd/run.rs / cmd/chat.rs / cmd/resolve.rs before
613        // build_agent_definition is called. By the time we reach this code path,
