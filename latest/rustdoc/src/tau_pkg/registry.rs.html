1//! Read-only accessors over the per-scope lockfile.
2//!
3//! [`list`] and [`get`] are thin wrappers over [`LockFile::load`] —
4//! they hide the lockfile-loading mechanism from external callers
5//! (tau-cli, future tau-runtime) so consumers don't reach into
6//! [`crate::lockfile`]'s internals.
7//!
8//! Note: every call re-reads the lockfile from disk. For repeated
9//! lookups, callers should `LockFile::load` directly and reuse the
10//! result.
11
12use tau_domain::PackageName;
13
14use crate::error::RegistryError;
15use crate::lockfile::{LockFile, LockedPackage};
16use crate::scope::Scope;
17
18/// List every installed package in `scope`, in lockfile order.
19///
20/// Returns `Ok(Vec::new())` if the lockfile doesn't exist (lazy
21/// creation per [`LockFile::load`]).
22///
23/// # Example
24///
25/// ```ignore
26/// // `Scope` is `#[non_exhaustive]`; constructed via Scope::resolve / global / new_project.
27/// use tau_pkg::{list, Scope};
28///
29/// let scope = Scope::global().unwrap();
30/// for pkg in list(&scope).unwrap() {
31///     println!("{}@{}", pkg.name, pkg.active_version);
32/// }
33/// ```
34pub fn list(scope: &Scope) -> Result<Vec<LockedPackage>, RegistryError> {
35    Ok(LockFile::load(&scope.lockfile_path())?.packages)
36}
37
38/// Look up a single installed package by name.
39///
40/// Returns `Ok(None)` if the package is not in the lockfile (or the
41/// lockfile doesn't exist — lazy creation returns the default empty
42/// lockfile).
43///
44/// # Example
45///
46/// ```ignore
47/// use tau_pkg::{get, Scope};
48///
49/// let scope = Scope::global().unwrap();
50/// let name: tau_domain::PackageName = "acme-tool".parse().unwrap();
51/// if let Some(pkg) = get(&scope, &name).unwrap() {
52///     println!("{} installed at {}", pkg.name, pkg.active_version);
53/// }
54/// ```
55pub fn get(scope: &Scope, name: &PackageName) -> Result<Option<LockedPackage>, RegistryError> {
56    Ok(LockFile::load(&scope.lockfile_path())?.find(name).cloned())
57}

59#[cfg(test)]
60mod tests {
61    use super::*;
62
63    use std::time::{Duration, UNIX_EPOCH};
64
65    use tempfile::TempDir;
66
67    use crate::lockfile::LockedVersion;
68
69    fn make_scope(tmp: &TempDir) -> Scope {
70        Scope::global_at(tmp.path().join("tau-home")).unwrap()
71    }
72
73    fn fixture_pkg(name: &str, version: &str) -> LockedPackage {
74        LockedPackage {
75            name: name.parse().unwrap(),
76            active_version: version.parse().unwrap(),
77            source: "https://x.com/y.git".parse().unwrap(),
78            installed_versions: vec![LockedVersion {
79                version: version.parse().unwrap(),
80                rev: None,
81                resolved_commit: "0123456789abcdef0123456789abcdef01234567".into(),
82                sha256: String::new(),
83                installed_at: UNIX_EPOCH + Duration::from_secs(1_700_000_000),
84            }],
85            plugin: None,
86            skill: None,
87            synthesized_from: None,
88        }
89    }
90
91    #[test]
92    fn list_returns_empty_when_lockfile_missing() {
93        let tmp = TempDir::new().unwrap();
94        let scope = make_scope(&tmp);
95        assert!(list(&scope).unwrap().is_empty());
96    }
97
98    #[test]
99    fn list_returns_all_packages_in_lockfile_order() {
100        let tmp = TempDir::new().unwrap();
101        let scope = make_scope(&tmp);
102
103        let mut lf = LockFile::default();
104        lf.upsert(fixture_pkg("aaa-pkg", "1.0.0"));
105        lf.upsert(fixture_pkg("bbb-pkg", "2.0.0"));
106        lf.save(&scope.lockfile_path()).unwrap();
107
108        let packages = list(&scope).unwrap();
109        assert_eq!(packages.len(), 2);
110        assert_eq!(packages[0].name.as_str(), "aaa-pkg");
111        assert_eq!(packages[1].name.as_str(), "bbb-pkg");
112    }
113
114    #[test]
115    fn get_returns_none_for_unknown_package() {
116        let tmp = TempDir::new().unwrap();
117        let scope = make_scope(&tmp);
118        let name: PackageName = "ghost".parse().unwrap();
119        assert!(get(&scope, &name).unwrap().is_none());
120    }
121
122    #[test]
123    fn get_returns_some_for_known_package() {
124        let tmp = TempDir::new().unwrap();
125        let scope = make_scope(&tmp);
126
127        let mut lf = LockFile::default();
128        lf.upsert(fixture_pkg("acme-tool", "1.2.3"));
129        lf.save(&scope.lockfile_path()).unwrap();
130
131        let name: PackageName = "acme-tool".parse().unwrap();
132        let pkg = get(&scope, &name).unwrap();
133        assert!(pkg.is_some());
134        assert_eq!(pkg.unwrap().active_version.to_string(), "1.2.3");
135    }
136}
