1//! Layer 2 install-time cross-check: spawn the plugin binary, perform the
2//! `meta.handshake` RPC, and compare the capabilities the binary claims
3//! (via `tool.describe_capabilities`) against what the manifest declares.
4//!
5//! For tool-port plugins the check is bidirectional — binary-claims-extra
6//! and manifest-declares-unused both hard-fail. For LLM-backend and storage
7//! plugins the binary-side wire mechanism is deferred per ADR-0016 Decision 1;
8//! those paths return the manifest's capability list verbatim.
9//!
10//! The public entry point is [`cross_check_plugin_capabilities`].
11

24const TOOL_DESCRIBE_CAPABILITIES_METHOD: &str = "tool.describe_capabilities";

26/// Timeout for the `meta.handshake` response. A plugin that spawns but never
27/// writes its handshake reply would otherwise stall `tau install` indefinitely.
28const HANDSHAKE_TIMEOUT: Duration = Duration::from_secs(10);
29
30/// Per-method timeout for each `tool.describe_capabilities` response.
31const DESCRIBE_TIMEOUT: Duration = Duration::from_secs(5);
32
33/// Errors that may arise during a plugin cross-check.
34///
35/// `#[non_exhaustive]`: future verification layers may add variants without
36/// breaking callers.
37#[non_exhaustive]
38#[derive(Debug, thiserror::Error)]
39pub enum CrossCheckError {
40    /// The plugin binary could not be launched.
41    #[error("plugin spawn failed: {0}")]
42    SpawnFailed(String),
43    /// The `meta.handshake` round-trip failed (encode / IO / decode /
44    /// protocol error).
45    #[error("plugin handshake failed: {0}")]
46    HandshakeFailed(String),
47    /// The binary reports a capability that the manifest does not declare.
48    #[error(
49        "plugin '{plugin}' declares capability {claimed:?} via tool.describe_capabilities \
50         but manifest does not include it"
51    )]
52    BinaryClaimsExtra {
53        /// Plugin name (from the handshake response).
54        plugin: String,
55        /// The extra capability the binary claims.
56        claimed: Capability,
57    },
58    /// The manifest declares a capability that the binary never requests.
59    #[error(
60        "manifest of '{plugin}' declares capability {declared:?} but binary does not request it"
61    )]
62    ManifestDeclaresUnused {
63        /// Plugin name (from the handshake response).
64        plugin: String,
65        /// The capability declared in the manifest but not requested by the binary.
66        declared: Capability,
67    },
68}

70/// Spawn the plugin binary, perform the `meta.handshake` RPC, enumerate
71/// capabilities via `tool.describe_capabilities` (tool-port plugins only),
72/// and return the deduplicated [`CapabilityShape`] set.
73///
74/// For LLM-backend and storage plugins the binary-side wire mechanism is
75/// deferred (ADR-0016 Decision 1); those paths return
76/// `manifest.capabilities()` mapped to [`CapabilityShape`] values.
77///
78/// # Errors
79///
80/// Returns [`CrossCheckError::SpawnFailed`] if the binary cannot be launched,
81/// [`CrossCheckError::HandshakeFailed`] if the handshake round-trip fails, and
82/// [`CrossCheckError::BinaryClaimsExtra`] /
83/// [`CrossCheckError::ManifestDeclaresUnused`] if the capability sets diverge.
84///
85/// **Tests:** the `diff_capabilities` helper + error display are unit-tested
86/// in this file. The end-to-end function `cross_check_plugin_capabilities`
87/// itself is exercised via real-binary integration tests in `tau-plugin-compat`
88/// (Tasks 6-9 of sub-project B's plan).
89pub async fn cross_check_plugin_capabilities(
90    binary_path: &Path,
91    manifest: &PackageManifest,
92) -> Result<Vec<CapabilityShape>, CrossCheckError> {
93    // ── 1. Spawn ─────────────────────────────────────────────────────────────
94    let mut child = Command::new(binary_path)
95        .stdin(std::process::Stdio::piped())
96        .stdout(std::process::Stdio::piped())
97        .stderr(std::process::Stdio::null())
98        .kill_on_drop(true) // ensure child is reaped on every early-return path
99        .spawn()
100        .map_err(|e| CrossCheckError::SpawnFailed(format!("{e}")))?;
101
102    let stdin = child
103        .stdin
104        .take()
105        .ok_or_else(|| CrossCheckError::SpawnFailed("child stdin not piped".to_string()))?;
106    let stdout = child
107        .stdout
108        .take()
109        .ok_or_else(|| CrossCheckError::SpawnFailed("child stdout not piped".to_string()))?;
110
111    let mut writer = FramedWriter::new(stdin);
112    let mut reader = FramedReader::new(stdout, FramerOptions::default());
113
114    // ── 2. Handshake ─────────────────────────────────────────────────────────
115    let port = manifest
116        .plugin()
117        .map(|p| p.provides)
118        .unwrap_or(PortKind::Tool);
119
120    let nanos = SystemTime::now()
121        .duration_since(UNIX_EPOCH)
122        .unwrap_or_default()
123        .as_nanos();
124
125    // Send `{}` rather than Null: plugin config structs typically don't
126    // accept null as the top-level value (only objects), so a Null here
127    // causes the SDK runner to fail config deserialization and exit
128    // silently — the host then sees EOF on the next read. An empty
129    // object lets every plugin with `#[derive(Default)]`-style configs
130    // construct their defaults.
131    let handshake_req = HandshakeRequest::new(
132        PROTOCOL_VERSION.to_string(),
133        port,
134        TraceContext::new(
135            format!("install-cross-check-{nanos}"),
136            "install".to_string(),
137            "cross-check".to_string(),
138        ),
139        serde_json::Value::Object(serde_json::Map::new()),
140    );
141
142    // The wire shape for `meta.handshake` params is a 1-element array
143    // (`Vec<HandshakeRequest>`) per the SDK's matching decoder; see
144    // `tau_plugin_sdk::handshake::drive_handshake` and the parallel
145    // call site at `tau_runtime::plugin_host::handshake`.
146    let params = rmp_serde::to_vec(&vec![&handshake_req])
147        .map_err(|e| CrossCheckError::HandshakeFailed(format!("encode handshake request: {e}")))?;
148
149    let req_frame = Frame::Request {
150        id: 1,
151        method: meta::HANDSHAKE_METHOD.to_string(),
152        params,
153    };
154    let req_bytes = req_frame
155        .encode()
156        .map_err(|e| CrossCheckError::HandshakeFailed(format!("frame encode: {e}")))?;
157
158    writer
159        .write_frame(&req_bytes)
160        .await
161        .map_err(|e| CrossCheckError::HandshakeFailed(format!("write frame: {e}")))?;
162
163    // ── 3. Read handshake response ────────────────────────────────────────────
164    let body = timeout(HANDSHAKE_TIMEOUT, reader.next_frame())
165        .await
166        .map_err(|_| {
167            CrossCheckError::HandshakeFailed("handshake response timed out after 10s".to_string())
168        })?
169        .map_err(|e| CrossCheckError::HandshakeFailed(format!("read frame: {e}")))?
170        .ok_or_else(|| {
171            CrossCheckError::HandshakeFailed("EOF before handshake response".to_string())
172        })?;
173
174    let frame = Frame::decode(&body)
175        .map_err(|e| CrossCheckError::HandshakeFailed(format!("decode frame: {e}")))?;
176
177    let response_bytes = match frame {
178        Frame::Response {
179            id: 1,
180            error: None,
181            result: Some(bytes),
182        } => bytes,
183        Frame::Response {
184            error: Some(env), ..
185        } => {
186            return Err(CrossCheckError::HandshakeFailed(format!(
187                "plugin returned RPC error {}: {}",
188                env.code, env.message
189            )));
190        }
191        Frame::Response {
192            id,
193            error: None,
194            result: None,
195            ..
196        } => {
197            return Err(CrossCheckError::HandshakeFailed(format!(
198                "handshake response id={id} has neither result nor error"
199            )));
200        }
201        Frame::Response { id, .. } => {
202            return Err(CrossCheckError::HandshakeFailed(format!(
203                "unexpected response id={id} (expected 1)"
204            )));
205        }
206        other => {
207            return Err(CrossCheckError::HandshakeFailed(format!(
208                "expected Response frame, got {other:?}"
209            )));
210        }
211    };
212
213    let response: HandshakeResponse = rmp_serde::from_slice(&response_bytes)
214        .map_err(|e| CrossCheckError::HandshakeFailed(format!("decode HandshakeResponse: {e}")))?;
215
216    let plugin_name = response.plugin_name.clone();
217
218    // ── 4. Enumerate capabilities ─────────────────────────────────────────────
219    let binary_caps: Vec<Capability> = match response.provides {
220        PortKind::Tool => {
221            enumerate_tool_capabilities(&mut writer, &mut reader, &response.methods).await?
222        }
223        PortKind::LlmBackend | PortKind::Storage => {
224            tracing::debug!(
225                plugin = %plugin_name,
226                provides = %response.provides,
227                "manifest-only capability path: binary-side wire mechanism deferred \
228                 per ADR-0016 Decision 1"
229            );
230            manifest.capabilities().to_vec()
231        }
232        other => {
233            tracing::debug!(
234                plugin = %plugin_name,
235                provides = %other,
236                "unrecognised port kind — returning manifest capabilities verbatim"
237            );
238            manifest.capabilities().to_vec()
239        }
240    };
241
242    // ── 5. Best-effort shutdown ───────────────────────────────────────────────
243    // Encoding an empty Vec to msgpack is infallible (produces [0x90]).
244    let empty_params = rmp_serde::to_vec::<Vec<()>>(&Vec::new())
245        .expect("encoding empty Vec to msgpack is infallible");
246    let shutdown = Frame::Notification {
247        method: meta::SHUTDOWN_METHOD.to_string(),
248        params: empty_params,
249    };
250    if let Ok(shutdown_bytes) = shutdown.encode() {
251        let _ = writer.write_frame(&shutdown_bytes).await;
252    }
253    let _ = child.wait().await;
254
255    // ── 6. Diff ───────────────────────────────────────────────────────────────
256    diff_capabilities(&plugin_name, &binary_caps, manifest.capabilities())?;
257
258    // ── 7. Map to CapabilityShape ─────────────────────────────────────────────
259    let shapes: Vec<CapabilityShape> = binary_caps
260        .iter()
261        .map(|c| c.required_shape())
262        .collect::<std::collections::HashSet<_>>()
263        .into_iter()
264        .collect();
265
266    Ok(shapes)
267}

269/// Issue `tool.describe_capabilities` for every method in `methods` that
270/// begins with `"tool."`, union the results, and deduplicate.
271async fn enumerate_tool_capabilities<W, R>(
272    writer: &mut FramedWriter<W>,
273    reader: &mut FramedReader<R>,
274    methods: &[String],
275) -> Result<Vec<Capability>, CrossCheckError>
276where
277    W: tokio::io::AsyncWrite + Unpin,
278    R: tokio::io::AsyncRead + Unpin,
279{
280    let mut all: Vec<Capability> = Vec::new();
281
282    let params_bytes = rmp_serde::to_vec::<Vec<()>>(&Vec::new()).map_err(|e| {
283        CrossCheckError::HandshakeFailed(format!("encode tool.describe_capabilities params: {e}"))
284    })?;
285
286    for (id, method) in (2_u32..).zip(methods.iter().filter(|m| m.starts_with("tool."))) {
287        let req_frame = Frame::Request {
288            id,
289            method: TOOL_DESCRIBE_CAPABILITIES_METHOD.to_string(),
290            params: params_bytes.clone(),
291        };
292        let req_bytes = req_frame
293            .encode()
294            .map_err(|e| CrossCheckError::HandshakeFailed(format!("frame encode: {e}")))?;
295
296        writer
297            .write_frame(&req_bytes)
298            .await
299            .map_err(|e| CrossCheckError::HandshakeFailed(format!("write frame: {e}")))?;
300
301        let body = timeout(DESCRIBE_TIMEOUT, reader.next_frame())
302            .await
303            .map_err(|_| {
304                CrossCheckError::HandshakeFailed(format!(
305                    "tool.describe_capabilities response for '{method}' timed out after 5s"
306                ))
307            })?
308            .map_err(|e| CrossCheckError::HandshakeFailed(format!("read frame: {e}")))?
309            .ok_or_else(|| {
310                CrossCheckError::HandshakeFailed(
311                    "EOF while reading tool.describe_capabilities response".to_string(),
312                )
313            })?;
314
315        let frame = Frame::decode(&body)
316            .map_err(|e| CrossCheckError::HandshakeFailed(format!("decode frame: {e}")))?;
317
318        match frame {
319            Frame::Response {
320                error: Some(env), ..
321            } => {
322                // Warn rather than hard-fail: plugins predating priority 12 may
323                // not implement `tool.describe_capabilities`. Upgrading to a
324                // hard-fail is tracked as a separate plan item. A malicious
325                // plugin could exploit this to hide capabilities, so the warn
326                // level ensures the gap is visible in logs.
327                tracing::warn!(
328                    method = %method,
329                    code = env.code,
330                    message = %env.message,
331                    "tool.describe_capabilities returned RPC error — skipping method \
332                     (backward-compat tolerance for pre-priority-12 plugins)"
333                );
334            }
335            Frame::Response {
336                result: Some(bytes),
337                ..
338            } => {
339                let caps: Vec<Capability> = rmp_serde::from_slice(&bytes).map_err(|e| {
340                    CrossCheckError::HandshakeFailed(format!(
341                        "decode Vec<Capability> from tool.describe_capabilities: {e}"
342                    ))
343                })?;
344                all.extend(caps);
345            }
346            Frame::Response {
347                error: None,
348                result: None,
349                ..
350            } => {
351                tracing::debug!(
352                    method = %method,
353                    "tool.describe_capabilities returned empty response — skipping"
354                );
355            }
356            other => {
357                tracing::debug!(
358                    method = %method,
359                    frame = ?other,
360                    "unexpected frame type for tool.describe_capabilities — skipping"
361                );
362            }
363        }
364    }
365
366    // Deduplicate via PartialEq (Capability is PartialEq but not Hash; O(n²)
367    // is fine for the small counts we expect, and avoids relying on Debug
368    // stability as a dedup key).
369    let mut deduped: Vec<Capability> = Vec::new();
370    for cap in all {
371        if !deduped.contains(&cap) {
372            deduped.push(cap);
373        }
374    }
375
376    Ok(deduped)
377}

379/// Bidirectional capability set-diff. Returns the first divergence found.
380///
381/// * [`CrossCheckError::BinaryClaimsExtra`] — binary has a capability the
382///   manifest does not declare.
383/// * [`CrossCheckError::ManifestDeclaresUnused`] — manifest declares a
384///   capability the binary never requests.
385///
386/// O(n*m); fine for the small capability counts we expect (typically < 10 per plugin).
387fn diff_capabilities(
388    plugin_name: &str,
389    binary: &[Capability],
390    manifest: &[Capability],
391) -> Result<(), CrossCheckError> {
392    for claimed in binary {
393        if !manifest.contains(claimed) {
394            return Err(CrossCheckError::BinaryClaimsExtra {
395                plugin: plugin_name.to_string(),
396                claimed: claimed.clone(),
397            });
398        }
399    }
400    for declared in manifest {
401        if !binary.contains(declared) {
402            return Err(CrossCheckError::ManifestDeclaresUnused {
403                plugin: plugin_name.to_string(),
404                declared: declared.clone(),
405            });
406        }
407    }
408    Ok(())
409}

415    /// Deserialize a `Capability` from a JSON literal.
416    /// Required because `FsCapability::Read` et al. are `#[non_exhaustive]`
417    /// variants — struct-expression construction is blocked outside the
418    /// defining crate (E0639).
419    fn cap_from_json(json: &str) -> Capability {
420        serde_json::from_str(json).expect("valid capability JSON")
421    }
422
423    fn fs_read() -> Capability {
424        cap_from_json(r#"{"kind":"fs.read","paths":["/tmp/**"]}"#)
425    }
426
427    fn net_http() -> Capability {
428        cap_from_json(r#"{"kind":"net.http","hosts":["api.example.com"],"methods":["GET"]}"#)
429    }

488    #[test]
489    fn cross_check_error_is_non_exhaustive_via_match() {
490        // In-crate tests can exhaust the match without a wildcard (same crate,
491        // so the compiler sees all current variants). The real proof that
492        // `#[non_exhaustive]` is present is that external crates CANNOT omit
493        // the wildcard. Here we add `#[allow(unreachable_patterns)]` to keep the
494        // wildcard arm as documentation of that contract without a compile error.
495        let err = CrossCheckError::SpawnFailed("bad".to_string());
496        #[allow(unreachable_patterns)]
497        let _handled = match err {
498            CrossCheckError::SpawnFailed(_) => "spawn",
499            CrossCheckError::HandshakeFailed(_) => "handshake",
500            CrossCheckError::BinaryClaimsExtra { .. } => "extra",
501            CrossCheckError::ManifestDeclaresUnused { .. } => "unused",
502            _ => "future-variant",
503        };
504    }
