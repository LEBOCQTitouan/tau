1//! Lockfile types — `tau-lock.toml` schema.
2//!
3//! The lockfile records every installed package per scope:
4//!
5//! - **Project scope:** `<project>/tau-lock.toml` (lives at the project
6//!   root and is **committed** to the project's git repository).
7//! - **Global scope:** `<scope.path()>/tau-lock.toml` (typically
8//!   `~/.tau/tau-lock.toml`; **local state**, not committed).
9//!
10//! TOML round-trip uses `humantime-serde` so timestamps are RFC-3339
11//! strings (human-readable in diffs). `schema_version` is bumped only
12//! on breaking changes; lockfiles with a newer version than this tau
13//! version supports are rejected via [`crate::RegistryError::SchemaTooNew`].
14//!
15//! The `sha256` slot on [`LockedVersion`] is populated by
16//! `install_with_options` after source materialization (Task 2).
17//! The `binary_sha256` slot on [`LockedPlugin`] is populated by
18//! `install_with_options` after `cargo build` succeeds (Task 2).
19//!
20//! [`LockFile::load`]/`save`/`find`/`upsert`/`remove` land in
21//! Task 7. This file (Task 6) defines only the data shapes + `Default`.
22

33/// Maximum `LockFile::schema_version` this tau version recognizes.
34/// A `tau-lock.toml` with a higher value is rejected by
35/// `LockFile::load` (Task 7) via `RegistryError::SchemaTooNew`.
36///
37/// History:
38/// - `1` — v0.1: `LockedPackage` had no `plugin` field.
39/// - `2` — Plugin loading: `LockedPackage::plugin: Option<LockedPlugin>`
40///   added. v1 lockfiles auto-upgrade to v2 on load (the `plugin` field
41///   defaults to `None` for legacy entries via `#[serde(default)]`).
42/// - `3` — Content hashing: `LockedPlugin::binary_sha256: String` and
43///   `LockedVersion::sha256` now populated by `install_with_options`.
44///   v2 lockfiles auto-upgrade to v3 on load (`binary_sha256` defaults
45///   to `""` for legacy entries via `#[serde(default)]`; these are
46///   flagged `unverified` by `tau verify`).
47/// - `4` — Sandbox shapes: `LockedPlugin::required_shapes: Vec<CapabilityShape>`
48///   added. v3 lockfiles auto-upgrade to v4 on load (`required_shapes`
49///   defaults to `vec![]` for legacy entries via `#[serde(default)]`;
50///   the runtime falls back to deriving shapes from the manifest at
51///   startup and logs a warning per affected plugin).
52/// - `5` — Skill metadata: `LockedPackage::skill: Option<LockedSkill>`
53///   added. v4 lockfiles auto-upgrade to v5 on load (`skill` defaults
54///   to `None` for legacy entries via `#[serde(default)]`; skill
55///   packages installed before the upgrade surface as "unverified"
56///   via `tau verify` until reinstalled).
57/// - `6` — Synthesized-manifest provenance:
58///   `LockedPackage::synthesized_from: Option<SynthesizedSource>` added.
59///   v5 lockfiles auto-upgrade to v6 on load (`synthesized_from` defaults
60///   to `None` for legacy entries via `#[serde(default)]`; packages
61///   installed before the upgrade retain `None` provenance, which is
62///   correct — they were installed from tau-native sources).
63pub const MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION: u32 = 6;

65/// Schema for `tau-lock.toml`.
66///
67/// Project scope: lives at `<project>/tau-lock.toml` (committed).
68/// Global scope: lives at `~/.tau/tau-lock.toml` (local state).
69///
70/// # Example
71///
72/// ```ignore
73/// // `LockFile` is `#[non_exhaustive]`; constructed via [`LockFile::default`].
74/// use tau_pkg::lockfile::LockFile;
75///
76/// let lf = LockFile::default();
77/// assert_eq!(lf.schema_version, 6);
78/// assert!(lf.packages.is_empty());
79/// ```
80#[non_exhaustive]
81#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
82pub struct LockFile {
83    /// Schema version. Currently `6`. Bumped on breaking changes only.
84    /// v1–v5 lockfiles are accepted on load and auto-upgraded to v6
85    /// on the next save (v1→v2: legacy entries get `plugin = None`;
86    /// v2→v3: `LockedPlugin` entries get `binary_sha256 = ""`
87    /// defaulted via `#[serde(default)]`; v3→v4: `LockedPlugin`
88    /// entries get `required_shapes = []` defaulted via
89    /// `#[serde(default)]` with a per-plugin warning emitted;
90    /// v4→v5: `LockedPackage` entries get `skill = None` defaulted
91    /// via `#[serde(default)]` with a once-per-process warn emitted;
92    /// v5→v6: `LockedPackage` entries get `synthesized_from = None`
93    /// defaulted via `#[serde(default)]`).
94    pub schema_version: u32,
95    /// `CARGO_PKG_VERSION` of the tau-pkg crate that last wrote this file.
96    pub generated_by_tau_version: String,
97    /// Timestamp of the last [`Self::default`] or `save()` call. Set to
98    /// `SystemTime::now()` on construction so a freshly-defaulted but
99    /// not-yet-saved `LockFile` already carries a value.
100    #[serde(with = "humantime_serde")]
101    pub generated_at: SystemTime,
102    /// Installed packages. Renamed to `[[package]]` in TOML output for
103    /// natural diff output.
104    #[serde(default, rename = "package")]
105    pub packages: Vec<LockedPackage>,
106}

120pub enum SynthesizedSource {
121    /// The package was installed from a vanilla Anthropic Agent Skills
122    /// source (SKILL.md only; no tau.toml in the source tree).
123    Anthropic,
124}

126/// One installed package's lockfile entry.
127///
128/// `active_version` is the version the runtime loads when no version
129/// pin is supplied. `installed_versions` records every version
130/// currently materialized on disk for this package (multi-version
131/// cohabitation per scope).
132///
133/// # Example
134///
135/// ```ignore
136/// // `LockedPackage` is `#[non_exhaustive]`; in tests, construct via
137/// // struct literal from within the crate. External callers receive
138/// // values from `LockFile::find` / `list` / `get`.
139/// ```
140#[non_exhaustive]
141#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
142pub struct LockedPackage {
143    /// Validated package name, from `tau_domain::PackageName`.
144    pub name: PackageName,
145    /// The version the runtime loads by default for this package.
146    pub active_version: Version,
147    /// Where the package was fetched from.
148    pub source: PackageSource,
149    /// Every version currently installed on disk. Renamed to
150    /// `[[package.versions]]` in TOML output.
151    #[serde(default, rename = "versions")]
152    pub installed_versions: Vec<LockedVersion>,
153    /// Plugin metadata recorded at install time. `None` for data-only
154    /// packages and for legacy v1 lockfile entries (which had no
155    /// `plugin` field; `#[serde(default)]` populates it as `None` on
156    /// auto-upgrade).
157    ///
158    /// Added in lockfile schema v2.
159    #[serde(default, skip_serializing_if = "Option::is_none")]
160    pub plugin: Option<LockedPlugin>,
161    /// Skill metadata recorded at install time. `None` for non-skill
162    /// packages and for legacy v4 lockfile entries (which had no
163    /// `skill` field; `#[serde(default)]` populates it as `None` on
164    /// auto-upgrade).
165    ///
166    /// Added in lockfile schema v5.
167    #[serde(default, skip_serializing_if = "Option::is_none")]
168    pub skill: Option<LockedSkill>,
169    /// Provenance: `Some(_)` if this package's manifest was synthesized
170    /// at install time from a non-tau source format (e.g. Anthropic
171    /// Agent Skills). `None` for packages installed from sources
172    /// that already had a `tau.toml`.
173    ///
174    /// Added in lockfile schema v6 (Skills-5). v5 entries deserialize
175    /// as `None`.
176    #[serde(default, skip_serializing_if = "Option::is_none")]
177    pub synthesized_from: Option<SynthesizedSource>,
178}

180/// Recorded build artifact for a plugin package.
181///
182/// Written by [`crate::install_with_options`] when the installed
183/// package's manifest carries a `[plugin]` table and the cargo build
184/// step succeeded. Consumed by `tau-runtime` to spawn the plugin
185/// subprocess at run time.
186///
187/// `#[non_exhaustive]`: future fields (e.g. `sha256` of the binary,
188/// build features, toolchain version) are non-breaking.
189///
190/// # Example
191///
192/// ```ignore
193/// // `LockedPlugin` is `#[non_exhaustive]`; constructed by the install
194/// // lifecycle (Task 12). External callers (notably tau-runtime
195/// // integration tests that synthesize a lockfile against pre-built
196/// // binaries) build it via [`LockedPlugin::new`].
197/// ```
198#[non_exhaustive]
199#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
200pub struct LockedPlugin {
201    /// Copy of the `[plugin]` table from the package's `tau.toml` at
202    /// install time. Frozen here so `tau-runtime` doesn't need to
203    /// re-read the manifest from the package source tree.
204    pub manifest: PluginManifest,
205    /// Canonical absolute path to the built binary
206    /// (e.g. `<pkg_dir>/target/release/<bin>`). Set via
207    /// `Path::canonicalize` so symlinks and relative components are
208    /// resolved at install time.
209    pub binary_path: PathBuf,
210    /// When the binary was built (the timestamp of the `cargo build`
211    /// step that produced it).
212    #[serde(with = "humantime_serde")]
213    pub built_at: SystemTime,
214    /// SHA-256 of the built binary at `binary_path`. Populated by
215    /// `install_with_options` after `cargo build` succeeds. Empty
216    /// for v2-leftover entries (informational `unverified` status
217    /// from `tau verify`, not drift).
218    ///
219    /// Added in lockfile schema v3.
220    #[serde(default)]
221    pub binary_sha256: String,
222    /// Required [`CapabilityShape`]s per plugin, populated at install
223    /// time by Layer 2 cross-check (manifest declarations → required
224    /// shapes). Empty vec on lockfiles older than v4 (auto-upgraded
225    /// with a warn logged via [`tracing::warn`]).
226    ///
227    /// Added in lockfile schema v4.
228    #[serde(default)]
229    pub required_shapes: Vec<CapabilityShape>,
230}

268/// Recorded install-time metadata for a `kind = "skill"` package.
269///
270/// Written by [`crate::install_with_options`] when the installed
271/// package's manifest has `kind = "skill"` and the SKILL.md
272/// validation in [`crate::skill_check`] passes. Consumed by:
273/// - `tau verify` (this crate) — compares `content_sha256` against
274///   the re-hashed SKILL.md to detect drift.
275/// - Skills-3 (`tau skill list / show`) — reads `frontmatter` for
276///   the summary view.

283pub struct LockedSkill {

285    /// encoded (lowercase). Empty for v4-leftover entries (informational
286    /// `unverified` status from `tau verify`, not drift).
287    pub content_sha256: String,
288
289    /// Snapshot of SKILL.md frontmatter (name + description).
290    pub frontmatter: SkillFrontmatterSnapshot,
291}

304/// One installed version's lockfile entry.
305///
306/// `rev` is opaque user input (branch name, tag, or 40-char SHA);
307/// `resolved_commit` is the 40-char SHA that `git rev-parse HEAD`
308/// produced after the clone. Together they support reproducible
309/// installs even when the user pinned a moving branch.
310///
311/// `sha256` is the SHA-256 of the installed source tree, computed by
312/// `install_with_options` after source materialization (Task 2).
313/// Empty for v2-leftover entries.
314///
315/// # Example
316///
317/// ```ignore
318/// // `LockedVersion` is `#[non_exhaustive]`; constructed by the install
319/// // lifecycle (Task 10) and consumed by `LockFile` accessors (Task 7).
320/// ```
321#[non_exhaustive]
322#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
323pub struct LockedVersion {
324    /// The version this entry refers to.
325    pub version: Version,
326    /// Branch name, tag, or SHA as supplied by the user (opaque).
327    pub rev: Option<String>,
328    /// Full 40-char commit SHA after `git rev-parse HEAD` at install time.
329    pub resolved_commit: String,
330    /// SHA-256 of the installed source tree at install time. Computed by
331    /// `install_with_options` via `tau_pkg::tree_hash` after source
332    /// materialization. Empty for v2-leftover entries (flagged
333    /// `unverified` by `tau verify`).
334    #[serde(default)]
335    pub sha256: String,
336    /// When this version was installed.
337    #[serde(with = "humantime_serde")]
338    pub installed_at: SystemTime,
339}

341/// Emit a `tracing::warn!` once per process when a plugin is loaded from
342/// a v3 lockfile that has no `required_shapes`.
343///
344/// Uses a single global `std::sync::Once` (not one per plugin name) to
345/// avoid log spam on every `LockFile::load` call during a long-running
346/// session. The first plugin that triggers this path wins the log line;
347/// subsequent missing-shapes plugins are silent. This is intentional —
348/// the generic "re-install to refresh" message is only useful once.
349fn warn_missing_required_shapes(plugin_bin: &str) {
350    // We can't have one Once per plugin name without a global map, which
351    // adds complexity. Instead we use a single global Once: the warning
352    // fires once per process, listing the first plugin that triggers it.
353    // Subsequent missing-shapes plugins are silent. This is the simplest
354    // correct implementation that avoids log spam.
355    static WARN_ONCE: Once = Once::new();
356    let bin = plugin_bin.to_owned();
357    WARN_ONCE.call_once(move || {
358        tracing::warn!(
359            plugin = %bin,
360            "required_shapes missing for plugin {}; falling back to manifest-derived shapes \
361             — re-install to refresh",
362            bin,
363        );
364    });
365}

367/// Emit a once-per-process warning that the lockfile was auto-upgraded
368/// from v4 to v5 (added `LockedSkill` field on `LockedPackage`). Any
369/// skill packages installed before the upgrade will surface as
370/// "unverified" via `tau verify` until reinstalled.
371fn warn_lockfile_pre_v5_once() {
372    use std::sync::Once;
373    static WARN_ONCE: Once = Once::new();
374    WARN_ONCE.call_once(|| {
375        tracing::warn!(
376            name = "tau_pkg.lockfile.v4_to_v5_auto_upgrade",
377            "lockfile auto-upgraded from v4 to v5; skill packages installed before \
378             the upgrade have no cached SKILL.md hash + frontmatter — \
379             re-run `tau install <skill>` to refresh"
380        );
381    });
382}

395impl LockFile {
396    /// Read the lockfile from `path`.
397    ///
398    /// Returns `LockFile::default()` if the file doesn't exist (lazy creation —
399    /// the first install in a scope creates the lockfile via [`Self::save`]).
400    ///
401    /// # Errors
402    ///
403    /// - [`RegistryError::Io`] — the file exists but could not be read.
404    /// - [`RegistryError::Parse`] — the file is not valid TOML or doesn't
405    ///   match the `LockFile` schema.
406    /// - [`RegistryError::SchemaTooNew`] — the file's `schema_version` exceeds
407    ///   [`MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION`].
408    ///
409    /// # Example
410    ///
411    /// ```ignore
412    /// use std::path::Path;
413    /// use tau_pkg::lockfile::LockFile;
414    ///
415    /// let lf = LockFile::load(Path::new("/nonexistent/tau-lock.toml")).unwrap();
416    /// assert!(lf.packages.is_empty()); // lazy creation
417    /// ```
418    pub fn load(path: &Path) -> Result<Self, RegistryError> {
419        match fs::metadata(path) {
420            Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
421                return Ok(LockFile::default());
422            }
423            Err(e) => {
424                return Err(RegistryError::Io {
425                    message: format!("reading lockfile metadata {}: {e}", path.display()),
426                });
427            }
428            Ok(_) => {}
429        }
430
431        let text = fs::read_to_string(path).map_err(|e| RegistryError::Io {
432            message: format!("reading lockfile {}: {e}", path.display()),
433        })?;
434
435        let mut parsed: LockFile = toml::from_str(&text).map_err(|e| RegistryError::Parse {
436            reason: e.to_string(),
437        })?;
438
439        if parsed.schema_version > MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION {
440            return Err(RegistryError::SchemaTooNew {
441                found: parsed.schema_version,
442                supported: MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION,
443            });
444        }
445
446        // Schema migrations — additive. Each `was_pre_vN` flag captures
447        // a lockfile that needs an additive field populated to a
448        // sensible default. `serde(default)` handles the in-memory
449        // population; this block emits the once-per-process warnings
450        // and bumps the recorded schema_version so the next save()
451        // writes the current version.
452        let was_pre_v4 = parsed.schema_version < 4;
453        let was_pre_v5 = parsed.schema_version < 5;
454        let was_pre_v6 = parsed.schema_version < 6;
455        if parsed.schema_version < MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION {
456            parsed.schema_version = MAX_SUPPORTED_LOCKFILE_SCHEMA_VERSION;
457        }
458
459        // v3 → v4 migration: `required_shapes` empty on plugin entries.
460        if was_pre_v4 {
461            for pkg in &parsed.packages {
462                if let Some(plugin) = &pkg.plugin {
463                    if plugin.required_shapes.is_empty() {
464                        warn_missing_required_shapes(&plugin.manifest.bin);
465                    }
466                }
467            }
468        }
469
470        // v4 → v5 migration: skill entries are absent on legacy
471        // lockfiles. Emit a once-per-process warn so users know to
472        // re-run `tau install` on any pre-v5 skill packages.
473        if was_pre_v5 {
474            // We can't distinguish skill packages from non-skill
475            // packages in a v4 lockfile (no kind discriminator on
476            // LockedPackage). Skills-3 will surface "unverified"
477            // status for entries that ARE skills but lack the cached
478            // frontmatter.
479            warn_lockfile_pre_v5_once();
480        }
481
482        // v5 → v6 migration: `synthesized_from` is absent on legacy
483        // lockfiles. The field defaults to `None` via `#[serde(default)]`
484        // which is the correct value for all packages installed before
485        // Skills-5 (they were all installed from tau-native sources).
486        // No warning needed — `None` is the silent, correct default.
487        let _ = was_pre_v6; // migration is purely serde-default; no warn required
488
489        Ok(parsed)
490    }
491
492    /// Atomically write the lockfile to `path`.
493    ///
494    /// Implementation: write-to-temp-then-rename + `sync_all`. Creates the
495    /// parent directory if it doesn't exist. A crash between the write and
496    /// the rename leaves the target either non-existent or fully written —
497    /// never zero bytes.
498    ///
499    /// Note: `generated_at` is set at construction time ([`Self::default`]),
500    /// not at save time. Callers that want a fresh timestamp must set
501    /// `self.generated_at = SystemTime::now()` before calling `save`.
502    ///
503    /// Note: we do not fsync the parent directory after `persist`. On ext4
504    /// (`data=ordered`) and APFS/HFS+ the rename is journaled; a parent
505    /// fsync would be belt-and-suspenders. Revisit if tau-pkg targets
506    /// FAT32 or other non-journaled mounts.
507    ///
508    /// # Errors
509    ///
510    /// - [`RegistryError::Io`] — parent directory creation, temp-file
511    ///   creation, write, fsync, or rename failed.
512    /// - [`RegistryError::Internal`] — TOML serialization failed (should
513    ///   never happen in practice).
514    ///
515    /// # Example
516    ///
517    /// ```ignore
518    /// use std::path::Path;
519    /// use tau_pkg::lockfile::LockFile;
520    ///
521    /// let lf = LockFile::default();
522    /// lf.save(Path::new("/tmp/tau-lock.toml")).unwrap();
523    /// ```
524    pub fn save(&self, path: &Path) -> Result<(), RegistryError> {
525        let parent = path.parent().unwrap_or_else(|| Path::new("."));
526
527        fs::create_dir_all(parent).map_err(|e| RegistryError::Io {
528            message: format!("creating lockfile directory {}: {e}", parent.display()),
529        })?;
530
531        let text = toml::to_string_pretty(self).map_err(|e| RegistryError::Internal {
532            message: format!("lockfile serialization: {e}"),
533        })?;
534
535        let tmp = tempfile::NamedTempFile::new_in(parent).map_err(|e| RegistryError::Io {
536            message: format!("creating temp file in {}: {e}", parent.display()),
537        })?;
538
539        fs::write(tmp.path(), text.as_bytes()).map_err(|e| RegistryError::Io {
540            message: format!("writing temp lockfile {}: {e}", tmp.path().display()),
541        })?;
542
543        tmp.as_file().sync_all().map_err(|e| RegistryError::Io {
544            message: format!("fsync lockfile {}: {e}", tmp.path().display()),
545        })?;
546
547        tmp.persist(path).map_err(|e| RegistryError::Io {
548            message: format!(
549                "persisting lockfile {} -> {}: {}",
550                e.file.path().display(),
551                path.display(),
552                e.error,
553            ),
554        })?;
555
556        Ok(())
557    }

559    /// Find a package by name.
560    ///
561    /// Linear scan; O(n) with tiny n (packages per scope).
562    /// Returns `None` if the package is not in the lockfile.
563    ///
564    /// # Example
565    ///
566    /// ```ignore
567    /// use tau_pkg::lockfile::LockFile;
568    ///
569    /// let lf = LockFile::default();
570    /// let name: tau_domain::PackageName = "acme-tool".parse().unwrap();
571    /// assert!(lf.find(&name).is_none());
572    /// ```
573    pub fn find(&self, name: &PackageName) -> Option<&LockedPackage> {
574        self.packages.iter().find(|p| p.name == *name)
575    }
576
577    /// Insert or update a package entry.
578    ///
579    /// If a package with the same name already exists, it is replaced
580    /// **in place** (preserving insertion order for other packages).
581    /// Otherwise the package is appended.
582    ///
583    /// Used by `install()` (Task 10).
584    ///
585    /// # Example
586    ///
587    /// ```ignore
588    /// use tau_pkg::lockfile::LockFile;
589    ///
590    /// let mut lf = LockFile::default();
591    /// // lf.upsert(pkg);  // pkg: LockedPackage
592    /// ```
593    pub fn upsert(&mut self, package: LockedPackage) {
594        if let Some(existing) = self.packages.iter_mut().find(|p| p.name == package.name) {
595            *existing = package;
596        } else {
597            self.packages.push(package);
598        }
599    }
600
601    /// Remove a package entry by name.
602    ///
603    /// Returns the removed entry if present, `None` otherwise.
604    /// Preserves insertion order of the remaining entries
605    /// (`Vec::remove`, not `swap_remove`).
606    ///
607    /// Used by `uninstall()` (Task 11).
608    ///
609    /// # Example
610    ///
611    /// ```ignore
612    /// use tau_pkg::lockfile::LockFile;
613    ///
614    /// let mut lf = LockFile::default();
615    /// let name: tau_domain::PackageName = "acme-tool".parse().unwrap();
616    /// assert!(lf.remove(&name).is_none());
617    /// ```
618    pub fn remove(&mut self, name: &PackageName) -> Option<LockedPackage> {
619        let pos = self.packages.iter().position(|p| p.name == *name)?;
620        Some(self.packages.remove(pos))
621    }
622}
