1//! [`Configure`] trait for plugins that consume static config from the
2//! handshake.
3//!
4//! Plugin authors who need to read the host's `HandshakeRequest.config`
5//! JSON value implement [`Configure`] in addition to their port trait
6//! (e.g. [`tau_ports::LlmBackend`]) and call one of the
7//! `run_*_with_config` runner flavors. The runner deserializes the
8//! handshake `config` field as `T::Config` and constructs the plugin
9//! via [`Configure::from_config`] before entering the dispatch loop.
10//!
11//! See `docs/superpowers/specs/2026-04-28-plugin-loading-design.md`
12//! §5.4.
13
14use thiserror::Error;
15
16/// Errors raised by [`Configure::from_config`] implementations.
17///
18/// `#[non_exhaustive]`: additive variants are non-breaking. Plugins
19/// should return one of the typed variants below; there is no generic
20/// `Internal` escape hatch.
21///
22/// # Example
23///
24/// ```ignore
25/// use tau_plugin_sdk::ConfigError;
26/// let err = ConfigError::MissingField("api_key");
27/// assert!(format!("{err}").contains("missing"));
28/// ```
29#[non_exhaustive]
30#[derive(Debug, Error)]
31pub enum ConfigError {
32    /// JSON config payload could not be deserialized as the plugin's
33    /// `Configure::Config` type.
34    #[error("config decode failed: {0}")]
35    Decode(#[from] serde_json::Error),
36
37    /// A required config field was missing.
38    #[error("missing required config field: {0}")]
39    MissingField(&'static str),
40
41    /// A config field had an unsupported value.
42    #[error("invalid value for config field {field}: {detail}")]
43    InvalidValue {
44        /// The name of the offending field.
45        field: &'static str,
46        /// Human-readable explanation of why the value was rejected.
47        detail: String,
48    },
49
50    /// A required environment variable was missing or malformed.
51    /// Distinct from [`ConfigError::MissingField`]: the variant carries
52    /// the env-var name as a runtime `String` so plugins with
53    /// customizable env-var-name configuration (e.g. an Anthropic
54    /// plugin reading `api_key_env: String` from handshake config) can
55    /// surface the actual name in the error message.
56    #[error("env var {name} unusable: {detail}")]
57    InvalidEnvVar {
58        /// Name of the environment variable that was checked.
59        name: String,
60        /// Human-readable explanation of why it was unusable.
61        detail: String,
62    },
63}
64
65/// Trait implemented by plugins that consume the handshake's `config`
66/// field. The runner deserializes the JSON config as
67/// [`Configure::Config`] and calls [`Configure::from_config`] before
68/// entering the dispatch loop.
69///
70/// Plugins that don't need static config call [`crate::run_llm_backend`]
71/// / [`crate::run_tool`] directly with a pre-constructed instance and
72/// don't need to implement this trait.
73///
74/// # Example
75///
76/// ```ignore
77/// use serde::Deserialize;
78/// use tau_plugin_sdk::{ConfigError, Configure};
79///
80/// #[derive(Deserialize)]
81/// struct MyConfig {
82///     api_key: String,
83/// }
84///
85/// struct MyPlugin {
86///     api_key: String,
87/// }
88///
89/// impl Configure for MyPlugin {
90///     type Config = MyConfig;
91///     fn from_config(config: Self::Config) -> Result<Self, ConfigError> {
92///         if config.api_key.is_empty() {
93///             return Err(ConfigError::MissingField("api_key"));
94///         }
95///         Ok(MyPlugin { api_key: config.api_key })
96///     }
97/// }
98/// ```
99pub trait Configure: Sized {
100    /// The plugin author's config shape. Deserialized from
101    /// [`tau_plugin_protocol::HandshakeRequest::config`] before the
102    /// runner enters the dispatch loop.
103    type Config: serde::de::DeserializeOwned;
104
105    /// Construct the plugin from the deserialized config.
106    ///
107    /// # Errors
108    ///
109    /// Return [`ConfigError::MissingField`] for required fields that
110    /// were absent or [`ConfigError::InvalidValue`] for fields that
111    /// were present but unacceptable.
112    fn from_config(config: Self::Config) -> Result<Self, ConfigError>;
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118
119    #[test]
120    fn invalid_env_var_displays_name_and_detail() {
121        let err = ConfigError::InvalidEnvVar {
122            name: "MY_ORG_ANTHROPIC_KEY".into(),
123            detail: "not set in environment".into(),
124        };
125        let s = format!("{err}");
126        assert!(s.contains("MY_ORG_ANTHROPIC_KEY"));
127        assert!(s.contains("not set in environment"));
128    }
129
130    #[test]
131    fn invalid_env_var_pattern_matches() {
132        let err = ConfigError::InvalidEnvVar {
133            name: "FOO".into(),
134            detail: "bar".into(),
135        };
136        assert!(matches!(
137            err,
138            ConfigError::InvalidEnvVar { ref name, .. } if name == "FOO"
139        ));
140    }
