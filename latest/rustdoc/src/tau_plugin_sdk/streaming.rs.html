1//! Helper: turn a `Stream<Item = Result<CompletionChunk, LlmError>>`
2//! into a series of `stream.chunk` notifications carrying the
3//! originating msgid, plus a final [`Frame::Response`] carrying
4//! `{ stop_reason, usage }` or an error envelope.
5//!
6//! See `docs/superpowers/specs/2026-04-28-plugin-loading-design.md`
7//! §4.6 for the wire-level streaming protocol.
8
9use futures_util::StreamExt;
10use serde::Serialize;
11use tau_plugin_protocol::{
12    error::{RpcErrorEnvelope, INTERNAL_ERROR},
13    Frame, FramedWriter,
14};
15use tokio::io::AsyncWrite;
16use tokio::pin;
17
18use crate::error::SdkError;
19
20/// Final summary serialized into the terminal `Frame::Response::result`
21/// at the end of a streaming dispatch.
22#[derive(Debug, Clone, Serialize)]
23struct StreamSummary {
24    stop_reason: Option<tau_ports::StopReason>,
25    usage: Option<tau_ports::TokenUsage>,
26}
27
28/// Pump an LLM completion stream out as `stream.chunk` notifications,
29/// terminating with a [`Frame::Response`] carrying the final summary
30/// or an error envelope.
31///
32/// The `msgid` is the originating `llm.stream` request id.
33/// Each [`tau_ports::CompletionChunk`] yielded by the stream is encoded
34/// as the second element of a `(msgid, chunk)` `stream.chunk`
35/// notification. When the stream ends cleanly, a single
36/// [`Frame::Response`] is sent carrying a JSON-shaped summary payload
37/// (`{ stop_reason, usage }`); both fields default to `None` if no
38/// terminal [`tau_ports::CompletionChunk::Finish`] was observed. If
39/// the stream yields an error, that final response carries an error
40/// envelope with code [`INTERNAL_ERROR`] and the dispatch terminates
41/// without exhausting any further items.
42pub async fn stream_completion<W>(
43    writer: &mut FramedWriter<W>,
44    msgid: u32,
45    stream: tau_ports::CompletionStream,
46) -> Result<(), SdkError>
47where
48    W: AsyncWrite + Unpin,
49{
50    pin!(stream);
51    let mut final_stop_reason: Option<tau_ports::StopReason> = None;
52    let mut final_usage: Option<tau_ports::TokenUsage> = None;
53
54    while let Some(item) = stream.next().await {
55        match item {
56            Ok(chunk) => {
57                // Capture the terminal Finish marker for the final summary.
58                if let tau_ports::CompletionChunk::Finish { stop_reason, usage } = &chunk {
59                    final_stop_reason = Some(*stop_reason);
60                    final_usage = *usage;
61                }
62                // Forward the chunk as a `stream.chunk` notification carrying
63                // the originating msgid + the chunk itself.
64                let params_bytes = rmp_serde::to_vec(&(msgid, &chunk))?;
65                let frame = Frame::Notification {
66                    method: "stream.chunk".to_string(),
67                    params: params_bytes,
68                };
69                writer.write_frame(&frame.encode()?).await?;
70            }
71            Err(llm_err) => {
72                // Stream errored mid-flight: terminate with an error response.
73                let envelope = RpcErrorEnvelope::new(
74                    INTERNAL_ERROR,
75                    format!("llm stream error: {llm_err}"),
76                    None,
77                );
78                let frame = Frame::Response {
79                    id: msgid,
80                    error: Some(envelope),
81                    result: None,
82                };
83                writer.write_frame(&frame.encode()?).await?;
84                return Ok(());
85            }
86        }
87    }
88
89    // Stream ended cleanly: send the final summary response.
90    let summary = StreamSummary {
91        stop_reason: final_stop_reason,
92        usage: final_usage,
93    };
94    let result_bytes = rmp_serde::to_vec(&summary)?;
95    let frame = Frame::Response {
96        id: msgid,
97        error: None,
98        result: Some(result_bytes),
99    };
100    writer.write_frame(&frame.encode()?).await?;
101    Ok(())
102}
