1//! Baseline SBPL allowlist.
2//!
3//! Every macOS process needs a minimum set of system path reads + Mach
4//! lookups to bootstrap libc + dyld. Without these, even `/bin/echo` fails
5//! to start under `(deny default)`.
6//!
7//! These rules were discovered empirically by running progressively-stricter
8//! profiles against `/bin/echo` and adding rules until exec succeeded.
9//!
10//! Lock-step with macOS releases: if a new macOS version moves a system
11//! library, integration tests will fail loudly and we update the baseline.
12
13/// Minimum SBPL rule block plugins need before any plan-derived rules.
14///
15/// Notes on each block:
16/// - `process*` allows the plugin to fork/exec children. sandbox-exec needs
17///   `process-exec` to launch the plugin itself; `process-fork` for any
18///   sub-shells the plugin spawns; etc. Children inherit the same profile,
19///   so this doesn't widen the sandbox.
20/// - `mach-lookup` is the single most important allow rule — Mach is
21///   macOS's IPC primitive; libc bootstrap fails immediately if it can't
22///   look up the bootstrap server.
23/// - `sysctl-read` is needed by libc to determine CPU/RAM/uname info.
24/// - `signal (target self)` lets the plugin signal itself (panic handler).
25/// - `file-read*` over the dyld + libc paths covers the dynamic linker's
26///   needs.
27/// - `/private/tmp` is a symlink target on macOS that resolves to /tmp;
28///   include both forms so the plugin can write tempfiles.
29pub const SBPL_BASELINE: &str = r#"
30;; ---- baseline: bootstrap libc + dyld ----
31(allow process*)
32(allow signal (target self))
33(allow sysctl-read)
34(allow mach-lookup)
35(allow ipc-posix-shm)
36(allow iokit-open)
37(allow file-ioctl)
38
39;; dyld + libc system paths
40(allow file-read*
41  (subpath "/usr/lib")
42  (subpath "/usr/share")
43  (subpath "/usr/libexec")
44  (subpath "/System/Library")
45  (subpath "/Library/Apple/System/Library")
46  (subpath "/Library/Apple/usr"))
47
48(allow file-read-data
49  (literal "/")
50  (literal "/private")
51  (literal "/private/etc")
52  (literal "/private/etc/localtime")
53  (literal "/private/etc/hosts")
54  (literal "/private/etc/resolv.conf")
55  (literal "/dev/null")
56  (literal "/dev/random")
57  (literal "/dev/urandom")
58  (literal "/dev/dtracehelper"))
59
60(allow file-read-metadata)
61"#;
