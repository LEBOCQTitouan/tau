1//! tau-net-bridge — proxy bridge running inside the strict-tier child's
2//! empty network namespace (sub-project H, ADR-0020).
3//!
4//! Workflow:
5//!   1. Bring `lo` up (rtnetlink); empty netns starts with lo DOWN
6//!   2. Bind a TCP listener on 127.0.0.1:8443
7//!   3. fork(2): parent of fork = bridge runtime; child of fork = plugin
8//!   4. Bridge: for each accepted TCP conn, dial the proxy Unix socket;
9//!      splice bytes both ways
10//!   5. When the plugin exits, the bridge exits with the same status
11
12#![cfg_attr(not(target_os = "linux"), allow(unused))]
13
14#[cfg(target_os = "linux")]
15fn main() -> std::io::Result<()> {
16    use std::net::{Ipv4Addr, SocketAddrV4, TcpListener};
17    use std::os::unix::process::CommandExt;
18    use std::process::Command;
19
20    let args: Vec<String> = std::env::args().collect();
21    let parsed = parse_args(&args)
22        .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidInput, format!("args: {e}")))?;
23
24    bring_lo_up()?;
25
26    let bind_addr = SocketAddrV4::new(Ipv4Addr::LOCALHOST, parsed.listen_port);
27    let listener = TcpListener::bind(bind_addr)?;
28    listener.set_nonblocking(false)?;
29
30    // SAFETY: fork is the canonical Unix way to split the bridge from the
31    // plugin process; both inherit fds and the netns/userns from us.
32    let pid = unsafe { libc::fork() };
33    if pid < 0 {
34        return Err(std::io::Error::last_os_error());
35    }
36    if pid == 0 {
37        // Child of fork: replace this process with the plugin
38        let mut cmd = Command::new(&parsed.plugin_argv[0]);
39        cmd.args(&parsed.plugin_argv[1..]);
40        // env was set by the parent (tau) before spawn — we inherit it.
41        return Err(cmd.exec());
42    }
43
44    // Parent of fork: run the bridge loop until the plugin exits
45    let proxy_sock = parsed.proxy_sock_path.clone();
46    std::thread::spawn(move || run_bridge_loop(listener, &proxy_sock));
47
48    // Wait for the plugin to exit; propagate its exit code
49    let mut status: libc::c_int = 0;
50    let waited = unsafe { libc::waitpid(pid, &mut status, 0) };
51    if waited < 0 {
52        return Err(std::io::Error::last_os_error());
53    }
54    if libc::WIFEXITED(status) {
55        std::process::exit(libc::WEXITSTATUS(status));
56    } else if libc::WIFSIGNALED(status) {
57        std::process::exit(128 + libc::WTERMSIG(status));
58    } else {
59        std::process::exit(1);
60    }
61}

69// Args and parse_args are pure logic — no Linux-specific types — so they
70// compile on all platforms, enabling the unit tests to run on macOS too.
71struct Args {
72    proxy_sock_path: std::path::PathBuf,
73    listen_port: u16,
74    plugin_argv: Vec<String>,
75}
76
77fn parse_args(argv: &[String]) -> Result<Args, &'static str> {
78    let mut proxy_sock = None;
79    let mut listen_addr = None;
80    let mut plugin_argv = Vec::new();
81    let mut i = 1;
82    while i < argv.len() {
83        let arg = &argv[i];
84        if let Some(v) = arg.strip_prefix("--proxy-sock=") {
85            proxy_sock = Some(std::path::PathBuf::from(v));
86        } else if let Some(v) = arg.strip_prefix("--listen=") {
87            listen_addr = Some(v.to_string());
88        } else if arg == "--" {
89            plugin_argv.extend(argv[i + 1..].iter().cloned());
90            break;
91        } else {
92            return Err("unexpected arg");
93        }
94        i += 1;
95    }
96    let proxy_sock_path = proxy_sock.ok_or("missing --proxy-sock=")?;
97    let listen = listen_addr.ok_or("missing --listen=")?;
98    let port: u16 = listen
99        .rsplit_once(':')
100        .and_then(|(_, p)| p.parse().ok())
101        .ok_or("invalid --listen= addr")?;
102    if plugin_argv.is_empty() {
103        return Err("missing -- <plugin> <args>");
104    }
105    Ok(Args {
106        proxy_sock_path,
107        listen_port: port,
108        plugin_argv,
109    })
110}

112#[cfg(target_os = "linux")]
113fn bring_lo_up() -> std::io::Result<()> {
114    use futures::TryStreamExt;
115    use rtnetlink::{new_connection, LinkUnspec};
116    use tokio::runtime::Builder;
117
118    // Best-effort: empty netns from Native adapter starts with lo down and we
119    // have CAP_NET_ADMIN-in-userns so set-up succeeds. Inside Docker containers
120    // (Container adapter), lo is already up but we don't have CAP_NET_ADMIN
121    // and the set-up call fails with EPERM. Either way, the listener bind on
122    // 127.0.0.1:8443 below is the load-bearing check: if lo isn't actually
123    // usable, the bind fails loudly. Don't fail the bridge if we can't bring
124    // lo up — log a warning and let the bind try.
125    let rt = Builder::new_current_thread().enable_all().build()?;
126    let attempt: std::io::Result<()> = rt.block_on(async {
127        let (connection, handle, _) = new_connection().map_err(std::io::Error::other)?;
128        tokio::spawn(connection);
129        let mut links = handle.link().get().match_name("lo".to_string()).execute();
130        let link = links
131            .try_next()
132            .await
133            .map_err(std::io::Error::other)?
134            .ok_or_else(|| std::io::Error::other("lo not found"))?;
135        handle
136            .link()
137            .set(LinkUnspec::new_with_index(link.header.index).up().build())
138            .execute()
139            .await
140            .map_err(std::io::Error::other)?;
141        Ok(())
142    });
143    if let Err(e) = attempt {
144        eprintln!("bridge: bring lo up failed (continuing — lo may already be up): {e}");
145    }
146    Ok(())
147}

149#[cfg(target_os = "linux")]
150fn run_bridge_loop(listener: std::net::TcpListener, proxy_sock: &std::path::Path) {
151    use std::os::unix::net::UnixStream;
152    for tcp_conn in listener.incoming() {
153        let Ok(tcp_conn) = tcp_conn else { continue };
154        let proxy_conn = match UnixStream::connect(proxy_sock) {
155            Ok(c) => c,
156            Err(e) => {
157                eprintln!("bridge: proxy connect failed: {e}");
158                continue;
159            }
160        };
161        std::thread::spawn(move || splice_bidirectional(tcp_conn, proxy_conn));
162    }
163}

165#[cfg(target_os = "linux")]
166fn splice_bidirectional(tcp: std::net::TcpStream, unix: std::os::unix::net::UnixStream) {
167    let tcp_clone = match tcp.try_clone() {
168        Ok(c) => c,
169        Err(_) => return,
170    };
171    let unix_clone = match unix.try_clone() {
172        Ok(c) => c,
173        Err(_) => return,
174    };
175    let h1 = std::thread::spawn(move || {
176        let mut a = &tcp;
177        let mut b = &unix;
178        let _ = std::io::copy(&mut a, &mut b);
179    });
180    let h2 = std::thread::spawn(move || {
181        let mut a = &unix_clone;
182        let mut b = &tcp_clone;
183        let _ = std::io::copy(&mut a, &mut b);
184    });
185    let _ = h1.join();
186    let _ = h2.join();
187}

193    #[test]
194    fn parse_args_well_formed() {
195        let argv: Vec<String> = vec![
196            "tau-net-bridge".into(),
197            "--proxy-sock=/tmp/x.sock".into(),
198            "--listen=127.0.0.1:8443".into(),
199            "--".into(),
200            "/usr/bin/plugin".into(),
201            "arg1".into(),
202        ];
203        let parsed = parse_args(&argv).expect("parse");
204        assert_eq!(
205            parsed.proxy_sock_path,
206            std::path::PathBuf::from("/tmp/x.sock")
207        );
208        assert_eq!(parsed.listen_port, 8443);
209        assert_eq!(parsed.plugin_argv, vec!["/usr/bin/plugin", "arg1"]);
